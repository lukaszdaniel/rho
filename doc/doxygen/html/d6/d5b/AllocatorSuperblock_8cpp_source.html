48 void rho::AllocatorSuperblock::allocateArena() {
49  void* arena = nullptr;
50  size_t space = s_arenasize; // Total acquired arena space.
51  if (posix_memalign(&arena, s_small_superblock_size, space) != 0) {
52  allocerr("failed to allocate small-object arena");
53  }
54  size_t num_superblock = space / s_small_superblock_size;
55  arena_superblock_start = reinterpret_cast<uintptr_t>(arena);
56  arena_superblock_end = arena_superblock_start
57  + num_superblock * s_small_superblock_size;
58  arena_superblock_next = arena_superblock_start;
59 
60  // Poison the whole small object arena.
61  ASAN_POISON_MEMORY_REGION(reinterpret_cast<void*>(arena), space);
62 }

64 rho::AllocatorSuperblock* rho::AllocatorSuperblock::newSuperblockFromArena(
65  unsigned block_size) {
66  if (arena_superblock_next >= arena_superblock_end) {
67  return nullptr;
68  }
69  void* pointer = reinterpret_cast<void*>(arena_superblock_next);
70  // The whole arena is poisoned on allocation, now we just unpoison this
71  // superblock header.
72  ASAN_UNPOISON_MEMORY_REGION(pointer, s_superblock_header_size);
73  unsigned superblock_size =
74  (s_small_superblock_size - s_superblock_header_size) / block_size;
75  unsigned bitset_entries = (superblock_size + 63) / 64;
76  AllocatorSuperblock* superblock =
77  new (pointer)AllocatorSuperblock(
78  sizeClassFromBlockSize(block_size), bitset_entries);
79  arena_superblock_next += s_small_superblock_size;
80  return superblock;
81 }

83 rho::AllocatorSuperblock* rho::AllocatorSuperblock::newLargeSuperblock(
84  unsigned size_log2) {
85  void* memory = new double[s_large_superblock_size / sizeof(double)];
86  unsigned bitset_entries =
87  ((1 << (s_large_superblock_size_log2 - size_log2)) + 63) / 64;
88  AllocatorSuperblock* superblock = new (memory)AllocatorSuperblock(
89  sizeClassFromSizeLog2(size_log2), bitset_entries);
90  GCNodeAllocator::s_alloctable->insertSuperblock(superblock,
91  s_large_superblock_size_log2);
92  // Poison all blocks in the superblock. They are unpoisoned one at a time
93  // later, when allocated.
94  ASAN_POISON_MEMORY_REGION(
95  reinterpret_cast<void*>(superblock->firstBlockPointer()),
96  s_large_superblock_size - s_superblock_header_size);
97  return superblock;
98 }

100 rho::AllocatorSuperblock* rho::AllocatorSuperblock::arenaSuperblockFromPointer(
101  uintptr_t candidate) {
102  if (candidate >= arena_superblock_start && candidate < arena_superblock_next) {
103  if ((candidate & (s_small_superblock_size - 1)) < s_superblock_header_size) {
104  // The pointer points inside the superblock header.
105  return nullptr;
106  } else {
107  // The pointer points inside a block in the superblock.
108  return reinterpret_cast<AllocatorSuperblock*>(
109  candidate & ~uintptr_t{s_small_superblock_size - 1});
110  }
111  } else {
112  return nullptr;
113  }
114 }

124 void* rho::AllocatorSuperblock::lookupBlock(uintptr_t candidate) const {
125  uintptr_t first_block = firstBlockPointer();
126  unsigned index = (candidate - first_block) / blockSize();
127  // Check that the block is actually allocated.
128  if (isBlockAllocated(index)) {
129  return reinterpret_cast<char*>(first_block) + index * blockSize();
130  } else {
131  return nullptr;

135 void* rho::AllocatorSuperblock::allocateBlock(size_t block_size) {
136  assert(block_size >= 32 && block_size <= 256
137  && "Only use allocateBlock() to allocate objects between "
138  "32 and 256 bytes.");
139  assert((block_size & 7) == 0
140  && "The size argument must be a multiple of 8 bytes");
141 
142  unsigned size_class = sizeClassFromBlockSize(block_size);
143  AllocatorSuperblock* superblock;
144  if (GCNodeAllocator::s_superblocks[size_class]) {
145  superblock = GCNodeAllocator::s_superblocks[size_class];
146  } else {
147  superblock = newSuperblockFromArena(block_size);
148  if (!superblock) {
149  return nullptr;
150  }
151  GCNodeAllocator::s_superblocks[size_class] = superblock;
152  }
153  return superblock->allocateNextUntouched();
154 }

156 void* rho::AllocatorSuperblock::allocateNextUntouched() {
157  unsigned block_size = blockSize();
158  unsigned num_blocks =
159  (superblockSize() - s_superblock_header_size) / block_size;
160  uint32_t index = m_next_untouched;
161  tagBlockAllocated(index);
162  m_next_untouched += 1;
163  if (m_next_untouched == num_blocks) {
164  GCNodeAllocator::s_superblocks[m_size_class] = nullptr;
165  }
166  void* result = reinterpret_cast<void*>(
167  firstBlockPointer() + (index * block_size));
168  ASAN_UNPOISON_MEMORY_REGION(result, block_size);
169  return result;
170 }

173  uintptr_t block = reinterpret_cast<uintptr_t>(pointer);
174  uintptr_t first_block = firstBlockPointer();
175  unsigned index = (block - first_block) / blockSize();

177  // Mark the block as not allocated.

185 void* rho::AllocatorSuperblock::allocateLarge(unsigned size_log2) {
186  assert(size_log2 >= 6 && size_log2 < GCNodeAllocator::s_num_medium_pools
187  && "Can not allocate objects smaller than "
188  "64 bytes using allocateLarge()");
189  unsigned size_class = sizeClassFromSizeLog2(size_log2);
190  AllocatorSuperblock* superblock;
191  if (GCNodeAllocator::s_superblocks[size_class]) {
192  // Reuse existing superblock for this allocation size.
193  // All superblocks in s_superblocks have at least one untouched block.
194  superblock = GCNodeAllocator::s_superblocks[size_class];
195  } else {
196  // Allocate a new large superblock.
197  superblock = newLargeSuperblock(size_log2);
198  GCNodeAllocator::s_superblocks[size_class] = superblock;
199  }
200  return superblock->allocateNextUntouched();
201 }

224 void rho::AllocatorSuperblock::applyToBlocks(std::function<void(void*)> fun)
225  const {
226  uintptr_t block = firstBlockPointer();
227  unsigned block_size = blockSize();
228  unsigned num_blocks =
229  (superblockSize() - s_superblock_header_size) / block_size;
230  unsigned int bitset_entries = (num_blocks + 63) / 64;
231  for (unsigned int i = 0; i < bitset_entries; ++i) {
232  if (m_free[i] != ~0ull) {
233  for (int index = 0; index < 64; ++index) {
234  if (!(m_free[i] & (uint64_t{1} << index))) {
235 #ifdef ALLOCATION_CHECK
236  // Extra consistency check.
237  if (!GCNodeAllocator::lookupPointer(reinterpret_cast<void*>(block))) {
238  allocerr("apply to all blocks iterating over non-alloc'd pointer");
239  }
240 #endif
241  fun(reinterpret_cast<void*>(block));
242  }
243  block += block_size;
244  }
245  } else {
246  block += block_size * 64;
247  }
248  }
249 }

259 void rho::AllocatorSuperblock::printSummary() const {
260  unsigned superblock_size =
261  (superblockSize() - s_superblock_header_size) / blockSize();
262  printf("Superblock Summary (blocksize=%d, num block=%d):\n",
263  blockSize(), superblockSize());
264  unsigned int bitset_entries = (superblock_size + 63) / 64;
265  for (unsigned int i = 0; i < bitset_entries; ++i) {
266  int num_free = 0;
267  for (int index = 0; index < 64; ++index) {
268  if (m_free[i] & (uint64_t{1} << index)) {
269  num_free += 1;
270  }
271  }
272  switch (num_free) {
273  case 0:
274  printf("#"); // Full.
275  break;
276  case 64:
277  printf("."); // Empty.
278  break;
279  default:
280  printf("%d", (64 - num_free) / 7);
281  break;
282  }
283  if (i + 1 < bitset_entries) {
284  printf(" ");
285  }
286  }
287  printf("\n");
288 }
