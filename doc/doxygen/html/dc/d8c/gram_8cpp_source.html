308 static int mbcs_get_next(int c, wchar_t *wc)
309 {
310  int res; char s[9];
311  size_t clen = 1;
312  mbstate_t mb_st;
313 
314  s[0] = (char) c;
315  /* This assumes (probably OK) that all MBCS embed ASCII as single-byte
316  lead bytes, including control chars */
317  if((unsigned int) c < 0x80) {
318  *wc = (wchar_t) c;
319  return 1;
320  }
321  if(utf8locale) {
322  clen = utf8clen((char) c);
323  for(size_t i = 1; i < clen; i++) {
324  c = xxgetc();
325  if(c == R_EOF) Rf_error(_("EOF whilst reading MBCS char at line %d"), ParseState.xxlineno);
326  s[i] = (char) c;
327  }
328  s[clen] ='\0'; /* x86 Solaris requires this */
329  res = (int) mbrtowc(wc, s, clen, NULL);
330  if(res == -1) Rf_error(_("invalid multibyte character in parser at line %d"), ParseState.xxlineno);
331  } else {
332  /* This is not necessarily correct for stateful MBCS */
333  while(clen <= MB_CUR_MAX) {
334  mbs_init(&mb_st);
335  res = (int) mbrtowc(wc, s, clen, &mb_st);
336  if(res >= 0) break;
337  if(res == -1)
338  Rf_error(_("invalid multibyte character in parser at line %d"), ParseState.xxlineno);
339  /* so res == -2 */
340  c = xxgetc();
341  if(c == R_EOF) Rf_error(_("EOF whilst reading MBCS char at line %d"), ParseState.xxlineno);
342  s[clen++] = (char) c;
343  } /* we've tried enough, so must be complete or invalid by now */
344  }
345  for(size_t i = clen - 1; i > 0; i--) xxungetc(s[i]);
346  return clen;
347 }
