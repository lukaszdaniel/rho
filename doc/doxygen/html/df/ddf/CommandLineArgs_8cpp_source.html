44 /*
45  This copies the command line arguments to the Rstart
46  structure. The memory is obtained from calloc, etc.
47  since these are permanent and it is not intended that
48  they be modified. This is why they are copied before
49  being processed and removed from the list.
50 
51  We might store these as a SEXP. I have no strong opinion
52  about this.
53  */
54 
55 /* Permanent copy of the command line arguments and the number
56  of them passed to the application.
57  These are populated via the routine R_set_command_line_arguments().
58 */
59 static int NumCommandLineArgs = 0;
60 static char **CommandLineArgs = nullptr;
61 
62 
63 void R_set_command_line_arguments(int argc, char **argv)
64 {
65  int i;
66 
67  NumCommandLineArgs = argc;
68  CommandLineArgs = static_cast<char**>(calloc(size_t(argc), sizeof(char*)));
69 
70  for(i = 0; i < argc; i++)
71  CommandLineArgs[i] = strdup(argv[i]);
72 }

75 /*
76  The .Internal which returns the command line arguments that are stored
77  in global variables.
78  */
79 HIDDEN SEXP do_commandArgs(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op)
80 {
81  int i;
82  SEXP vals;
83 
84  /* need protection as mkChar allocates */
85  vals = PROTECT(Rf_allocVector(STRSXP, NumCommandLineArgs));
86  for(i = 0; i < NumCommandLineArgs; i++)
87  SET_STRING_ELT(vals, i, Rf_mkChar(CommandLineArgs[i]));
88  UNPROTECT(1);
89  return vals;
90 }

96 void R_common_command_line(int *pac, char **argv, Rstart Rp)
97 {
98  int ac = *pac, newac = 1; /* argv[0] is process name */
99  char *p, **av = argv, msg[1024];
100  Rboolean processing = TRUE;
101 
102  R_RestoreHistory = 1;
103  while(--ac) {
104  if(processing && **++av == '-') {
105  if (streql(*av, "--version")) {
106  Rf_PrintVersion(msg, 1024);
107  R_ShowMessage(msg);
108  exit(0);
109  }
110  else if(streql(*av, "--args")) {
111  /* copy this through for further processing */
112  argv[newac++] = *av;
113  processing = FALSE;
114  }
115  else if(streql(*av, "--save")) {
116  Rp->SaveAction = SA_SAVE;
117  }
118  else if(streql(*av, "--no-save")) {
119  Rp->SaveAction = SA_NOSAVE;
120  }
121  else if(streql(*av, "--restore")) {
122  Rp->RestoreAction = SA_RESTORE;
123  }
124  else if(streql(*av, "--no-restore")) {
125  Rp->RestoreAction = SA_NORESTORE;
126  R_RestoreHistory = 0;
127  }
128  else if(streql(*av, "--no-restore-data")) {
129  Rp->RestoreAction = SA_NORESTORE;
130  }
131  else if(streql(*av, "--no-restore-history")) {
132  R_RestoreHistory = 0;
133  }
134  else if (streql(*av, "--silent") ||
135  streql(*av, "--quiet") ||
136  streql(*av, "-q")) {
137  Rp->R_Quiet = TRUE;
138  }
139  else if (streql(*av, "--vanilla")) {
140  Rp->SaveAction = SA_NOSAVE; /* --no-save */
141  Rp->RestoreAction = SA_NORESTORE; /* --no-restore */
142  R_RestoreHistory = 0; // --no-restore-history (= part of --no-restore)
143  Rp->LoadSiteFile = FALSE; /* --no-site-file */
144  Rp->LoadInitFile = FALSE; /* --no-init-file */
145  Rp->NoRenviron = TRUE; // --no-environ
146 #ifdef Win32
147  R_LoadRconsole = FALSE;
148 #endif
149  }
150  else if (streql(*av, "--no-environ")) {
151  Rp->NoRenviron = TRUE;
152  }
153  else if (streql(*av, "--verbose")) {
154  Rp->R_Verbose = TRUE;
155  }
156  else if (streql(*av, "--slave") ||
157  streql(*av, "-s")) {
158  Rp->R_Quiet = TRUE;
159  Rp->R_Slave = TRUE;
160  Rp->SaveAction = SA_NOSAVE;
161  }
162  else if (streql(*av, "--no-site-file")) {
163  Rp->LoadSiteFile = FALSE;
164  }
165  else if (streql(*av, "--no-init-file")) {
166  Rp->LoadInitFile = FALSE;
167  }
168  else if (streql(*av, "--debug-init")) {
169  Rp->DebugInitFile = TRUE;
170  }
171  else if (streqln(*av, "--encoding", 10)) {
172  if(strlen(*av) < 12) {
173  if(ac > 1) {ac--; av++; p = *av;} else p = nullptr;
174  } else p = &(*av)[11];
175  if (p == nullptr) {
176  R_ShowMessage(_("WARNING: no value given for --encoding"));
177  } else {
178  strncpy(R_StdinEnc, p, 30);
179  R_StdinEnc[30] = '\0';
180  }
181  }
182 #ifdef Win32
183  else if (streql(*av, "--no-Rconsole")) {
184  R_LoadRconsole = 0;
185  }
186 #endif
187  else if (streql(*av, "-save") ||
188  streql(*av, "-nosave") ||
189  streql(*av, "-restore") ||
190  streql(*av, "-norestore") ||
191  streql(*av, "-noreadline") ||
192  streql(*av, "-quiet") ||
193  streql(*av, "-nsize") ||
194  streql(*av, "-vsize") ||
195  streqln(*av, "--max-nsize", 11) ||
196  streqln(*av, "--max-vsize", 11) ||
197  streql(*av, "-V") ||
198  streql(*av, "-n") ||
199  streql(*av, "-v")) {
200  snprintf(msg, 1024,
201  _("WARNING: option '%s' no longer supported"), *av);
202  R_ShowMessage(msg);
203  }
204  else if (streqln(*av, "--min-vsize", 11)) {
205  if(strlen(*av) < 13) {
206  if(ac > 1) {ac--; av++; p = *av;} else p = nullptr;
207  } else p = &(*av)[12];
208  if (p == nullptr) {
209  snprintf(msg, 1024,
210  _("WARNING: no value given for '%s'"), *av);
211  R_ShowMessage(msg);
212  break;
213  }
214  int ierr;
215  R_size_t value;
216  value = R_Decode2Long(p, &ierr);
217  if(ierr) {
218  if(ierr < 0)
219  snprintf(msg, 1024,
220  _("WARNING: '%s' value is invalid: ignored"),
221  *av);
222  else
223  sprintf(msg,
224  _("WARNING: %s: too large and ignored"),
225  *av);
226  R_ShowMessage(msg);
227 
228  } else {
229  Rp->vsize = value;
230  }
231  }
232  else if(streqln(*av, "--min-nsize", 11) ||
233  streqln(*av, "--max-ppsize", 12)) {
234  // rho silently ignores these parameters for compatibility
235  // with CR.
236  }
237  else { /* unknown -option */
238  argv[newac++] = *av;
239  }
240  }
241  else {
242  argv[newac++] = *av;
243  }
244  }
245  *pac = newac;
246  return;
247 }
