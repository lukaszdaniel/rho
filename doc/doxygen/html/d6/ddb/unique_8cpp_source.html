119 static size_t rhash(SEXP x, R_xlen_t indx, HashData *d)
120 {
121  /* There is a problem with signed 0s under IEC60559 */
122  double tmp = (REAL(x)[indx] == 0.0) ? 0.0 : REAL(x)[indx];
123  /* need to use both 32-byte chunks or endianness is an issue */
124  /* we want all NaNs except NA equal, and all NAs equal */
125  if (R_IsNA(tmp)) tmp = NA_REAL;
126  else if (R_IsNaN(tmp)) tmp = R_NaN;
127 #if 2*SIZEOF_INT == SIZEOF_DOUBLE
128  {
129  union foo tmpu;
130  tmpu.d = tmp;
131  return scatter(tmpu.u[0] + tmpu.u[1], d);
132  }
133 #else
134  return scatter(*((unsigned int *) (&tmp)), d);
135 #endif
136 }

138 static Rcomplex unify_complex_na(Rcomplex z) {
139  Rcomplex ans;
140  ans.r = (z.r == 0.0) ? 0.0 : z.r;
141  ans.i = (z.i == 0.0) ? 0.0 : z.i;
142  if (R_IsNA(ans.r) || R_IsNA(ans.i))
143  ans.r = ans.i = NA_REAL;
144  else if (R_IsNaN(ans.r) || R_IsNaN(ans.i))
145  ans.r = ans.i = R_NaN;
146  return ans;
147 }

149 static size_t chash(SEXP x, R_xlen_t indx, HashData *d)
150 {
151  Rcomplex tmp = unify_complex_na(COMPLEX(x)[indx]);
152 
153 #if 2*SIZEOF_INT == SIZEOF_DOUBLE
154  {
155  unsigned int u;
156  union foo tmpu;
157  tmpu.d = tmp.r;
158  u = tmpu.u[0] ^ tmpu.u[1];
159  tmpu.d = tmp.i;
160  u ^= tmpu.u[0] ^ tmpu.u[1];
161  return scatter(u, d);
162  }
163 #else
164  return scatter((*((unsigned int *)(&tmp.r)) ^
165  (*((unsigned int *)(&tmp.i)))), d);
166 #endif
167 }

171 static size_t cshash(SEXP x, R_xlen_t indx, HashData *d)
172 {
173  intptr_t z = intptr_t( STRING_ELT(x, indx));
174  unsigned int z1 = static_cast<unsigned int>((z & 0xffffffff)), z2 = 0;
175 #if SIZEOF_LONG == 8
176  z2 = static_cast<unsigned int>(z/0x100000000L);
177 #endif
178  return scatter(z1 ^ z2, d);

181 static size_t shash(SEXP x, R_xlen_t indx, HashData *d)
182 {
183  unsigned int k;
184  const char *p;
185  const void *vmax = vmaxget();
186  if(!d->useUTF8 && d->useCache) return cshash(x, indx, d);
187  /* Not having d->useCache really should not happen anymore. */
188  p = Rf_translateCharUTF8(STRING_ELT(x, indx));
189  k = 0;
190  while (*p++)
191  k = 11 * k + static_cast<unsigned int>(*p); /* was 8 but 11 isn't a power of 2 */
192  vmaxset(vmax); /* discard any memory used by translateChar */
193  return scatter(k, d);
194 }

220 /* This is differentiating {NA,1}, {NA,0}, {NA, NaN}, {NA, NA},
221  * but R's print() and format() render all as "NA" */
222 static int cplx_eq(Rcomplex x, Rcomplex y)
223 {
224  if (!ISNAN(x.r) && !ISNAN(x.i) && !ISNAN(y.r) && !ISNAN(y.i))
225  return x.r == y.r && x.i == y.i;
226  else if (R_IsNA(x.r) || R_IsNA(x.i)) // x is NA
227  return (R_IsNA(y.r) || R_IsNA(y.i)) ? 1 : 0;
228  else if (R_IsNA(y.r) || R_IsNA(y.i)) // y is NA but x is not
229  return 0;
230  // else : none is NA but there's at least one NaN; hence ISNAN(.) == R_IsNaN(.)
231  return
232  (((ISNAN(x.r) && ISNAN(y.r)) || (!ISNAN(x.r) && !ISNAN(y.r) && x.r == y.r)) && // Re
233  ((ISNAN(x.i) && ISNAN(y.i)) || (!ISNAN(x.i) && !ISNAN(y.i) && x.i == y.i)) // Im
234  ) ? 1 : 0;
235 }

267 static size_t vhash(SEXP x, R_xlen_t indx, HashData *d)
268 {
269  int i;
270  unsigned int key;
271  SEXP _this = VECTOR_ELT(x, indx);
272 
273  key = OBJECT(_this) + 2*TYPEOF(_this) + 100U*static_cast<unsigned int>(Rf_length(_this));
274  /* maybe we should also look at attributes, but that slows us down */
275  switch (TYPEOF(_this)) {
276  case LGLSXP:
277  /* This is not too clever: pack into 32-bits and then scatter? */
278  for(i = 0; i < LENGTH(_this); i++) {
279  key ^= lhash(_this, i, d);
280  key *= 97;
281  }
282  break;
283  case INTSXP:
284  for(i = 0; i < LENGTH(_this); i++) {
285  key ^= ihash(_this, i, d);
286  key *= 97;
287  }
288  break;
289  case REALSXP:
290  for(i = 0; i < LENGTH(_this); i++) {
291  key ^= rhash(_this, i, d);
292  key *= 97;
293  }
294  break;
295  case CPLXSXP:
296  for(i = 0; i < LENGTH(_this); i++) {
297  key ^= chash(_this, i, d);
298  key *= 97;
299  }
300  break;
301  case STRSXP:
302  for(i = 0; i < LENGTH(_this); i++) {
303  key ^= shash(_this, i, d);
304  key *= 97;
305  }
306  break;
307  case RAWSXP:
308  for(i = 0; i < LENGTH(_this); i++) {
309  key ^= scatter(static_cast<unsigned int>(rawhash(_this, i, d)), d);
310  key *= 97;
311  }
312  break;
313  case VECSXP:
314  for(i = 0; i < LENGTH(_this); i++) {
315  key ^= vhash(_this, i, d);
316  key *= 97;
317  }
318  break;
319  default:
320  break;
321  }
322  return scatter(key, d);
323 }

331 /*
332  Choose M to be the smallest power of 2
333  not less than 2*n and set K = log2(M).
334  Need K >= 1 and hence M >= 2, and 2^M <= 2^31 -1, hence n <= 2^29.
335 
336  Dec 2004: modified from 4*n to 2*n, since in the worst case we have
337  a 50% full table, and that is still rather efficient -- see
338  R. Sedgewick (1998) Algorithms in C++ 3rd edition p.606.
339 */
340 static void MKsetup(R_xlen_t n, HashData *d, R_xlen_t nmax)
341 {
342 #ifdef LONG_VECTOR_SUPPORT
343  /* M = 2^32 is safe, hence n <= 2^31 -1 */
344  if(n < 0) /* protect against overflow to -ve */
345  Rf_error(_("length %d is too large for hashing"), n);
346 #else
347  if(n < 0 || n >= 1073741824) /* protect against overflow to -ve */
348  Rf_error(_("length %d is too large for hashing"), n);
349 #endif
350 
351  if (nmax != NA_INTEGER && nmax != 1) n = nmax;
352  size_t n2 = 2U * (size_t) n;
353  d->M = 2;
354  d->K = 1;
355  while (d->M < n2) {
356  d->M *= 2;
357  d->K++;
358  }
359  d->nmax = n;
360 }

362 #define IMAX 4294967296L
363 static void HashTableSetup(SEXP x, HashData *d, R_xlen_t nmax)
364 {
365  d->useUTF8 = FALSE;
366  d->useCache = TRUE;
367  switch (TYPEOF(x)) {
368  case LGLSXP:
369  d->hash = lhash;
370  d->equal = lequal;
371  d->nmax = d->M = 4;
372  d->K = 2; /* unused */
373  break;
374  case INTSXP:
375  {
376  d->hash = ihash;
377  d->equal = iequal;
378 #ifdef LONG_VECTOR_SUPPORT
379  R_xlen_t nn = XLENGTH(x);
380  if (nn > IMAX) nn = IMAX;
381  MKsetup(nn, d, nmax);
382 #else
383  MKsetup(LENGTH(x), d, nmax);
384 #endif
385  }
386  break;
387  case REALSXP:
388  d->hash = rhash;
389  d->equal = requal;
390  MKsetup(XLENGTH(x), d, nmax);
391  break;
392  case CPLXSXP:
393  d->hash = chash;
394  d->equal = cequal;
395  MKsetup(XLENGTH(x), d, nmax);
396  break;
397  case STRSXP:
398  d->hash = shash;
399  d->equal = sequal;
400  MKsetup(XLENGTH(x), d, nmax);
401  break;
402  case RAWSXP:
403  d->hash = rawhash;
404  d->equal = rawequal;
405  d->nmax = d->M = 256;
406  d->K = 8; /* unused */
407  break;
408  case VECSXP:
409  d->hash = vhash;
410  d->equal = vequal;
411  MKsetup(XLENGTH(x), d, nmax);
412  break;
413  default:
414  UNIMPLEMENTED_TYPE("HashTableSetup", x);
415  }
416 #ifdef LONG_VECTOR_SUPPORT
417  d->isLong = RHOCONSTRUCT(Rboolean, IS_LONG_VEC(x));
418  if (d->isLong) {
419  d->HashTable = Rf_allocVector(REALSXP, R_xlen_t(d->M));
420  for (size_t i = 0; i < d->M; i++) REAL(d->HashTable)[i] = NIL;
421  } else
422 #endif
423  {
424  d->HashTable = Rf_allocVector(INTSXP, R_xlen_t(d->M));
425  for (size_t i = 0; i < d->M; i++) INTEGER(d->HashTable)[i] = NIL;
426  }
427 }

433 static int isDuplicated(SEXP x, R_xlen_t indx, HashData *d)
434 {
435 #ifdef LONG_VECTOR_SUPPORT
436  if (d->isLong) {
437  double *h = REAL(d->HashTable);
438  size_t i = d->hash(x, indx, d);
439  while (h[i] != NIL) {
440  if (d->equal(x, R_xlen_t(h[i]), x, indx))
441  return h[i] >= 0 ? 1 : 0;
442  i = (i + 1) % d->M;
443  }
444  if (d->nmax-- < 0) Rf_error("hash table is full");
445  h[i] = double(indx);
446  } else
447 #endif
448  {
449  int *h = INTEGER(d->HashTable);
450  size_t i = d->hash(x, indx, d);
451  while (h[i] != NIL) {
452  if (d->equal(x, h[i], x, indx))
453  return h[i] >= 0 ? 1 : 0;
454  i = (i + 1) % d->M;
455  }
456  if (d->nmax-- < 0) Rf_error("hash table is full");
457  h[i] = int(indx);
458  }
459  return 0;
460 }
