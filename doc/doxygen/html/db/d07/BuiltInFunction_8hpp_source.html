339  template <typename... Args>
340  RObject *invokeNativeCall(const Expression *call, Environment *env,
341  const PairList *tags, Args... args) const
342  {
343  assert(getCallingConvention() == CallingConvention::FixedNative || getCallingConvention() == CallingConvention::VarArgsNative);
344  assert(sizeof...(Args) == arity() || arity() == -1);
345  if (functionNeedsInternalDispatch() && argsHaveClassToDispatchOn(args...))
346  {
347  std::initializer_list<RObject *> args_array = {args...};
348  auto dispatched = RealInternalDispatch(
349  call, args_array.size(), args_array.begin(), tags, env);
350  if (dispatched.first)
351  return dispatched.second;
352  }
353  return invokeNativeCallWithoutDispatch(call, args...);
354  }

356  template <typename... Args>
357  RObject *invokeNativeCallWithoutDispatch(
358  const Expression *call, Args... args) const
359  {
360  assert(sizeof...(Args) == arity() || arity() == -1);
361  if (getCallingConvention() == CallingConvention::FixedNative)
362  {
363  auto fn = reinterpret_cast<RObject *(*)(Expression *,
364  const BuiltInFunction *, Args...)>(m_function.fixed_native);
365  return (*fn)(const_cast<Expression *>(call), this, args...);
366  }
367  else
368  {
369  assert(getCallingConvention() == CallingConvention::VarArgsNative);
370  return (m_function.varargs_native)(const_cast<Expression *>(call), this, sizeof...(Args), args...);
371  }

414  /* @brief Attempt to dispatch this call to a method.
415  *
416  * @param call The call being dispatched.
417  * @param env The environment to call the method from.
418  * @param evaluated_args The arguments to pass to the method.
419  *
420  * @return The first value is whether or not the dispatch succeeded.
421  * The second value is the return value if it did.
422  *
423  * @note If the dispatch fails, then evaluated_args is unchanged.

425  std::pair<bool, RObject *> InternalDispatch(const Expression *call,
426  Environment *env, ArgList &&evaluated_args) const
427  {
428  assert(evaluated_args.status() == ArgList::EVALUATED);
429 
430  // NB: functionNeedsInternalDispatch() might not be true, since
431  // we could be doing dispatch for a special etc.
432  assert(m_dispatch_type != DispatchType::NONE);
433  if (argsHaveClassToDispatchOn(evaluated_args))
434  {
435  return RealInternalDispatch(call, env, std::move(evaluated_args));
436  }
437  return std::make_pair(false, nullptr);

551  bool argsHaveClassToDispatchOn(
552  RObject *const *evaluated_args, int num_args) const
553  {
554  switch (num_args)
555  {
556  case 0:
557  return false;
558  case 1:

561  return argsHaveClassToDispatchOn(
562  evaluated_args[0], evaluated_args[1]);
563  }

566  bool argsHaveClassToDispatchOn(const ArgList &evaluated_args) const
567  {
568  switch (evaluated_args.size())
569  {
570  case 0:
571  return false;
572  case 1:
573  return argsHaveClassToDispatchOn(evaluated_args.get(0));
574  default:
575  return argsHaveClassToDispatchOn(
576  evaluated_args.get(0), evaluated_args.get(1));
577  }
578  }

580  // Internal dispatch is used a lot, but there most of the time
581  // no dispatch is needed because no objects are involved.
582  // This quickly detects those cases.
583  bool argsHaveClassToDispatchOn(
584  const RObject *arg1 = nullptr, const RObject *arg2 = nullptr, ...) const
585  {
586  assert(m_dispatch_type != DispatchType::NONE);
587  if (arg1 && arg1->hasClass())
588  return true;
589  // 'Ops' dispatch on the second argument as well.
590  if (m_dispatch_type == DispatchType::GROUP_OPS && arg2 && arg2->hasClass())
591  return true;
592  return false;
593  }
