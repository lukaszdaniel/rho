38 void StackChecker::checkAvailableStackSpace(size_t required_bytes)
39 {
40  if (R_CStackLimit == (uintptr_t)-1)
41  return;
42 
43  char dummy;
44  intptr_t usage = R_CStackDir * (R_CStackStart - (uintptr_t)&dummy);
45  if (usage + required_bytes > R_CStackLimit - R_CStackLimit / 16)
46  {
47  handleStackSpaceExceeded();
48  }
49 }

51 void StackChecker::setDepthLimit(unsigned int depth)
52 {
53  if (depth < R_MIN_EXPRESSIONS_OPT || depth > R_MAX_EXPRESSIONS_OPT)
54  {
55  Rf_error(_("'expressions' parameter invalid, allowed %d...%d"),
56  R_MIN_EXPRESSIONS_OPT, R_MAX_EXPRESSIONS_OPT);
57  }
58  s_depth_limit = depth;
59 }

61 void StackChecker::handleStackDepthExceeded()
62 {
63  DisableStackCheckingScope no_stack_checking;
64  Rf_errorcall(nullptr, _("evaluation nested too deeply: "
65  "infinite recursion / options(expressions=)?"));
66 }

68 void StackChecker::handleStackSpaceExceeded()
69 {
70  // We'll need to use the remaining stack space for error recovery,
71  // so temporarily disable stack checking.
72  DisableStackCheckingScope no_stack_checking;
73 
74  // Do not translate this, to save stack space.
75  Rf_errorcall(R_NilValue, "C stack usage is too close to the limit");
76 }

78 DisableStackCheckingScope::DisableStackCheckingScope()
79 {
80  m_previous_limit = StackChecker::depthLimit();
81  StackChecker::setDepthLimit(R_MAX_EXPRESSIONS_OPT);

83  m_previous_stack_limit = R_CStackLimit;
