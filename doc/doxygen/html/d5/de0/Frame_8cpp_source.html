336 Frame::Binding* Frame::obtainBinding(const Symbol* symbol)
337 {
338  Frame::Binding* binding = nullptr;
339  if (isLocked()) {
340  // If the frame is locked, we can only return pre-existing bindings.
341  Frame::Binding* binding = Frame::binding(symbol);
342  if (!binding) {
343  Rf_error(_("cannot add bindings to a locked frame"));
344  }
345  return binding;
346  }
347 
348  if (m_descriptor)
349  {
350  // Use the pressigned location.
351  int location = m_descriptor->getLocation(symbol);
352  if (location != -1) {
353  binding = m_bindings + location;
354  m_used_bindings_size = std::max(m_used_bindings_size,
355  (unsigned char)(location + 1));
356  }
357  } else
358  {
359  // If the binding exists, return that.
360  binding = Frame::binding(symbol);
361  if (binding)
362  return binding;
363 
364  // Otherwise return the first unused space in the array if any.
365  for (unsigned char i = 0; i < m_bindings_size; ++i) {
366  if (!m_bindings[i].isSet()) {
367  // Found an unused spot.
368  m_used_bindings_size = std::max(m_used_bindings_size,
369  (unsigned char)(i + 1));
370  binding = &m_bindings[i];
371  break;
372  }
373  }
374  }
375 
376  // If all else fails, go to the overflow.
377  if (!binding) {
378  if (!m_overflow) {
379  m_overflow = new map;
380  }
381  binding = &((*m_overflow)[symbol]);
382  }
383 
384  if (!binding->frame()) {
385  initializeBinding(binding, symbol);
386  }
387  return binding;
388 }

461  bool isMissingArgument(const Symbol* sym, Frame* frame)
462  {
463  RObject* rawval;
464  if (sym->isDotDotSymbol()) {
465  unsigned int ddv = sym->dotDotIndex();
466  Frame::Binding* bdg = frame->binding(DotsSymbol);
467  if (!bdg)
468  return false; // This is what CR does. Is it really right?
469  ConsCell* cc = SEXP_downcast<ConsCell*>(bdg->rawValue());
470  while (cc && ddv > 1) {
471  cc = cc->tail();
472  --ddv;
473  }
474  if (!cc)
475  return true;
476  rawval = cc->car();
477  } else {
478  // Not a ..n symbol:
479  if (sym == Symbol::missingArgument())
480  return true;
481  Frame::Binding* bdg = frame->binding(sym);
482  if (!bdg)
483  return false;
484  rawval = bdg->rawValue();
485  if (bdg->origin() == Frame::Binding::MISSING
486  || rawval == Symbol::missingArgument())
487  return true;
488  if (bdg->isActive())
489  return false;
490  }
491  if (rawval && rawval->sexptype() == PROMSXP) {
492  Promise* prom = static_cast<Promise*>(rawval);
493  return prom->isMissingSymbol();
494  }
495  return false;
496  }
