187 /* FIXME: consider using
188  tmp = (LDOUBLE)x1 - floor(q) * (LDOUBLE)x2;
189  */
190 static double myfmod(double x1, double x2)
191 {
192  if (x2 == 0.0) return R_NaN;
193  double q = x1 / x2, tmp = x1 - floor(q) * x2;
194  if(R_FINITE(q) && (fabs(q) > 1/R_AccuracyInfo.eps))
195  warning(_("probable complete loss of accuracy in modulus"));
196  q = floor(tmp/x2);
197  return tmp - q * x2;
198 }

200 static double myfloor(double x1, double x2)
201 {
202  double q = x1 / x2, tmp;
203 
204  if (x2 == 0.0) return q;
205  tmp = x1 - floor(q) * x2;
206  return floor(q) + floor(tmp/x2);
207 }

209 double R_pow(double x, double y) /* = x ^ y */
210 {
211  /* squaring is the most common of the specially handled cases so
212  check for it first. */
213  if(y == 2.0)
214  return x * x;
215  if(x == 1. || y == 0.)
216  return(1.);
217  if(x == 0.) {
218  if(y > 0.) return(0.);
219  else if(y < 0) return(R_PosInf);
220  else return(y); /* NA or NaN, we assert */
221  }
222  if (R_FINITE(x) && R_FINITE(y)) {
223  /* There was a special case for y == 0.5 here, but
224  gcc 4.3.0 -g -O2 mis-compiled it. Showed up with
225  100^0.5 as 3.162278, example(pbirthday) failed. */
226 #ifdef USE_POWL_IN_R_POW
227  return powl(x, y);
228 #else
229  return pow(x, y);
230 #endif
231  }
232  if (ISNAN(x) || ISNAN(y))
233  return(x + y);
234  if(!R_FINITE(x)) {
235  if(x > 0) /* Inf ^ y */
236  return (y < 0.)? 0. : R_PosInf;
237  else { /* (-Inf) ^ y */
238  if(R_FINITE(y) && y == floor(y)) /* (-Inf) ^ n */
239  return (y < 0.) ? 0. : (myfmod(y, 2.) != 0 ? x : -x);
240  }
241  }
242  if(!R_FINITE(y)) {
243  if(x >= 0) {
244  if(y > 0) /* y == +Inf */
245  return (x >= 1) ? R_PosInf : 0.;
246  else /* y == -Inf */
247  return (x < 1) ? R_PosInf : 0.;
248  }
249  }
250  return R_NaN; // all other cases: (-Inf)^{+-Inf, non-int}; (neg)^{+-Inf}
251 }

253 double R_pow_di(double x, int n)
254 {
255  double xn = 1.0;
256 
257  if (ISNAN(x)) return x;
258  if (n == NA_INTEGER) return NA_REAL;
259 
260  if (n != 0) {
261  if (!R_FINITE(x)) return R_POW(x, double(n));
262 
263  bool is_neg = (n < 0);
264  if(is_neg) n = -n;
265  for(;;) {
266  if(n & 01) xn *= x;
267  if(n >>= 1) x *= x; else break;
268  }
269  if(is_neg) xn = 1. / xn;
270  }
271  return xn;
272 }

344 SEXP attribute_hidden R_binary(SEXP call, SEXP op, SEXP xarg, SEXP yarg)
345 {
346  ARITHOP_TYPE oper = ARITHOP_TYPE( PRIMVAL(op));
347 
348  GCStackRoot<VectorBase> x(FIXUP_NULL_AND_CHECK_TYPES(xarg));
349  GCStackRoot<VectorBase> y(FIXUP_NULL_AND_CHECK_TYPES(yarg));
350  Rboolean xarray = Rf_isArray(x), yarray = Rf_isArray(y);
351  R_xlen_t nx = XLENGTH(x);
352  R_xlen_t ny = XLENGTH(y);
353 
354 #define R_ARITHMETIC_ARRAY_1_SPECIAL
355 
356 #ifdef R_ARITHMETIC_ARRAY_1_SPECIAL
357  /* If either x or y is a matrix with length 1 and the other is a
358  vector of a different length, we want to coerce the matrix to be a vector.
359  Do we want to? We don't do it! BDR 2004-03-06
360 
361  From 3.4.0 (Sep. 2016), this signals a warning,
362  and in the future we will disable these 2 clauses,
363  so it will give an error.
364  */
365 
366  /* FIXME: Danger Will Robinson.
367  * ----- We might be trashing arguments here.
368  */
369  if (xarray != yarray) {
370  if (xarray && nx==1 && ny!=1) {
371  if(ny != 0)
372  Rf_warningcall(call,
373  _(
374  "Recycling array of length 1 in array-vector arithmetic is deprecated.\n\
375  Use c() or as.vector() instead.\n"));
376  x = static_cast<VectorBase*>(Rf_duplicate(x));
377  Rf_setAttrib(x, R_DimSymbol, R_NilValue);
378  }
379  if (yarray && ny==1 && nx!=1) {
380  if(nx != 0)
381  Rf_warningcall(call,
382  _("Recycling array of length 1 in vector-array arithmetic is deprecated.\n\
383  Use c() or as.vector() instead.\n"));
384  y = static_cast<VectorBase*>(Rf_duplicate(y));
385  Rf_setAttrib(y, R_DimSymbol, R_NilValue);
386  }
387  }
388 #endif
389 
390  checkOperandsConformable(x, y);
391 
392  GCStackRoot<> val;
393  /* need to preserve object here, as *_binary copies class attributes */
394  if (TYPEOF(x) == CPLXSXP || TYPEOF(y) == CPLXSXP) {
395  if (nx > 0 && ny > 0 &&
396  ((nx > ny) ? nx % ny : ny % nx) != 0) // mismatch
397  Rf_warningcall(call,
398  _("longer object length is not a multiple of shorter object length"));
399 
400  x = COERCE_IF_NEEDED(x, CPLXSXP);
401  y = COERCE_IF_NEEDED(y, CPLXSXP);
402  val = complex_binary(oper, x, y);
403  BinaryArithmeticAttributeCopier::copyAttributes(
404  SEXP_downcast<VectorBase*>(val.get()), x, y);
405  }
406  else if (TYPEOF(x) == REALSXP || TYPEOF(y) == REALSXP) {
407  /* real_binary can handle REALSXP or INTSXP operand, but not LGLSXP. */
408  /* Can get a LGLSXP. In base-Ex.R on 24 Oct '06, got 8 of these. */
409  if (TYPEOF(x) != INTSXP) x = COERCE_IF_NEEDED(x, REALSXP);
410  if (TYPEOF(y) != INTSXP) y = COERCE_IF_NEEDED(y, REALSXP);
411  val = real_binary(oper, x, y);
412  }
413  else {
414  val = integer_binary(oper, x, y, call);
415  }
416 
417  return val;
418 }

497  int integer_plus(int lhs, int rhs, Rboolean* naflag) {
498  if (lhs == NA_INTEGER || rhs == NA_INTEGER) {
499  return NA_INTEGER;
500  }
501  if ((lhs > 0 && INT_MAX - lhs < rhs)
502  || (lhs < 0 && INT_MAX + lhs < -rhs)) {
503  // Integer overflow.
504  *naflag = TRUE;
505  return NA_INTEGER;
506  }
507  return lhs + rhs;
508  }

510  int integer_minus(int lhs, int rhs, Rboolean* naflag) {
511  if (rhs == NA_INTEGER)
512  return NA_INTEGER;
513  return integer_plus(lhs, -rhs, naflag);
514  }

516  int integer_times(int lhs, int rhs, Rboolean* naflag) {
517  if (lhs == NA_INTEGER || rhs == NA_INTEGER) {
518  return NA_INTEGER;
519  }
520  // This relies on the assumption that a double can represent all the
521  // possible values of an integer. This isn't true for 64-bit integers.
522  static_assert(sizeof(int) <= 4,
523  "integer_times assumes 32 bit integers which isn't true on this platform");
524  double result = static_cast<double>(lhs) * static_cast<double>(rhs);
525  if (std::abs(result) > INT_MAX) {
526  // Integer overflow.
527  *naflag = TRUE;
528  return NA_INTEGER;
529  }
530  return static_cast<int>(result);
531  }

533  double integer_divide(int lhs, int rhs) {
534  if (lhs == NA_INTEGER || rhs == NA_INTEGER) {
535  return NA_REAL;
536  }

540  double integer_pow(int lhs, int rhs) {
541  if(lhs == 1 || rhs == 0)
542  return 1;
543  if (lhs == NA_INTEGER || rhs == NA_INTEGER) {
544  return NA_REAL;
545  }
546  return R_POW(static_cast<double>(lhs), static_cast<double>(rhs));

549  int integer_mod(int lhs, int rhs) {
550  if (lhs == NA_INTEGER || rhs == NA_INTEGER || rhs == 0)
551  return NA_INTEGER;
552  return (lhs >= 0 && rhs > 0) ? lhs % rhs :
553  static_cast<int>(myfmod(lhs, rhs));
554  }

556  int integer_idiv(int lhs, int rhs) {
557  /* This had x %/% 0 == 0 prior to 2.14.1, but
558  it seems conventionally to be undefined */
559  if (lhs == NA_INTEGER || rhs == NA_INTEGER || rhs == 0)
560  return NA_INTEGER;
561  return static_cast<int>(floor(double(lhs) / static_cast<double>(rhs)));
562  }

583 static SEXP integer_binary(ARITHOP_TYPE code, SEXP s1, SEXP s2, SEXP lcall)
584 {
585  Rboolean naflag = FALSE;
586  VectorBase* ans = nullptr;
587 
588  switch (code) {
589  case PLUSOP:
590  ans = apply_integer_binary(
591  [&](int lhs, int rhs) { return integer_plus(lhs, rhs, &naflag); },
592  s1, s2);
593  break;
594  case MINUSOP:
595  ans = apply_integer_binary(
596  [&](int lhs, int rhs) { return integer_minus(lhs, rhs, &naflag); },
597  s1, s2);
598  break;
599  case TIMESOP:
600  ans = apply_integer_binary(
601  [&](int lhs, int rhs) { return integer_times(lhs, rhs, &naflag); },
602  s1, s2);
603  break;
604  case DIVOP:
605  ans = apply_integer_binary(
606  [](int lhs, int rhs) { return integer_divide(lhs, rhs); },
607  s1, s2);
608  break;
609  case POWOP:
610  ans = apply_integer_binary(
611  [](int lhs, int rhs) { return integer_pow(lhs, rhs); },
612  s1, s2);
613  break;
614  case MODOP:
615  ans = apply_integer_binary(
616  [](int lhs, int rhs) { return integer_mod(lhs, rhs); },
617  s1, s2);
618  break;
619  case IDIVOP:
620  ans = apply_integer_binary(
621  [](int lhs, int rhs) { return integer_idiv(lhs, rhs); },
622  s1, s2);
623  break;
624  }
625  if (naflag)
626  Rf_warningcall(lcall, _("NAs produced by integer overflow"));
627 
628  return ans;
629 }

637 template<class Op>
638 static RealVector* apply_real_binary(Op op, SEXP lhs, SEXP rhs)
639 {
640  using namespace VectorOps;
641 
642  SEXPTYPE lhs_type = TYPEOF(lhs);
643  SEXPTYPE rhs_type = TYPEOF(rhs);
644 
645  if(lhs_type == REALSXP && rhs_type == REALSXP) {
646  return applyBinaryOperator(
647  op,
648  BinaryArithmeticAttributeCopier(),
649  SEXP_downcast<RealVector*>(lhs),
650  SEXP_downcast<RealVector*>(rhs));
651  } else if(lhs_type == INTSXP) {
652  return applyBinaryOperator(
653  [=](int lhs, double rhs) { return op(intToReal(lhs), rhs); },
654  BinaryArithmeticAttributeCopier(),
655  SEXP_downcast<IntVector*>(lhs),
656  SEXP_downcast<RealVector*>(rhs));
657  } else {
658  assert(rhs_type == INTSXP);
659  return applyBinaryOperator(
660  [=](double lhs, int rhs) { return op(lhs, intToReal(rhs)); },
661  BinaryArithmeticAttributeCopier(),
662  SEXP_downcast<RealVector*>(lhs),
663  SEXP_downcast<IntVector*>(rhs));
664  }

668 {
669  switch (code) {
670  case PLUSOP:
671  return apply_real_binary(
672  [](double lhs, double rhs) { return lhs + rhs; },

674  case MINUSOP:
675  return apply_real_binary(

677  s1, s2);
678  case TIMESOP:
679  return apply_real_binary(
680  [](double lhs, double rhs) { return lhs * rhs; },

682  case DIVOP:

684  [](double lhs, double rhs) { return lhs / rhs; },

686  case POWOP:
687  return apply_real_binary(
688  [](double lhs, double rhs) { return R_POW(lhs, rhs); },
