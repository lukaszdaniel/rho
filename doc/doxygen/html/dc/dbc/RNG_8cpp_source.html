109 static double fixup(double x)
110 {
111  /* ensure 0 and 1 are never returned */
112  if(x <= 0.0) return 0.5*i2_32m1;
113  if((1.0 - x) <= 0.0) return 1.0 - 0.5*i2_32m1;
114  return x;
115 }

118 double unif_rand(void)
119 {
120  double value;
121 
122  switch(RNG_kind) {
123 
124  case WICHMANN_HILL:
125  I1 = I1 * 171 % 30269;
126  I2 = I2 * 172 % 30307;
127  I3 = I3 * 170 % 30323;
128  value = I1 / 30269.0 + I2 / 30307.0 + I3 / 30323.0;
129  return fixup(value - int(value));/* in [0,1) */
130 
131  case MARSAGLIA_MULTICARRY:/* 0177777(octal) == 65535(decimal)*/
132  I1= 36969*(I1 & 0177777) + (I1>>16);
133  I2= 18000*(I2 & 0177777) + (I2>>16);
134  return fixup(((I1 << 16)^(I2 & 0177777)) * i2_32m1); /* in [0,1) */
135 
136  case SUPER_DUPER:
137  /* This is Reeds et al (1984) implementation;
138  * modified using __unsigned__ seeds instead of signed ones
139  */
140  I1 ^= ((I1 >> 15) & 0377777); /* Tausworthe */
141  I1 ^= I1 << 17;
142  I2 *= 69069; /* Congruential */
143  return fixup((I1^I2) * i2_32m1); /* in [0,1) */
144 
145  case MERSENNE_TWISTER:
146  return fixup(MT_genrand());
147 
148  case KNUTH_TAOCP:
149  case KNUTH_TAOCP2:
150  return fixup(KT_next() * KT);
151 
152  case USER_UNIF:
153  return *(static_cast<double *>(User_unif_fun()));
154 
155  case LECUYER_CMRG:
156  {
157  /* Based loosely on the GPL-ed version of
158  http://www.iro.umontreal.ca/~lecuyer/myftp/streams00/c2010/RngStream.c
159  but using int_least64_t, which C99 guarantees.
160  */
161  int_least64_t k; // rho change
162  int_least64_t p1, p2;
163 
164 #define II(i) (RNG_Table[RNG_kind].i_seed[i])
165 #define m1 4294967087
166 #define m2 4294944443
167 #define normc 2.328306549295727688e-10
168 #define a12 int_least64_t(1403580)
169 #define a13n int_least64_t(810728)
170 #define a21 int_least64_t(527612)
171 #define a23n int_least64_t(1370589)
172 
173  p1 = a12 * static_cast<unsigned int>(II(1)) - a13n * static_cast<unsigned int>(II(0));
174  /* p1 % m1 would surely do */
175  k = int((p1 / m1));
176  p1 -= k * m1;
177  if (p1 < 0.0) p1 += m1;
178  II(0) = II(1); II(1) = II(2); II(2) = Int32( p1);
179 
180  p2 = a21 * static_cast<unsigned int>(II(5)) - a23n * static_cast<unsigned int>(II(3));
181  k = int((p2 / m2));
182  p2 -= k * m2;
183  if (p2 < 0.0) p2 += m2;
184  II(3) = II(4); II(4) = II(5); II(5) = Int32( p2);
185 
186  return double(((p1 > p2) ? (p1 - p2) : (p1 - p2 + m1))) * normc;
187  }
188  default:
189  Rf_error(_("unif_rand: unimplemented RNG kind %d"), RNG_kind);
190  return -1.;
191  }

194 /* we must mask global variable here, as I1-I3 hide RNG_kind
195  and we want the argument */
196 static void FixupSeeds(RNGtype RNG_kind, int initial)
197 {
198 /* Depending on RNG, set 0 values to non-0, etc. */
199 
200  int j, notallzero = 0;
201 
202  /* Set 0 to 1 :
203  for(j = 0; j <= RNG_Table[RNG_kind].n_seed - 1; j++)
204  if(!RNG_Table[RNG_kind].i_seed[j]) RNG_Table[RNG_kind].i_seed[j]++; */
205 
206  switch(RNG_kind) {
207  case WICHMANN_HILL:
208  I1 = I1 % 30269; I2 = I2 % 30307; I3 = I3 % 30323;
209 
210  /* map values equal to 0 mod modulus to 1. */
211  if(I1 == 0) I1 = 1;
212  if(I2 == 0) I2 = 1;
213  if(I3 == 0) I3 = 1;
214  return;
215 
216  case SUPER_DUPER:
217  if(I1 == 0) I1 = 1;
218  /* I2 = Congruential: must be ODD */
219  I2 |= 1;
220  break;
221 
222  case MARSAGLIA_MULTICARRY:
223  if(I1 == 0) I1 = 1;
224  if(I2 == 0) I2 = 1;
225  break;
226 
227  case MERSENNE_TWISTER:
228  if(initial) I1 = 624;
229  /* No action unless user has corrupted .Random.seed */
230  if(I1 <= 0) I1 = 624;
231  /* check for all zeroes */
232  for (j = 1; j <= 624; j++)
233  if(RNG_Table[RNG_kind].i_seed[j] != 0) {
234  notallzero = 1;
235  break;
236  }
237  if(!notallzero) Randomize(RNG_kind);
238  break;
239 
240  case KNUTH_TAOCP:
241  case KNUTH_TAOCP2:
242  if(KT_pos <= 0) KT_pos = 100;
243  /* check for all zeroes */
244  for (j = 0; j < 100; j++)
245  if(RNG_Table[RNG_kind].i_seed[j] != 0) {
246  notallzero = 1;
247  break;
248  }
249  if(!notallzero) Randomize(RNG_kind);
250  break;
251  case USER_UNIF:
252  break;
253  case LECUYER_CMRG:
254  /* first set: not all zero, in [0, m1)
255  second set: not all zero, in [0, m2) */
256  {
257  unsigned int tmp;
258  int allOK = 1;
259  for (j = 0; j < 3; j++) {
260  tmp = RNG_Table[RNG_kind].i_seed[j];
261  if(tmp != 0) notallzero = 1;
262  if (tmp >= m1) allOK = 0;
263  }
264  if(!notallzero || !allOK) Randomize(RNG_kind);
265  for (j = 3; j < 6; j++) {
266  tmp = RNG_Table[RNG_kind].i_seed[j];
267  if(tmp != 0) notallzero = 1;
268  if (tmp >= m2) allOK = 0;
269  }
270  if(!notallzero || !allOK) Randomize(RNG_kind);
271  }
272  break;
273  default:
274  Rf_error(_("FixupSeeds: unimplemented RNG kind %d"), RNG_kind);
275  }
276 }

278 static void RNG_Init(RNGtype kind, Int32 seed)
279 {
280  int j;
281 
282  BM_norm_keep = 0.0; /* zap Box-Muller history */
283 
284  /* Initial scrambling */
285  for(j = 0; j < 50; j++)
286  seed = (69069 * seed + 1);
287  switch(kind) {
288  case WICHMANN_HILL:
289  case MARSAGLIA_MULTICARRY:
290  case SUPER_DUPER:
291  case MERSENNE_TWISTER:
292  /* i_seed[0] is mti, *but* this is needed for historical consistency */
293  for(j = 0; j < RNG_Table[kind].n_seed; j++) {
294  seed = (69069 * seed + 1);
295  RNG_Table[kind].i_seed[j] = seed;
296  }
297  FixupSeeds(kind, 1);
298  break;
299  case KNUTH_TAOCP:
300  RNG_Init_R_KT(seed);
301  break;
302  case KNUTH_TAOCP2:
303  RNG_Init_KT2(seed);
304  break;
305  case LECUYER_CMRG:
306  for(j = 0; j < RNG_Table[kind].n_seed; j++) {
307  seed = (69069 * seed + 1);
308  while(seed >= m2) seed = (69069 * seed + 1);
309  RNG_Table[kind].i_seed[j] = seed;
310  }
311  break;
312  case USER_UNIF:
313  User_unif_fun = R_FindSymbol("user_unif_rand", "", nullptr);
314  if (!User_unif_fun) Rf_error(_("'user_unif_rand' not in load table"));
315  User_unif_init = reinterpret_cast<UnifInitFun>(R_FindSymbol("user_unif_init", "", nullptr));
316  if (User_unif_init) (void) User_unif_init(seed);
317  User_unif_nseed = R_FindSymbol("user_unif_nseed", "", nullptr);
318  User_unif_seedloc = R_FindSymbol("user_unif_seedloc", "", nullptr);
319  if (User_unif_seedloc) {
320  int ns = 0;
321  if (!User_unif_nseed) {
322  Rf_warning(_("cannot read seeds unless 'user_unif_nseed' is supplied"));
323  break;
324  }
325  ns = *(static_cast<int *>(User_unif_nseed()));
326  if (ns < 0 || ns > 625) {
327  Rf_warning(_("seed length must be in 0...625; ignored"));
328  break;
329  }
330  RNG_Table[kind].n_seed = ns;
331  RNG_Table[kind].i_seed = static_cast<Int32 *>(User_unif_seedloc());
332  }
333  break;
334  default:
335  Rf_error(_("RNG_Init: unimplemented RNG kind %d"), kind);
336  }
337 }

353 static Rboolean GetRNGkind(SEXP seeds)
354 {
355  /* Load RNG_kind, N01_kind from .Random.seed if present */
356  int tmp, *is;
357  RNGtype newRNG; N01type newN01;
358 
359  if (Rf_isNull(seeds))
360  seeds = GetSeedsFromVar();
361  if (seeds == R_UnboundValue) return TRUE;
362  if (!Rf_isInteger(seeds)) {
363  if (seeds == R_MissingArg) /* How can this happen? */
364  Rf_error(_("'.Random.seed' is a missing argument with no default"));
365  Rf_warning(_("'.Random.seed' is not an integer vector but of type '%s', so ignored"),
366  Rf_type2char(TYPEOF(seeds)));
367  goto invalid;
368  }
369  is = INTEGER(seeds);
370  tmp = is[0];
371  /* avoid overflow here: max current value is 705 */
372  if (tmp == R_NaInt || tmp < 0 || tmp > 1000) {
373  Rf_warning(_("'.Random.seed[1]' is not a valid integer, so ignored"));
374  goto invalid;
375  }
376  newRNG = (RNGtype) (tmp % 100);
377  newN01 = (N01type) (tmp / 100);
378  if (newN01 > KINDERMAN_RAMAGE) {
379  Rf_warning(_("'.Random.seed[1]' is not a valid Normal type, so ignored"));
380  goto invalid;
381  }
382  switch(newRNG) {
383  case WICHMANN_HILL:
384  case MARSAGLIA_MULTICARRY:
385  case SUPER_DUPER:
386  case MERSENNE_TWISTER:
387  case KNUTH_TAOCP:
388  case KNUTH_TAOCP2:
389  case LECUYER_CMRG:
390  break;
391  case USER_UNIF:
392  if(!User_unif_fun) {
393  Rf_warning(_("'.Random.seed[1] = 5' but no user-supplied generator, so ignored"));
394  goto invalid;
395  }
396  break;
397  default:
398  Rf_warning(_("'.Random.seed[1]' is not a valid RNG kind so ignored"));
399  goto invalid;
400  }
401  RNG_kind = newRNG; N01_kind = newN01;
402  return FALSE;
403 invalid:
404  RNG_kind = RNG_DEFAULT; N01_kind = N01_DEFAULT;
405  Randomize(RNG_kind);
406  PutRNGstate(); // write out to .Random.seed
407  return TRUE;
408 }

411 void GetRNGstate()
412 {
413  /* Get .Random.seed into proper variables */
414  int len_seed;
415  SEXP seeds;
416 
417  seeds = GetSeedsFromVar();
418  if (seeds == R_UnboundValue) {
419  Randomize(RNG_kind);
420  } else {
421  /* this might re-set the generator */
422  if(GetRNGkind(seeds)) return;
423  len_seed = RNG_Table[RNG_kind].n_seed;
424  /* Not sure whether this test is needed: wrong for USER_UNIF */
425  if(XLENGTH(seeds) > 1 && XLENGTH(seeds) < len_seed + 1)
426  Rf_error(_("'.Random.seed' has wrong length"));
427  if(XLENGTH(seeds) == 1 && RNG_kind != USER_UNIF)
428  Randomize(RNG_kind);
429  else {
430  int j, *is = INTEGER(seeds);
431  for(j = 1; j <= len_seed; j++)
432  RNG_Table[RNG_kind].i_seed[j - 1] = Int32(is[j]);
433  FixupSeeds(RNG_kind, 0);
434  }
435  }
436 }

438 void PutRNGstate()
439 {
440  /* Copy out seeds to .Random.seed */
441  int len_seed, j;
442  SEXP seeds;
443 
444  if (RNG_kind > LECUYER_CMRG || N01_kind > KINDERMAN_RAMAGE) {
445  Rf_warning("Internal .Random.seed is corrupt: not saving");
446  return;
447  }
448 
449  len_seed = RNG_Table[RNG_kind].n_seed;
450 
451  PROTECT(seeds = Rf_allocVector(INTSXP, len_seed + 1));
452 
453  INTEGER(seeds)[0] = RNG_kind + 100 * N01_kind;
454  for(j = 0; j < len_seed; j++)
455  INTEGER(seeds)[j+1] = int(RNG_Table[RNG_kind].i_seed[j]);
456 
457  /* assign only in the workspace */
458  Rf_defineVar(Symbols::SeedsSymbol, seeds, R_GlobalEnv);
459  UNPROTECT(1);
460 }

462 static void RNGkind(RNGtype newkind)
463 {
464 /* Choose a new kind of RNG.
465  * Initialize its seed by calling the old RNG's unif_rand()
466  */
467  if (newkind == (RNGtype)-1) newkind = RNG_DEFAULT;
468  switch(newkind) {
469  case WICHMANN_HILL:
470  case MARSAGLIA_MULTICARRY:
471  case SUPER_DUPER:
472  case MERSENNE_TWISTER:
473  case KNUTH_TAOCP:
474  case USER_UNIF:
475  case KNUTH_TAOCP2:
476  case LECUYER_CMRG:
477  break;
478  default:
479  Rf_error(_("RNGkind: unimplemented RNG kind %d"), newkind);
480  }
481  GetRNGstate();
482  // precaution against corruption as per package randtoolbox
483  double u = unif_rand();
484  if (u < 0.0 || u > 1.0) {
485  Rf_warning("someone corrupted the random-number generator: re-initializing");
486  RNG_Init(newkind, TimeToSeed());
487  } else
488  RNG_Init(newkind, (Int32) (u * UINT_MAX));
489  RNG_kind = newkind;
490  PutRNGstate();
491 }

493 static void Norm_kind(N01type kind)
494 {
495  /* N01type is an enumeration type, so this will probably get
496  mapped to an unsigned integer type. */
497  if (kind == (N01type)-1) kind = N01_DEFAULT;
498  if (kind > KINDERMAN_RAMAGE)
499  Rf_error(_("invalid Normal type in 'RNGkind'"));
500  if (kind == USER_NORM) {
501  User_norm_fun = R_FindSymbol("user_norm_rand", "", nullptr);
502  if (!User_norm_fun) Rf_error(_("'user_norm_rand' not in load table"));
503  }
504  GetRNGstate(); /* might not be initialized */
505  if (kind == BOX_MULLER)
506  BM_norm_keep = 0.0; /* zap Box-Muller history */
507  N01_kind = kind;
508  PutRNGstate();
509 }

514 HIDDEN SEXP do_RNGkind (/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* kind_, rho::RObject* normal_kind_)
515 {
516  SEXP ans, rng, norm;
517 
518  GetRNGstate(); /* might not be initialized */
519  PROTECT(ans = Rf_allocVector(INTSXP, 2));
520  INTEGER(ans)[0] = RNG_kind;
521  INTEGER(ans)[1] = N01_kind;
522  rng = kind_;
523  norm = normal_kind_;
524  GetRNGkind(nullptr); /* pull from .Random.seed if present */
525  if(!Rf_isNull(rng)) { /* set a new RNG kind */
526  RNGkind(RNGtype( Rf_asInteger(rng)));
527  }
528  if(!Rf_isNull(norm)) { /* set a new normal kind */
529  Norm_kind(N01type(Rf_asInteger(norm)));
530  }
531  UNPROTECT(1);
532  return ans;
533 }

537 {

541  if(!Rf_isNull(seed_)) {
542  seed = Rf_asInteger(seed_);
543  if (seed == R_NaInt)
544  Rf_error(_("supplied seed is not a valid integer"));
545  } else seed = int(TimeToSeed());
546  skind = kind_;
547  nkind = normal_kind_;

552  RNG_Init(RNG_kind, Int32(seed)); /* zaps BM history */
553  PutRNGstate();
554  return nullptr;

576 /* A C-program for MT19937: Real number version([0,1)-interval)
577  (1999/10/28)
578  genrand() generates one pseudorandom real number (double)
579  which is uniformly distributed on [0,1)-interval, for each
580  call. sgenrand(seed) sets initial values to the working area
581  of 624 words. Before genrand(), sgenrand(seed) must be
582  called once. (seed is any 32-bit integer.)
583  Integer generator is obtained by modifying two lines.
584  Coded by Takuji Nishimura, considering the suggestions by
585  Topher Cooper and Marc Rieffel in July-Aug. 1997.

632 /* Initialization by "sgenrand()" is an example. Theoretically,
633  there are 2^19937-1 possible states as an intial state.
634  Essential bits in "seed_array[]" is following 19937 bits:
635  (seed_array[0]&UPPER_MASK), seed_array[1], ..., seed_array[N-1].
636  (seed_array[0]&LOWER_MASK) is discarded.
637  Theoretically,
638  (seed_array[0]&UPPER_MASK), seed_array[1], ..., seed_array[N-1]
639  can take any values except all zeros. */
640 
641 static double MT_genrand(void)
642 {
643  Int32 y;
644  static Int32 mag01[2]={0x0, MATRIX_A};
645  /* mag01[x] = x * MATRIX_A for x=0,1 */
646 
647  mti = int(dummy[0]);
648 
649  if (mti >= N) { /* generate N words at one time */
650  int kk;
651 
652  if (mti == N+1) /* if sgenrand() has not been called, */
653  MT_sgenrand(4357); /* a default initial seed is used */
654 
655  for (kk = 0; kk < N - M; kk++) {
656  y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
657  mt[kk] = mt[kk+M] ^ (y >> 1) ^ mag01[y & 0x1];
658  }
659  for (; kk < N - 1; kk++) {
660  y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
661  mt[kk] = mt[kk+(M-N)] ^ (y >> 1) ^ mag01[y & 0x1];
662  }
663  y = (mt[N-1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
664  mt[N-1] = mt[M-1] ^ (y >> 1) ^ mag01[y & 0x1];
665 
666  mti = 0;
667  }
668 
669  y = mt[mti++];
670  y ^= TEMPERING_SHIFT_U(y);
671  y ^= TEMPERING_SHIFT_S(y) & TEMPERING_MASK_B;
672  y ^= TEMPERING_SHIFT_T(y) & TEMPERING_MASK_C;
673  y ^= TEMPERING_SHIFT_L(y);
674  dummy[0] = Int32(mti);
675 
676  return ( double(y) * 2.3283064365386963e-10 ); /* reals: [0,1)-interval */
677 }

703 {

735 static void ran_start(long seed)
736 {
737  int t,j;
738  long x[KK+KK-1]; /* the preparation buffer */
739  long ss=(seed+2)&(MM-2);
740  for (j=0;j<KK;j++) {
741  x[j]=ss; /* bootstrap the buffer */
742  ss<<=1; if (ss>=MM) ss-=MM-2; /* cyclic shift 29 bits */
743  }
744  x[1]++; /* make x[1] (and only x[1]) odd */
745  for (ss=seed&(MM-1),t=TT-1; t; ) {
746  for (j=KK-1;j>0;j--) x[j+j]=x[j], x[j+j-1]=0; /* "square" */
747  for (j=KK+KK-2;j>=KK;j--)
748  x[j-(KK-LL)]=mod_diff(x[j-(KK-LL)],x[j]),
749  x[j-KK]=mod_diff(x[j-KK],x[j]);
750  if (is_odd(ss)) { /* "multiply by z" */
751  for (j=KK;j>0;j--) x[j]=x[j-1];
752  x[0]=x[KK]; /* shift the buffer cyclically */
753  x[LL]=mod_diff(x[LL],x[KK]);
754  }
755  if (ss) ss>>=1; else t--;
756  }
757  for (j=0;j<LL;j++) ran_x[j+KK-LL]=x[j];
758  for (;j<KK;j++) ran_x[j-LL]=x[j];
759  for (j=0;j<10;j++) ran_array(x,KK+KK-1); /* warm things up */
760  ran_arr_ptr=&ran_arr_sentinel;
761 }
