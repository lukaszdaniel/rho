104 /* Use the widest available unsigned type if uint64_t is not
105  available. The algorithm below extracts a number less than 62**6
106  (approximately 2**35.725) from uint64_t, so ancient hosts where
107  uintmax_t is only 32 bits lose about 3.725 bits of randomness,
108  which is better than not having mkstemp at all. */
109 #if !defined UINT64_MAX && !defined uint64_t
110 # define uint64_t uintmax_t
111 #endif

113 /* These are the characters used in temporary filenames. */
114 static const char letters[] =
115 "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
116 
117 /* Generate a temporary file name based on TMPL. TMPL must match the
118  rules for mk[s]temp (i.e. end in "XXXXXX"). The name constructed
119  does not exist at the time of the call to __gen_tempname. TMPL is
120  overwritten with the result.
121 
122  KIND is:
123  __GT_DIR: create a directory, which will be mode 0700.
124 
125  We use a clever algorithm to get hard-to-predict names. */
126 static int
127 gen_tempname (char *tmpl)
128 {
129  int len;
130  char *XXXXXX;
131  static uint64_t value;
132  uint64_t random_time_bits;
133  int count, fd = -1;
134  int save_errno = errno;
135 
136  len = int( strlen(tmpl));
137  if (len < 6 || strcmp (&tmpl[len - 6], "XXXXXX"))
138  {
139  __set_errno (EINVAL);
140  return -1;
141  }
142 
143  /* This is where the Xs start. */
144  XXXXXX = &tmpl[len - 6];
145 
146  /* Get some more or less random data. We need 36 bits. */
147  random_time_bits = Rf_TimeToSeed();
148  value += (random_time_bits << 8) ^ getpid ();
149 
150  for (count = 0; count < TMP_MAX; value += 7777, ++count)
151  {
152  uint64_t v = value;
153 
154  /* Fill in the random bits. */
155  XXXXXX[0] = letters[v % 62];
156  v /= 62;
157  XXXXXX[1] = letters[v % 62];
158  v /= 62;
159  XXXXXX[2] = letters[v % 62];
160  v /= 62;
161  XXXXXX[3] = letters[v % 62];
162  v /= 62;
163  XXXXXX[4] = letters[v % 62];
164  v /= 62;
165  XXXXXX[5] = letters[v % 62];
166 
167 #ifdef Win32
168  fd = mkdir (tmpl);
169 #else
170  fd = mkdir (tmpl, S_IRUSR | S_IWUSR | S_IXUSR);
171 #endif
172 
173  if (fd >= 0)
174  {
175  __set_errno (save_errno);
176  return fd;
177  }
178  else if (errno != EEXIST)
179  return -1;
180  }
181 
182  /* We got out of the loop because we ran out of combinations to try. */
183  __set_errno (EEXIST);
184  return -1;
185 }

187 /* Generate a unique temporary directory from TEMPLATE.
188  The last six characters of TEMPLATE must be "XXXXXX";
189  they are replaced with a string that makes the filename unique.
190  The directory is created, mode 700, and its name is returned.
191  (This function comes from OpenBSD.) */
192 char *
193 mkdtemp (char *Template)
194 #ifdef __cplusplus
195  throw()
196 #endif
197 {
198  if (gen_tempname (Template))
199  return NULL;
200  else
201  return Template;
202 }
