97 RObject* Closure::execute(Environment* env) const
98 {
99  RObject* ans;
100  Evaluator::maybeCheckForUserInterrupts();
101  Environment::ReturnScope returnscope(env);
102  Closure::DebugScope debugscope(this);
103  try {
104  ++m_num_invokes;
105 #ifdef ENABLE_LLVM_JIT
106  if (m_compiled_body
107  && m_compiled_body->hasMatchingFrameLayout(env)) {
108  PlainContext boctxt;
109  ans = m_compiled_body->evalInEnvironment(env);
110  } else {
111  if (!m_compiled_body && m_num_invokes == 100) {
112  // Compile the body, but stay in the interpreter because the
113  // frame hasn't been setup for a compiled function.
114  // TODO(kmillar): recompile functions as needed.
115  compile();
116  }
117 #endif
118  BailoutContext boctxt;
119  ans = Evaluator::evaluate(m_body, env);
120 #ifdef ENABLE_LLVM_JIT
121  }
122 #endif
123  if (ans && ans->sexptype() == BAILSXP) {
124  ReturnBailout* rbo = dynamic_cast<ReturnBailout*>(ans);
125  if (!rbo || rbo->environment() != env)
126  abort();
127  R_Visible = Rboolean(rbo->printResult());
128  ans = rbo->value();
129  }
130  }
131  catch (ReturnException& rx) {
132  if (rx.environment() != env)
133  throw;
134  ans = rx.value();
135  }
136  return ans;
137 }
