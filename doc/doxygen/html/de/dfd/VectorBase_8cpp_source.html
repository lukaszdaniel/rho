78 PairList *VectorBase::resizeAttributes(const PairList *attributes,
79  std::size_t new_size)
80 {
81  GCStackRoot<PairList> ans(PairList::cons(nullptr)); // dummy first link
82  PairList *op = ans;
83  for (const PairList *ip = attributes; ip; ip = ip->tail())
84  {
85  const RObject *tag = ip->tag();
86  RObject *value = ip->car();
87 
88  if (tag == NamesSymbol)
89  {
90  StringVector *names = SEXP_downcast<StringVector *>(value);
91  size_t old_size = names->size();
92  names = VectorBase::resize(names, new_size);
93  // resize() pads with NA, but we want blank strings instead.
94  for (size_t i = old_size; i < new_size; i++)
95  (*names)[i] = String::blank();
96  value = names;
97  }
98  if (tag != DimSymbol && tag != DimNamesSymbol)
99  {
100  op->setTail(PairList::cons(value, nullptr, tag));
101  op = op->tail();
102  }
103  }
104  return ans->tail();
105 }

145 void VectorBase::tooBig(std::size_t bytes)
146 {
147  double dsize = double(bytes) / 1024.0;
148  if (dsize > 1024.0 * 1024.0)

153  dsize / 1024.0);
