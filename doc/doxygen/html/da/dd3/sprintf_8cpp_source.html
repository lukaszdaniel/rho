51 static const char *findspec(const char *str)
52 {
53  /* This is not strict about checking where '.' is allowed.
54  It should allow - + ' ' # 0 as flags
55  m m. .n n.m as width/precision
56  */
57  const char *p = str;
58 
59  if(*p != '%') return p;
60  for(p++; ; p++) {
61  if(*p == '-' || *p == '+' || *p == ' ' || *p == '#' || *p == '.' ) continue;
62  /* '*' will currently have got substituted before this */
63  if(*p == '*' || (*p >= '0' && *p <= '9')) continue;
64  break;
65  }
66  return p;
67 }

71 static Rboolean checkfmt(const char *fmt, const char *pattern)
72 {
73  const char *p =fmt;
74 
75  if(*p != '%') return TRUE;
76  p = findspec(fmt);
77  return strcspn(p, pattern) ? TRUE : FALSE;
78 }

177  chunk = strcspn(curFormat + 1, "diosfeEgGxXaA") + 2;

437  }

440  chunk = (ch) ? size_t (ch - curFormat) : strlen(curFormat);
441  strncpy(bit, curFormat, chunk);
442  bit[chunk] = '\0';
443  }

446  strlen(ss) + 1, &outbuff));
447  strcat(outputString, ss);
448  } else {

450  strlen(bit) + 1, &outbuff));

452  }
453  } /* end for ( each chunk ) */

457  nprotect++;
