67 void ProvenanceTracker::CommandScope::monitorWrite(const Frame::Binding &bdg)
68 {
69  const Symbol *sym = bdg.symbol();
70  GCStackRoot<Provenance> prov(new Provenance(sym, m_chronicle));
71  if (m_xenogenetic)
72  prov->setXenogenous(bdg.rawValue()); // Maybe ought to clone value
73  Frame::Binding &ncbdg = const_cast<Frame::Binding &>(bdg);
74  ncbdg.setProvenance(prov);
75  m_chronicle->writeBinding(prov);
76 }
