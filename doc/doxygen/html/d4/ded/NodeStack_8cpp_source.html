52 void NodeStack::eraseTopmost(RObject* node)
53 {
54 #ifndef NDEBUG
55  if (m_innermost_scope
56  && m_vector.size() == m_innermost_scope->startSize())
57  throw std::logic_error("NodeStack::eraseTopmost(): "
58  "too many pops in this scope.");
59 #endif
60  std::vector<RObject*>::reverse_iterator rit
61  = find(m_vector.rbegin(), m_vector.rend(), node);
62  if (rit == m_vector.rend())
63  throw std::invalid_argument("NodeStack::unprotectPtr:"
64  " pointer not found.");
65  // See Josuttis p.267 for the need for -1 :
66  std::vector<RObject*>::iterator it = rit.base() - 1;
67  if (it - m_vector.begin() < int(m_protected_count)) {
68  GCNode::decRefCount(node);
69  --m_protected_count;
70  }
71  m_vector.erase(it);
72 }

89 void NodeStack::protectAll()
90 {
91  std::vector<RObject*>::iterator start
92  = m_vector.begin() + std::ptrdiff_t(m_protected_count);
93  std::vector<RObject*>::iterator end = m_vector.end();
94  for (std::vector<RObject*>::iterator it = start; it != end; ++it)
