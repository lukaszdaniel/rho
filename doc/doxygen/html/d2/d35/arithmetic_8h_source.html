57 /* Note that the behaviour of log(0) required is not necessarily that
58  mandated by C99 (-HUGE_VAL), and the behaviour of log(x < 0) is
59  optional in C99. Some systems return -Inf for log(x < 0), e.g.
60  libsunmath on Solaris.
61 */
62 static R_INLINE double logbase(double x, double base)
63 {
64 #ifdef HAVE_LOG10
65  if(base == 10) return x > 0 ? log10(x) : x == 0 ? R_NegInf : R_NaN;
66 #endif
67 #ifdef HAVE_LOG2
68  if(base == 2) return x > 0 ? log2(x) : x == 0 ? R_NegInf : R_NaN;
69 #endif
70  return R_log(x) / R_log(base);
71 }

75 /* for binary operations */
76 /* adapted from Radford Neal's pqR */
77 static R_INLINE SEXP R_allocOrReuseVector(SEXP s1, SEXP s2,
78  SEXPTYPE type , R_xlen_t n)
79 {
80  R_xlen_t n1 = XLENGTH(s1);
81  R_xlen_t n2 = XLENGTH(s2);
82 
83  /* Try to use space for 2nd arg if both same length, so 1st argument's
84  attributes will then take precedence when copied. */
85 
86  if (n == n2) {
87  if (TYPEOF(s2) == type && NO_REFERENCES(s2)) {
88  if (ATTRIB(s2) != R_NilValue)
89  /* need to remove 'names' attribute if present to
90  match what copyMostAttrib does. copyMostAttributes
91  also skips 'dim' and 'dimnames' but those, here
92  since those, if present, will be replaced by
93  attribute cleanup code in R_Binary) */
94  Rf_setAttrib(s2, R_NamesSymbol, R_NilValue);
95  return s2;
96  }
97  else
98  /* Can use 1st arg's space only if 2nd arg has no attributes, else
99  we may not get attributes of result right. */
100  if (n == n1 && TYPEOF(s1) == type && NO_REFERENCES(s1)
101  && ATTRIB(s2) == R_NilValue)
102  return s1;
103  }
104  else if (n == n1 && TYPEOF(s1) == type && NO_REFERENCES(s1))
105  return s1;
106 
107  return Rf_allocVector(type, n);
108 }
