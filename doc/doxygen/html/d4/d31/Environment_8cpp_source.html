122 Frame::Binding* Environment::findBinding(const Symbol* symbol)
123 {
124  bool cache_miss = false;
125  Environment* env = this;
126 #ifdef CHECK_CACHE
127  Frame::Binding* cache_binding = 0;
128 #endif
129  Cache* search_path_cache = searchPathCache();
130  while (env) {
131  if (env->isSearchPathCachePortal()) {
132  // TODO: the cache isn't effective for S3 methods.
133  Cache::iterator it = search_path_cache->find(symbol);
134  if (it == search_path_cache->end())
135  cache_miss = true;
136 #ifdef CHECK_CACHE
137  else cache_binding = it->second;
138 #else
139  else return it->second;
140 #endif
141  }
142  Frame::Binding* bdg = env->frame()->binding(symbol);
143  if (bdg) {
144 #ifdef CHECK_CACHE
145  if (cache_binding && cache_binding != bdg)
146  abort();
147 #endif
148  if (cache_miss)
149  (*search_path_cache)[symbol] = bdg;
150  return bdg;
151  }
152  env = env->enclosingEnvironment();
153  }
154  return nullptr;
155 }

161 void Environment::flushFromSearchPathCache(const Symbol* sym)
162 {
163  Cache* search_path_cache = searchPathCache();
164 
165  if (sym)
166  search_path_cache->erase(sym);
167  else {
168  // Clear the cache, but retain the current number of buckets:
169  size_t buckets = search_path_cache->bucket_count();
170  search_path_cache->clear();
171  search_path_cache->rehash(buckets);
172  }
173 }

229 void Environment::setOnSearchPath(bool status) {
230  if (status == m_on_search_path)
231  return;
232 
233  m_on_search_path = status;
234  if (!m_frame)
235  return;
236 
237  if (status)
238  m_frame->incCacheCount();
239  else
240  m_frame->decCacheCount();
241 
242  // Invalidate cache entries.
243  std::vector<const Symbol*> symbols = frame()->symbols(true);
244  for (std::vector<const Symbol*>::const_iterator symbol = symbols.begin();
245  symbol != symbols.end(); ++symbol) {
246  flushFromSearchPathCache(*symbol);
247  }

281 Environment* Environment::attachToSearchPath(int pos, StringVector* name)
282 {
283  // Duplicate the environment.
284  GCStackRoot<Frame> frame(static_cast<Frame*>(m_frame->clone()));
285  GCStackRoot<Environment> new_env(new Environment(nullptr, frame));
286  new_env->setAttribute(NameSymbol, name);
287 
288  // Iterate through the search path to the environment just before where we
289  // want to insert.
290  // This will be either pos - 1 or the environment prior to base().
291  Environment* where = global();
292  for (int n = 1; n < pos - 1 && where->enclosingEnvironment() != base(); n++)
293  where = where->enclosingEnvironment();
294 
295  // Insert the new environment after where.
296  new_env->m_enclosing = where->m_enclosing;
297  where->m_enclosing = new_env;
298  new_env->setOnSearchPath(true);
299 
300  return new_env;
301 }

303 Environment* Environment::detachFromSearchPath(int pos) {
304  if (pos == 1)
305  error(_("invalid '%s' argument"), "pos");
306 
307  // Iterate through the search path to the environment just before where we
308  // want to detach.
309  Environment* where = global();
310  for (int n = 1; n < pos - 1 && where->enclosingEnvironment() != empty(); n++)
311  where = where->enclosingEnvironment();
312 
313  Environment *env_to_detach = where->enclosingEnvironment();
314  if (env_to_detach == base())
315  error(_("detaching \"package:base\" is not allowed"));
316  if (env_to_detach == empty())
317  error(_("invalid '%s' argument"), "pos");
318 
319  // Detach the environment after where.
320  where->m_enclosing = env_to_detach->m_enclosing;
321  env_to_detach->m_enclosing = nullptr;
322  env_to_detach->setOnSearchPath(false);
323 
324  return env_to_detach;
