64 static int isum(SEXP sx, isum_INT *value, Rboolean narm, SEXP call)
65 {
66  LONG_INT s = 0; // at least 64-bit
67  int updated = 0;
68 #ifdef LONG_VECTOR_SUPPORT
69  int ii = R_INT_MIN; // need > 2^32 entries to overflow; checking earlier is a waste
70 /* NOTE: cannot use 64-bit *value to pass R_NaInt: that is "regular" 64bit int
71  * -> pass the NA information via return value ('updated').
72  * After the first 2^32 entries, only check every 1000th time (related to GET_REGION_BUFSIZE=512 ?)
73  * Assume LONG_INT_MAX >= 2^63-1 >=~ 9.223e18 > (1000 * 9000..0L = 9 * 10^18)
74  */
75 # define ISUM_OVERFLOW_CHECK do { \
76  if (ii++ > 1000) { \
77  if (s > 9000000000000000L || s < -9000000000000000L) { \
78  DbgP2("|OVERFLOW triggered: s=%ld|", s); \
79  /* *value = s; no use, TODO continue from 'k' */ \
80  return 42; /* was overflow, NA; now switch to irsum()*/ \
81  } \
82  ii = 0; \
83  } \
84  } while (0)
85 #else
86 # define ISUM_OVERFLOW_CHECK do { } while(0)
87 #endif
88 
89  /**** assumes INTEGER(sx) and LOGICAL(sx) are identical!! */
90  ITERATE_BY_REGION(sx, x, i, nbatch, int, INTEGER, {
91  for (int k = 0; k < nbatch; k++) {
92  if (x[k] != R_NaInt) {
93  if(!updated) updated = 1;
94  s += x[k];
95  ISUM_OVERFLOW_CHECK;
96  } else if (!narm) {
97  // updated = R_NaInt;
98  return R_NaInt;
99  }
100  }
101  });
102  *value = s;
103  return updated;
104 #undef ISUM_OVERFLOW_CHECK
105 }
106 #else // no LONG_INT : should never be used with a C99/C11 compiler
107 # define isum_INT int

400 static Rboolean cprod(SEXP sx, Rcomplex *value, Rboolean narm)
401 {
402  Rcomplex *x = COMPLEX(sx);
403  R_xlen_t n = XLENGTH(sx);
404  LDOUBLE sr = 1.0, si = 0.0;
405  Rboolean updated = FALSE;
406  for (R_xlen_t k = 0; k < n; k++) {
407  if (!narm || (!std::isnan(x[k].r) && !std::isnan(x[k].i))) {
408  if(!updated) updated = TRUE;
409  LDOUBLE tr = sr, ti = si;
410  sr = tr * x[k].r - ti * x[k].i;
411  si = tr * x[k].i + ti * x[k].r;
412  }
413  }
414  value->r = double(sr);
415  value->i = double(si);
416 
417  return updated;
418 }

421 HIDDEN
422 SEXP fixup_NaRm(SEXP args)
423 {
424  SEXP t;
425 
426  /* Need to make sure na.rm is last and exists */
427  GCStackRoot<> na_value(Rf_ScalarLogical(FALSE));
428  for(SEXP a = args, prev = nullptr; a != nullptr; a = CDR(a)) {
429  if(TAG(a) == Symbols::NaRmSymbol) {
430  if(CDR(a) == nullptr) return args;
431  na_value = CAR(a);
432  if(prev == nullptr) args = CDR(a);
433  else SETCDR(prev, CDR(a));
434  }
435  prev = a;
436  }
437 
438  t = PairList::cons(na_value, nullptr);
439  PROTECT(t);
440  SET_TAG(t, Symbols::NaRmSymbol);
441  if (args == nullptr)
442  args = t;
443  else {
444  SEXP r = args;
445  while (CDR(r) != nullptr) r = CDR(r);
446  SETCDR(r, t);
447  }
448  UNPROTECT(1);
449  return args;
450 }

460 R_INLINE static SEXP logical_mean(SEXP x)
461 {
462  R_xlen_t n = XLENGTH(x);
463  LDOUBLE s = 0.0;
464  for (R_xlen_t i = 0; i < n; i++) {
465  int xi = LOGICAL_ELT(x, i);
466  if(xi == R_NaLog)
467  return Rf_ScalarReal(R_NaReal);
468  s += xi;
469  }
470  return Rf_ScalarReal((double) (s/n));
471 }

473 R_INLINE static SEXP integer_mean(SEXP x)
474 {
475  R_xlen_t n = XLENGTH(x);
476  LDOUBLE s = 0.0;
477  for (R_xlen_t i = 0; i < n; i++) {
478  int xi = INTEGER_ELT(x, i);
479  if(xi == R_NaInt)

486 R_INLINE static SEXP real_mean(SEXP x)
487 {
488  R_xlen_t n = XLENGTH(x);
489  LDOUBLE s = 0.0;
490  ITERATE_BY_REGION(x, dx, i, nbatch, double, REAL, {
491  for (R_xlen_t k = 0; k < nbatch; k++)
492  s += dx[k];
493  });
494  s /= n;
495  if (std::isfinite((double) s)) {
496  LDOUBLE t = 0.0;
497  ITERATE_BY_REGION(x, dx, i, nbatch, double, REAL, {
498  for (R_xlen_t k = 0; k < nbatch; k++)
499  t += (dx[k] - s);
500  });
501  s += t/n;
502  }
503  return Rf_ScalarReal((double) s);
504 }

506 R_INLINE static SEXP complex_mean(SEXP x)
507 {
508  R_xlen_t n = XLENGTH(x);
509  LDOUBLE s = 0.0, si = 0.0;
510  Rcomplex *px = COMPLEX(x);
511  for (R_xlen_t i = 0; i < n; i++) {
512  Rcomplex xi = px[i];
513  s += xi.r;
514  si += xi.i;
515  }
516  s /= n; si /= n;
517  if( std::isfinite((double)s) && std::isfinite((double)si) ) {
518  LDOUBLE t = 0.0, ti = 0.0;
519  for (R_xlen_t i = 0; i < n; i++) {
520  Rcomplex xi = px[i];
521  t += xi.r - s;
522  ti += xi.i - si;
523  }
524  s += t/n; si += ti/n;
525  }
526  Rcomplex val = { double(s), double(si) };
527  return Rf_ScalarComplex(val);
528 }

530 HIDDEN SEXP do_summary(SEXP call, SEXP op, SEXP args, SEXP env)
531 {
532  checkArity(op, args);
533  if(PRIMVAL(op) == 1) { /* mean */
534  SEXP x = CAR(args);
535  switch(TYPEOF(x)) {
536  case LGLSXP: return logical_mean(x);
537  case INTSXP: return integer_mean(x);
538  case REALSXP: return real_mean(x);
539  case CPLXSXP: return complex_mean(x);
540  default:
541  Rf_error(R_MSG_type, Rf_type2char(TYPEOF(x)));
542  return nullptr; // -Wall on clang 4.2
543  }
544  }
545 
546  SEXP ans;
547  /* match to foo(..., na.rm=FALSE) */
548  PROTECT(args = fixup_NaRm(args));
549  Expression* call2 = new Expression(CAR(call), SEXP_downcast<PairList*>(args));
550 
551  ArgList arglist(SEXP_downcast<PairList*>(args), ArgList::EVALUATED);
552  auto dispatched = SEXP_downcast<BuiltInFunction*>(op)->InternalDispatch(
553  call2, SEXP_downcast<Environment*>(env), std::move(arglist));
554  if (dispatched.first) {
555  UNPROTECT(1);
556  return(dispatched.second);
557  }
558 
559 #ifdef DEBUG_Summary
560  REprintf("C do_summary(op%s, *): did NOT dispatch\n", PRIMNAME(op));
561 #endif
562 
563  ans = Rf_matchArgExact(Symbols::NaRmSymbol, &args);
564  Rboolean narm = Rboolean(Rf_asLogical(ans));
565 
566  if (ALTREP(CAR(args)) && CDDR(args) == nullptr &&
567  (CDR(args) == nullptr || TAG(CDR(args)) == R_NaRmSymbol)) {
568  SEXP toret = nullptr;
569  SEXP vec = CAR(args);
570  switch(PRIMVAL(op)) {
571  case 0:
572  if(TYPEOF(vec) == INTSXP)
573  toret = ALTINTEGER_SUM(vec, narm);
574  else if (TYPEOF(vec) == REALSXP)
575  toret = ALTREAL_SUM(vec, narm);
576  break;
577  case 2:
578  if(TYPEOF(vec) == INTSXP)
579  toret = ALTINTEGER_MIN(vec, narm);
580  else if (TYPEOF(vec) == REALSXP)
581  toret = ALTREAL_MIN(vec, narm);
582  break;
583  case 3:
584  if(TYPEOF(vec) == INTSXP)
585  toret = ALTINTEGER_MAX(vec, narm);
586  else if (TYPEOF(vec) == REALSXP)
587  toret = ALTREAL_MAX(vec, narm);
588  break;
589  default:
590  break;
591  }
592  if(toret != nullptr) {
593  UNPROTECT(1); /* args */
594  return toret;
595  }
596  }
597 
598  Rboolean int_a, real_a, complex_a,
599  empty = TRUE;// <==> only zero-length arguments, or NA with na.rm=T
600  int updated = 0; //
601  /* updated = R_NaInt if encountered NA,
602  updated != 0 , as soon as (i)tmp (do_summary),
603  or *value ([ir]min / max) is assigned; */
604  SEXP a;
605  double tmp = 0.0, s;
606  Rcomplex ztmp, zcum={0.0, 0.0} /* -Wall */;
607  int itmp = 0, icum = 0, warn = 0 /* dummy */;
608  Rboolean use_isum = TRUE; // indicating if isum() should used; otherwise irsum()
609  isum_INT iLtmp = (isum_INT)0, iLcum = iLtmp; // for isum() only
610  SEXPTYPE ans_type;/* only INTEGER, REAL, COMPLEX or STRSXP here */
611 
612  int iop = PRIMVAL(op);
613  switch(iop) {
614  case 0:/* sum */
615  /* we need to find out if _all_ the arguments are integer or logical
616  in advance, as we might overflow before we find out. NULL is
617  documented to be the same as integer(0).
618  */
619  a = args;
620  complex_a = real_a = FALSE;
621  while (a != nullptr) {
622  switch(TYPEOF(CAR(a))) {
623  case INTSXP:
624  case LGLSXP:
625  case NILSXP:
626  break;
627  case REALSXP:
628  real_a = TRUE;
629  break;
630  case CPLXSXP:
631  complex_a = TRUE;
632  break;
633  default:
634  a = CAR(a);
635  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
636  return nullptr;
637  }
638  a = CDR(a);
639  }
640  if(complex_a) {
641  ans_type = CPLXSXP;
642  } else if(real_a) {
643  ans_type = REALSXP;
644  } else {
645  ans_type = INTSXP; iLcum = (isum_INT)0;
646  }
647  DbgP3("do_summary: sum(.. na.rm=%d): ans_type = %s\n",
648  narm, Rf_type2char(ans_type));
649  zcum.r = zcum.i = 0.; icum = 0;
650  break;
651 
652  case 2:/* min */
653  DbgP2("do_summary: min(.. na.rm=%d) ", narm);
654  ans_type = INTSXP;
655  zcum.r = R_PosInf;
656  icum = INT_MAX;
657  break;
658 
659  case 3:/* max */
660  DbgP2("do_summary: max(.. na.rm=%d) ", narm);
661  ans_type = INTSXP;
662  zcum.r = R_NegInf;;
663  icum = R_INT_MIN;
664  break;
665 
666  case 4:/* prod */
667  ans_type = REALSXP;
668  zcum.r = 1.;
669  zcum.i = 0.;
670  break;
671 
672  default:
673  Rf_errorcall(call,
674  _("internal error ('op = %d' in do_summary).\t Call a Guru"),
675  iop);
676  return nullptr;/*-Wall */
677  }
678 
679  SEXP stmp = R_NaString,
680  scum = PROTECT(R_NaString);
681  /*-- now loop over all arguments. Do the 'op' switch INSIDE : */
682  while (args != nullptr) {
683  a = CAR(args);
684  int_a = FALSE;// int_a = TRUE <--> a is INTEGER
685  real_a = FALSE;
686 
687  if(Rf_xlength(a) > 0) {
688  updated = 0;/*- GLOBAL -*/

691  case 2:/* min */
692  case 3:/* max */
693

695  case LGLSXP:
696  case INTSXP:
697  int_a = TRUE;
698  if (iop == 2) updated = imin(a, &itmp, narm);
699  else updated = imax(a, &itmp, narm);
700  break;
701  case REALSXP:
702  real_a = TRUE;
703  if(ans_type == INTSXP) {/* change to REAL */
704  ans_type = REALSXP;
705  if(!empty) zcum.r = Int2Real(icum);
706  }
707  if (iop == 2) updated = rmin(a, &tmp, narm);
708  else updated = rmax(a, &tmp, narm);
709  break;
710  case STRSXP:
711  if(!empty && ans_type == INTSXP) {
712  scum = Rf_StringFromInteger(icum, &warn);
713  UNPROTECT(1); /* scum */
714  PROTECT(scum);
715  } else if(!empty && ans_type == REALSXP) {
716  scum = Rf_StringFromReal(zcum.r, &warn);
717  UNPROTECT(1); /* scum */
718  PROTECT(scum);
719  }
720  ans_type = STRSXP;
721  if (iop == 2) updated = smin(a, &stmp, narm);
722  else updated = smax(a, &stmp, narm);
723  break;
724  default:
725  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
726  return nullptr;
727  }
728 
729  if(updated) {/* 'a' had non-NA elements; --> "add" tmp or itmp*/
730  DbgP1(" updated:");
731  if(ans_type == INTSXP) {
732  DbgP3(" INT: (old)icum= %ld, itmp=%ld\n", icum,itmp);
733  if (icum == R_NaInt); /* NA trumps anything */
734  else if (itmp == R_NaInt ||
735  (iop == 2 && itmp < icum) || /* min */
736  (iop == 3 && itmp > icum)) /* max */
737  icum = itmp;
738  } else if(ans_type == REALSXP) {
739  if (int_a) tmp = Int2Real(itmp);
740  DbgP3(" REAL: (old)cum= %g, tmp=%g\n", zcum.r,tmp);
741  if (ISNA(zcum.r)); /* NA trumps anything */
742  else if (std::isnan(tmp)) {
743  if (ISNA(tmp)) zcum.r = tmp;
744  else zcum.r += tmp;/* NA or NaN */
745  } else if(
746  (iop == 2 && tmp < zcum.r) ||
747  (iop == 3 && tmp > zcum.r)) zcum.r = tmp;
748  } else if(ans_type == STRSXP) {
749  if(int_a)
750  stmp = Rf_StringFromInteger(itmp, &warn);
751  else if(real_a)
752  stmp = Rf_StringFromReal(tmp, &warn);
753 
754  if(empty)
755  scum = stmp;
756  else if (scum != R_NaString) {
757  PROTECT(stmp);
758  if(stmp == R_NaString ||
759  (iop == 2 && stmp != scum && Scollate(stmp, scum) < 0) ||
760  (iop == 3 && stmp != scum && Scollate(stmp, scum) > 0) )
761  scum = stmp;
762  UNPROTECT(1); /* stmp */
763  }
764  UNPROTECT(1); /* scum */
765  PROTECT(scum);
766  }

769  -- if there are no non-missing elements.
770  */
771  DbgP2(" NOT updated [!! RARE !!]: int_a=%s\n", int_a ? "TRUE" : "FALSE");
772  }
773 
774  break;/*--- end of min() / max() ---*/
775 
776  case 0:/* sum */

778  case LGLSXP:
779  case INTSXP:
780 #ifdef LONG_INT

782  isum(a, &iLtmp, narm, call) :
783  risum(a, &tmp, narm));

794  DbgP3(" .. switching type to REAL, tmp=%g, zcum.r=%g",
795  tmp, zcum.r);
796  }
797  else if(updated) {

801  if(s > INT_MAX || s < R_INT_MIN ||
802  iLtmp < -LONG_INT_MAX || LONG_INT_MAX < iLtmp) {
803  ans_type = REALSXP;

807  use_isum = FALSE;

809  zcum.r = s;

812  else {
813  iLcum += iLtmp;

817  } else { // dealt with R_NaInt already above
818  zcum.r += use_isum ? (double)iLtmp : tmp;

823 #else

835  } else
836  zcum.r += Int2Real(iLtmp);
837  }
838 #endif
839  break;
840  case REALSXP:

842  ans_type = REALSXP;
843  if(!empty) zcum.r = Int2Real(iLcum);
844  }
845  updated = rsum(a, &tmp, narm);
846  if(updated) {
847  zcum.r += tmp;
848  }
849  break;
850  case CPLXSXP:
851  if(ans_type == INTSXP) {
852  ans_type = CPLXSXP;
853  if(!empty) zcum.r = Int2Real(iLcum);
854  } else if (ans_type == REALSXP)
855  ans_type = CPLXSXP;
856  updated = csum(a, &ztmp, narm);
857  if(updated) {
858  zcum.r += ztmp.r;
859  zcum.i += ztmp.i;
860  }
861  break;
862  default:
863  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
864  return nullptr;
865  }
866 
867  break;/* sum() part */
868 
869  case 4:/* prod */
870 
871  switch(TYPEOF(a)) {
872  case LGLSXP:
873  case INTSXP:
874  case REALSXP:
875  if(TYPEOF(a) == REALSXP)
876  updated = rprod(a, &tmp, narm);
877  else
878  updated = iprod(a, &tmp, narm);
879  if(updated) {
880  zcum.r *= tmp;
881  zcum.i *= tmp;
882  }
883  break;
884  case CPLXSXP:
885  ans_type = CPLXSXP;
886  updated = cprod(a, &ztmp, narm);
887  if(updated) {
888  Rcomplex z;
889  z.r = zcum.r;
890  z.i = zcum.i;
891  zcum.r = z.r * ztmp.r - z.i * ztmp.i;
892  zcum.i = z.r * ztmp.i + z.i * ztmp.r;
893  }
894  break;
895  default:
896  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
897  return nullptr;
898  }
899 
900  break;/* prod() part */
901 
902  } /* switch(iop) */
903 
904  } else { /* len(a)=0 */
905  /* Even though this has length zero it can still be invalid,
906  e.g. list() or raw() */
907  switch(TYPEOF(a)) {
908  case LGLSXP:
909  case INTSXP:
910  case REALSXP:
911  case NILSXP: /* OK historically, e.g. PR#1283 */
912  break;
913  case CPLXSXP:
914  if (iop == 2 || iop == 3)
915  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
916  return nullptr;
917  break;
918  case STRSXP:
919  if (iop == 2 || iop == 3) {
920  if(!empty && ans_type == INTSXP) {
921  scum = Rf_StringFromInteger(icum, &warn);
922  UNPROTECT(1); /* scum */
923  PROTECT(scum);
924  } else if(!empty && ans_type == REALSXP) {
925  scum = Rf_StringFromReal(zcum.r, &warn);
926  UNPROTECT(1); /* scum */
927  PROTECT(scum);
928  }
929  ans_type = STRSXP;
930  break;
931  }
932  default:
933  Rf_errorcall(call, R_MSG_type, Rf_type2char(TYPEOF(a)));
934  return nullptr;
935  }
936  if(ans_type < TYPEOF(a) && ans_type != CPLXSXP) {
937  if(!empty && ans_type == INTSXP)
938  zcum.r = Int2Real(icum);
939  ans_type = TYPEOF(a);
940  }
941  }
942  DbgP3(" .. upd.=%d, empty=%d", updated, (int)empty);
943  if(empty && updated) empty=FALSE;
944  DbgP2(", new empty=%d\n", (int)empty);
945  args = CDR(args);
946  } /*-- while(..) loop over args */
947 
948  /*-------------------------------------------------------*/
949  if(empty && (iop == 2 || iop == 3)) {
950  if(ans_type == STRSXP) {
951  Rf_warningcall(call, _("no non-missing arguments, returning NA"));
952  } else {
953  if(iop == 2)
954  Rf_warningcall(call, _("no non-missing arguments to min; returning Inf"));
955  else
956  Rf_warningcall(call, _("no non-missing arguments to max; returning -Inf"));
957  ans_type = REALSXP;
958  }
959  }
960 
961  switch(ans_type) {
962  case INTSXP: ans = (iop == 0) ? Rf_ScalarInteger(iLcum) : Rf_ScalarInteger(icum);break;
963  case REALSXP: ans = Rf_ScalarReal(zcum.r); break;
964  case CPLXSXP: ans = Rf_ScalarComplex(zcum);break;
965  case STRSXP: ans = Rf_ScalarString(scum); break;
966  default: break; // -Wswitch
967  }
968  UNPROTECT(2); /* scum, args */
969  return ans;
970 
971 na_answer: /* only sum(INTSXP, ...) case currently used */
972  switch(ans_type) {
973  case INTSXP: ans = Rf_ScalarInteger(R_NaInt); break;
974  case REALSXP: ans = Rf_ScalarReal(R_NaReal); break;
975  case CPLXSXP: ans = Rf_ScalarComplex({R_NaReal, R_NaReal}); break;
976  case STRSXP: ans = Rf_ScalarString(R_NaString); break;
977  default: break; // -Wswitch
978  }
979  UNPROTECT(2); /* scum, args */
980  return ans;
981 }/* do_summary */

1009 HIDDEN SEXP do_first_min(/*const*/ Expression* call, const BuiltInFunction* op, RObject* x_)
1010 {

1012  R_xlen_t i, n, indx = -1;
1013 
1014  if (!Rf_isNumeric(sx)) {
1015  sx = Rf_coerceVector(sx, REALSXP);
1016  }
1017  n = XLENGTH(sx);
1018  switch(TYPEOF(sx)) {
1019  case LGLSXP: // with only (TRUE, FALSE, NA) -- may be fast
1020  {
1021  int *r = LOGICAL(sx);
1022  if(op->variant() == 0) { /* which.min */
1023  for (i = 0; i < n; i++)
1024  if (r[i] == FALSE) {
1025  indx = i; break; // found FALSE: done
1026  } else if (indx == -1 && r[i] != R_NaLog) {
1027  indx = i; // first TRUE

1030  for (i = 0; i < n; i++)
1031  if (r[i] == TRUE) {
1032  indx = i; break; // found TRUE: done
1033  } else if (indx == -1 && r[i] != R_NaLog) {
1034  indx = i; // first FALSE
1035  }
1036  }
1037  }
1038  break;
1039 
1040  case INTSXP:
1041  {
1042  int s, *r = INTEGER(sx);
1043  if(op->variant() == 0) { /* which.min */
1044  s = INT_MAX;
1045  for (i = 0; i < n; i++)
1046  if (r[i] != R_NaInt && (r[i] < s || indx == -1)) {
1047  s = r[i]; indx = i;
1048  }
1049  } else { /* which.max */
1050  s = INT_MIN;
1051  for (i = 0; i < n; i++)
1052  if (r[i] != R_NaInt && (r[i] > s || indx == -1)) {
1053  s = r[i]; indx = i;
1054  }
1055  }
1056  }
1057  break;
1058 
1059  case REALSXP:
1060  {
1061  double s, *r = REAL(sx);
1062  if(op->variant() == 0) { /* which.min */
1063  s = R_PosInf;
1064  for (i = 0; i < n; i++)
1065  if ( !std::isnan(r[i]) && (r[i] < s || indx == -1) ) {
1066  s = r[i]; indx = i;
1067  }
1068  } else { /* which.max */
1069  s = R_NegInf;
1070  for (i = 0; i < n; i++)
1071  if ( !std::isnan(r[i]) && (r[i] > s || indx == -1) ) {
1072  s = r[i]; indx = i;
1073  }
1074  }
1075  }
1076  break;
1077  default: Rf_error(_("invalid type")); break;
1078  } // switch()
1079 
1080 
1081  i = (indx != -1);
1082  bool large = (indx + 1) > INT_MAX;
1083  PROTECT(ans = Rf_allocVector(large ? REALSXP : INTSXP, i ? 1 : 0));
1084  if (i) {
1085  if(large)
1086  REAL(ans)[0] = (double)indx + 1;
1087  else
1088  INTEGER(ans)[0] = (int)indx + 1;
1089  if (Rf_getAttrib(sx, Symbols::NamesSymbol) != nullptr) { /* preserve names */
1090  SEXP ansnam;
1091  PROTECT(ansnam =
1092  Rf_ScalarString(STRING_ELT(Rf_getAttrib(sx, Symbols::NamesSymbol), indx)));
1093  Rf_setAttrib(ans, Symbols::NamesSymbol, ansnam);
1094  UNPROTECT(1);
1095  }
1096  }
1097  return ans;
1098 }

1223  && tmp < ra[i]) ||
