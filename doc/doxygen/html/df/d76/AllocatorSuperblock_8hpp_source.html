61  AllocatorSuperblock(unsigned size_class, unsigned bitset_entries):
62  m_size_class(size_class),
63  m_next_untouched(0) {
64  // Here we mark all bitset entries as free so that we don't have to do
65  // precise range checking while iterating over currently allocated blocks
66  // when the number of blocks is not evenly divisible by 64.
67  for (unsigned int i = 0; i < bitset_entries; ++i) {
68  m_free[i] = ~0ull;
69  }
70  }
