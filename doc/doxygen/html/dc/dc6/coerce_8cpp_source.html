359 SEXP Rf_PairToVectorList(SEXP x)
360 {
361  SEXP xptr, xnew, xnames;
362  int i, len = 0, named = 0;
363  for (xptr = x ; xptr != R_NilValue ; xptr = CDR(xptr)) {
364  named = named | (TAG(xptr) != R_NilValue);
365  len++;
366  }
367  PROTECT(x);
368  PROTECT(xnew = Rf_allocVector(VECSXP, len));
369  for (i = 0, xptr = x; i < len; i++, xptr = CDR(xptr)) {
370  if (NAMED(x) > NAMED(CAR(xptr)))
371  SET_NAMED(CAR(xptr), NAMED(x));
372  SET_VECTOR_ELT(xnew, i, CAR(xptr));
373  }
374  if (named) {
375  PROTECT(xnames = Rf_allocVector(STRSXP, len));
376  xptr = x;
377  for (i = 0, xptr = x; i < len; i++, xptr = CDR(xptr)) {
378  if(TAG(xptr) == R_NilValue)
379  SET_STRING_ELT(xnames, i, R_BlankString);
380  else
381  SET_STRING_ELT(xnames, i, PRINTNAME(TAG(xptr)));
382  }
383  Rf_setAttrib(xnew, R_NamesSymbol, xnames);
384  UNPROTECT(1);
385  }
386  Rf_copyMostAttrib(x, xnew);
387  UNPROTECT(2);
388  return xnew;
389 }

416 static SEXP coerceToSymbol(SEXP v)
417 {
418  SEXP ans = R_NilValue;
419  int warn = 0;
420  if (Rf_length(v) <= 0)
421  Rf_error(_("invalid data of mode '%s' (too short)"),
422  Rf_type2char(TYPEOF(v)));
423  PROTECT(v);
424  switch(TYPEOF(v)) {
425  case LGLSXP:
426  ans = Rf_StringFromLogical(LOGICAL(v)[0], &warn);
427  break;
428  case INTSXP:
429  ans = Rf_StringFromInteger(INTEGER(v)[0], &warn);
430  break;
431  case REALSXP:
432  ans = Rf_StringFromReal(REAL(v)[0], &warn);
433  break;
434  case CPLXSXP:
435  ans = Rf_StringFromComplex(COMPLEX(v)[0], &warn);
436  break;
437  case STRSXP:
438  ans = STRING_ELT(v, 0);
439  break;
440  case RAWSXP:
441  ans = StringFromRaw(RAW(v)[0], &warn);
442  break;
443  default:
444  UNIMPLEMENTED_TYPE("coerceToSymbol", v);
445  }
446  if (warn) Rf_CoercionWarning(warn);/*2000/10/23*/
447  ans = Rf_installTrChar(ans);
448  UNPROTECT(1);
449  return ans;
450 }
