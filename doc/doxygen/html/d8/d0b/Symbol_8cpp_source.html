63 Symbol::Symbol(const String* the_name)
64  : RObject(SYMSXP), m_dd_index(-1), m_is_special_symbol(false)
65 {
66  m_name = the_name;
67  // If this is a ..n symbol, extract the value of n.
68  // boost::regex_match (libboost_regex1_36_0-1.36.0-9.5) doesn't
69  // seem comfortable with empty strings, hence the size check.
70  if (m_name && m_name->size() > 2) {
71  // Versions of GCC prior to 4.9 don't support std::regex, so use
72  // boost::regex instead.
73  static const boost::regex *regex = new boost::regex("\\.\\.(\\d+)");
74 
75  string name(m_name->c_str());
76  boost::smatch dd_match;
77  if (boost::regex_match(name, dd_match, *regex)) {
78  istringstream iss(dd_match[1]);
79  int n;
80  iss >> n;
81  m_dd_index = n;
82  }
83  }
84 }

104 RObject* Symbol::evaluate(Environment* env)
105 {
106  if (this == DotsSymbol)
107  Rf_error(_("'...' used in an incorrect context"));
108  GCStackRoot<> val;
109  if (isDotDotSymbol()) {
110  val = Rf_ddfindVar(this, env);
111  // printf("I'm in isDotDotSymbol()\n");
112  } else {
113  Frame::Binding* bdg = env->findBinding(this);
114  if (bdg)
115  val = bdg->unforcedValue();
116  else if (this == missingArgument())
117  val = this; // This reproduces CR behaviour
118  else val = unboundValue();
119  }
120  if (!val)
121  return nullptr;
122  if (val == unboundValue())
123  Rf_error(_("object '%s' not found 2"), name()->c_str());
124  if (val == missingArgument() && !isDotDotSymbol()) {
125  if (m_name && m_name != String::blank())
126  Rf_error(_("argument \"%s\" is missing, with no default"),
127  name()->c_str());
128  else Rf_error(_("argument is missing, with no default"));
129  }
130  if (val->sexptype() == PROMSXP) {
131  val = Rf_eval(val, env);
132  SET_NAMED(val, 2);
133  }
134  else if (NAMED(val) < 1)
135  SET_NAMED(val, 1);
136  return val;
137 }
