107 bool WeakRef::check()
108 {
109 #ifndef NDEBUG
110  // Check sizes:
111  size_t total_size = getLive()->size() + getFinalizationPending()->size()
112  + getTombstone()->size();
113  if (total_size != size_t(s_count)) {
114  cerr << "WeakRef::check() : tally error\n"
115  << "live size: " << getLive()->size()
116  << "\nfinalization pending size: "
117  << getFinalizationPending()->size()
118  << "\ntombstone size: " << getTombstone()->size()
119  << "\ns_count: " << s_count << '\n';
120  abort();
121  }
122  // Check the live list:
123  for (const WeakRef* wr: *getLive()) {
124  if (wr->m_ready_to_finalize) {
125  cerr << "Node on live list set READY_TO_FINALIZE\n";
126  abort();
127  }
128  if (!wr->m_key) {
129  cerr << "Node on live list with null key\n";
130  abort();
131  }
132  }
133  // Check finalization pending:
134  for (const WeakRef* wr: *getFinalizationPending()) {
135  if (!wr->m_ready_to_finalize) {
136  cerr << "Node on finalization pending list not READY_TO_FINALIZE\n";
137  abort();
138  }
139  if (!wr->m_key) {
140  cerr << "Node on finalization pending list with null key\n";
141  abort();
142  }
143  if (!wr->m_Rfinalizer && !wr->m_Cfinalizer) {
144  cerr << "Node on finalization pending list without finalizer\n";
145  abort();
146  }
147  }
148  // Check tombstone:
149  for (const WeakRef* wr: *getTombstone()) {
150  if (wr->m_ready_to_finalize) {
151  cerr << "Node on tombstone list set READY_TO_FINALIZE\n";
152  abort();
153  }
154  if (wr->m_key) {
155  cerr << "Node on tombstone list with non-null key\n";
156  abort();
157  }
158  }
159 #endif
160  return true;
161 }

172 void WeakRef::finalize()
173 {
174  R_CFinalizer_t Cfin = m_Cfinalizer;
175  GCStackRoot<> key(m_key);
176  GCStackRoot<FunctionBase> Rfin(m_Rfinalizer);
177  // Do this now to ensure that finalizer is run only once, even if
178  // an error occurs:
179  tombstone();
180  if (Cfin) Cfin(key);
181  else if (Rfin) {
182  GCStackRoot<PairList> tail(new PairList(key));
183  GCStackRoot<Expression> e(new Expression(Rfin, tail));
184  Evaluator::evaluate(e, Environment::global());
185  }
186 }

206 void WeakRef::markThru()
207 {
208  WeakRef::check();
209  WRList newlive;
210 
211  WRList* live = getLive();
212  WRList* finalization_pending = getFinalizationPending();
213 
214  // Step 2-3 of algorithm. Mark the value and R finalizer if the
215  // key is marked.
216  {
217  unsigned int marks_applied;
218  do {
219  GCNode::CountingMarker marker;
220  WRList::iterator lit = live->begin();
221  while (lit != live->end()) {
222  WeakRef* wr = *lit++;
223  RObject* key = wr->key();
224  if (key->isMarked()) {
225  RObject* value = wr->value();
226  if (value)
227  marker(value);
228  FunctionBase* Rfinalizer = wr->m_Rfinalizer;
229  if (Rfinalizer)
230  marker(Rfinalizer);
231  wr->transfer(live, &newlive);
232  }
233  }
234  marks_applied = marker.marksApplied();
235  } while (marks_applied > 0);
236  }
237  // Step 4 of algorithm. Process references with unmarked keys.
238  {
239  GCNode::Marker marker;
240  WRList::iterator lit = live->begin();
241  while (lit != live->end()) {
242  WeakRef* wr = *lit++;
243  FunctionBase* Rfinalizer = wr->m_Rfinalizer;
244  if (Rfinalizer)
245  marker(Rfinalizer);
246  if (Rfinalizer || wr->m_Cfinalizer) {
247  marker(wr);
248  marker(wr->m_key);
249  wr->m_ready_to_finalize = true;
250  wr->transfer(live, finalization_pending);
251  }
252  else {
253  wr->tombstone();
254  // Expose to reference-counting collection:
255  wr->m_self = nullptr;
256  }
257  }
258  }
259  // Step 5 of algorithm. Mark all live references with reachable keys.
260  {
261  GCNode::Marker marker;
262  live->splice(live->end(), newlive);
263  for (WeakRef* wr: *live) {
264  marker(wr);
265  }
266  }
267 }

277 void WeakRef::runExitFinalizers()
278 {
279  WeakRef::check();
280  WRList* live = getLive();
281  WRList* finalization_pending = getFinalizationPending();
282 
283  WRList::iterator lit = live->begin();
284  while (lit != live->end()) {
285  WeakRef* wr = *lit++;
286  if (wr->m_finalize_on_exit) {
287  wr->m_ready_to_finalize = true;
288  wr->transfer(live, finalization_pending);
289  }
290  }
291  runFinalizers();

294 bool WeakRef::runFinalizers()
295 {
296  WRList* finalization_pending = getFinalizationPending();
297  if (finalization_pending->empty())
298  return false;
299 
300  // Prevent this function from running again when already in progress.
301  static Rboolean running = FALSE;
302  if (running) return FALSE;
303  running = TRUE;
304 
305  try {
306  WeakRef::check();
307 
308  WRList::iterator lit = finalization_pending->begin();
309  while (lit != finalization_pending->end()) {
310  WeakRef* wr = *lit++;
311  GCStackRoot<> topExp(R_CurrentExpr);
312  size_t savestack = ProtectStack::size();
313  {
314  // An Evaluator is declared for the finalizer to
315  // insure that any errors that might occur do not spill into
316  // the call that triggered the collection:
317  Evaluator evalr;
318  try {
319  wr->finalize();
320  }
321  catch (CommandTerminated) {
322  }
323  // Expose WeakRef to reference-counting collection:
324  wr->m_self = nullptr;
325  }
326  ProtectStack::restoreSize(savestack);
327  R_CurrentExpr = topExp;
328  }
329  running = FALSE;
330  } catch (...) {
331  running = FALSE;
332  throw;
333  }
334  return true;
335 }
