49 /* Use R_alloc as this might get interrupted */
50 static char *Rconn_getline2(Rconnection con, char *buf, int bufsize)
51 {
52  int c, nbuf = 0;
53  while((c = Rconn_fgetc(con)) != R_EOF) {
54  if(nbuf+1 >= bufsize) { // allow for terminator below
55  bufsize *= 2;
56  char *buf2 = R_alloc(bufsize, sizeof(char));
57  memcpy(buf2, buf, nbuf);
58  buf = buf2;
59  }
60  if(c != '\n'){
61  buf[nbuf++] = (char) c;
62  } else {
63  buf[nbuf++] = '\0';
64  break;
65  }
66  }
67  if (!nbuf)
68  return nullptr;
69  /* Make sure it is null-terminated even if file did not end with
70  * newline.
71  */
72  if(buf[nbuf-1]) buf[nbuf] = '\0';
73  return buf;
74 }

76 HIDDEN SEXP do_readDCF(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* file_, rho::RObject* fields_, rho::RObject* keep_white_)
77 {
78  int nwhat, nret, nc, nr, m, k, lastm, need, i, n_eblanklines = 0;
79  bool blank_skip, field_skip = false;
80  int whatlen, dynwhat, buflen = 8096; // was 100, but that re-alloced often
81  char *line, *buf;
82  regex_t blankline, contline, trailblank, regline, eblankline;
83  regmatch_t regmatch[1];
84  SEXP file, what, what2, retval, retval2, dims, dimnames;
85  Rconnection con = nullptr;
86  bool wasopen, is_eblankline;
87 
88  SEXP fold_excludes;
89  bool field_fold = true, has_fold_excludes;
90  const char *field_name;
91  int offset = 0; /* -Wall */
92 
93  file = file_;
94  con = getConnection(Rf_asInteger(file));
95  wasopen = con->isopen;
96  if(!wasopen && !con->open(con))
97  Rf_error(_("cannot open the connection"));
98 
99  try {
100  if(!con->canread) Rf_error(_("cannot read from this connection"));
101 
102  PROTECT(what = Rf_coerceVector(fields_, STRSXP)); /* argument fields */
103  nwhat = LENGTH(what);
104  dynwhat = (nwhat == 0);
105 
106  PROTECT(fold_excludes = Rf_coerceVector(keep_white_, STRSXP));
107  has_fold_excludes = (LENGTH(fold_excludes) > 0);
108 
109  buf = static_cast<char *>(malloc(buflen));
110  if(!buf) Rf_error(_("could not allocate memory for 'read.dcf'"));
111  nret = 20;
112  /* it is easier if we first have a record per column */
113  PROTECT(retval = allocMatrixNA(STRSXP, LENGTH(what), nret));
114 
115  /* These used to use [:blank:] but that can match \xa0 as part of
116  a UTF-8 character (and is nbspace on Windows). */
117  tre_regcomp(&blankline, "^[[:blank:]]*$", REG_NOSUB & REG_EXTENDED);
118  tre_regcomp(&trailblank, "[ \t]+$", REG_EXTENDED);
119  tre_regcomp(&contline, "^[[:blank:]]+", REG_EXTENDED);
120  tre_regcomp(&regline, "^[^:]+:[[:blank:]]*", REG_EXTENDED);
121  tre_regcomp(&eblankline, "^[[:space:]]+\\.[[:space:]]*$", REG_EXTENDED);
122 
123  k = 0;
124  lastm = -1; /* index of the field currently being recorded */
125  blank_skip = true;
126  rho::RAllocStack::Scope rscope;
127  char buf0[MAXELTSIZE];
128  while((line = Rconn_getline2(con, buf0, MAXELTSIZE))) {
129  if(strlen(line) == 0 ||
130  tre_regexecb(&blankline, line, 0, nullptr, 0) == 0) {
131  /* A blank line. The first one after a record ends a new
132  * record, subsequent ones are skipped */
133  if(!blank_skip) {
134  k++;
135  if(k > nret - 1){
136  nret *= 2;
137  PROTECT(retval2 = allocMatrixNA(STRSXP, LENGTH(what), nret));
138  transferVector(retval2, retval);
139  UNPROTECT_PTR(retval);
140  retval = retval2;
141  }
142  blank_skip = true;
143  lastm = -1;
144  field_skip = false;
145  field_fold = true;
146  n_eblanklines = 0;
147  }
148  } else {
149  blank_skip = false;
150  if(tre_regexecb(&contline, line, 1, regmatch, 0) == 0) {
151  /* A continuation line: wrong if at the beginning of a
152  record. */
153  if((lastm == -1) && !field_skip) {
154  line[20] = '\0';
155  Rf_error(_("Found continuation line starting '%s ...' at begin of record."),
156  line);
157  }
158  if(lastm >= 0) {
159  need = int(strlen(R_CHAR(STRING_ELT(retval,
160  lastm + nwhat * k)))) + 2;
161  if(tre_regexecb(&eblankline, line, 0, nullptr, 0) == 0) {
162  is_eblankline = true;
163  if(field_fold) {
164  n_eblanklines++;
165  continue;
166  }
167  } else {
168  is_eblankline = false;
169  if(field_fold) {
170  offset = regmatch[0].rm_eo;
171  /* Also remove trailing whitespace. */
172  if((tre_regexecb(&trailblank, line, 1,
173  regmatch, 0) == 0))
174  line[regmatch[0].rm_so] = '\0';
175  } else {
176  offset = 0;
177  }
178  need += int(strlen(line + offset)) + n_eblanklines;
179  }
180  if(buflen < need) {
181  char *tmp = static_cast<char *>(realloc(buf, need));
182  if(!tmp) {
183  free(buf);
184  Rf_error(_("could not allocate memory for 'read.dcf'"));
185  } else buf = tmp;
186  buflen = need;
187  }
188  strcpy(buf, R_CHAR(STRING_ELT(retval, lastm + nwhat * k)));
189  if(strlen(buf) || !field_fold)
190  strcat(buf, "\n");
191  if(!is_eblankline) {
192  if(n_eblanklines > 0) {
193  for(i = 0; i < n_eblanklines; i++) {
194  strcat(buf, "\n");
195  }
196  n_eblanklines = 0;
197  }
198  strcat(buf, line + offset);
199  }
200  SET_STRING_ELT(retval, lastm + nwhat * k, Rf_mkChar(buf));
201  }
202  } else {
203  if(tre_regexecb(&regline, line, 1, regmatch, 0) == 0){
204  for(m = 0; m < nwhat; m++){
205  whatlen = int(strlen(R_CHAR(STRING_ELT(what, m))));
206  if(static_cast<int>(strlen(line)) > whatlen &&
207  line[whatlen] == ':' &&
208  strncmp(R_CHAR(STRING_ELT(what, m)),
209  line, whatlen) == 0) {
210  /* An already known field we are recording. */
211  lastm = m;
212  field_skip = false;
213  field_name = R_CHAR(STRING_ELT(what, lastm));
214  if(has_fold_excludes) {
215  field_fold =
216  field_is_foldable_p(field_name,
217  fold_excludes);
218  }
219  if(field_fold) {
220  offset = regmatch[0].rm_eo;
221  /* Also remove trailing whitespace. */
222  if((tre_regexecb(&trailblank, line, 1,
223  regmatch, 0) == 0))
224  line[regmatch[0].rm_so] = '\0';
225  } else {
226  offset = 0;
227  }
228  SET_STRING_ELT(retval, m + nwhat * k,
229  Rf_mkChar(line + offset));
230  break;
231  } else {
232  /* This is a field, but not one prespecified */
233  lastm = -1;
234  field_skip = true;
235  }
236  }
237  if(dynwhat && (lastm == -1)) {
238  /* A previously unseen field and we are
239  * recording all fields */
240  field_skip = false;
241  PROTECT(what2 = Rf_allocVector(STRSXP, nwhat+1));
242  PROTECT(retval2 = allocMatrixNA(STRSXP,
243  Rf_nrows(retval)+1,
244  Rf_ncols(retval)));
245  if(nwhat > 0) {
246  Rf_copyVector(what2, what);
247  for(nr = 0; nr < Rf_nrows(retval); nr++){
248  for(nc = 0; nc < Rf_ncols(retval); nc++){
249  SET_STRING_ELT(retval2, nr+nc*Rf_nrows(retval2),
250  STRING_ELT(retval,
251  nr+nc*Rf_nrows(retval)));
252  }
253  }
254  }
255  UNPROTECT_PTR(retval);
256  UNPROTECT_PTR(what);
257  retval = retval2;
258  what = what2;
259  /* Make sure enough space was used */
260  need = int((Rf_strchr(line, ':') - line + 1));
261  if(buflen < need){
262  char *tmp = static_cast<char *>(realloc(buf, need));
263  if(!tmp) {
264  free(buf);
265  Rf_error(_("could not allocate memory for 'read.dcf'"));
266  } else buf = tmp;
267  buflen = need;
268  }
269  strncpy(buf, line, Rf_strchr(line, ':') - line);
270  buf[Rf_strchr(line, ':') - line] = '\0';
271  SET_STRING_ELT(what, nwhat, Rf_mkChar(buf));
272  nwhat++;
273  /* lastm uses C indexing, hence nwhat - 1 */
274  lastm = nwhat - 1;
275  field_name = R_CHAR(STRING_ELT(what, lastm));
276  if(has_fold_excludes) {
277  field_fold =
278  field_is_foldable_p(field_name,
279  fold_excludes);
280  }
281  offset = regmatch[0].rm_eo;
282  if(field_fold) {
283  /* Also remove trailing whitespace. */
284  if((tre_regexecb(&trailblank, line, 1,
285  regmatch, 0) == 0))
286  line[regmatch[0].rm_so] = '\0';
287  }
288  SET_STRING_ELT(retval, lastm + nwhat * k,
289  Rf_mkChar(line + offset));
290  }
291  } else {
292  /* Must be a regular line with no tag ... */
293  line[20] = '\0';
294  Rf_error(_("Line starting '%s ...' is malformed!"), line);
295  }
296  }
297  }
298  }
299  if(!wasopen) con->close(con);
300  } catch (...) {
301  if (!wasopen && con->isopen)
302  con->close(con);
303  throw;
304  }
305  free(buf);
306  tre_regfree(&blankline);
307  tre_regfree(&contline);
308  tre_regfree(&trailblank);
309  tre_regfree(&regline);
310  tre_regfree(&eblankline);
311 
312  if(!blank_skip) k++;
313 
314  /* and now transpose the whole matrix */
315  PROTECT(retval2 = allocMatrixNA(STRSXP, k, LENGTH(what)));
316  Rf_copyMatrix(retval2, retval, TRUE);
317 
318  PROTECT(dimnames = Rf_allocVector(VECSXP, 2));
319  PROTECT(dims = Rf_allocVector(INTSXP, 2));
320  INTEGER(dims)[0] = k;
321  INTEGER(dims)[1] = LENGTH(what);
322  SET_VECTOR_ELT(dimnames, 1, what);
323  Rf_setAttrib(retval2, Symbols::DimSymbol, dims);
324  Rf_setAttrib(retval2, Symbols::DimNamesSymbol, dimnames);
325  UNPROTECT(6);
326  return(retval2);
327 }

331 {
