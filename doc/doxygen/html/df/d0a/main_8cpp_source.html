222 {
223  ParseStatus status;

235  case PARSE_OK:

241  SET_SYMVALUE(R_LastvalueSymbol, R_CurrentExpr);
242  UNPROTECT(1);

251  break;
252  case PARSE_EOF:

255  break;
256  case PARSE_INCOMPLETE:
257  /* can't happen: just here to quieten -Wall */

260  }

293 /*
294  This is a reorganization of the REPL (Read-Eval-Print Loop) to separate
295  the loop from the actions of the body. The motivation is to make the iteration
296  code (Rf_ReplIteration) available as a separately callable routine
297  to avoid cutting and pasting it when one wants a single iteration
298  of the loop. This is needed as we allow different implementations
299  of event loops. Currently (summer 2002), we have a package in
300  preparation that uses Rf_ReplIteration within either the
301  Tcl or Gtk event loop and allows either (or both) loops to
302  be used as a replacement for R's loop and take over the event
303  handling for the R process.
304 
305  The modifications here are intended to leave the semantics of the REPL
306  unchanged, just separate into routines. So the variables that maintain
307  the state across iterations of the loop are organized into a structure
308  and passed to Rf_ReplIteration() from Rf_ReplConsole().
309 */

345 int
346 Rf_ReplIteration(SEXP rho, unsigned int savestack, R_ReplState *state)
347 {
348  int c, browsevalue;
349  SEXP value, thisExpr;
350  Rboolean wasDisplayed = FALSE;
351  int browselevel = int(Browser::numberActive());
352 
353  if(!*state->bufp) {
354  R_Busy(0);
355  if (R_ReadConsole(R_PromptString(browselevel, state->prompt_type),
356  state->buf, CONSOLE_BUFFER_SIZE, 1) == 0)
357  return(-1);
358  state->bufp = state->buf;
359  }
360 #ifdef SHELL_ESCAPE /* not default */
361  if (*state->bufp == '!') {
362  R_system(&(state->buf[1]));
363  state->buf[0] = '\0';
364  return(0);
365  }
366 #endif /* SHELL_ESCAPE */
367  while((c = *state->bufp++)) {
368  R_IoBufferPutc(c, &R_ConsoleIob);
369  if(c == ';' || c == '\n') break;
370  }
371 
372  ProtectStack::restoreSize(savestack);
373  R_CurrentExpr = R_Parse1Buffer(&R_ConsoleIob, 0, &state->status);
374 
375  switch(state->status) {
376 
377  case PARSE_NULL:
378 
379  /* The intention here is to break on CR but not on other
380  null statements: see PR#9063 */
381  if (browselevel && !R_DisableNLinBrowser
382  && streql(reinterpret_cast<char *>(state->buf), "\n")) return -1;
383  R_IoBufferWriteReset(&R_ConsoleIob);
384  state->prompt_type = 1;
385  return 1;
386 
387  case PARSE_OK:
388  {
389  R_IoBufferReadReset(&R_ConsoleIob);
390  R_CurrentExpr = R_Parse1Buffer(&R_ConsoleIob, 1, &state->status);
391 #ifdef PROVENANCE_TRACKING
392  ProvenanceTracker::CommandScope scope(R_CurrentExpr);
393 #endif
394  if (browselevel) {
395  browsevalue = ParseBrowser(R_CurrentExpr, rho);
396  if(browsevalue == 1) return -1;
397  if(browsevalue == 2) {
398  R_IoBufferWriteReset(&R_ConsoleIob);
399  return 0;
400  }
401  }
402  /* PR#15770 We don't want to step into expressions entered at the debug prompt.
403  The 'S' will be changed back to 's' after the next eval. */
404  if (R_BrowserLastCommand == 's') R_BrowserLastCommand = 'S';
405  R_Visible = FALSE;
406  R_EvalDepth = 0;
407  resetTimeLimits();
408  PROTECT(thisExpr = R_CurrentExpr);
409  R_Busy(1);
410  value = Rf_eval(thisExpr, rho);
411  SET_SYMVALUE(R_LastvalueSymbol, value);
412  wasDisplayed = R_Visible;
413  if (R_Visible)
414  Rf_PrintValueEnv(value, rho);
415  if (R_CollectWarnings)
416  PrintWarnings();
417  Rf_callToplevelHandlers(thisExpr, value, TRUE, wasDisplayed);
418  R_CurrentExpr = value; /* Necessary? Doubt it. */
419  UNPROTECT(1);
420  if (R_BrowserLastCommand == 'S') R_BrowserLastCommand = 's';
421  R_IoBufferWriteReset(&R_ConsoleIob);
422  state->prompt_type = 1;
423  return(1);
424  }
425 
426  case PARSE_ERROR:
427 
428  state->prompt_type = 1;
429  parseError(R_NilValue, 0);
430  R_IoBufferWriteReset(&R_ConsoleIob);
431  return(1);
432 
433  case PARSE_INCOMPLETE:
434 
435  R_IoBufferReadReset(&R_ConsoleIob);
436  state->prompt_type = 2;
437  return(2);
438 
439  case PARSE_EOF:
440 
441  return(-1);
442  break;
443  }
444 
445  return(0);
446 }
