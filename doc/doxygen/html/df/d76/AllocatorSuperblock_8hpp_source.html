62  AllocatorSuperblock(unsigned size_class, unsigned bitset_entries) : m_size_class(size_class),
63  m_next_untouched(0)
64  {
65  // Here we mark all bitset entries as free so that we don't have to do
66  // precise range checking while iterating over currently allocated blocks
67  // when the number of blocks is not evenly divisible by 64.
68  for (unsigned int i = 0; i < bitset_entries; ++i)
69  {
70  m_free[i] = ~0ull;
71  }
