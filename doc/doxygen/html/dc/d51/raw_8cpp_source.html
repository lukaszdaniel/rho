55 SEXP attribute_hidden do_rawToChar(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* multiple_)
56 {
57  SEXP ans, x = x_;
58 
59  if (!isRaw(x))
60  error(_("argument 'x' must be a raw vector"));
61  int multiple = asLogical(multiple_);
62  if (multiple == NA_LOGICAL)
63  error(_("argument 'multiple' must be TRUE or FALSE"));
64  if (multiple) {
65  R_xlen_t i, nc = XLENGTH(x);
66  char buf[2];
67  buf[1] = '\0';
68  PROTECT(ans = allocVector(STRSXP, nc));
69  for (i = 0; i < nc; i++) {
70  buf[0] = char( RAW(x)[i]);
71  SET_STRING_ELT(ans, i, Rf_mkChar(buf));
72  }
73  /* do we want to copy e.g. names here? */
74  UNPROTECT(1);
75  return ans;
76  } else {
77  int i, j, nc = LENGTH(x);
78  /* String is not necessarily 0-terminated and may contain nuls.
79  Strip trailing nuls */
80  for (i = 0, j = -1; i < nc; i++) if(RAW(x)[i]) j = i;
81  nc = j + 1;
82  return Rf_ScalarString(
83  Rf_mkCharLenCE(reinterpret_cast<const char *>(RAW(x)), j+1, CE_NATIVE));
84  }
85 }

93 
94  if (!isRaw(x))
95  Rf_error(_("argument 'x' must be a raw vector"));
96  if (shift == NA_INTEGER || shift < -8 || shift > 8)
97  Rf_error(_("argument 'shift' must be a small integer"));
98  PROTECT(ans = Rf_duplicate(x));
99  if (shift > 0)
100  for (R_xlen_t i = 0; i < XLENGTH(x); i++)

104  RAW(ans)[i] >>= (-shift);
105  UNPROTECT(1);
106  return ans;

146 SEXP attribute_hidden do_packBits(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* type_)
147 {
148  SEXP ans, x = x_, stype = type_;
149  Rboolean useRaw;
150  R_xlen_t i, len = XLENGTH(x), slen;
151  int fac;
152 
153  if (TYPEOF(x) != RAWSXP && TYPEOF(x) != LGLSXP && TYPEOF(x) != INTSXP)
154  Rf_error(_("argument 'x' must be raw, integer or logical"));
155  if (!Rf_isString(stype) || LENGTH(stype) != 1)
156  Rf_error(_("argument '%s' must be a character string"), "type");
157  useRaw = RHOCONSTRUCT(Rboolean, strcmp(R_CHAR(STRING_ELT(stype, 0)), "integer"));
158  fac = useRaw ? 8 : 32;
159  if (len% fac)
160  Rf_error(_("argument 'x' must be a multiple of %d long"), fac);
161  slen = len/fac;
162  PROTECT(ans = Rf_allocVector(useRaw ? RAWSXP : INTSXP, slen));
163  for (i = 0; i < slen; i++)
164  if (useRaw) {
165  Rbyte btmp = 0;
166  for (int k = 7; k >= 0; k--) {
167  btmp <<= 1;
168  if (isRaw(x))
169  btmp |= RAW(x)[8*i + k] & 0x1;
170  else if (Rf_isLogical(x) || Rf_isInteger(x)) {
171  int j = INTEGER(x)[8*i+k];
172  if (j == NA_INTEGER)
173  Rf_error(_("argument 'x' must not contain NAs"));
174  btmp |= j & 0x1;
175  }
176  }
177  RAW(ans)[i] = btmp;
178  } else {
179  unsigned int itmp = 0;
180  for (int k = 31; k >= 0; k--) {
181  itmp <<= 1;
182  if (isRaw(x))
183  itmp |= RAW(x)[32*i + k] & 0x1;
184  else if (Rf_isLogical(x) || Rf_isInteger(x)) {
185  int j = INTEGER(x)[32*i+k];
186  if (j == NA_INTEGER)
187  Rf_error(_("argument 'x' must not contain NAs"));
188  itmp |= j & 0x1;
189  }
190  }
191  INTEGER(ans)[i] = int( itmp);
192  }
193  UNPROTECT(1);
194  return ans;
195 }

197 static int mbrtoint(int *w, const char *s)
198 {
199  unsigned int byte;
200  byte = *(reinterpret_cast<unsigned char *>(RHO_C_CAST(char*, s)));
201 
202  if (byte == 0) {
203  *w = 0;
204  return 0;
205  } else if (byte < 0xC0) {
206  *w = int( byte);
207  return 1;
208  } else if (byte < 0xE0) {
209  if (!s[1]) return -2;
210  if ((s[1] & 0xC0) == 0x80) {
211  *w = int ((((byte & 0x1F) << 6) | (s[1] & 0x3F)));
212  return 2;
213  } else return -1;
214  } else if (byte < 0xF0) {
215  if (!s[1] || !s[2]) return -2;
216  if (((s[1] & 0xC0) == 0x80) && ((s[2] & 0xC0) == 0x80)) {
217  *w = int (((byte & 0x0F) << 12)
218  | ((s[1] & 0x3F) << 6) | (s[2] & 0x3F));
219  byte = *w;
220  if (byte >= 0xD800 && byte <= 0xDFFF) return -1; /* surrogate */
221  if (byte == 0xFFFE || byte == 0xFFFF) return -1;
222  return 3;
223  } else return -1;
224  } else if (byte < 0xF8) {
225  if (!s[1] || !s[2] || !s[3]) return -2;
226  if (((s[1] & 0xC0) == 0x80)
227  && ((s[2] & 0xC0) == 0x80)
228  && ((s[3] & 0xC0) == 0x80)) {
229  *w = int (((byte & 0x07) << 18)
230  | ((s[1] & 0x3F) << 12)
231  | ((s[2] & 0x3F) << 6)
232  | (s[3] & 0x3F));
233  byte = *w;
234  return 4;
235  } else return -1;
236  } else if (byte < 0xFC) {
237  if (!s[1] || !s[2] || !s[3] || !s[4]) return -2;
238  if (((s[1] & 0xC0) == 0x80)
239  && ((s[2] & 0xC0) == 0x80)
240  && ((s[3] & 0xC0) == 0x80)
241  && ((s[4] & 0xC0) == 0x80)) {
242  *w = int (((byte & 0x03) << 24)
243  | ((s[1] & 0x3F) << 18)
244  | ((s[2] & 0x3F) << 12)
245  | ((s[3] & 0x3F) << 6)
246  | (s[4] & 0x3F));
247  byte = *w;
248  return 5;
249  } else return -1;
250  } else {
251  if (!s[1] || !s[2] || !s[3] || !s[4] || !s[5]) return -2;
252  if (((s[1] & 0xC0) == 0x80)
253  && ((s[2] & 0xC0) == 0x80)
254  && ((s[3] & 0xC0) == 0x80)
255  && ((s[4] & 0xC0) == 0x80)
256  && ((s[5] & 0xC0) == 0x80)) {
257  *w = int (((byte & 0x01) << 30)
258  | ((s[1] & 0x3F) << 24)
259  | ((s[2] & 0x3F) << 18)
260  | ((s[3] & 0x3F) << 12)
261  | ((s[5] & 0x3F) << 6)
262  | (s[5] & 0x3F));
263  byte = *w;
264  return 6;
265  } else return -1;
266  }
267  /* return -2; not reached */
268 }

270 SEXP attribute_hidden do_utf8ToInt(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_)
271 {
272  SEXP ans, x = x_;
273  int tmp, used = 0; /* -Wall */
274  R_xlen_t i, j, nc;
275 
276  if (!Rf_isString(x) || LENGTH(x) == 0)
277  Rf_error(_("argument must be a character vector of length 1"));
278  if (LENGTH(x) > 1)
279  Rf_warning(_("argument should be a character vector of length 1\nall but the first element will be ignored"));
280  if (STRING_ELT(x, 0) == rho::String::NA()) return Rf_ScalarInteger(NA_INTEGER);
281  const char *s = R_CHAR(STRING_ELT(x, 0));
282  if (!utf8Valid(s)) return Rf_ScalarInteger(NA_INTEGER);
283  nc = XLENGTH(STRING_ELT(x, 0)); /* ints will be shorter */
284  int *ians = static_cast<int *>(RHO_alloc(nc, sizeof(int)));
285  for (i = 0, j = 0; i < nc; i++) {
286  used = mbrtoint(&tmp, s);
287  if (used <= 0) break;
288  ians[j++] = tmp;
289  s += used;
290  }
291  if (used < 0) error(_("invalid UTF-8 string"));
292  ans = Rf_allocVector(INTSXP, j);
293  if (j) memcpy(INTEGER(ans), ians, sizeof(int) * j);
294  return ans;
295 }

299  { 0x7f, 0x7ff, 0xffff, 0x1fffff, 0x3ffffff, 0x7fffffff};

302 static size_t inttomb(char *s, const int wc)
303 {
304  int i, j;
305  unsigned int cvalue = wc;
306  char buf[10], *b;
307 
308  b = s ? s : buf;
309  if (cvalue == 0) {*b = 0; return 0;}
310  for (i = 0; i < RHOCONSTRUCT(int, sizeof(utf8_table1)/sizeof(int)); i++)
311  if (RHOCONSTRUCT(int, cvalue) <= utf8_table1[i]) break;
312  b += i;
313  for (j = i; j > 0; j--) {
314  *b-- = char((0x80 | (cvalue & 0x3f)));
315  cvalue >>= 6;
316  }
317  *b = char((utf8_table2[i] | cvalue));
318  return i + 1;
319 }

324 {

326  int multiple;
327  size_t used, len;
328  char buf[10], *tmp;
329 
330  PROTECT(x = Rf_coerceVector(x_, INTSXP));
331  if (!Rf_isInteger(x))
332  Rf_error(_("argument 'x' must be an integer vector"));
333  multiple = Rf_asLogical(multiple_);
334  if (multiple == NA_LOGICAL)
335  Rf_error(_("argument 'multiple' must be TRUE or FALSE"));
336  if (multiple) {
337  R_xlen_t i, nc = XLENGTH(x);
338  PROTECT(ans = Rf_allocVector(STRSXP, nc));
339  for (i = 0; i < nc; i++) {
340  if (INTEGER(x)[i] == NA_INTEGER)
341  SET_STRING_ELT(ans, i, rho::String::NA());
342  else {
343  used = inttomb(buf, INTEGER(x)[i]);
344  buf[used] = '\0';
345  SET_STRING_ELT(ans, i, Rf_mkCharCE(buf, CE_UTF8));
346  }
347  }
348  /* do we want to copy e.g. names here? */
349  } else {
350  int i, nc = LENGTH(x);
351  Rboolean haveNA = FALSE;
352  /* Note that this gives zero length for input '0', so it is omitted */
353  for (i = 0, len = 0; i < nc; i++) {
354  if (INTEGER(x)[i] == NA_INTEGER) { haveNA = TRUE; break; }
355  len += inttomb(nullptr, INTEGER(x)[i]);
356  }
357  if (haveNA) {
358  UNPROTECT(1);
359  return Rf_ScalarString(rho::String::NA());
360  }
361  if (len >= 10000) {
362  tmp = Calloc(len+1, char);
363  } else {
364  R_CheckStack2(len+1);
365  tmp = RHO_S_CAST(char*, alloca(len+1)); tmp[len] = '\0';
366  }
367  for (i = 0, len = 0; i < nc; i++) {
368  used = inttomb(buf, INTEGER(x)[i]);
369  strncpy(tmp + len, buf, used);
370  len += used;
371  }
372  ans = PROTECT(Rf_ScalarString(Rf_mkCharLenCE(tmp, int( len), CE_UTF8)));
373  if(len >= 10000) Free(tmp);
374  }
375  UNPROTECT(2);
376  return ans;
377 }
