28 /* =========
29  * Printing:
30  * =========
31  *
32  * All printing in R is done via the functions Rprintf and REprintf
33  * or their (v) versions Rvprintf and REvprintf.
34  * These routines work exactly like (v)printf(3). Rprintf writes to
35  * ``standard output''. It is redirected by the sink() function,
36  * and is suitable for ordinary output. REprintf writes to
37  * ``standard error'' and is useful for error messages and warnings.
38  * It is not redirected by sink().
39  *
40  * See ./format.cpp for the format_FOO_ functions which provide
41  * ~~~~~~~~~~ the length, width, etc.. that are used here.
42  * See ./print.cpp for do_printdefault, do_prmatrix, etc.
43  *
44  *
45  * Here, the following UTILITIES are provided:
46  *
47  * The utilities EncodeLogical, EncodeInteger, EncodeReal
48  * and EncodeString can be used to convert R objects to a form suitable
49  * for printing. These print the values passed in a formatted form
50  * or, in the case of NA values, an NA indicator. EncodeString takes
51  * care of printing all the standard ANSI escapes \a, \t \n etc.
52  * so that these appear in their backslash form in the string. There
53  * is also a routine called Rstrlen which computes the length of the
54  * string in its escaped rather than literal form.
55  *
56  * Finally there is a routine called EncodeElement which will encode
57  * a single R-vector element. This is used in deparse and write.table.
58  */

97 R_size_t R_Decode2Long(char* p, int* ierr)
98 {
99  R_size_t v = strtol(p, &p, 10);
100  *ierr = 0;
101  if(p[0] == '\0') return v;
102  /* else look for letter-code ending : */
103  if(R_Verbose)
104  REprintf("R_Decode2Long(): v=%ld\n", v);
105  // NOTE: currently, positive *ierr are not differentiated in the callers:
106  if(p[0] == 'G') {
107  if((Giga * double(v)) > double(R_SIZE_T_MAX)) { *ierr = 4; return(v); }
108  return R_size_t(Giga) * v;
109  }
110  else if(p[0] == 'M') {
111  if((Mega * double(v)) > double(R_SIZE_T_MAX)) { *ierr = 1; return(v); }
112  return R_size_t(Mega) * v;
113  }
114  else if(p[0] == 'K') {
115  if((1024 * double(v)) > double(R_SIZE_T_MAX)) { *ierr = 2; return(v); }
116  return (1024*v);
117  }
118  else if(p[0] == 'k') {
119  if((1000 * double(v)) > double(R_SIZE_T_MAX)) { *ierr = 3; return(v); }
120  return (1000*v);
121  }
122  else {
123  *ierr = -1;
124  return(v);
125  }
126 }

189 const char* EncodeReal0(double x, int w, int d, int e, const char* dec)
190 {
191  static char buff[NB], buff2[2*NB];
192  char fmt[20], *out = buff;
193 
194  /* IEEE allows signed zeros (yuck!) */
195  if (x == 0.0) x = 0.0;
196  if (!std::isfinite(x)) {
197  if(ISNA(x)) snprintf(buff, NB, "%*s", min(w, int(NB-1)), R_CHAR(R_print.na_string));
198  else if(std::isnan(x)) snprintf(buff, NB, "%*s", min(w, int(NB-1)), "NaN");
199  else if(x > 0) snprintf(buff, NB, "%*s", min(w, int(NB-1)), "Inf");
200  else snprintf(buff, NB, "%*s", min(w, int(NB-1)), "-Inf");
201  }
202  else if (e) {
203  if(d) {
204  sprintf(fmt,"%%#%d.%de", min(w, int(NB-1)), d);
205  snprintf(buff, NB, fmt, x);
206  }
207  else {
208  sprintf(fmt,"%%%d.%de", min(w, int(NB-1)), d);
209  snprintf(buff, NB, fmt, x);
210  }
211  }
212  else { /* e = 0 */
213  sprintf(fmt,"%%%d.%df", min(w, int(NB-1)), d);
214  snprintf(buff, NB, fmt, x);
215  }
216  buff[NB-1] = '\0';
217 
218  if(strcmp(dec, ".")) {
219  char *p, *q;
220  for(p = buff, q = buff2; *p; p++) {
221  if(*p == '.') for(const char *r = dec; *r; r++) *q++ = *r;
222  else *q++ = *p;
223  }
224  *q = '\0';
225  out = buff2;
226  }
227 
228  return out;
229 }

241  else if(x > 0) snprintf(buff, NB, "%*s", min(w, int(NB-1)), "Inf");
242  else snprintf(buff, NB, "%*s", min(w, int(NB-1)), "-Inf");
243  }
244  else if (e) {
245  if(d) {
246  sprintf(fmt,"%%#%d.%de", min(w, int(NB-1)), d);
247  snprintf(buff, NB, fmt, x);
248  }
249  else {
250  sprintf(fmt,"%%%d.%de", min(w, int(NB-1)), d);
251  snprintf(buff, NB, fmt, x);
252  }
253  }
254  else { /* e = 0 */
255  sprintf(fmt,"%%%d.%df", min(w, int(NB-1)), d);
256  snprintf(buff, NB, fmt, x);
257  }
258  buff[NB-1] = '\0';
259 
260  // Drop trailing zeroes
261  for (char *p = buff; *p; p++) {
262  if(*p == '.') {

268  while((*(replace++) = *(p++)))
269  ;
270  break;
271  }
272  }
273 
274  if(strcmp(dec, ".")) {
275  char *p, *q;
276  for(p = buff, q = buff2; *p; p++) {
277  if(*p == '.') for(const char *r = dec; *r; r++) *q++ = *r;
278  else *q++ = *p;
279  }
280  *q = '\0';
281  out = buff2;
282  }
283 
284  return out;
285 }

333 constexpr size_t NB3 = NB + 3;
334 const char* EncodeComplex(Rcomplex x, int wr, int dr, int er, int wi, int di, int ei, const char* dec)
335 {
336  static char buff[NB3];
337 
338  /* IEEE allows signed zeros; strip these here */
339  if (x.r == 0.0) x.r = 0.0;
340  if (x.i == 0.0) x.i = 0.0;
341 
342  if (ISNA(x.r) || ISNA(x.i)) {
343  snprintf(buff, NB,
344  "%*s", /* was "%*s%*s", R_print.gap, "", */
345  min(wr+wi+2, int(NB-1)), R_CHAR(R_print.na_string));
346  } else {
347  char Re[NB];
348  const char *Im, *tmp;
349  int flagNegIm = 0;
350  Rcomplex y;
351  /* formatComplex rounded, but this does not, and we need to
352  keep it that way so we don't get strange trailing zeros.
353  But we do want to avoid printing small exponentials that
354  are probably garbage.
355  */
356  z_prec_r(&y, &x, R_print.digits);
357  /* EncodeReal has static buffer, so copy */
358  tmp = EncodeReal0(y.r == 0. ? y.r : x.r, wr, dr, er, dec);
359  strcpy(Re, tmp);
360  if ( (flagNegIm = (x.i < 0)) ) x.i = -x.i;
361  Im = EncodeReal0(y.i == 0. ? y.i : x.i, wi, di, ei, dec);
362  snprintf(buff, NB3, "%s%s%si", Re, flagNegIm ? "-" : "+", Im);
363  }
364  buff[NB3-1] = '\0';
365  return buff;
366 }

386 /* strlen() using escaped rather than literal form.
387  In MBCS locales it works in characters, and reports in display width.
388  Rstrwid is also used in printarray.cpp.
389 
390  This supported embedded nuls when we had those.

392 HIDDEN
393 int Rstrwid(const char *str, int slen, cetype_t ienc, int quote)
394 {
395  const char* p = str;
396  int len = 0, i;
397 
398  if(ienc == CE_BYTES) { // not currently used for that encoding
399  for (i = 0; i < slen; i++) {
400  unsigned char k = str[i];
401  if (k >= 0x20 && k < 0x80) len += 1;
402  else len += 4;
403  }
404  return len;
405  }
406  /* Future-proof: currently that is all Rstrlen calls it with,
407  and printarray has CE_NATIVE explicitly */
408  if(ienc > 2) // CE_NATIVE, CE_UTF8, CE_BYTES are supported
409  Rf_warning("unsupported encoding (%d) in Rstrwid", ienc);
410  if(mbcslocale || ienc == CE_UTF8) {
411  int res;
412  mbstate_t mb_st;
413  wchar_t wc;
414  Rwchar_t k; /* not wint_t as it might be signed */
415 
416  if(ienc != CE_UTF8) mbs_init(&mb_st);
417  for (i = 0; i < slen; i++) {
418  res = (ienc == CE_UTF8) ? int(Rf_utf8toucs(&wc, p)):
419  int(mbrtowc(&wc, p, MB_CUR_MAX, nullptr));
420  if(res >= 0) {
421  if (ienc == CE_UTF8 && IS_HIGH_SURROGATE(wc))
422  k = utf8toucs32(wc, p);
423  else
424  k = wc;
425  if(0x20 <= k && k < 0x7f && iswprint((wint_t)k)) {
426  switch(wc) {
427  case L'\\':
428  len += 2;
429  break;
430  case L'\'':
431  case L'"':
432  case L'`':
433  len += (quote == *p) ? 2 : 1;
434  break;
435  default:
436  len++; /* assumes these are all width 1 */
437  break;
438  }
439  p++;
440  } else if (k < 0x80) {
441  switch(wc) {
442  case L'\a':
443  case L'\b':
444  case L'\f':
445  case L'\n':
446  case L'\r':
447  case L'\t':
448  case L'\v':
449  case L'\0':
450  len += 2; break;
451  default:
452  /* print in octal */
453  len += 4; break;
454  }
455  p++;
456  } else {
457  len += iswprint(wint_t(k)) ? Ri18n_wcwidth(wc) :
458  (k > 0xffff ? 10 : 6);
459  i += (res - 1);
460  p += res;
461  }
462  } else {
463  len += 4;
464  p++;
465  }
466  }
467  } else // not MBCS nor marked as UTF-8
468  for (i = 0; i < slen; i++) {
469  if(static_cast<unsigned char>(*p) < 0x80) {
470  /* ASCII */
471  if(isprint(int(*p))) {
472  switch(*p) {
473  case '\\':
474  len += 2; break;
475  case '\'':
476  case '"':
477  case '`':
478  len += (quote == *p)? 2 : 1; break;
479  default:
480  len++; break;
481  }
482  } else switch(*p) {
483  case '\a':
484  case '\b':
485  case '\f':
486  case '\n':
487  case '\r':
488  case '\t':
489  case '\v':
490  case '\0':
491  len += 2; break;
492  default:
493  /* print in octal */
494  len += 4; break;
495  }
496  p++;
497  } else { /* 8 bit char */
498 #ifdef Win32 /* It seems Windows does not know what is printable! */
499  len++;
500 #else
501  len += isprint(int(*p)) ? 1 : 4;
502 #endif
503  p++;
504  }
505  }
506 
507  return len;
508 }

534 HIDDEN
535 const char* Rf_EncodeString(SEXP s, int w, int quote, Rprt_adj justify)
536 {
537  int b, b0, i, j, cnt;
538  const char *p; char *q, buf[11];
539  cetype_t ienc = Rf_getCharCE(s);
540  Rboolean useUTF8 = Rboolean(w < 0);
541  const void *vmax = vmaxget();
542 
543  if (w < 0) w = w + 1000000;
544 
545  /* We have to do something like this as the result is returned, and
546  passed on by EncodeElement -- so no way could be end user be
547  responsible for freeing it. However, this is not thread-safe. */
548 
549  static R_StringBuffer gBuffer = { nullptr, 0, BUFSIZE };
550  R_StringBuffer* buffer = &gBuffer;
551 
552  if (s == R_NaString) {
553  p = quote ? R_CHAR(R_print.na_string) : R_CHAR(R_print.na_string_noquote);
554  cnt = i = int(quote ? strlen(R_CHAR(R_print.na_string)) :
555  strlen(R_CHAR(R_print.na_string_noquote)));
556  quote = 0;
557  } else {
558 #ifdef Win32
559  if(WinUTF8out) {
560  if(ienc == CE_UTF8) {
561  p = R_CHAR(s);
562  i = Rstrlen(s, quote);
563  cnt = LENGTH(s);
564  } else {
565  p = Rf_translateCharUTF8(s);
566  if(p == R_CHAR(s)) {
567  i = Rstrlen(s, quote);
568  cnt = LENGTH(s);
569  } else {
570  cnt = strlen(p);
571  i = Rstrwid(p, cnt, CE_UTF8, quote);
572  }
573  ienc = CE_UTF8;
574  }
575  } else
576 #endif
577  {
578  if(IS_BYTES(s)) {
579  ienc = CE_NATIVE;
580  p = R_CHAR(s);
581  cnt = int(strlen(p));
582  const char *q;
583  char *pp = R_alloc(4*cnt+1, 1), *qq = pp, buf[5];
584  for (q = p; *q; q++) {
585  unsigned char k = static_cast<unsigned char>(*q);
586  if (k >= 0x20 && k < 0x80) {
587  *qq++ = *q;
588  if (quote && *q == '"') cnt++;
589  } else {
590  snprintf(buf, 5, "\\x%02x", k);
591  for(j = 0; j < 4; j++) *qq++ = buf[j];
592  cnt += 3;
593  }
594  }
595  *qq = '\0';
596  p = pp;
597  i = cnt;
598  } else if (useUTF8 && ienc == CE_UTF8) {
599  p = R_CHAR(s);
600  i = Rstrlen(s, quote);
601  cnt = LENGTH(s);
602  } else {
603  ienc = CE_NATIVE;
604  p = Rf_translateChar(s);
605  if(p == R_CHAR(s)) {
606  i = Rstrlen(s, quote);
607  cnt = LENGTH(s);
608  } else {
609  cnt = int(strlen(p));
610  i = Rstrwid(p, cnt, CE_NATIVE, quote);
611  }
612  }
613  }
614  }

616  /* We need enough space for the encoded string, including escapes.
617  Octal encoding turns one byte into four.
618  \u encoding can turn a multibyte into six or ten,
619  but it turns 2/3 into 6, and 4 (and perhaps 5/6) into 10.
620  Let's be wasteful here (the worst case appears to be an MBCS with
621  one byte for an upper-plane Unicode point output as ten bytes,
622  but I doubt that such an MBCS exists: two bytes is plausible).
623 
624  +2 allows for quotes, +6 for UTF_8 escapes.
625  */
