124 SEXP attribute_hidden do_provenance (SEXP call, SEXP op, SEXP args, SEXP rho)
125 {
126 #ifndef PROVENANCE_TRACKING

131  int n;

138  Environment* env=static_cast<Environment*>(rho);

143  if (!provenance)

145  const Provenance::Set& children=provenance->children();
146 
147  GCStackRoot<ListVector> list(ListVector::create(nfields));
148  GCStackRoot<StringVector> timestamp(StringVector::create(1));
149  GCStackRoot<StringVector> names(StringVector::create(nfields));
150 
151  (*timestamp)[0]=const_cast<String*>(provenance->getTime());
152 
153  (*names)[0]=const_cast<String*>(String::obtain("command"));
154  (*names)[1]=const_cast<String*>(String::obtain("symbol"));
155  (*names)[2]=const_cast<String*>(String::obtain("timestamp"));
156  (*names)[3]=const_cast<String*>(String::obtain("parents"));
157  (*names)[4]=const_cast<String*>(String::obtain("children"));
158 
159  (*list)[0] = const_cast<RObject*>(provenance->command());
160  (*list)[1] = const_cast<Symbol*>(provenance->symbol());
161  (*list)[2]=timestamp;
162  // Handle parents:
163  {
164  std::pair<CommandChronicle::ParentVector::const_iterator,
165  CommandChronicle::ParentVector::const_iterator>
166  pr = provenance->parents();
167  size_t sz = pr.second - pr.first;
168  StringVector* sv = StringVector::create(sz);
169  (*list)[3] = sv;
170  unsigned int i = 0;
171  for (CommandChronicle::ParentVector::const_iterator it = pr.first;
172  it != pr.second; ++it) {
173  const Provenance* p = *it;
174  (*sv)[i++] = const_cast<String*>(p->symbol()->name());
175  }
176  }
177  if (!children.empty()) {
178  StringVector* sv = StringVector::create(children.size());
179  (*list)[4] = sv;
180  unsigned int i = 0;
181  for (Provenance::Set::const_iterator it = children.begin();
182  it != children.end(); ++it) {
183  const Provenance* p = *it;
184  (*sv)[i++] = const_cast<String*>(p->symbol()->name());
185  }
186  }
187 
188  setAttrib(list,R_NamesSymbol,names);
189 
190  return list;
191 #endif // PROVENANCE_TRACKING

215 do_provenance_graph(/*const*/ Expression* call, const BuiltInFunction* op, Environment* rho, RObject* const* args, int num_args, const PairList* tags)
216 {
217 #ifndef PROVENANCE_TRACKING
218  Rf_error(_("provenance tracking not implemented in this build"));
219  return nullptr;
220 #else
221  int nargs = num_args;
222  if (nargs != 1)
223  Rf_error(_("%d arguments passed to 'provenance.graph' which requires 1"),
224  nargs);
225  // SEXP arg1 = CAR((RObject*) args);
226  const RObject* arg1 = args[0];
227  if (!arg1 || arg1->sexptype() != STRSXP)
228  Rf_error(_("invalid 'names' argument"));
229 
230  Environment* env = static_cast<Environment*>(rho);
231  Provenance::Set provs;
232  const StringVector* sv = static_cast<const StringVector*>(arg1);
233  for (size_t i = 0; i < sv->size(); i++) {
234  const char* name = (*sv)[i]->c_str();
235  Symbol* sym = Symbol::obtain(name);
236  Frame::Binding* bdg = env->findBinding(sym);
237  if (!bdg)
238  Rf_error(_("symbol '%s' not found"), name);
239  else {
240  Provenance* prov = const_cast<Provenance*>(bdg->provenance());
241  if (!prov)
242  Rf_warning(_("'%s' does not have provenance information"),
243  name);
244  else provs.insert(prov);
245  }
246  }
247 
248  Provenance::Set* ancestors = Provenance::ancestors(provs);
249 
250  GCStackRoot<ListVector> ans(ListVector::create(7));
251  std::map<const Provenance*, unsigned int> ancestor_index;
252  std::vector<std::pair<unsigned int, const RObject*> > xenogenous_bdgs;
253 
254  // Assemble information on graph nodes:
255  {
256  size_t n = ancestors->size();
257  GCStackRoot<ListVector> symbols(ListVector::create(n));
258  GCStackRoot<ListVector> commands(ListVector::create(n));
259  GCStackRoot<RealVector> timestamps(RealVector::create(n));
260  size_t i = 0;
261  for (Provenance::Set::iterator it = ancestors->begin();
262  it != ancestors->end(); ++it) {
263  const Provenance* p = *it;
264  (*symbols)[i] = const_cast<Symbol*>(p->symbol());
265  (*commands)[i] = const_cast<RObject*>(p->command());
266  (*timestamps)[i] = p->timestamp();
267  ++i;
268  ancestor_index[p] = i;
269  if (p->isXenogenous())
270  xenogenous_bdgs.push_back(std::make_pair(i, p->value()));
271  }
272 
273  (*ans)[0] = symbols;
274  (*ans)[1] = commands;
275  (*ans)[2] = timestamps;
276  }
277 
278  // Record information on xenogenous bindings:
279  {
280  size_t xn = xenogenous_bdgs.size();
281  GCStackRoot<IntVector> xenogenous(IntVector::create(xn));
282  GCStackRoot<ListVector> values(ListVector::create(xn));
283  for (unsigned int i = 0; i < xn; ++i) {
284  std::pair<unsigned int, const RObject*>& pr = xenogenous_bdgs[i];
285  (*xenogenous)[i] = pr.first;
286  (*values)[i] = const_cast<RObject*>(pr.second);
287  }
288  (*ans)[3] = xenogenous;
289  (*ans)[4] = values;
290  }
291 
292  // Assemble information on graph edges:
293  {
294  typedef std::set<std::pair<unsigned int, unsigned int> > EdgeSet;
295  EdgeSet edges;
296  for (Provenance::Set::iterator it = ancestors->begin();
297  it != ancestors->end(); ++it) {
298  const Provenance* child = *it;
299  unsigned int child_idx = ancestor_index[child];
300  std::pair<CommandChronicle::ParentVector::const_iterator,
301  CommandChronicle::ParentVector::const_iterator>
302  pr = child->parents();
303  for (CommandChronicle::ParentVector::const_iterator it = pr.first;
304  it != pr.second; ++it) {
305  const Provenance* parent = *it;
306  unsigned int parent_idx = ancestor_index[parent];
307  edges.insert(std::make_pair(parent_idx, child_idx));
308  }
309  }
310 
311  size_t en = edges.size();
312  GCStackRoot<IntVector> parents(IntVector::create(en));
313  GCStackRoot<IntVector> children(IntVector::create(en));
314  unsigned int i = 0;
315  for (EdgeSet::const_iterator it = edges.begin(); it != edges.end(); ++it) {
316  const std::pair<unsigned int, unsigned int>& edge = *it;
317  (*parents)[i] = edge.first;
318  (*children)[i] = edge.second;
319  ++i;
320  }
321 
322  (*ans)[5] = parents;
323  (*ans)[6] = children;
324  }
325  delete ancestors;
326  return ans;
327 #endif // PROVENANCE_TRACKING
328 }
