190 SEXP attribute_hidden do_gc(/*const*/ Expression* call, const BuiltInFunction* op, RObject* verbose_, RObject* reset_)
191 {
192  std::ostream* report_os
193  = GCManager::setReporting(Rf_asLogical(verbose_) ? &std::cerr : nullptr);
194  bool reset_max = Rf_asLogical(reset_);
195  GCManager::gc();
196  R_RunPendingFinalizers();
197  GCManager::setReporting(report_os);
198  GCStackRoot<> value(Rf_allocVector(REALSXP, 6));
199  REAL(value)[0] = GCNode::numNodes();
200  REAL(value)[1] = NA_REAL;
201  REAL(value)[2] = GCManager::maxNodes();
202  /* next four are in 0.1MB, rounded up */
203  REAL(value)[3] = 0.1*ceil(10. * MemoryBank::bytesAllocated()/Mega);
204  REAL(value)[4] = 0.1*ceil(10. * GCManager::triggerLevel()/Mega);
205  REAL(value)[5] = 0.1*ceil(10. * GCManager::maxBytes()/Mega);
206  if (reset_max) GCManager::resetMaxTallies();
207  return value;
208 }

238 static void gc_end_timing(void)
239 {
240  if (gctime_enabled) {
241  double times[5], delta;
242  R_getProcTime(times);
243  delta = R_getClockIncrement();
244 
245  /* add delta to compensate for timer resolution:
246  NB: as all current Unix-alike systems use getrusage,
247  this may over-compensate.
248  */
249  gctimes[0] += times[0] - gcstarttimes[0] + delta;
250  gctimes[1] += times[1] - gcstarttimes[1] + delta;
251  gctimes[2] += times[2] - gcstarttimes[2] + delta;
252  gctimes[3] += times[3] - gcstarttimes[3];
253  gctimes[4] += times[4] - gcstarttimes[4];
254  }

272  NewEnvironment
273 
274  Create an environment by extending "rho" with a frame obtained by
275  pairing the variable names given by the tags on "namelist" with
276  the values given by the elements of "valuelist".
277 
278  This definition allows the namelist argument to be shorter than the
279  valuelist; in this case the remaining values must be named already.
280  (This is useful in cases where the entire valuelist is already
281  named--namelist can then be R_NilValue.)
282 */
283 SEXP Rf_NewEnvironment(SEXP namelist, SEXP valuelist, SEXP rho)
284 {
285  SEXP v = valuelist;
286  SEXP n = namelist;
287  while (v != R_NilValue && n != R_NilValue) {
288  SET_TAG(v, TAG(n));
289  v = CDR(v);
290  n = CDR(n);
291  }
292 
293  GCStackRoot<> namelistr(namelist);
294  GCStackRoot<PairList> namevalr(SEXP_downcast<PairList*>(valuelist));
295  GCStackRoot<Environment> rhor(SEXP_downcast<Environment*>(rho));
296  // +5 to leave some room for local variables:
297  GCStackRoot<Frame> frame(Frame::normalFrame(Rf_length(namevalr) + 5));
298  frameReadPairList(frame, namevalr);
299  return new Environment(rhor, frame);
300 }

305 SEXP Rf_allocVector3(SEXPTYPE type, R_xlen_t length, void*)
306 {
307  SEXP s = nullptr; // -Wall
308 
309  if (length > R_XLEN_T_MAX)
310  Rf_error(_("vector is too large")); /**** put length into message */
311  else if (length < 0 )
312  Rf_error(_("negative length vectors are not allowed"));
313 
314  /* number of vector cells to allocate */
315  switch (type) {
316  case NILSXP:
317  return nullptr;
318  case RAWSXP:
319  s = RawVector::create(length);
320  break;
321  case CHARSXP:
322  Rf_error("use of allocVector(CHARSXP ...) is defunct\n");
323  break;
324  case LGLSXP:
325  s = LogicalVector::create(length);
326  break;
327  case INTSXP:
328  s = IntVector::create(length);
329  break;
330  case REALSXP:
331  s = RealVector::create(length);
332  break;
333  case CPLXSXP:
334  s = ComplexVector::create(length);
335  break;
336  case STRSXP:
337  s = StringVector::create(length);
338  break;
339  case EXPRSXP:
340  s = ExpressionVector::create(length);
341  break;
342  case VECSXP:
343  s = ListVector::create(length);
344  break;
345  case LANGSXP:
346  {
347  if (length == 0)
348  return nullptr;
349 #ifdef LONG_VECTOR_SUPPORT
350  if (length > R_SHORT_LEN_MAX) Rf_error("invalid length for pairlist");
351 #endif
352  GCStackRoot<PairList> tl(PairList::make(length - 1));
353  s = new CachingExpression(nullptr, tl);
354  break;
355  }
356  case LISTSXP:
357 #ifdef LONG_VECTOR_SUPPORT
358  if (length > R_SHORT_LEN_MAX) Rf_error("invalid length for pairlist");
359 #endif
360  return Rf_allocList(int( length));
361  default:
362  Rf_error(_("invalid type/length (%s/%d) in vector allocation"),
363  Rf_type2char(type), length);
364  return nullptr; // -Wall
365  }
366  return s;
367 }

370  SEXP res = R_NilValue;
371  SEXP n;

594 void *R_AllocStringBuffer(std::size_t blen, R_StringBuffer *buf)
595 {
596  std::size_t blen1, bsize = buf->defaultSize;
597 
598  /* for backwards compatibility, this used to free the buffer */
599  if(blen == std::size_t(-1)) {
600  Rf_error("R_AllocStringBuffer( (size_t)-1 ) is no longer allowed");
601  }
602 
603  if(blen * sizeof(char) < buf->bufsize) return buf->data;
604  blen1 = blen = (blen + 1) * sizeof(char);
605  blen = (blen / bsize) * bsize;
606  if(blen < blen1) blen += bsize;
607 
608  if(buf->data == nullptr) {
609  buf->data = static_cast<char *>(malloc(blen));
610  buf->data[0] = '\0';
611  } else
612  buf->data = static_cast<char *>(realloc(buf->data, blen));
613  buf->bufsize = blen;
614  if(!buf->data) {
615  buf->bufsize = 0;
616  /* don't translate internal error message */
617  Rf_error("could not allocate memory (%u Mb) in C function 'R_AllocStringBuffer'",
618  static_cast<unsigned int>(blen)/1024/1024);
619  }
620  return buf->data;
621 }
