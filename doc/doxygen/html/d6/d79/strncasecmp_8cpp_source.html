35 int strncasecmp(const char *s1, const char *s2, size_t n)
36 {
37  char c1, c2;
38 
39  for (size_t i = 0; i < n; i++) {
40  c1 = s1[i]; c2 = s2[i];
41  c1 = isupper(c1) ? char(tolower(c1)) : c1;
42  c2 = isupper(c2) ? char(tolower(c2)) : c2;
43  if (c1 == '\0') return ((c2 == '\0') ? 0 : -1);
44  if (c2 == '\0') return 1;
45  if (c1 < c2) return -1;
46  if (c1 > c2) return 1;
47  }
48  return 0;
49 }
