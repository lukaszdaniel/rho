41 R_len_t Rf_length(SEXP s)
42 {
43  if (Rf_isVector(s))
44  return LENGTH(s);
45  switch (TYPEOF(s)) {

50  case DOTSXP:

107 SEXP Rf_mkCLOSXP(SEXP formals, SEXP body, SEXP rho)
108 {
109  GCStackRoot<PairList> formrt(SEXP_downcast<PairList*>(formals));
110  GCStackRoot<> bodyrt(body);
111  GCStackRoot<Environment> envrt(rho ? SEXP_downcast<Environment*>(rho)
112  : Environment::global());
113  switch (TYPEOF(body)) {
114  case CLOSXP:
115  case BUILTINSXP:
116  case SPECIALSXP:
117  case DOTSXP:
118  case ANYSXP:
119  Rf_error(_("invalid body argument for 'function'"));
120  break;
121  default:
122  break;
123  }
124  return new Closure(formrt, bodyrt, envrt);
