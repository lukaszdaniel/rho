28 /*
29  This is an effort to merge the 3 different dynload.c files in the
30  distribution from the unix/, macintosh/dll/ and gnuwin32/ directories.
31  The aim is to consolidate these different implementations into
32  i) a generic or platform-independent common core
33  ii) platform-dependent routines that are registered
34  as function pointers.
35  The reason for using function pointers rather than explicit
36  linking of symbols is
37  a) to avoid confusion in the linking
38  b) to allow for easily overriding these in embedded applications
39  in which a host application needs to control how R finds
40  symbols. This may be necessary for security reasons.
41  */

43 /* Dynamic Loading Support
44  *
45  * This module provides support for run-time loading of shared objects
46  * access to symbols within such objects via .C and .Fortran. This is
47  * done under Unix with dlopen, dlclose and dlsym (the exception is
48  * hpux, where we use compatibility code provided by Luke Tierney).
49  * There are two cases:
50  *
51  *
52  * 1. The dlopen interface is available.
53  *
54  * In this case all symbol location in packages is done using the
55  * dlopen routines. We maintain a list of currently loaded shared
56  * objects in an array called "LoadedDLL" with the number of currently
57  * loaded objects being "CountDLL". To locate a symbol, we probe
58  * the loaded objects in order until the symbol is located. If we
59  * do not find a symbol in the loaded objects, we search the
60  * executable itself. This search is not very efficient, but this
61  * probably pales into insignificance when compared with the
62  * inefficiencies in the R interpreter.
63  *
64  * Loading and unloading of shared objects is done via the routines
65  * AddDLL and DeleteDLL. These routines maintain the list of
66  * currently loaded objects. When an object is added, any existing
67  * reference to that object is deleted and then the object is
68  * inserted at the start of the search list. This way, symbols in
69  * more recently loaded objects are found first.
70  *
71  *
72  * Accessing native routines in base (the R executable).
73  *
74  * In this case, we use the registration mechanism and the DllInfo array
75  * in ../main/Rdynload.cpp to locate functions in the executable. We do this
76  * by straight linear search through the table.
77  * Note that the base routines registered are listed in
78  * ../main/registration.c
79  * and are registered during the initialization of the R engine.
80  * (This replaces the previous mechanism that built a table
81  * from ../appl/ROUTINES using Perl/sed).
82  *
83  *
84  * If speed is ever an issue in the lookup of registered symbols, we can
85  * store the registered routines in a hashtable or binary tree as they
86  * are being registered.
87  */

181 /* Allocate LoadedDLL. Errors are reported via R_Suicide, because this is
182  called too early during startup to use Rf_error(.) */
183 static void initLoadedDLL()
184 {
185  if (CountDLL != 0 || LoadedDLL != NULL)
186  R_Suicide("DLL table corruption detected"); /* not translated */
187 
188  /* Note that it is likely that dlopen will use up at least one file
189  descriptor for each DLL loaded (it may load further dynamically
190  linked libraries), so we do not want to get close to the fd limit
191  (which may be as low as 256). By default, the maximum number of DLLs
192  that can be loaded is 100. When the fd limit is known, we allow
193  increasing the maximum number of DLLs via environment variable up to
194  60% of the limit on open files, but to no more than 1000.
195  */
196  int maxlimit;
197  int fdlimit = R_GetFDLimit();
198  if (fdlimit > 0) { /* fd limit known */
199  maxlimit = (int) (0.6 * fdlimit);
200  if (maxlimit > 1000) maxlimit = 1000;
201  if (maxlimit < 100)
202  R_Suicide(_("the limit on the number of open files is too low"));
203  } else
204  maxlimit = 100;
205 
206  char *req = getenv("R_MAX_NUM_DLLS");
207  if (req != NULL) {
208  int reqlimit = atoi(req);
209  if (reqlimit < 100)
210  R_Suicide(_("R_MAX_NUM_DLLS must be at least 100"));
211  if (reqlimit > maxlimit) {
212  if (maxlimit == 1000)
213  R_Suicide(_("MAX_NUM_DLLS cannot be bigger than 1000"));
214 
215  char msg[128];
216  snprintf(msg, 128,
217  _("MAX_NUM_DLLS bigger than %d may exhaust open files limit"),
218  maxlimit);
219  R_Suicide(msg);
220  }
221  MaxNumDLLs = reqlimit;
222  } else
223  MaxNumDLLs = 100;
224 
225  /* memory is set to zero */
226  LoadedDLL = (DllInfo *) calloc(MaxNumDLLs, sizeof(DllInfo));
227  if (LoadedDLL == NULL)
228  R_Suicide(_("could not allocate space for DLL table"));
229 }

299 /*
300  Explicitly register the native routines for use in .Call(), .C() and
301  .Fortran() functions. These registered values are used to resolve
302  symbols in an object that makes a call to this routine, rather than
303  the usual dynamic resolution done by dlsym() or the equivalent on
304  the different platforms.
305  */
306 int
307 R_registerRoutines(DllInfo *info, const R_CMethodDef * const croutines,
308  const R_CallMethodDef * const callRoutines,
309  const R_FortranMethodDef * const fortranRoutines,
310  const R_ExternalMethodDef * const externalRoutines)
311 {
312  int i, num;
313 
314  if(info == nullptr)
315  Rf_error(_("R_RegisterRoutines called with invalid DllInfo object."));
316 
317  /* Default is to look in registered and then dynamic (unless
318  the is no handle such as in "base" or "embedded")
319  Potentially change in the future to be only registered
320  if there are any registered values.
321  */
322  info->useDynamicLookup = (info->handle) ? TRUE : FALSE;
323  info->forceSymbols = FALSE;
324 
325  if(croutines) {
326  for(num = 0; croutines[num].name != nullptr; num++) {;}
327  info->CSymbols = static_cast<Rf_DotCSymbol*>(calloc(size_t(num),
328  sizeof(Rf_DotCSymbol)));
329  info->numCSymbols = num;
330  for(i = 0; i < num; i++) {
331  R_addCRoutine(info, croutines+i, info->CSymbols + i);
332  }
333  }
334 
335  if(fortranRoutines) {
336  for(num = 0; fortranRoutines[num].name != nullptr; num++) {;}
337  info->FortranSymbols =
338  static_cast<Rf_DotFortranSymbol*>(calloc(size_t(num),
339  sizeof(Rf_DotFortranSymbol)));
340  info->numFortranSymbols = num;
341 
342  for(i = 0; i < num; i++)
343  R_addFortranRoutine(info, fortranRoutines+i,
344  info->FortranSymbols + i);
345  }
346 
347  if(callRoutines) {
348  for(num = 0; callRoutines[num].name != nullptr; num++) {;}
349  info->CallSymbols =
350  static_cast<Rf_DotCallSymbol*>(calloc(size_t(num), sizeof(Rf_DotCallSymbol)));
351  info->numCallSymbols = num;
352  for(i = 0; i < num; i++)
353  R_addCallRoutine(info, callRoutines+i, info->CallSymbols + i);
354  }
355 
356  if(externalRoutines) {
357  for(num = 0; externalRoutines[num].name != nullptr; num++) {;}
358  info->ExternalSymbols =
359  static_cast<Rf_DotExternalSymbol*>(calloc(size_t(num),
360  sizeof(Rf_DotExternalSymbol)));
361  info->numExternalSymbols = num;
362 
363  for(i = 0; i < num; i++)
364  R_addExternalRoutine(info, externalRoutines+i,
365  info->ExternalSymbols + i);
366  }
367 
368  return(1);
369 }

502 static int DeleteDLL(const char *path)
503 {
504  int i, loc;
505 
506  for (i = 0; i < CountDLL; i++) {
507  if (streql(path, LoadedDLL[i].path)) {
508  loc = i;
509  goto found;
510  }
511  }
512  return 0;
513 found:
514 #ifdef CACHE_DLL_SYM
515  if(R_osDynSymbol->deleteCachedSymbols)
516  R_osDynSymbol->deleteCachedSymbols(&LoadedDLL[loc]);
517 #endif
518  R_callDLLUnload(&LoadedDLL[loc]);
519  R_osDynSymbol->closeLibrary(LoadedDLL[loc].handle);
520  Rf_freeDllInfo(LoadedDLL+loc);
521  /* FIXME: why not use memcpy here? */
522  for(i = loc + 1 ; i < CountDLL ; i++) {
523  LoadedDLL[i - 1].path = LoadedDLL[i].path;
524  LoadedDLL[i - 1].name = LoadedDLL[i].name;
525  LoadedDLL[i - 1].handle = LoadedDLL[i].handle;
526  LoadedDLL[i - 1].useDynamicLookup = LoadedDLL[i].useDynamicLookup;
527  LoadedDLL[i - 1].numCSymbols = LoadedDLL[i].numCSymbols;
528  LoadedDLL[i - 1].numCallSymbols = LoadedDLL[i].numCallSymbols;
529  LoadedDLL[i - 1].numFortranSymbols = LoadedDLL[i].numFortranSymbols;
530  LoadedDLL[i - 1].numExternalSymbols = LoadedDLL[i].numExternalSymbols;
531  LoadedDLL[i - 1].CSymbols = LoadedDLL[i].CSymbols;
532  LoadedDLL[i - 1].CallSymbols = LoadedDLL[i].CallSymbols;
533  LoadedDLL[i - 1].FortranSymbols = LoadedDLL[i].FortranSymbols;
534  LoadedDLL[i - 1].ExternalSymbols = LoadedDLL[i].ExternalSymbols;
535  LoadedDLL[i - 1].forceSymbols = LoadedDLL[i].forceSymbols;
536  }
537  CountDLL--;
538  return 1;
539 }

573 static DllInfo* AddDLL(const char *path, int asLocal, int now,
574  const char *DLLsearchpath)
575 {
576  HINSTANCE handle;
577  DllInfo *info = nullptr;
578 
579  DeleteDLL(path);
580  if(CountDLL == MaxNumDLLs) {
581  strcpy(DLLerror, _("maximal number of DLLs reached..."));
582  return nullptr;
583  }
584 
585  handle = R_osDynSymbol->loadLibrary(path, asLocal, now, DLLsearchpath);
586 
587  if(handle == nullptr) {
588  R_osDynSymbol->getError(DLLerror, DLLerrBUFSIZE);
589  return nullptr;
590  }
591 
592  info = R_RegisterDLL(handle, path);
593 
594  /* Now look for an initializing routine named R_init_<object name>.
595  If it is present, we call it. It should take a reference to the
596  DllInfo object currently being initialized.
597  */
598  if(info) {
599  const char *nm = info->name;
600  size_t len = strlen(nm) + 9;
601  char tmp[len]; // R_init_ + underscore + null
602  DllInfoInitCall f;
603 #ifdef HAVE_NO_SYMBOL_UNDERSCORE
604  snprintf(tmp, len, "%s%s","R_init_", info->name);
605 #else
606  snprintf(tmp, len, "_%s%s","R_init_", info->name);
607 #endif
608  f = DllInfoInitCall( R_osDynSymbol->dlsym(info, tmp));
609  /* If that failed, might have used the package name with
610  . replaced by _ (as . it not valid in symbol names). */
611  if(!f) {
612  /* This is potentially unsafe in MBCSs, as '.' might be
613  part of a character: but is not in UTF-8 */
614  for(char *p = tmp; *p; p++) if(*p == '.') *p = '_';
615  f = DllInfoInitCall( R_osDynSymbol->dlsym(info, tmp));
616  }
617  if(f) f(info);
618  }
619 
620  return info;
621 }

624 static DllInfo *R_RegisterDLL(HINSTANCE handle, const char *path)
625 {
626  char *dpath, DLLname[PATH_MAX], *p;
627  DllInfo *info;
628 
629  dpath = static_cast<char *>(malloc(strlen(path)+1));
630  if(dpath == nullptr) {
631  strcpy(DLLerror, _("could not allocate space for 'path'"));
632  R_osDynSymbol->closeLibrary(handle);
633  return nullptr;
634  }
635  strcpy(dpath, path);
636 
637  if(R_osDynSymbol->fixPath) R_osDynSymbol->fixPath(dpath);
638 
639  /* keep only basename from path */
640  p = Rf_strrchr(dpath, FILESEP[0]);
641  if(!p) p = dpath; else p++;
642  if(strlen(p) < PATH_MAX) strcpy(DLLname, p);
643  else Rf_error(_("DLLname '%s' is too long"), p);
644 
645  /* remove SHLIB_EXT if present */
646  p = DLLname + strlen(DLLname) - strlen(SHLIB_EXT);
647 #ifdef Win32 /* case-insensitive file system */
648  if(p > DLLname && stricmp(p, SHLIB_EXT) == 0) *p = '\0';
649 #else
650  if(p > DLLname && streql(p, SHLIB_EXT)) *p = '\0';
651 #endif
652 
653  if (addDLL(dpath, DLLname, handle)) {
