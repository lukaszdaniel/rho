57 SEXP attribute_hidden do_version(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op)
58 {
59  SEXP value, names;
60  char buf[128];
61 
62  PROTECT(value = Rf_allocVector(VECSXP,15));
63  PROTECT(names = Rf_allocVector(STRSXP,15));
64 
65  SET_STRING_ELT(names, 0, Rf_mkChar("platform"));
66  SET_VECTOR_ELT(value, 0, Rf_mkString(R_PLATFORM));
67  SET_STRING_ELT(names, 1, Rf_mkChar("arch"));
68  SET_VECTOR_ELT(value, 1, Rf_mkString(R_CPU));
69  SET_STRING_ELT(names, 2, Rf_mkChar("os"));
70  SET_VECTOR_ELT(value, 2, Rf_mkString(R_OS));
71 
72  snprintf(buf, 128, "%s, %s", R_CPU, R_OS);
73  SET_STRING_ELT(names, 3, Rf_mkChar("system"));
74  SET_VECTOR_ELT(value, 3, Rf_mkString(buf));
75 
76  SET_STRING_ELT(names, 4, Rf_mkChar("status"));
77  SET_VECTOR_ELT(value, 4, Rf_mkString(R_STATUS));
78  SET_STRING_ELT(names, 5, Rf_mkChar("major"));
79  SET_VECTOR_ELT(value, 5, Rf_mkString(R_MAJOR));
80  SET_STRING_ELT(names, 6, Rf_mkChar("minor"));
81  SET_VECTOR_ELT(value, 6, Rf_mkString(R_MINOR));
82  SET_STRING_ELT(names, 7, Rf_mkChar("year"));
83  SET_VECTOR_ELT(value, 7, Rf_mkString(R_YEAR));
84  SET_STRING_ELT(names, 8, Rf_mkChar("month"));
85  SET_VECTOR_ELT(value, 8, Rf_mkString(R_MONTH));
86  SET_STRING_ELT(names, 9, Rf_mkChar("day"));
87  SET_VECTOR_ELT(value, 9, Rf_mkString(R_DAY));
88  SET_STRING_ELT(names, 10, Rf_mkChar("svn rev"));
89 
90  snprintf(buf, 128, "%d", R_SVN_BASEREVISION);
91  SET_VECTOR_ELT(value, 10, Rf_mkString(buf));
92  SET_STRING_ELT(names, 11, Rf_mkChar("git rev"));
93 
94  snprintf(buf, 128, "%s", R_GIT_REVISION);
95  SET_VECTOR_ELT(value, 11, Rf_mkString(buf));
96  SET_STRING_ELT(names, 12, Rf_mkChar("language"));
97  SET_VECTOR_ELT(value, 12, Rf_mkString("R"));
98 
99  Rf_PrintRhoVersionString(buf, 128);
100  SET_STRING_ELT(names, 13, Rf_mkChar("version.string"));
101  SET_VECTOR_ELT(value, 13, Rf_mkString(buf));
102  SET_STRING_ELT(names, 14, Rf_mkChar("nickname"));
103  SET_VECTOR_ELT(value, 14, Rf_mkString(R_NICK));
104 
105  Rf_setAttrib(value, R_NamesSymbol, names);
106  UNPROTECT(2);
107  return value;
108 }

122 void attribute_hidden Rf_PrintVersionString(char *s, size_t len)
123 {
124  if(R_SVN_BASEREVISION <= 0) {// 'svn info' failed in ../../Makefile.in
125  snprintf(s, len, "R version %s.%s %s (%s-%s-%s)",
126  R_MAJOR, R_MINOR, R_STATUS, R_BASEYEAR, R_BASEMONTH, R_BASEDAY);
127  } else if(strlen(R_STATUS) == 0) {
128  snprintf(s, len, "R version %s.%s (%s-%s-%s)",
129  R_MAJOR, R_MINOR, R_BASEYEAR, R_BASEMONTH, R_BASEDAY);
130  } else if(streql(R_STATUS, "Under development (unstable)")) {
131  snprintf(s, len, "R %s (%s-%s-%s r%d)",
132  R_STATUS, R_BASEYEAR, R_BASEMONTH, R_BASEDAY, R_SVN_BASEREVISION);
133  } else {
134  snprintf(s, len, "R version %s.%s %s (%s-%s-%s r%d)",
135  R_MAJOR, R_MINOR, R_STATUS, R_BASEYEAR, R_BASEMONTH, R_BASEDAY,
136  R_SVN_BASEREVISION);
137  }
138 }

160 void attribute_hidden PrintVersion_part_1(char *s, size_t len)
161 {
162 #define SPRINTF_2(_FMT, _OBJ) snprintf(tmp, 128, _FMT, _OBJ); strcat(s, tmp)
163  char tmp[128];
164  Rf_PrintRhoVersionString(s, len);
165  strcat(s, " -- \"R(C) -> Rho(C++)\"\n");
166  strcat(s, "Rho Copyright (C) 2008-2014 Andrew R. Runnalls\n\n");
167 
168  Rf_PrintVersionString(tmp, len);
169  if(strlen(R_NICK) != 0) {
170  char nick[128];
171  snprintf(nick, 128, " -- \"%s\"", R_NICK);
172  strcat(tmp, nick);
173  }
174  strcat(s, tmp);
175  SPRINTF_2("\nCopyright (C) %s The R Foundation for Statistical Computing\n", R_YEAR);
176 /* strcat(s, "ISBN 3-900051-07-0\n"); */
177  SPRINTF_2("Platform: %s", R_PLATFORM);
178  if(strlen(R_ARCH)) { SPRINTF_2("/%s", R_ARCH); }
179  SPRINTF_2(" (%d-bit)\n", 8*int(sizeof(void *)));
180 }
