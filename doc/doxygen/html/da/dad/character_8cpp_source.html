157 /* R strings are limited to 2^31 - 1 bytes on all platforms */
158 int R_nchar(SEXP string, nchar_type type_,
159  Rboolean allowNA, Rboolean keepNA, const char* msg_name)
160 {
161  if (string == NA_STRING)
162  return keepNA ? NA_INTEGER : 2;
163  // else :
164  switch(type_) {
165  case Bytes:
166  return LENGTH(string);
167  break;
168  case Chars:
169  if (IS_UTF8(string)) {
170  const char *p = CHAR(string);
171  if (!utf8Valid(p)) {
172  if (!allowNA)
173  error(_("invalid multibyte string, %s"), msg_name);
174  return NA_INTEGER;
175  } else {
176  int nc = 0;
177  for( ; *p; p += utf8clen(*p)) nc++;
178  return nc;
179  }
180  } else if (IS_BYTES(string)) {
181  if (!allowNA) /* could do chars 0 */
182  error(_("number of characters is not computable in \"bytes\" encoding, %s"),
183  msg_name);
184  return NA_INTEGER;
185  } else if (mbcslocale) {
186  int nc = (int) mbstowcs(NULL, translateChar(string), 0);
187  if (!allowNA && nc < 0)
188  error(_("invalid multibyte string, %s"), msg_name);
189  return (nc >= 0 ? nc : NA_INTEGER);
190  } else
191  return ((int) strlen(translateChar(string)));
192  break;
193  case Width:
194  if (IS_UTF8(string)) {
195  const char *p = CHAR(string);
196  if (!utf8Valid(p)) {
197  if (!allowNA)
198  error(_("invalid multibyte string, %s"), msg_name);
199  return NA_INTEGER;
200  } else {
201  wchar_t wc1;
202  Rwchar_t ucs;
203  int nc = 0;
204  for( ; *p; p += utf8clen(*p)) {
205  utf8toucs(&wc1, p);
206  if (IS_HIGH_SURROGATE(wc1))
207  ucs = utf8toucs32(wc1, p);
208  else
209  ucs = wc1;
210  nc += Ri18n_wcwidth(ucs);
211  }
212  return nc;
213  }
214  } else if (IS_BYTES(string)) {
215  if (!allowNA) /* could do width 0 */
216  error(_("width is not computable for %s in \"bytes\" encoding"),
217  msg_name);
218  return NA_INTEGER;
219  } else if (mbcslocale) {
220  const char *xi = translateChar(string);
221  int nc = (int) mbstowcs(NULL, xi, 0);
222  if (nc >= 0) {
223  const void *vmax = vmaxget();
224  wchar_t *wc = (wchar_t *)
225  R_AllocStringBuffer((nc+1)*sizeof(wchar_t), &cbuff);
226  mbstowcs(wc, xi, nc + 1);
227  int nci18n = Ri18n_wcswidth(wc, 2147483647);
228  vmaxset(vmax);
229  return (nci18n < 1) ? nc : nci18n;
230  } else if (allowNA)
231  error(_("invalid multibyte string, %s"), msg_name);
232  else
233  return NA_INTEGER;
234  } else
235  return (int) strlen(translateChar(string));
236 
237  } // switch
238  return NA_INTEGER; // -Wall
239 } // R_nchar()

284 static void substr(char *buf, const char *str, int ienc, int sa, int so)
285 {
286 /* Store the substring str [sa:so] into buf[] */
287  int i, j, used;
288 
289  if (ienc == CE_UTF8) {
290  const char *end = str + strlen(str);
291  for (i = 0; i < so && str < end; i++) {
292  int used = utf8clen(*str);
293  if (i < sa - 1) { str += used; continue; }
294  for (j = 0; j < used; j++) *buf++ = *str++;
295  }
296  } else if (ienc == CE_LATIN1 || ienc == CE_BYTES) {
297  for (str += (sa - 1), i = sa; i <= so; i++) *buf++ = *str++;
298  } else {
299  if (mbcslocale && !strIsASCII(str)) {
300  const char *end = str + strlen(str);
301  mbstate_t mb_st;
302  mbs_init(&mb_st);
303  for (i = 1; i < sa; i++) str += Rf_mbrtowc(nullptr, str, MB_CUR_MAX, &mb_st);
304  for (i = sa; i <= so && str < end; i++) {
305  used = int( Rf_mbrtowc(nullptr, str, MB_CUR_MAX, &mb_st));
306  for (j = 0; j < used; j++) *buf++ = *str++;
307  }
308  } else
309  for (str += (sa - 1), i = sa; i <= so; i++) *buf++ = *str++;
310  }
311  *buf = '\0';
312 }

364 do_startsWith(Expression* call, const BuiltInFunction* op,
365  RObject* x,
366  RObject* Xfix /* 'prefix' or 'suffix' */)
367 {
368  if (!isString(x) || !isString(Xfix))
369  error(_("non-character object(s)"));
370  R_xlen_t
371  n1 = XLENGTH(x),
372  n2 = XLENGTH(Xfix),
373  n = (n1 > 0 && n2 > 0) ? ((n1 >= n2) ? n1 : n2) : 0;
374  if (n == 0) return allocVector(LGLSXP, 0);
375  SEXP ans = PROTECT(allocVector(LGLSXP, n));
376 
377  typedef const char * cp;
378  if (n2 == 1) { // optimize the most common case
379  SEXP el = STRING_ELT(Xfix, 0);
380  if (el == NA_STRING) {
381  for (R_xlen_t i = 0; i < n1; i++)
382  LOGICAL(ans)[i] = NA_LOGICAL;
383  } else {
384  // ASCII matching will do for ASCII Xfix except in non-UTF-8 MBCS
385  Rboolean need_translate = TRUE;
386  if (strIsASCII(CHAR(el)) && (utf8locale || !mbcslocale))
387  need_translate = FALSE;
388  cp y0 = need_translate ? translateCharUTF8(el) : CHAR(el);
389  int ylen = (int) strlen(y0);
390  for (R_xlen_t i = 0; i < n1; i++) {
391  SEXP el = STRING_ELT(x, i);
392  if (el == NA_STRING) {
393  LOGICAL(ans)[i] = NA_LOGICAL;
394  } else {
395  cp x0 = need_translate ? translateCharUTF8(el) : CHAR(el);
396  if(op->variant() == 0) { // startsWith
397  LOGICAL(ans)[i] = streqln(x0, y0, ylen);
398  } else { // endsWith
399  int off = (int)strlen(x0) - ylen;
400  if (off < 0)
401  LOGICAL(ans)[i] = 0;
402  else {
403  LOGICAL(ans)[i] = memcmp(x0 + off, y0, ylen) == 0;
404  }
405  }
406  }
407  }
408  }
409  } else { // n2 > 1
410  // convert both inputs to UTF-8
411  cp *x0 = (cp *) R_alloc(n1, sizeof(char *));
412  cp *y0 = (cp *) R_alloc(n2, sizeof(char *));
413  // and record lengths, -1 for NA
414  int *x1 = (int *) R_alloc(n1, sizeof(int *));
415  int *y1 = (int *) R_alloc(n2, sizeof(int *));
416  for (R_xlen_t i = 0; i < n1; i++) {
417  SEXP el = STRING_ELT(x, i);
418  if (el == NA_STRING)
419  x1[i] = -1;
420  else {
421  x0[i] = translateCharUTF8(el);
422  x1[i] = (int) strlen(x0[i]);
423  }
424  }
425  for (R_xlen_t i = 0; i < n2; i++) {
426  SEXP el = STRING_ELT(Xfix, i);
427  if (el == NA_STRING)
428  y1[i] = -1;
429  else {
430  y0[i] = translateCharUTF8(el);
431  y1[i] = (int) strlen(y0[i]);
432  }
433  }
434  R_xlen_t i, i1, i2;
435  if(op->variant() == 0) { // 0 = startsWith, 1 = endsWith
436  MOD_ITERATE2(n, n1, n2, i, i1, i2, {
437  if (x1[i1] < 0 || y1[i2] < 0)
438  LOGICAL(ans)[i] = NA_LOGICAL;
439  else if (x1[i1] < y1[i2])
440  LOGICAL(ans)[i] = 0;
441  else // memcmp should be faster than strncmp
442  LOGICAL(ans)[i] =
443  memcmp(x0[i1], y0[i2], y1[i2]) == 0;
444  });
445  } else { // endsWith
446  MOD_ITERATE2(n, n1, n2, i, i1, i2, {
447  if (x1[i1] < 0 || y1[i2] < 0)
448  LOGICAL(ans)[i] = NA_LOGICAL;
449  else {
450  int off = x1[i1] - y1[i2];
451  if (off < 0)
452  LOGICAL(ans)[i] = 0;
453  else {
454  LOGICAL(ans)[i] =
455  memcmp(x0[i1] + off, y0[i2], y1[i2]) == 0;
456  }
457  }
458  });
459  }
460  }
461  UNPROTECT(1);
462  return ans;
463 }

467 substrset(char *buf, const char *const str, cetype_t ienc, size_t sa, size_t so)
468 {
469  /* Replace the substring buf[sa:so] by str[] */
470  //int i;
471  size_t in = 0, out = 0;
472 
473  if (ienc == CE_UTF8) {
474  for (size_t i = 1; i < sa; i++) buf += utf8clen(*buf);
475  for (size_t i = sa; i <= so && in < strlen(str); i++) {
476  in += utf8clen(str[in]);
477  out += utf8clen(buf[out]);
478  if (!str[in]) break;
479  }
480  if (in != out) memmove(buf+in, buf+out, strlen(buf+out)+1);
481  memcpy(buf, str, in);
482  } else if (ienc == CE_LATIN1 || ienc == CE_BYTES) {
483  in = strlen(str);
484  out = so - sa + 1;
485  memcpy(buf + sa - 1, str, (in < out) ? in : out);
486  } else {
487  /* This cannot work for stateful encodings */
488  if (mbcslocale) {
489  for (size_t i = 1; i < sa; i++) buf += Rf_mbrtowc(nullptr, buf, MB_CUR_MAX, nullptr);
490  /* now work out how many bytes to replace by how many */
491  for (size_t i = sa; i <= so && in < strlen(str); i++) {
492  in += Rf_mbrtowc(nullptr, str+in, MB_CUR_MAX, nullptr);
493  out += Rf_mbrtowc(nullptr, buf+out, MB_CUR_MAX, nullptr);
494  if (!str[in]) break;
495  }
496  if (in != out) memmove(buf+in, buf+out, strlen(buf+out)+1);
497  memcpy(buf, str, in);
498  } else {
499  in = strlen(str);
500  out = so - sa + 1;
501  memcpy(buf + sa - 1, str, (in < out) ? in : out);
502  }
503  }

626 static SEXP stripchars(const char * const inchar, size_t minlen, int usecl)
627 {
628  int i, j;
629  size_t nspace = 0;
630  char *s = cbuff.data;
631 
632  /* The R wrapper removed leading and trailing spces */
633  mystrcpy(s, inchar);
634  if (strlen(s) < minlen) goto donesc;
635 
636  /* The for() loops never touch the first character */
637 
638  /* record spaces for removal later (as they act as word boundaries) */
639  for (i = UPPER(s), j = 1; i > 0; i--) {
640  if (isspace((int)s[i])) {
641  if (j) s[i] = '\0'; // trailing space
642  else nspace++;
643  } else j = 0;
644  if (strlen(s) - nspace <= minlen)
645  goto donesc;
646  }
647 
648  if(usecl) {
649  /* remove l/case vowels,
650  which are not at the beginning of a word but are at the end */
651  for (i = UPPER(s); i > 0; i--) {
652  if (LC_VOWEL(s, i) && LASTCHAR(s, i))
653  mystrcpy(s + i, s + i + 1);
654  if (strlen(s) - nspace <= minlen)
655  goto donesc;
656  }
657 
658  /* remove those not at the beginning of a word */
659  for (i = UPPER(s); i > 0; i--) {
660  if (LC_VOWEL(s, i) && !FIRSTCHAR(s, i))
661  mystrcpy(s + i, s + i + 1);
662  if (strlen(s) - nspace <= minlen)
663  goto donesc;
664  }
665 
666  /* Now do the same for remaining l/case chars */
667  for (i = UPPER(s); i > 0; i--) {
668  if (islower((int)s[i]) && LASTCHAR(s, i))
669  mystrcpy(s + i, s + i + 1);
670  if (strlen(s) - nspace <= minlen)
671  goto donesc;
672  }
673 
674  for (i = UPPER(s); i > 0; i--) {
675  if (islower((int)s[i]) && !FIRSTCHAR(s, i))
676  mystrcpy(s + i, s + i + 1);
677  if (strlen(s) - nspace <= minlen)
678  goto donesc;
679  }
680  }
681 
682  /* all else has failed so we use brute force */
683 
684  for (i = UPPER(s); i > 0; i--) {
685  if (!FIRSTCHAR(s, i) && !isspace((int)s[i]))
686  mystrcpy(s + i, s + i + 1);
687  if (strlen(s) - nspace <= minlen)
688  goto donesc;
689  }

693  size_t upper = strlen(s);
694  if (upper > minlen)
695  for (i = upper - 1; i > 0; i--)
696  if (isspace((int)s[i]))
697  mystrcpy(s + i, s + i + 1);
698  }
699 
700  return mkChar(s);
701 }
