232  Frame::Binding *Frame::obtainBinding(const Symbol *symbol)
233  {
234  Frame::Binding *binding = nullptr;
235  if (isLocked())
236  {
237  // If the frame is locked, we can only return pre-existing bindings.
238  Frame::Binding *binding = Frame::binding(symbol);
239  if (!binding)
240  {
241  Rf_error(_("cannot add bindings to a locked frame"));
242  }
243  return binding;
244  }
245 
246  if (m_descriptor)
247  {
248  // Use the pressigned location.
249  int location = m_descriptor->getLocation(symbol);
250  if (location != -1)
251  {
252  binding = m_bindings + location;
253  m_used_bindings_size = std::max(m_used_bindings_size,
254  (unsigned char)(location + 1));
255  }
256  }
257  else
258  {
259  // If the binding exists, return that.
260  binding = Frame::binding(symbol);
261  if (binding)
262  return binding;
263 
264  // Otherwise return the first unused space in the array if any.
265  for (unsigned char i = 0; i < m_bindings_size; ++i)
266  {
267  if (!m_bindings[i].isSet())
268  {
269  // Found an unused spot.
270  m_used_bindings_size = std::max(m_used_bindings_size,
271  (unsigned char)(i + 1));
272  binding = &m_bindings[i];
273  break;
274  }
275  }
276  }
277 
278  // If all else fails, go to the overflow.
279  if (!binding)
280  {
281  if (!m_overflow)
282  {
283  m_overflow = new map;
284  }
285  binding = &((*m_overflow)[symbol]);
286  }
287 
288  if (!binding->frame())
289  {
290  initializeBinding(binding, symbol);
291  }
292  return binding;
293  }

549  bool isMissingArgument(const Symbol *sym, Frame *frame)
550  {
551  RObject *rawval;
552  if (sym->isDotDotSymbol())
553  {
554  unsigned int ddv = sym->dotDotIndex();
555  Frame::Binding *bdg = frame->binding(DotsSymbol);
556  if (!bdg)
557  return false; // This is what CR does. Is it really right?
558  ConsCell *cc = SEXP_downcast<ConsCell *>(bdg->rawValue());
559  while (cc && ddv > 1)
560  {
561  cc = cc->tail();
562  --ddv;
563  }
564  if (!cc)
565  return true;
566  rawval = cc->car();
567  }
568  else
569  {
570  // Not a ..n symbol:
571  if (sym == Symbol::missingArgument())
572  return true;
573  Frame::Binding *bdg = frame->binding(sym);
574  if (!bdg)
575  return false;
576  rawval = bdg->rawValue();
577  if (bdg->origin() == Frame::Binding::MISSING || rawval == Symbol::missingArgument())
578  return true;
579  if (bdg->isActive())
580  return false;
581  }
582  if (rawval && rawval->sexptype() == PROMSXP)
583  {
584  Promise *prom = static_cast<Promise *>(rawval);
585  return prom->isMissingSymbol();
586  }
587  return false;
588  }
