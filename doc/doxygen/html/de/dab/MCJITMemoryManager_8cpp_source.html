49 uint64_t MCJITMemoryManager::getSymbolAddress(const std::string& name)
50 {
51  if (startsWith(name, symbol_prefix)) {
52  std::string symbol_name = name.substr(symbol_prefix.length());
53  return reinterpret_cast<uint64_t>(Symbol::obtain(symbol_name));
54  }
55 
56  if (startsWith(name, primitive_prefix)) {
57  // This returns the location of the BuiltInFunction.
58  std::string builtin_name = name.substr(primitive_prefix.length());
59  return reinterpret_cast<uint64_t>(
60  BuiltInFunction::obtainPrimitive(builtin_name));
61  }
62  if (startsWith(name, internal_prefix)) {
63  // This returns the location of the BuiltInFunction.
64  std::string builtin_name = name.substr(internal_prefix.length());
65  return reinterpret_cast<uint64_t>(
66  BuiltInFunction::obtainInternal(builtin_name));
67  }
68 
69  auto mapping = m_mappings.find(name);
70  if (mapping != m_mappings.end()) {
71  return reinterpret_cast<uint64_t>(mapping->second.first);
72  }
73 
74  return RTDyldMemoryManager::getSymbolAddress(name);
75 }

114 GlobalVariable* MCJITMemoryManager::addGlobal(Type* type, void* address,
115  bool is_constant,
116  std::string prefix)
117 {
118  std::string name = prefix;
119  if (prefix.empty()) {
120  prefix = "rho.global";
121  name = addCounter(prefix);
122  }
123  // Find a unique name in the global table.
124  std::pair<void*, GlobalVariable*>* item;
125  while(1) {
126  item = &m_mappings[name];
127  if (item->second && item->first != address) {
128  // This name was already assigned to a different object. Make a new
129  // name and try again.
130  name = addCounter(prefix);
131  continue;
132  }
133  break;
134  }
135 
136  if (!item->second) {
137  // New symbol in the global table.
138  item->first = address;
139  item->second = new GlobalVariable(*m_module, type, is_constant,
140  GlobalVariable::ExternalLinkage,
141  nullptr, name);
142  }
143  return item->second;
144 }
