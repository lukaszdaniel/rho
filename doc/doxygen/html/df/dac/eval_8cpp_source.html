76 /* A simple mechanism for profiling R code. When R_PROFILING is
77  enabled, eval will write out the call stack every PROFSAMPLE
78  microseconds using the SIGPROF handler triggered by timer signals
79  from the ITIMER_PROF timer. Since this is the same timer used by C
80  profiling, the two cannot be used together. Output is written to
81  the file PROFOUTNAME. This is a plain text file. The first line
82  of the file contains the value of PROFSAMPLE. The remaining lines
83  each give the call stack found at a sampling point with the inner
84  most function first.
85 
86  To enable profiling, recompile eval.cpp with R_PROFILING defined. It
87  would be possible to selectively turn profiling on and off from R
88  and to specify the file name from R as well, but for now I won't
89  bother.
90 
91  The stack is traced by walking back along the context stack, just
92  like the traceback creation in jump_to_toplevel. One drawback of
93  this approach is that it does not show BUILTIN's since they don't
94  get a context. With recent changes to pos.to.env it seems possible
95  to insert a context around BUILTIN calls to that they show up in
96  the trace. Since there is a cost in establishing these contexts,
97  they are only inserted when profiling is enabled. [BDR: we have since
98  also added contexts for the BUILTIN calls to foreign code.]
99 
100  One possible advantage of not tracing BUILTIN's is that then
101  profiling adds no cost when the timer is turned off. This would be
102  useful if we want to allow profiling to be turned on and off from
103  within R.
104 
105  One thing that makes interpreting profiling output tricky is lazy
106  evaluation. When an expression f(g(x)) is profiled, lazy
107  evaluation will cause g to be called inside the call to f, so it
108  will appear as if g is called by f.
109 
110  L. T. */

147 static int getFilenum(const char* filename) {
148  int fnum;
149 
150  for (fnum = 0; fnum < R_Line_Profiling-1
151  && strcmp(filename, R_Srcfiles[fnum]); fnum++);
152 
153  if (fnum == R_Line_Profiling-1) {
154  size_t len = strlen(filename);
155  if (fnum >= RHOCONSTRUCT(int, R_Srcfile_bufcount)) { /* too many files */
156  R_Profiling_Error = 1;
157  return 0;
158  }
159  if (R_Srcfiles[fnum] - reinterpret_cast<char*>(RAW(R_Srcfiles_buffer)) + len + 1 > (long unsigned int) Rf_length(R_Srcfiles_buffer)) {
160  /* out of space in the buffer */
161  R_Profiling_Error = 2;
162  return 0;
163  }
164  strcpy(R_Srcfiles[fnum], filename);
165  R_Srcfiles[fnum+1] = R_Srcfiles[fnum] + len + 1;
166  *(R_Srcfiles[fnum+1]) = '\0';
167  R_Line_Profiling++;
168  }
169 
170  return fnum + 1;

397  int mem_profiling, int gc_profiling,
398  int line_profiling, int numfiles, int bufsize)
399 {

401  struct itimerval itv;
402 #else

409  if(R_ProfileOutfile != nullptr) R_EndProfiling();
410  R_ProfileOutfile = RC_fopen(filename, append ? "a" : "w", TRUE);
411  if (R_ProfileOutfile == nullptr)

413  Rf_translateChar(filename));
414  if(mem_profiling)
415  fprintf(R_ProfileOutfile, "memory profiling: ");
416  if(gc_profiling)
417  fprintf(R_ProfileOutfile, "GC profiling: ");

428  R_GC_Profiling = gc_profiling;

465 }
