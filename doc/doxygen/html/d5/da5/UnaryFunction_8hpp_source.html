121  template<typename Op, typename AttributeCopier,
122  typename InputType,
123  typename OutputType = VectorOpReturnType<Op, InputType>>
124  OutputType* applyUnaryOperator(Op op,
125  AttributeCopier attribute_copier,
126  const InputType* input)
127  {
