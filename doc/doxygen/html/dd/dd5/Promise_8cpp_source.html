85 RObject* PromiseData::evaluate()
86 {
87  if (m_is_pointer_to_promise) {
88  return getThis()->evaluate();
89  }
90  if (m_value == Symbol::unboundValue()) {
91  // Force promise:
92  if (m_interrupted) {
93  Rf_warning(_("restarting interrupted promise evaluation"));
94  m_interrupted = false;
95  }
96  else if (m_under_evaluation)
97  Rf_error(_("promise already under evaluation: recursive default argument reference or earlier problems?"));
98  m_under_evaluation = true;
99  try {
100  IncrementStackDepthScope scope;
101  PlainContext cntxt;
102  RObject* val = Evaluator::evaluate(
103  const_cast<RObject*>(m_valgen.get()),
104  m_environment.get());
105  setValue(val);
106  }
107  catch (...) {
108  m_interrupted = true;
109  throw;
110  }
111  m_under_evaluation = false;
112  }
113  return m_value;
114 }

116 bool PromiseData::isMissingSymbol() const
117 {

119  return getThis()->isMissingSymbol();
120  }

122  bool ans = false;
123  /* This is wrong but I'm not clear why - arr
124  if (m_value == Symbol::missingArgument())
125  return true;
126  */
127  if (m_value == Symbol::unboundValue() && m_valgen) {
128  const RObject* prexpr = m_valgen;
129  if (prexpr->sexptype() == SYMSXP) {
130  // According to Luke Tierney's comment to R_isMissing() in CR,
131  // if a cycle is found then a missing argument has been
132  // encountered, so the return value is true.
133  if (m_under_evaluation)
134  return true;
