306 template <typename T, SEXPTYPE ST>
307 void* rho::FixedVector<T, ST>::allocate(size_type sz)
308 {
309  // We allocate enough space for sz + 1 elements so that the stack scanner
310  // recognizes end() as belonging to this object.
311  size_type blocksize = (sz + 1) * sizeof(T);
312  // Check for integer overflow:
313  if (blocksize / sizeof(T) != sz + 1)
314  Rf_error(_("request to create impossibly large vector."));
315 
316  size_type headersize = sizeof(FixedVector);
317 
318  try {
319  return GCNode::operator new(blocksize + headersize - sizeof(T));
320  } catch (std::bad_alloc& e) {
321  tooBig(blocksize);
322  return nullptr;
323  }
324 }

375  }
