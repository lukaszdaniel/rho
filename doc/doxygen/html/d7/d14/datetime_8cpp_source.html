147 /*
148  Adjust a struct tm to be a valid date-time.
149  Return 0 if valid, -1 if invalid and uncorrectable, or a positive
150  integer approximating the number of corrections needed.
151  */
152 static int validate_tm(stm *tm)
153 {
154  int tmp, res = 0;
155 
156  if (tm->tm_sec < 0 || tm->tm_sec > 60) { /* 61 POSIX, 60 draft ISO C */
157  res++;
158  tmp = tm->tm_sec/60;
159  tm->tm_sec -= 60 * tmp; tm->tm_min += tmp;
160  if(tm->tm_sec < 0) {tm->tm_sec += 60; tm->tm_min--;}
161  }
162 
163  if (tm->tm_min < 0 || tm->tm_min > 59) {
164  res++;
165  tmp = tm->tm_min/60;
166  tm->tm_min -= 60 * tmp; tm->tm_hour += tmp;
167  if(tm->tm_min < 0) {tm->tm_min += 60; tm->tm_hour--;}
168  }
169 
170  if(tm->tm_hour == 24 && tm->tm_min == 0 && tm->tm_sec == 0) {
171  tm->tm_hour = 0; tm->tm_mday++;
172  if(tm->tm_mon >= 0 && tm->tm_mon <= 11) {
173  if(tm->tm_mday > days_in_month[tm->tm_mon] +
174  ((tm->tm_mon==1 && isleap(1900+tm->tm_year) ? 1 : 0))) {
175  tm->tm_mon++; tm->tm_mday = 1;
176  if(tm->tm_mon == 12) {
177  tm->tm_year++; tm->tm_mon = 0;
178  }
179  }
180  }
181  }
182  if (tm->tm_hour < 0 || tm->tm_hour > 23) {
183  res++;
184  tmp = tm->tm_hour/24;
185  tm->tm_hour -= 24 * tmp; tm->tm_mday += tmp;
186  if(tm->tm_hour < 0) {tm->tm_hour += 24; tm->tm_mday--;}
187  }
188 
189  /* defer fixing mday until we know the year */
190  if (tm->tm_mon < 0 || tm->tm_mon > 11) {
191  res++;
192  tmp = tm->tm_mon/12;
193  tm->tm_mon -= 12 * tmp; tm->tm_year += tmp;
194  if(tm->tm_mon < 0) {tm->tm_mon += 12; tm->tm_year--;}
195  }
196 
197  /* A limit on the loops of about 3000x round */
198  if(tm->tm_mday < -1000000 || tm->tm_mday > 1000000) return -1;
199 
200  if(abs(tm->tm_mday) > 366) {
201  res++;
202  /* first spin back until January */
203  while(tm->tm_mon > 0) {
204  --tm->tm_mon;
205  tm->tm_mday += days_in_month[tm->tm_mon] +
206  ((tm->tm_mon==1 && isleap(1900+tm->tm_year))? 1 : 0);
207  }
208  /* then spin on/back by years */
209  while(tm->tm_mday < 1) {
210  --tm->tm_year;
211  tm->tm_mday += 365 + (isleap(1900+tm->tm_year)? 1 : 0);
212  }
213  while(tm->tm_mday >
214  (tmp = 365 + (isleap(1900+tm->tm_year)? 1 : 0))) {
215  tm->tm_mday -= tmp; tm->tm_year++;
216  }
217  }
218 
219  while(tm->tm_mday < 1) {
220  res++;
221  if(--tm->tm_mon < 0) {tm->tm_mon += 12; tm->tm_year--;}
222  tm->tm_mday += days_in_month[tm->tm_mon] +
223  ((tm->tm_mon==1 && isleap(1900+tm->tm_year))? 1 : 0);
224  }
225 
226  while(tm->tm_mday >
227  (tmp = days_in_month[tm->tm_mon] +
228  ((tm->tm_mon==1 && isleap(1900+tm->tm_year))? 1 : 0))) {
229  res++;
230  if(++tm->tm_mon > 11) {tm->tm_mon -= 12; tm->tm_year++;}
231  tm->tm_mday -= tmp;
232  }
233  return res;
234 }

237 /* Substitute for mktime -- no checking, always in GMT */
238 static double mktime00 (stm *tm)
239 {
240  int day = 0;
241  int year, year0;
242  double excess = 0.0;
243 
244  day = tm->tm_mday - 1;
245  year0 = 1900 + tm->tm_year;
246  /* safety check for unbounded loops */
247  if (year0 > 3000) {
248  excess = int(year0/2000) - 1;
249  year0 -= int(excess * 2000);
250  } else if (year0 < 0) {
251  excess = -1 - int(-year0/2000);
252  year0 -= int(excess * 2000);
253  }
254 
255  for(int i = 0; i < tm->tm_mon; i++) day += days_in_month[i];
256  if (tm->tm_mon > 1 && isleap(year0)) day++;
257  tm->tm_yday = day;
258 
259  if (year0 > 1970) {
260  for (year = 1970; year < year0; year++)
261  day += days_in_year(year);
262  } else if (year0 < 1970) {
263  for (year = 1969; year >= year0; year--)
264  day -= days_in_year(year);
265  }
266 
267  /* weekday: Epoch day was a Thursday */
268  if ((tm->tm_wday = (day + 4) % 7) < 0) tm->tm_wday += 7;
269 
270  return tm->tm_sec + (tm->tm_min * 60) + (tm->tm_hour * 3600)
271  + (day + excess * 730485) * 86400.0;
272 }

338 static double guess_offset (stm *tm)
339 {
340  double offset, offset1, offset2;
341  int i, wday, year, oldmonth, oldisdst, oldmday;
342  stm oldtm;
343  /*
344  Adjust as best we can for timezones: if isdst is unknown, use
345  the smaller offset at same day in Jan or July of a valid year.
346  We don't know the timezone rules, but if we choose a year with
347  July 1 on the same day of the week we will likely get guess
348  right (since they are usually on Sunday mornings not in Jan/Feb).
349 
350  Update for 2.7.0: no one had DST before 1916, so just use the offset
351  in 1902, if available.
352  */
353 
354  memcpy(&oldtm, tm, sizeof(stm));
355  if(!have_broken_mktime() && tm->tm_year < 2) { /* no DST */
356  tm->tm_year = 2;
357  mktime(tm);
358  offset1 = double(mktime(tm)) - mktime00(tm);
359  memcpy(tm, &oldtm, sizeof(stm));
360  tm->tm_isdst = 0;
361  return offset1;
362  }
363  oldmonth = tm->tm_mon;
364  oldmday = tm->tm_mday;
365  /* We know there was no DST prior to 1916 */
366  oldisdst = (tm->tm_year < 16) ? 0 : tm->tm_isdst;
367 
368  /* so now look for a suitable year */
369  tm->tm_mon = 6;
370  tm->tm_mday = 1;
371  tm->tm_isdst = -1;
372  mktime00(tm); /* to get wday valid */
373  wday = tm->tm_wday;
374  if (oldtm.tm_year > 137) { /* in the unknown future */
375  for(i = 130; i < 137; i++) { /* These cover all the possibilities */
376  tm->tm_year = i;
377  mktime(tm);
378  if(tm->tm_wday == wday) break;
379  }
380  } else { /* a benighted OS with date before 1970 */
381  /* We could not use 1970 because of the Windows bug with
382  1970-01-01 east of GMT. */
383  for(i = 71; i < 82; i++) { /* These cover all the possibilities */
384  tm->tm_year = i;
385  mktime(tm);
386  if(tm->tm_wday == wday) break;
387  }
388  }
389  year = i;
390 
391  /* Now look up offset in January */
392  tm->tm_mday = oldmday;
393  tm->tm_mon = 0;
394  tm->tm_year = year;
395  tm->tm_isdst = -1;
396  offset1 = double(mktime(tm)) - mktime00(tm);
397  /* and in July */
398  tm->tm_year = year;
399  tm->tm_mon = 6;
400  tm->tm_isdst = -1;
401  offset2 = double(mktime(tm)) - mktime00(tm);
402  if(oldisdst > 0) {
403  offset = (offset1 > offset2) ? offset2 : offset1;
404  } else {
405  offset = (offset1 > offset2) ? offset1 : offset2;
406  }
407  /* now try to guess dst if unknown */
408  tm->tm_mon = oldmonth;
409  tm->tm_isdst = -1;
410  if(oldisdst < 0) {
411  offset1 = double(mktime(tm)) - mktime00(tm);
412  oldisdst = (offset1 < offset) ? 1:0;
413  if(oldisdst) offset = offset1;
414  }
415  /* restore all as mktime might alter it */
416  memcpy(tm, &oldtm, sizeof(stm));
417  /* and then set isdst */
418  tm->tm_isdst = oldisdst;
419  return offset;
420 }

459 static stm * localtime0(const double *tp, const int local, stm *ltm)
460 {
461  double d = *tp;
462  int y, tmp;
463  time_t t;
464 
465  Rboolean OK;
466 /* as mktime is broken, do not trust localtime */
467 #ifdef HAVE_WORKING_64BIT_MKTIME
468  if (sizeof(time_t) == 8)
469  OK = Rboolean(!have_broken_mktime() || d > 0.);
470  else
471 #endif
472  OK = Rboolean(d < 2147483647.0 &&
473  d > (have_broken_mktime() ? 0. : -2147483647.0));
474  if(OK) {
475  t = (time_t) d;
476  /* if d is negative and non-integer then t will be off by one day
477  since we really need floor(). But floor() is slow, so we just
478  fix t instead as needed. */
479  if (d < 0.0 && double(t) != d) t--;
480 #ifndef HAVE_POSIX_LEAPSECONDS
481  for(int y = 0; y < n_leapseconds; y++) if(t > leapseconds[y] + y - 1) t++;
482 #endif
483 #ifdef HAVE_LOCALTIME_R
484  return local ? localtime_r(&t, ltm) : gmtime_r(&t, ltm);
485 #else
486  return local ? localtime(&t) : gmtime(&t);
487 #endif
488  }

490  /* internal substitute code.
491  Like localtime, this returns a pointer to a static struct tm */
492 
493  int day = (int) floor(d/86400.0);
494  int left = (int) (d - day * 86400.0 + 1e-6); // allow for fractional secs
495 
496  static stm ltm0, *res = &ltm0;
497  memset(res, 0, sizeof(stm));
498  /* hour, min, and sec */
499  res->tm_hour = left / 3600;
500  left %= 3600;
501  res->tm_min = left / 60;
502  res->tm_sec = left % 60;
503 
504  /* weekday: 1970-01-01 was a Thursday */
505  if ((res->tm_wday = ((4 + day) % 7)) < 0) res->tm_wday += 7;
506 
507  /* year & day within year */
508  y = 1970;
509  if (day >= 0)
510  for ( ; day >= (tmp = days_in_year(y)); day -= tmp, y++);
511  else
512  for ( ; day < 0; --y, day += days_in_year(y) );
513 
514  y = res->tm_year = y - 1900;
515  res->tm_yday = day;
516 
517  /* month within year */
518  int mon;
519  for (mon = 0;
520  day >= (tmp = (days_in_month[mon]) + ((mon==1 && isleap(y+1900))?1:0));
521  day -= tmp, mon++);
522  res->tm_mon = mon;
523  res->tm_mday = day + 1;
524 
525  if(local) {
526  double shift;
527  /* daylight saving time is unknown */
528  res->tm_isdst = -1;
529 
530  /* Try to fix up time zone differences: cf PR#15480 */
531  int sdiff = (int)guess_offset(res);
532  int diff = sdiff/60;
533  // just in case secs are out of range and might affect this.
534  shift = 60.*res->tm_hour + res->tm_min + res->tm_sec/60.;
535  res->tm_min -= diff;
536  res->tm_sec -= (sdiff % 60);
537  validate_tm(res);
538  res->tm_isdst = -1;
539  /* now this might be a different day */
540  if(shift - diff < 0.) {
541  res->tm_yday--;
542  res->tm_wday--;
543  }
544  else if(shift - diff >= 24. * 60.) {
545  res->tm_yday++;
546  res->tm_wday++;
547  }
548  int sdiff2 = (int)guess_offset(res);
549  int diff2 = sdiff2/60;
550  if(diff2 != diff) {
551  res->tm_min += (diff - diff2);
552  res->tm_sec += (sdiff % 60) - (sdiff2 % 60);
553  validate_tm(res);
554  }
555 #ifdef HAVE_TM_GMTOFF
556  res->tm_gmtoff = -sdiff2;
557 #endif
558  // No DST before 1916
559  if(res->tm_year < 16) res->tm_isdst = 0;
560  return res;
561  } else {
562  res->tm_isdst = 0; /* no dst in GMT */
563  return res;
564  }
565 }

624 {
625  /* set mon and mday which glibc does not always set.
626  Use current year/... if none has been specified.
627 
628  Specifying mon but not mday nor yday is invalid.
629  */
630  time_t t = time(nullptr);

632  int tmp;
633 #ifndef HAVE_POSIX_LEAPSECONDS
634  t -= n_leapseconds;
635 #endif
636 #ifdef HAVE_LOCALTIME_R

638  tm0 = localtime_r(&t, &tm2);
639 #else

641 #endif

643  if(tm->tm_mon != R_NaInt && tm->tm_mday != R_NaInt) return;

649  ((mon==1 && isleap(1900+tm->tm_year))? 1 : 0))) {

652  }

654  tm->tm_mday = yday + 1;
655  } else {
656  if(tm->tm_mday == R_NaInt) {
657  if(tm->tm_mon != R_NaInt) {
658  *invalid = 1;
659  return;
660  } else tm->tm_mday = tm0->tm_mday;
661  }
662  if(tm->tm_mon == R_NaInt) tm->tm_mon = tm0->tm_mon;
