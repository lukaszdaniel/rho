43 /* duplicate - object duplication */
44 
45 /* Because we try to maintain the illusion of call by
46  * value, we often need to duplicate entire data
47  * objects. There are a couple of points to note.
48  * First, duplication of list-like objects is done
49  * iteratively to prevent growth of the pointer
50  * protection stack, and second, the duplication of
51  * promises requires that the promises be forced and
52  * the value duplicated. */
53 
54 #define COPY_TRUELENGTH(to, from) \
55  do { \
56  if (!IS_GROWABLE(from)) \
57  SET_TRUELENGTH(to, XTRUELENGTH(from)); \
58  } while (0)

60 /* This macro pulls out the common code in copying an atomic vector.
61  The special handling of the scalar case (__n__ == 1) seems to make
62  a small but measurable difference, at least for some cases
63  and when (as in R 2.15.x) a for() loop was used.
64 */
65 #ifdef __APPLE__
66 /* it seems macOS builds did not copy >= 2^32 bytes fully */
67 #define DUPLICATE_ATOMIC_VECTOR(type, fun, to, from, deep) \
68  do { \
69  R_xlen_t __n__ = XLENGTH(from); \
70  PROTECT(from); \
71  PROTECT(to = allocVector(TYPEOF(from), __n__)); \
72  if (__n__ == 1) \
73  fun(to)[0] = fun(from)[0]; \
74  else { \
75  R_xlen_t __this; \
76  type *__to = fun(to), *__from = fun(from); \
77  do { \
78  __this = (__n__ < 1000000) ? __n__ : 1000000; \
79  memcpy(__to, __from, __this * sizeof(type)); \
80  __n__ -= __this; \
81  __to += __this; \
82  __from += __this; \
83  } while (__n__ > 0); \
84  } \
85  DUPLICATE_ATTRIB(to, from, deep); \
86  COPY_TRUELENGTH(to, from); \
87  UNPROTECT(2); \
88  } while (0)
89 #else
90 #define DUPLICATE_ATOMIC_VECTOR(type, fun, to, from, deep) \
91  do { \
92  R_xlen_t __n__ = XLENGTH(from); \
93  PROTECT(from); \
94  PROTECT(to = Rf_allocVector(TYPEOF(from), __n__)); \
95  if (__n__ == 1) \
96  fun(to)[0] = fun(from)[0]; \
97  else \
98  memcpy(fun(to), fun(from), __n__ * sizeof(type)); \
99  DUPLICATE_ATTRIB(to, from, deep); \
100  COPY_TRUELENGTH(to, from); \
101  UNPROTECT(2); \
102  } while (0)
103 #endif

105 /* The following macros avoid the cost of going through calls to the
106  assignment functions (and duplicate in the case of ATTRIB) when the
107  ATTRIB or TAG value to be stored is nullptr, the value the field
108  will have been set to by the allocation function */
109 #define SHALLOW_DUPLICATE_ATTRIB(to, from, deep) \
110  do { \
111  SEXP __a__ = ATTRIB(from); \
112  if (__a__ != nullptr) { \
113  SET_ATTRIB(to, shallow_duplicate1(__a__, deep)); \
114  SET_OBJECT(to, OBJECT(from)); \
115  IS_S4_OBJECT(from) ? SET_S4_OBJECT(to) : UNSET_S4_OBJECT(to); \
116  } \
117  } while (0)
118 
119 #define DUPLICATE_ATTRIB(to, from, deep) \
120  do { \
121  SEXP __a__ = ATTRIB(from); \
122  if (__a__ != nullptr) { \
123  SET_ATTRIB(to, duplicate1(__a__, deep)); \
124  SET_OBJECT(to, OBJECT(from)); \
125  IS_S4_OBJECT(from) ? SET_S4_OBJECT(to) : UNSET_S4_OBJECT(to); \
126  } \
127  } while (0)

129 #define COPY_TAG(to, from) \
130  do { \
131  SEXP __tag__ = TAG(from); \
132  if (__tag__ != nullptr) \
133  SET_TAG(to, __tag__); \
134  } while (0)

162 SEXP Rf_duplicate(SEXP s){
163  if (!s) return nullptr;

165 #ifdef R_PROFILING
166  duplicate_counter++;
167 #endif

169  if (!t) return s;

214 /* Detect cycles that would be created by assigning 'child' as a
215  component of 's' in a complex assignment without duplicating
216  'child'. This is called quite often but almost always returns
217  FALSE. Could be made more efficient, at least with partial
218  inlining, but probably not worth while until it starts showing up
219  significantly in profiling. Based on code from Michael Lawrence. */
220 Rboolean R_cycle_detected(SEXP s, SEXP child) {
221  if (s == child) {
222  switch (TYPEOF(child)) {
223  case NILSXP:
224  case SYMSXP:
225  case ENVSXP:
226  case SPECIALSXP:
227  case BUILTINSXP:
228  case EXTPTRSXP:
229  case BCODESXP:
230  case WEAKREFSXP:
231  /* it's a cycle but one that is OK */
232  return FALSE;
233  default:
234  return TRUE;
235  }
236  }
237  if (ATTRIB(child) != nullptr) {
238  if (R_cycle_detected(s, ATTRIB(child)))
239  return TRUE;
240  }
241  if (Rf_isPairList(child)) {
242  SEXP el = child;
243  while(el != nullptr) {
244  if (s == el || R_cycle_detected(s, CAR(el)))
245  return TRUE;
246  if (ATTRIB(el) != nullptr && R_cycle_detected(s, ATTRIB(el)))
247  return TRUE;
248  el = CDR(el);
249  }
250  } else if (Rf_isVectorList(child)) {
251  for(int i = 0 ; i < Rf_length(child); i++)
252  if (R_cycle_detected(s, VECTOR_ELT(child, i)))
253  return TRUE;
254  }
255  return FALSE;
256 }

260 void Rf_copyVector(SEXP s, SEXP t)
261 {
262  SEXPTYPE sT = TYPEOF(s), tT = TYPEOF(t);
263  if (sT != tT)
264  Rf_error("vector types do not match in copyVector");
265  R_xlen_t ns = XLENGTH(s), nt = XLENGTH(t);
266  switch (sT) {
267  case STRSXP:
268  xcopyStringWithRecycle(s, t, 0, ns, nt);
269  break;
270  case LGLSXP:
271  xcopyWithRecycle(LOGICAL(s), LOGICAL(t), 0, ns, nt);
272  break;
273  case INTSXP:
274  xcopyWithRecycle(INTEGER(s), INTEGER(t), 0, ns, nt);
275  break;
276  case REALSXP:
277  xcopyWithRecycle(REAL(s), REAL(t), 0, ns, nt);
278  break;
279  case CPLXSXP:
280  xcopyWithRecycle(COMPLEX(s), COMPLEX(t), 0, ns, nt);
281  break;
282  case EXPRSXP:
283  case VECSXP:
284  xcopyVectorWithRecycle(s, t, 0, ns, nt);
285  break;
286  case RAWSXP:
287  xcopyWithRecycle(RAW(s), RAW(t), 0, ns, nt);
288  break;
289  default:
290  UNIMPLEMENTED_TYPE("copyVector", s);
291  }
292 }

294 void Rf_copyListMatrix(SEXP s, SEXP t, Rboolean byrow)
295 {
296  SEXP pt, tmp;
297  int i, j, nr, nc;
298  R_xlen_t ns;
299 
300  nr = Rf_nrows(s);
301  nc = Rf_ncols(s);
302  ns = (R_xlen_t(nr)) * nc;
303  pt = t;
304  if(byrow) {
305  R_xlen_t NR = nr;
306  PROTECT(tmp = Rf_allocVector(STRSXP, ns));
307  for (i = 0; i < nr; i++)
308  for (j = 0; j < nc; j++) {
309  SET_STRING_ELT(tmp, i + j * NR, Rf_duplicate(CAR(pt)));
310  pt = CDR(pt);
311  if(pt == nullptr) pt = t;
312  }
313  for (i = 0; i < ns; i++) {
314  SETCAR(s, STRING_ELT(tmp, i++));
315  s = CDR(s);
316  }
317  UNPROTECT(1);
318  }
319  else {
320  for (i = 0; i < ns; i++) {
321  SETCAR(s, Rf_duplicate(CAR(pt)));
322  s = CDR(s);
323  pt = CDR(pt);
324  if(pt == nullptr) pt = t;
325  }
326  }
327 }

329 void Rf_copyMatrix(SEXP s, SEXP t, Rboolean byrow)
330 {
331  int nr = Rf_nrows(s), nc = Rf_ncols(s);
332  R_xlen_t nt = XLENGTH(t);
333 
334  if (byrow) {
335  switch (TYPEOF(s)) {
336  case STRSXP:
337  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
338  SET_STRING_ELT(s, didx, STRING_ELT(t, sidx));
339  break;
340  case LGLSXP:
341  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
342  LOGICAL(s)[didx] = LOGICAL(t)[sidx];
343  break;
344  case INTSXP:
345  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
346  INTEGER(s)[didx] = INTEGER(t)[sidx];
347  break;
348  case REALSXP:
349  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
350  REAL(s)[didx] = REAL(t)[sidx];
351  break;
352  case CPLXSXP:
353  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
354  COMPLEX(s)[didx] = COMPLEX(t)[sidx];
355  break;
356  case EXPRSXP:
357  case VECSXP:
358  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
359  SET_VECTOR_ELT(s, didx, VECTOR_ELT(t, sidx));
360  break;
361  case RAWSXP:
362  FILL_MATRIX_BYROW_ITERATE(0, nr, nc, nt)
363  RAW(s)[didx] = RAW(t)[sidx];
364  break;
365  default:
366  UNIMPLEMENTED_TYPE("copyMatrix", s);
367  }
368  }
369  else
370  Rf_copyVector(s, t);
371 }

373 #define COPY_ELT_WITH_RECYCLE(TNAME, GETELT, SETELT) \
374 HIDDEN void \
375 xcopy##TNAME##WithRecycle(SEXP dst, SEXP src, R_xlen_t dstart, R_xlen_t n, R_xlen_t nsrc) { \
376  \
377  if (nsrc >= n) { /* no recycle needed */ \
378  for(R_xlen_t i = 0; i < n; i++) \
379  SETELT(dst, dstart + i, GETELT(src, i)); \
380  return; \
381  } \
382  if (nsrc == 1) { \
383  SEXP val = GETELT(src, 0); \
384  for(R_xlen_t i = 0; i < n; i++) \
385  SETELT(dst, dstart + i, val); \
386  return; \
387  } \
388  \
389  /* recycle needed */ \
390  R_xlen_t sidx = 0; \
391  for(R_xlen_t i = 0; i < n; i++, sidx++) { \
392  if (sidx == nsrc) sidx = 0; \
393  SETELT(dst, dstart + i, GETELT(src, sidx)); \
394  } \
395 }
