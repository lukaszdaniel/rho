69 static void
70 RegisterClass(SEXP class_, SEXPTYPE type, const char *cname, const char *pname,
71  DllInfo *dll)
72 {
73  PROTECT(class_);
74  if (Registry == nullptr) {
75  Registry = PairList::cons(nullptr, nullptr);
76  R_PreserveObject(Registry);
77  }
78 
79  SEXP csym = rho::Symbol::obtain(cname);
80  SEXP psym = rho::Symbol::obtain(pname);
81  SEXP stype = PROTECT(Rf_ScalarInteger(type));
82  SEXP iptr = R_MakeExternalPtr(dll, nullptr, nullptr);
83  SEXP entry = LookupClassEntry(csym, psym);
84  if (entry == nullptr) {
85  entry = Rf_list4(class_, psym, stype, iptr);
86  SET_TAG(entry, csym);
87  SETCDR(Registry, Rf_cons(entry, CDR(Registry)));
88  }
89  else {
90  SETCAR(entry, class_);
91  SETCAR(CDR(CDR(entry)), stype);
92  SETCAR(CDR(CDR(CDR(entry))), iptr);
93  }
94  SET_ALTREP_CLASS_SERIALIZED_CLASS(class_, csym, psym, stype);
95  UNPROTECT(2); /* class_, stype */
96 }

245 static SEXP ALTREP_UNSERIALIZE_CLASS(SEXP info)
246 {
247  if (TYPEOF(info) == LISTSXP) {
248  SEXP csym = ALTREP_SERIALIZED_CLASS_CLSSYM(info);
249  SEXP psym = ALTREP_SERIALIZED_CLASS_PKGSYM(info);
250  SEXP class_ = LookupClass(csym, psym);
251  if (class_ == nullptr) {
252  SEXP pname = Rf_ScalarString(PRINTNAME(psym));
253  R_tryCatchError(find_namespace, pname,
254  handle_namespace_error, nullptr);
255  class_ = LookupClass(csym, psym);
256  }
257  return class_;
258  }
259  return nullptr;
260 }

262 HIDDEN SEXP
263 ALTREP_UNSERIALIZE_EX(SEXP info, SEXP state, SEXP attr, int objf, int levs)
264 {
265  SEXP csym = ALTREP_SERIALIZED_CLASS_CLSSYM(info);
266  SEXP psym = ALTREP_SERIALIZED_CLASS_PKGSYM(info);
267  SEXPTYPE type = ALTREP_SERIALIZED_CLASS_TYPE(info);
268 
269  /* look up the class in the registry and handle failure */
270  SEXP class_ = ALTREP_UNSERIALIZE_CLASS(info);
271  if (class_ == nullptr) {
272  switch(type) {
273  case LGLSXP:
274  case INTSXP:
275  case REALSXP:
276  case CPLXSXP:
277  case STRSXP:
278  case RAWSXP:
279  case VECSXP:
280  case EXPRSXP:
281  Rf_warning("cannot unserialize ALTVEC object of class '%s' from package '%s'; returning length zero vector",
282  R_CHAR(PRINTNAME(csym)), R_CHAR(PRINTNAME(psym)));
283  return Rf_allocVector(type, 0);
284  default:
285  Rf_error("cannot unserialize this ALTREP object");
286  }
287  }
288 
289  /* check the registered and unserialized types match */
290  SEXPTYPE rtype = ALTREP_CLASS_BASE_TYPE(class_);
291  if (type != rtype)
292  Rf_warning("serialized class '%s' from package '%s' has type %s; registered class has type %s",
293  R_CHAR(PRINTNAME(csym)), R_CHAR(PRINTNAME(psym)),
294  Rf_type2char(type), Rf_type2char(rtype));
295 
296  /* dispatch to a class method */
297  altrep_methods_t *m = static_cast<altrep_methods_t *>(CLASS_METHODS_TABLE(class_));
298  SEXP val = m->UnserializeEX(class_, state, attr, objf, levs);
299  return val;
300 }

566 static SEXP altrep_DuplicateEX_default(SEXP x, Rboolean deep)
567 {
568  SEXP ans = ALTREP_DUPLICATE(x, deep);
569 
570  if (ans != nullptr &&
571  ans != x) { /* leave attributes alone if returning original */
572  /* handle attributes generically */
573  SEXP attr = ATTRIB(x);
574  if (attr != nullptr) {
575  PROTECT(ans);
576  SET_ATTRIB(ans, deep ? Rf_duplicate(attr) : Rf_shallow_duplicate(attr));
577  SET_OBJECT(ans, OBJECT(x));
578  IS_S4_OBJECT(x) ? SET_S4_OBJECT(ans) : UNSET_S4_OBJECT(ans);
579  UNPROTECT(1);
580  }
581  }
582  return ans;
583 }

981 {
982  if (COMPACT_SEQ_EXPANDED(x) == nullptr) {
983  /* no need to re-run if expanded data exists */
984  PROTECT(x);

998  /* compact sequences n1 : n2 with n1 > n2 */
999  for (int i = 0; i < n; i++)
1000  data[i] = n1 - i;
1001  }
1002  else
1003  Rf_error("compact sequences with increment %d not supported yet", inc);
1004

1006  UNPROTECT(1);
