42 do_mapply(/*const*/ Expression* call, const BuiltInFunction* op, Environment* rho, RObject* const* args, int num_args, const PairList* tags)
43 {
44  SEXP f = args[0], varyingArgs = args[1], constantArgs = args[2];
45  int m, zero = 0;
46  R_xlen_t *lengths, *counters, longest = 0;
47 
48  m = Rf_length(varyingArgs);
49  SEXP vnames = PROTECT(Rf_getAttrib(varyingArgs, R_NamesSymbol));
50  Rboolean named = RHOCONSTRUCT(Rboolean, vnames != nullptr);
51 
52  lengths = static_cast<R_xlen_t *>(RHO_alloc(m, sizeof(R_xlen_t)));
53  for (int i = 0; i < m; i++) {
54  SEXP tmp1 = VECTOR_ELT(varyingArgs, i);
55  lengths[i] = dispatch_xlength(tmp1, call, rho);
56  if (lengths[i] == 0) zero++;
57  if (lengths[i] > longest) longest = lengths[i];
58  }
59  if (zero && longest)
60  Rf_error(_("zero-length inputs cannot be mixed with those of non-zero length"));
61 
62  counters = static_cast<R_xlen_t *>(RHO_alloc(m, sizeof(R_xlen_t)));
63  if (m) memset(counters, 0, m * sizeof(R_xlen_t));
64 
65  SEXP mindex = PROTECT(Rf_allocVector(VECSXP, m));
66  SEXP nindex = PROTECT(Rf_allocVector(VECSXP, m));
67 
68  /* build a call like
69  f(dots[[1]][[4]], dots[[2]][[4]], dots[[3]][[4]], d=7)
70  */
71 
72  PairList* fargs = R_NilValue; // -Wall
73  if (constantArgs == R_NilValue)
74  ;
75  else if (Rf_isVectorList(constantArgs))
76  fargs = static_cast<PairList*>(Rf_VectorToPairList(constantArgs));
77  else
78  Rf_error(_("argument 'MoreArgs' of 'mapply' is not a list"));
79  PROTECT_INDEX fi;
80  PROTECT_WITH_INDEX(fargs, &fi);
81 
82  Rboolean realIndx = RHOCONSTRUCT(Rboolean, longest > INT_MAX);
83  static Symbol* Dots = Symbol::obtain("dots");
84  for (int j = m - 1; j >= 0; j--) {
85  SET_VECTOR_ELT(mindex, j, Rf_ScalarInteger(j + 1));
86  SET_VECTOR_ELT(nindex, j, Rf_allocVector(realIndx ? REALSXP : INTSXP, 1));
87  SEXP tmp1 = PROTECT(Rf_lang3(R_Bracket2Symbol, Dots, VECTOR_ELT(mindex, j)));
88  SEXP tmp2 = PROTECT(Rf_lang3(R_Bracket2Symbol, tmp1, VECTOR_ELT(nindex, j)));
89  REPROTECT(fargs = PairList::cons(tmp2, fargs), fi);
90  UNPROTECT(2);
91  if (named && R_CHAR(STRING_ELT(vnames, j))[0] != '\0')
92  SET_TAG(fargs, Rf_installTrChar(STRING_ELT(vnames, j)));
93  }
94 
95  Expression* fcall = new Expression(f, fargs);
96 
97  SEXP ans = PROTECT(Rf_allocVector(VECSXP, longest));
98 
99  for (int i = 0; i < longest; i++) {
100  for (int j = 0; j < m; j++) {
101  counters[j] = (++counters[j] > lengths[j]) ? 1 : counters[j];
102  if (realIndx)
103  REAL(VECTOR_ELT(nindex, j))[0] = double(counters[j]);
104  else
105  INTEGER(VECTOR_ELT(nindex, j))[0] = int(counters[j]);
106  }
107  SEXP tmp = R_forceAndCall(fcall, m, rho);
108  if (MAYBE_REFERENCED(tmp))
109  tmp = Rf_lazy_duplicate(tmp);
110  SET_VECTOR_ELT(ans, i, tmp);
111  }
112 
113  for (int j = 0; j < m; j++)
114  if (counters[j] != lengths[j])
115  Rf_warning(_("longer argument not a multiple of length of shorter"));
116 
117  UNPROTECT(5);
118  return ans;
119 }
