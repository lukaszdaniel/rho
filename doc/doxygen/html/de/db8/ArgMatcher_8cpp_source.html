51 ArgMatcher::ArgMatcher(const PairList* formals)
52  : m_dots_position(-1)
53 {
54  m_formals = formals;
55 
56  for (const PairList* f = formals; f; f = f->tail()) {
57  const Symbol* sym = dynamic_cast<const Symbol*>(f->tag());
58  if (!sym)
59  Rf_error(_("invalid formal arguments for 'function'"));
60  if (sym == DotsSymbol) {
61  if (has3Dots())
62  Rf_error(_("formals list contains more than one '...'"));
63  if (f->car() != Symbol::missingArgument())
64  Rf_error(_("'...' formal cannot have a default value"));
65  m_dots_position = m_formal_data.size();
66  } else {
67  pair<FormalMap::const_iterator, bool> pr =
68  m_formal_index.insert(make_pair(sym->name(),
69  m_formal_data.size()));
70  if (!pr.second)
71  Rf_error(_("duplicated name in formals list"));
72  }
73 
74  FormalData fdata = {sym, has3Dots(), f->car(),
75  static_cast<unsigned>(m_formal_data.size()) };
76  m_formal_data.push_back(fdata);
77  }
78 }

126 class ArgMatcher::MatchCallback {
127 public:
128  typedef ArgMatcher::FormalData FormalData;
129  typedef ArgMatcher::SuppliedData SuppliedData;
130  typedef ArgMatcher::SuppliedList SuppliedList;
131  typedef boost::iterator_range<std::vector<int>::const_iterator> ArgIndices;
132 
133  virtual void matchedArgument(const FormalData& formal,
134  int arg_index,
135  RObject* value) = 0;
136 
137  virtual void defaultValue(const FormalData& formal) = 0;

139  virtual void dottedArgs(const FormalData& formal,
140  ArgIndices arg_indices,

142 
143  static DottedArgs* makeDottedArgs(ArgIndices arg_indices,
144  const ArgList& all_args) {
145  DottedArgs* result = nullptr;
146  ConsCell* dots = nullptr;
147  for (int index : arg_indices) {
148  RObject* value = all_args.get(index);
149  const RObject* tag = all_args.getTag(index);
150  if (dots) {
151  PairList* next_item = new PairList(value, nullptr, tag);
152  dots->setTail(next_item);
153  dots = next_item;
154  } else {
155  dots = result = new DottedArgs(value, nullptr, tag);
156  }
157  }
158  return result;
159  }

162 void ArgMatcher::matchWithCache(const ArgList& supplied,
163  MatchCallback* callback,
164  const ArgMatchCache* matching) const
165 {
166  for (size_t findex = 0; findex < m_formal_data.size(); findex++) {
167  const FormalData& fdata = m_formal_data[findex];
168  int sindex = matching->getSindex(findex);
169  if (sindex >= 0) {
170  // This assumes that random access in an ArgList is O(1).
171  callback->matchedArgument(fdata, sindex, supplied.get(sindex));
172  }
173  else if (fdata.symbol != R_DotsSymbol) {
174  callback->defaultValue(fdata);
175  }
176  else {
177  callback->dottedArgs(fdata, matching->getDotArgs(), supplied);
178  }
179  }

417 void ArgMatcher::matchByPosition(const ArgList& supplied,
418  MatchCallback* callback) const
419 {
420  const size_t num_formals
421  = m_dots_position == -1 ? m_formal_data.size() : m_dots_position;
422  unsigned int supplied_index = 0;
423  size_t formals_index = 0;
424  ArgList::const_iterator s = supplied.getArgs().begin();
425  const ArgList::const_iterator end = supplied.getArgs().end();
426  for ( ;
427  (s != end) && (formals_index < num_formals);
428  ++s, ++formals_index, ++supplied_index)
429  {
430  callback->matchedArgument(m_formal_data[formals_index],
431  supplied_index, s->car());
432  }
433  // Set unmatched arguments to their default values.
434  for (; formals_index < m_formal_data.size(); ++formals_index) {
435  if (int(formals_index) != m_dots_position)
436  callback->defaultValue(m_formal_data[formals_index]);
437  }
438 
439  // Any remaining supplied args are either rolled into ... or
440  // there's an error:
441  if (has3Dots()) {
442  // Pass unmatched arguments.
443  vector<int> dotted_arg_indices;
444  for (; s != end; ++supplied_index, ++s) {
445  dotted_arg_indices.push_back(supplied_index);
446  }
447  callback->dottedArgs(m_formal_data[m_dots_position],
448  boost::make_iterator_range(dotted_arg_indices),
449  supplied);
450  } else {
451  if (s != end) {
452  unusedArgsError(&*s);
453  }
454  }
455 }

457 void ArgMatcher::match(const ArgList& supplied,
458  MatchCallback* callback) const
459 {
460  // Short-circuit if none of the arguments are named.
461  if (!supplied.hasTags()) {
462  matchByPosition(supplied, callback);
463  return;
464  }
465 
466  vector<MatchStatus, Allocator<MatchStatus> >
467  formals_status(m_formal_data.size(), UNMATCHED);
468  SuppliedList supplied_list;
469  // Exact matches by tag:
470  {
471  unsigned int sindex = 0;
472  for (const ConsCell& s : supplied.getArgs()) {
473  const Symbol* tag = static_cast<const Symbol*>(s.tag());
474  const String* name = (tag ? tag->name() : nullptr);
475  RObject* value = s.car();
476  FormalMap::const_iterator fmit
477  = (name ? m_formal_index.lower_bound(name)
478  : m_formal_index.end());
479  if (fmit != m_formal_index.end() && (*fmit).first == name) {
480  // Exact tag match:
481  unsigned int findex = (*fmit).second;
482  formals_status[findex] = EXACT_TAG;
483  callback->matchedArgument(m_formal_data[findex], sindex,
484  value);
485  } else {
486  // No exact tag match, so place supplied arg on list:
487  SuppliedData supplied_data
488  = {tag, value, fmit, sindex};
489  supplied_list.push_back(supplied_data);
490  }
491  ++sindex;
492  }
493  }
494  // Partial matches by tag:
495  {
496  SuppliedList::iterator slit = supplied_list.begin();
497  while (slit != supplied_list.end()) {
498  SuppliedList::iterator next = slit;
499  ++next;
500  const SuppliedData& supplied_data = *slit;
501  const String* supplied_name
502  = (supplied_data.tag ? supplied_data.tag->name() : nullptr);
503  FormalMap::const_iterator fmit = supplied_data.fm_iter;
504  // Within m_formal_index, skip formals formals following
505  // '...' and formals with exact matches:
506  while (fmit != m_formal_index.end()
507  && (m_formal_data[(*fmit).second].follows_dots
508  || formals_status[(*fmit).second] == EXACT_TAG))
509  ++fmit;
510  if (fmit != m_formal_index.end()
511  && isPrefix(supplied_name, (*fmit).first)) {
512  // This is a potential partial match. Remember the formal:
513  unsigned int findex = (*fmit).second;
514  // Has this formal already been partially matched? :
515  if (formals_status[(*fmit).second] == PARTIAL_TAG)
516  Rf_error(_("formal argument '%s' matched by multiple actual arguments"),
517  (*fmit).first->c_str());
518  // Does supplied arg partially match anything else? :
519  do {
520  ++fmit;
521  } while (fmit != m_formal_index.end()
522  && formals_status[(*fmit).second] == EXACT_TAG);
523  if (fmit != m_formal_index.end()
524  && isPrefix(supplied_name, (*fmit).first))
525  Rf_error(_("argument %d matches multiple formal arguments"),
526  supplied_data.index + 1);
527  // Partial match is OK:
528  if (s_warn_on_partial_match)
529  Rf_warning(_("partial argument match of '%s' to '%s'"),
530  supplied_name->c_str(),
531  (*fmit).first->c_str());
532  formals_status[findex] = PARTIAL_TAG;
533  callback->matchedArgument(m_formal_data[findex], supplied_data.index, supplied_data.value);
534  supplied_list.erase(slit);
535  }
536  slit = next;
537  }
538  }
539  // Positional matching and default values:
540  {
541  const size_t numformals = m_formal_data.size();
542  SuppliedList::iterator slit = supplied_list.begin();
543  for (unsigned int findex = 0; findex < numformals; ++findex) {
544  if (formals_status[findex] == UNMATCHED) {
545  const FormalData& fdata = m_formal_data[findex];
546  // Skip supplied arguments with tags:
547  while (slit != supplied_list.end() && (*slit).tag)
548  ++slit;
549  if (slit != supplied_list.end()
550  && !fdata.follows_dots) {
551  // Handle positional match:
552  const SuppliedData& supplied_data = *slit;
553  formals_status[findex] = POSITIONAL;
554  callback->matchedArgument(fdata, supplied_data.index,
555  supplied_data.value);
556  supplied_list.erase(slit++);
557  } else if (int(findex) != m_dots_position) {
558  callback->defaultValue(fdata);
559  }
560  }
561  }
562  }
563 
564  // Any remaining supplied args are either rolled into ... or
565  // there's an error:
566  if (has3Dots()) {
567  vector<int> dotted_arg_indices;
568  for (const auto& sitem : supplied_list) {
569  dotted_arg_indices.push_back(sitem.index);
570  }
571  callback->dottedArgs(m_formal_data[m_dots_position],
572  boost::make_iterator_range(dotted_arg_indices),
573  supplied);
574  }
575  else if (!supplied_list.empty())
576  unusedArgsError(supplied_list);
577 }

579 void ArgMatcher::propagateFormalBindings(const Environment* fromenv,
580  Environment* toenv) const
581 {
582  const Frame* fromf = fromenv->frame();
583  for (const FormalData& fdata : m_formal_data) {
584  const Symbol* symbol = fdata.symbol;
585  const Frame::Binding* frombdg = fromf->binding(symbol);
586  if (!frombdg)
587  Rf_error(_("could not find symbol \"%s\" in environment of the generic function"),
588  symbol->name()->c_str());

590  if (frombdg->origin() == Frame::Binding::EXPLICIT) {
591  makeBinding(toenv, fdata, Frame::Binding::EXPLICIT, val);
592  } else {

594  makeBinding(toenv, fdata, Frame::Binding::DEFAULTED,
595  fdata.default_value);

598  // m_formal_data excludes '...', so:
599  if (has3Dots()) {
600  const Frame::Binding* frombdg = fromf->binding(DotsSymbol);
601  toenv->frame()->importBinding(frombdg);
