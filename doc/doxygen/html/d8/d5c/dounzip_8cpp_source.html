118 static int
119 extract_one(unzFile uf, const char *const dest, const char * const filename,
120  SEXP names, int *nnames, int overwrite, int junk, int setTime)
121 {
122  int err = UNZ_OK;
123  FILE *fout;
124  char outname[PATH_MAX], dirs[PATH_MAX], buf[BUF_SIZE], *p, *pp;
125  char *fn, fn0[PATH_MAX];
126 
127  err = unzOpenCurrentFile(uf);
128  if (err != UNZ_OK) return err;
129  if (strlen(dest) > PATH_MAX - 1) return 1;
130  strcpy(outname, dest);
131  strcat(outname, FILESEP);
132  unz_file_info64 file_info;
133  char filename_inzip[PATH_MAX];
134  err = unzGetCurrentFileInfo64(uf, &file_info, filename_inzip,
135  sizeof(filename_inzip), nullptr, 0, nullptr, 0);
136  fn = filename_inzip; /* might be UTF-8 ... */
137  if (filename) {
138  if (strlen(dest) + strlen(filename) > PATH_MAX - 2) return 1;
139  strncpy(fn0, filename, PATH_MAX);
140  fn0[PATH_MAX - 1] = '\0';
141  fn = fn0;
142  }
143 #ifdef Win32
144  R_fixslash(fn);
145 #endif
146  if (junk && strlen(fn) >= 2) { /* need / and basename */
147  p = Rf_strrchr(fn, '/');
148  if (p) fn = p+1;
149  }
150  strcat(outname, fn);
151 
152 #ifdef Win32
153  R_fixslash(outname); /* ensure path separator is / */
154 #endif
155  p = outname + strlen(outname) - 1;
156  if (*p == '/') { /* Directories are stored with trailing slash */
157  if (!junk) {
158  *p = '\0';
159  if (!R_FileExists(outname)) {
160  /* make parents as required: have already checked dest exists */
161  pp = outname + strlen(dest) + 1;
162  while((p = Rf_strchr(pp, '/'))) {
163  strcpy(dirs, outname);
164  dirs[p - outname] = '\0';
165  if (!R_FileExists(dirs)) R_mkdir(dirs);
166  pp = p + 1;
167  }
168  err = R_mkdir(outname);
169  }
170  }
171  } else {
172  /* make parents as required: have already checked dest exists */
173  pp = outname + strlen(dest) + 1;
174  while((p = Rf_strchr(pp, '/'))) {
175  strcpy(dirs, outname);
176  dirs[p - outname] = '\0';
177  /* Rprintf("dirs is %s\n", dirs); */
178  if (!R_FileExists(dirs)) R_mkdir(dirs);
179  pp = p + 1;
180  }
181  /* Rprintf("extracting %s\n", outname); */
182  if (!overwrite && R_FileExists(outname)) {
183  Rf_warning(_(" not overwriting file '%s"), outname);
184  unzCloseCurrentFile(uf);
185  return err;
186  }
187  fout = R_fopen(outname, "wb");
188  int serrno = errno;
189  if (!fout) {
190  unzCloseCurrentFile(uf);
191  Rf_error(_("cannot open file '%s': %s"), outname, strerror(serrno));
192  return 3; /* not reached */
193  }
194  while (1) {
195  err = unzReadCurrentFile(uf, buf, BUF_SIZE);
196  /* Rprintf("read %d bytes\n", err); */
197  if (err <= 0) break;
198  if (fwrite(buf, err, 1, fout) != 1) { err = -200; break; }
199  if (err < BUF_SIZE) { err = 0; break; }
200  }
201  fclose(fout);
202  SET_STRING_ELT(names, (*nnames)++, Rf_mkChar(outname));
203  }
204  unzCloseCurrentFile(uf);
205 #ifdef Win32
206  if (setTime) setFileTime(outname, file_info.dosDate);
207 #else
208  if (setTime) setFileTime(outname, file_info.tmu_date);
209 #endif
210  return err;
211 }

214 static int
215 zipunzip(const char *zipname, const char *dest, int nfiles, const char **files,
216  SEXP *pnames, int *nnames, int overwrite, int junk, int setTime)
217 {
218  int i, err = UNZ_OK;
219  unzFile uf;
220  SEXP names = *pnames;
221 
222  uf = unzOpen64(zipname);
223  if (!uf) return 1;
224  if (nfiles == 0) { /* all files */
225  unz_global_info64 gi;
226  unzGetGlobalInfo64(uf, &gi);
227  for (i = 0; i < int(gi.number_entry); i++) {
228  if (i > 0) if ((err = unzGoToNextFile(uf)) != UNZ_OK) break;
229  if (*nnames+1 >= LENGTH(names)) {
230  SEXP onames = names;
231  names = Rf_allocVector(STRSXP, 2*LENGTH(names));
232  UNPROTECT(1);
233  PROTECT(names);
234  Rf_copyVector(names, onames);
235  }
236  if ((err = extract_one(uf, dest, nullptr, names, nnames,
237  overwrite, junk, setTime)) != UNZ_OK) break;
238 #ifdef Win32
239  R_ProcessEvents();
240 #else
241  R_CheckUserInterrupt();
242 #endif
243  }
244  } else {
245  for (i = 0; i < nfiles; i++) {
246  if ((err = unzLocateFile(uf, files[i], 1)) != UNZ_OK) break;
247  if ((err = extract_one(uf, dest, files[i], names, nnames,
248  overwrite, junk, setTime)) != UNZ_OK) break;
249 #ifdef Win32
250  R_ProcessEvents();
251 #else
252  R_CheckUserInterrupt();
253 #endif
254  }
255  }
256  *pnames = names;
257  unzClose(uf);
258  return err;
259 }

261 static SEXP ziplist(const char *zipname)
262 {
263  SEXP ans = nullptr, names, lengths, dates;
264  unzFile uf;
265  uLong i;
266  unz_global_info64 gi;
267  int err, nfiles;
268 
269  uf = unzOpen64(zipname);
270  if (!uf) Rf_error(_("zip file '%s' cannot be opened"), zipname);
271 
272  gi.number_entry = 0; /* =Wall */
273  err = unzGetGlobalInfo64 (uf, &gi);
274  if (err != UNZ_OK)
275  Rf_error("error %d with zipfile in unzGetGlobalInfo", err);
276  nfiles = int(gi.number_entry);
277  /* name, length, datetime */
278  PROTECT(ans = Rf_allocVector(VECSXP, 3));
279  SET_VECTOR_ELT(ans, 0, names = Rf_allocVector(STRSXP, nfiles));
280  SET_VECTOR_ELT(ans, 1, lengths = Rf_allocVector(REALSXP, nfiles));
281  SET_VECTOR_ELT(ans, 2, dates = Rf_allocVector(STRSXP, nfiles));
282 
283  for (i = 0; int(i) < nfiles; i++) {
284  char filename_inzip[PATH_MAX], date[50];
285  unz_file_info64 file_info;
286 
287  err = unzGetCurrentFileInfo64(uf, &file_info, filename_inzip,
288  sizeof(filename_inzip), nullptr, 0, nullptr, 0);
289  if (err != UNZ_OK)
290  Rf_error("error %d with zipfile in unzGetCurrentFileInfo\n", err);
291  /* In theory at least bit 11 of the flag tells us that the
292  filename is in UTF-8, so FIXME */
293  SET_STRING_ELT(names, i, Rf_mkChar(filename_inzip));
294  REAL(lengths)[i] = double(file_info.uncompressed_size);
295  snprintf(date, 50, "%d-%02d-%02d %02d:%02d",
296  file_info.tmu_date.tm_year,
297  file_info.tmu_date.tm_mon + 1,
298  file_info.tmu_date.tm_mday,
299  file_info.tmu_date.tm_hour,
300  file_info.tmu_date.tm_min);
301  SET_STRING_ELT(dates, i, Rf_mkChar(date));
302 
303  if (int(i) < nfiles - 1) {
304  err = unzGoToNextFile(uf);
305  if (err != UNZ_OK)
306  Rf_error("error %d with zipfile in unzGoToNextFile\n",err);
307  }
308  }
309  unzClose(uf);
310 
311  UNPROTECT(1);
312  return ans;
313 }

317 SEXP Runzip(SEXP args)
318 {
319  SEXP fn, ans, names = nullptr;
320  char zipname[PATH_MAX], dest[PATH_MAX];
321  const char *p, **topics = nullptr;
322  int i, ntopics, list, overwrite, junk, setTime, rc, nnames = 0;
323  const void *vmax = vmaxget();
324 
325  if (!Rf_isString(CAR(args)) || LENGTH(CAR(args)) != 1)
326  Rf_error(_("invalid zip name argument"));
327  p = R_ExpandFileName(Rf_translateChar(STRING_ELT(CAR(args), 0)));
328  if (strlen(p) > PATH_MAX - 1)
329  Rf_error(_("zip path is too long"));
330  strcpy(zipname, p);
331  args = CDR(args);
332  fn = CAR(args);
333  ntopics = Rf_length(fn);
334  if (ntopics > 0) {
335  if (!Rf_isString(fn))
336  Rf_error(_("invalid '%s' argument"), "files");
337  topics = static_cast<const char **>(RHO_alloc(ntopics, sizeof(char *)));
338  for (i = 0; i < ntopics; i++)
339  topics[i] = Rf_translateChar(STRING_ELT(fn, i));
340  }
341  args = CDR(args);
342  if (!Rf_isString(CAR(args)) || LENGTH(CAR(args)) != 1)
343  Rf_error(_("invalid '%s' argument"), "exdir");
344  p = R_ExpandFileName(Rf_translateChar(STRING_ELT(CAR(args), 0)));
345  if (strlen(p) > PATH_MAX - 1)
346  Rf_error(_("'exdir' is too long"));
347  strcpy(dest, p);
348  if (!R_FileExists(dest))
349  Rf_error(_("'exdir' does not exist"));
350  args = CDR(args);
351  list = Rf_asLogical(CAR(args));
352  if (list == R_NaLog)
353  Rf_error(_("invalid '%s' argument"), "list");
354  if (list) return(ziplist(zipname));
355  args = CDR(args);
356  overwrite = Rf_asLogical(CAR(args));
357  if (overwrite == R_NaLog)
358  Rf_error(_("invalid '%s' argument"), "overwrite");
359  args = CDR(args);
360  junk = Rf_asLogical(CAR(args));
361  if (junk == R_NaLog)
362  Rf_error(_("invalid '%s' argument"), "junkpaths");
363  args = CDR(args);
364  setTime = Rf_asLogical(CAR(args));
365  if (setTime == R_NaLog)
366  Rf_error(_("invalid '%s' argument"), "setTime");
367 
368  if (ntopics > 0)
369  PROTECT(names = Rf_allocVector(STRSXP, ntopics));
370  else
371  PROTECT(names = Rf_allocVector(STRSXP, 5000));
372  rc = zipunzip(zipname, dest, ntopics, topics, &names, &nnames,
373  overwrite, junk, setTime);
374  if (rc != UNZ_OK)
375  switch(rc) {
376  case UNZ_END_OF_LIST_OF_FILE:
377  Rf_warning(_("requested file not found in the zip file"));
378  break;
379  case UNZ_BADZIPFILE:
380  Rf_warning(_("zip file is corrupt"));
381  break;
382  case UNZ_CRCERROR:
383  Rf_warning(_("CRC error in zip file"));
384  break;
385  case UNZ_PARAMERROR:
386  case UNZ_INTERNALERROR:
387  Rf_warning("internal error in 'unz' code");
388  break;
389  case -200:
390  Rf_warning(_("write error in extracting from zip file"));
391  break;
392  default:
393  Rf_warning(_("error %d in extracting from zip file"), rc);
394  }
395  PROTECT(ans = Rf_ScalarInteger(rc));
396  PROTECT(names = Rf_lengthgets(names, nnames));
397  Rf_setAttrib(ans, Rf_install("extracted"), names);
398  UNPROTECT(3);
399  vmaxset(vmax);
400  return ans;
401 }

411 static Rboolean unz_open(Rconnection con)
412 {
413  unzFile uf;
414  char path[2*PATH_MAX], *p;
415  const char *tmp;
416 
417  if(con->mode[0] != 'r') {
418  Rf_warning(_("unz connections can only be opened for reading"));
419  return FALSE;
420  }
421  tmp = R_ExpandFileName(con->description);
422  if (strlen(tmp) > PATH_MAX - 1) {
423  Rf_warning(_("zip path is too long"));
424  return FALSE;
425  }
426  strcpy(path, tmp);
427  p = Rf_strrchr(path, ':');
428  if(!p) {
429  Rf_warning(_("invalid description of 'unz' connection"));
430  return FALSE;
431  }
432  *p = '\0';
433  uf = unzOpen64(path);
434  if(!uf) {
435  Rf_warning(_("cannot open zip file '%s'"), path);
436  return FALSE;
437  }
438  if (unzLocateFile(uf, p+1, 1) != UNZ_OK) {
439  Rf_warning(_("cannot locate file '%s' in zip file '%s'"), p+1, path);
440  unzClose(uf);
441  return FALSE;
442  }
443  unzOpenCurrentFile(uf);
444  (static_cast<Runzconn>((con->connprivate)))->uf = uf;
445  con->isopen = TRUE;
446  con->canwrite = FALSE;
447  con->canread = TRUE;
448  if(strlen(con->mode) >= 2 && con->mode[1] == 'b') con->text = FALSE;
449  else con->text = TRUE;
450  /* set_iconv(); not yet */
451  con->save = -1000;
452  return TRUE;
453 }

856  ZPOS64_T uPosFound = 0;

869  return 0;

873  {
874  size_t uReadSize;
875  ZPOS64_T uReadPos ;
