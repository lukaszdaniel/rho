81 SEXP attribute_hidden complex_unary(ARITHOP_TYPE code, SEXP s1, SEXP call)
82 {
83  R_xlen_t i, n;
84  SEXP ans;
85 
86  switch(code) {
87  case PLUSOP:
88  return s1;
89  case MINUSOP:
90  ans = NO_REFERENCES(s1) ? s1 : Rf_duplicate(s1);
91  n = XLENGTH(s1);
92  for (i = 0; i < n; i++) {
93  Rcomplex x = COMPLEX(s1)[i];
94  COMPLEX(ans)[i].r = -x.r;
95  COMPLEX(ans)[i].i = -x.i;
96  }
97  return ans;
98  default:
99  Rf_errorcall(call, _("invalid complex unary operator"));
100  }
101  return R_NilValue; /* -Wall */
102 }

104 static R_INLINE double complex R_cpow_n(double complex X, int k)
105 {
106  if(k == 0) return (double complex) 1.;
107  else if(k == 1) return X;
108  else if(k < 0) return 1. / R_cpow_n(X, -k);
109  else {/* k > 0 */
110  double complex z = (double complex) 1.;;
111  while (k > 0) {
112  if (k & 1) z = z * X;
113  if (k == 1) break;
114  k >>= 1; /* efficient division by 2; now have k >= 1 */
115  X = X * X;
116  }
117  return z;
118  }
119 }

124 /* reason for this:
125  1) X^n (e.g. for n = +/- 2, 3) is unnecessarily inaccurate in glibc;
126  cut-off 65536 : guided from empirical speed measurements
127 
128  2) On Mingw (but not Mingw-w64) the system cpow is explicitly linked
129  against the (slow) MSVCRT pow, and gets (0+0i)^Y as 0+0i for all Y.
130 
131  3) PPC macOS crashed on powers of 0+0i (at least under Rosetta).
132  Really 0i^-1 should by Inf+NaNi, but getting that portably seems too hard.
133  (C1x's CMPLX will eventually be possible.)
134 */

136 static double complex mycpow (double complex X, double complex Y)
137 {
138  double complex Z;
139  double yr = creal(Y), yi = cimag(Y);
140  int k;
141  if (X == 0.0) {
142  if (yi == 0.0) Z = R_pow(0.0, yr); else Z = R_NaN + R_NaN*I;
143  } else if (yi == 0.0 && yr == (k = (int) yr) && abs(k) <= 65536)
144  Z = R_cpow_n(X, k);
145  else
146 #ifdef HAVE_CPOW
147  Z = cpow(X, Y);
148 #else
149  {
150  /* Used for FreeBSD and MingGW, hence mainly with gcc */
151  double rho, r, i, theta;
152  r = hypot(creal(X), cimag(X));
153  i = atan2(cimag(X), creal(X));
154  theta = i * yr;
155  if (yi == 0.0)
156  rho = pow(r, yr);
157  else {
158  /* rearrangement of cexp(X * clog(Y)) */
159  r = log(r);
160  theta += r * yi;
161  rho = exp(r * yr - i * yi);
162  }
163 #ifdef __GNUC__
164  __real__ Z = rho * cos(theta);
165  __imag__ Z = rho * sin(theta);
166 #else
167  Z = rho * cos(theta) + (rho * sin(theta)) * I;
168 #endif
169  }
170 #endif
171  return Z;
172 }

176 SEXP attribute_hidden complex_binary(ARITHOP_TYPE code, SEXP s1, SEXP s2)
177 {
178  R_xlen_t i, i1, i2, n, n1, n2;
179  SEXP ans;
180 
181  /* Note: "s1" and "s2" are protected in the calling code. */
182  n1 = XLENGTH(s1);
183  n2 = XLENGTH(s2);
184  /* S4-compatibility change: if n1 or n2 is 0, result is of length 0 */
185  if (n1 == 0 || n2 == 0) return(Rf_allocVector(CPLXSXP, 0));
186 
187  n = (n1 > n2) ? n1 : n2;
188  ans = R_allocOrReuseVector(s1, s2, CPLXSXP, n);
189  PROTECT(ans);
190 
191  switch (code) {
192  case PLUSOP:
193  MOD_ITERATE2_CHECK(NINTERRUPT, n, n1, n2, i, i1, i2, {
194  Rcomplex x1 = COMPLEX(s1)[i1];
195  Rcomplex x2 = COMPLEX(s2)[i2];
196  COMPLEX(ans)[i].r = x1.r + x2.r;
197  COMPLEX(ans)[i].i = x1.i + x2.i;
198  });
199  break;
200  case MINUSOP:
201  MOD_ITERATE2_CHECK(NINTERRUPT, n, n1, n2, i, i1, i2, {
202  Rcomplex x1 = COMPLEX(s1)[i1];
203  Rcomplex x2 = COMPLEX(s2)[i2];
204  COMPLEX(ans)[i].r = x1.r - x2.r;
205  COMPLEX(ans)[i].i = x1.i - x2.i;
206  });
207  break;
208  case TIMESOP:
209  MOD_ITERATE2_CHECK(NINTERRUPT, n, n1, n2, i, i1, i2, {
210  SET_C99_COMPLEX(COMPLEX(ans), i,
211  C99_COMPLEX2(s1, i1) * C99_COMPLEX2(s2, i2));
212  });
213  break;
214  case DIVOP:
215  MOD_ITERATE2_CHECK(NINTERRUPT, n, n1, n2, i, i1, i2, {
216  SET_C99_COMPLEX(COMPLEX(ans), i,
217  C99_COMPLEX2(s1, i1) / C99_COMPLEX2(s2, i2));
218  });
219  break;
220  case POWOP:
221  MOD_ITERATE2_CHECK(NINTERRUPT, n, n1, n2, i, i1, i2, {
222  SET_C99_COMPLEX(COMPLEX(ans), i,
223  mycpow(C99_COMPLEX2(s1, i1), C99_COMPLEX2(s2, i2)));
224  });
225  break;
226  default:
227  Rf_error(_("unimplemented complex operation"));
228  }
229  UNPROTECT(1);
230 
231  /* quick return if there are no attributes */
232  if (ATTRIB(s1) == R_NilValue && ATTRIB(s2) == R_NilValue)
233  return ans;
234 
235  /* Copy attributes from longer argument. */
236 
237  if (ans != s2 && n == n2 && ATTRIB(s2) != R_NilValue)
238  Rf_copyMostAttrib(s2, ans);
239  if (ans != s1 && n == n1 && ATTRIB(s1) != R_NilValue)
240  Rf_copyMostAttrib(s1, ans); /* Done 2nd so s1's attrs overwrite s2's */
241 
242  return ans;
243 }

245 SEXP attribute_hidden do_cmathfuns(SEXP call, SEXP op, SEXP args, SEXP env)
246 {
247  SEXP x, y = R_NilValue; /* -Wall*/
248  R_xlen_t i, n;
249 
250  if (Rf_DispatchGroup("Complex", call, op, args, env, &x))
251  return x;
252  x = CAR(args);
253  if (Rf_isComplex(x)) {
254  n = XLENGTH(x);
255  switch(PRIMVAL(op)) {
256  case 1: /* Re */
257  y = Rf_allocVector(REALSXP, n);
258  for(i = 0 ; i < n ; i++)
259  REAL(y)[i] = COMPLEX(x)[i].r;
260  break;
261  case 2: /* Im */
262  y = Rf_allocVector(REALSXP, n);
263  for(i = 0 ; i < n ; i++)
264  REAL(y)[i] = COMPLEX(x)[i].i;
265  break;
266  case 3: /* Mod */
267  case 6: /* abs */
268  y = Rf_allocVector(REALSXP, n);
269  for(i = 0 ; i < n ; i++)
270 #if HAVE_CABS
271  REAL(y)[i] = cabs(C99_COMPLEX2(x, i));
272 #else
273  REAL(y)[i] = hypot(COMPLEX(x)[i].r, COMPLEX(x)[i].i);
274 #endif
275  break;
276  case 4: /* Arg */
277  y = Rf_allocVector(REALSXP, n);
278  for(i = 0 ; i < n ; i++)
279 #if HAVE_CARG
280  REAL(y)[i] = carg(C99_COMPLEX2(x, i));
281 #else
282  REAL(y)[i] = atan2(COMPLEX(x)[i].i, COMPLEX(x)[i].r);
283 #endif
284  break;
285  case 5: /* Conj */
286  y = NO_REFERENCES(x) ? x : Rf_allocVector(CPLXSXP, n);
287  for(i = 0 ; i < n ; i++) {
288  COMPLEX(y)[i].r = COMPLEX(x)[i].r;
289  COMPLEX(y)[i].i = -COMPLEX(x)[i].i;
290  }
291  break;
292  }
293  }
294  else if(Rf_isNumeric(x)) { /* so no complex numbers involved */
295  n = XLENGTH(x);
296  if(Rf_isReal(x)) PROTECT(x);
297  else PROTECT(x = Rf_coerceVector(x, REALSXP));
298  y = NO_REFERENCES(x) ? x : Rf_allocVector(REALSXP, n);
299 
300  switch(PRIMVAL(op)) {
301  case 1: /* Re */
302  case 5: /* Conj */
303  for(i = 0 ; i < n ; i++)
304  REAL(y)[i] = REAL(x)[i];
305  break;
306  case 2: /* Im */
307  for(i = 0 ; i < n ; i++)
308  REAL(y)[i] = 0.0;
309  break;
310  case 4: /* Arg */
311  for(i = 0 ; i < n ; i++)
312  if(ISNAN(REAL(x)[i]))
313  REAL(y)[i] = REAL(x)[i];
314  else if (REAL(x)[i] >= 0)
315  REAL(y)[i] = 0;
316  else
317  REAL(y)[i] = M_PI;
318  break;
319  case 3: /* Mod */
320  case 6: /* abs */
321  for(i = 0 ; i < n ; i++)
322  REAL(y)[i] = fabs(REAL(x)[i]);
323  break;
324  }
325  UNPROTECT(1);
326  }
327  else Rf_errorcall(call, _("non-numeric argument to function"));
328 
329  if (x != y && ATTRIB(x) != R_NilValue) {
330  PROTECT(x);
331  PROTECT(y);
332  SHALLOW_DUPLICATE_ATTRIB(y, x);
333  UNPROTECT(2);
334  }
335  return y;
336 }

339 #define MAX_DIGITS 22
340 void attribute_hidden z_prec_r(Rcomplex *r, Rcomplex *x, double digits)
341 {
342  double m = 0.0, m1, m2;
343  int dig, mag;
344 
345  r->r = x->r; r->i = x->i;
346  m1 = fabs(x->r); m2 = fabs(x->i);
347  if(R_FINITE(m1)) m = m1;
348  if(R_FINITE(m2) && m2 > m) m = m2;
349  if (m == 0.0) return;
350  if (!R_FINITE(digits)) {
351  if(digits > 0) return; else {r->r = r->i = 0.0; return ;}
352  }
353  dig = (int)floor(digits+0.5);
354  if (dig > MAX_DIGITS) return; else if (dig < 1) dig = 1;
355  mag = (int)floor(log10(m));
356  dig = dig - mag - 1;
357  if (dig > 306) {
358  double pow10 = 1.0e4;
359  digits = (double)(dig - 4);
360  r->r = Rf_fround(pow10 * x->r, digits)/pow10;
361  r->i = Rf_fround(pow10 * x->i, digits)/pow10;
362  } else {
363  digits = (double)(dig);
364  r->r = Rf_fround(x->r, digits);
365  r->i = Rf_fround(x->i, digits);
366  }
367 }

423 #ifndef HAVE_CTAN
424 #define ctan R_ctan
425 static double complex ctan(double complex z)
426 {
427  /* A&S 4.3.57 */
428  double x2, y2, den, ri;
429  x2 = 2.0 * creal(z);
430  y2 = 2.0 * cimag(z);
431  den = cos(x2) + cosh(y2);
432  /* any threshold between -log(DBL_EPSILON) and log(DBL_XMAX) will do*/
433  if (ISNAN(y2) || fabs(y2) < 50.0) ri = sinh(y2)/den;
434  else ri = (y2 < 0 ? -1.0 : 1.0);
435  return sin(x2)/den + ri * I;
436 }
437 #endif

439 #ifndef HAVE_CASIN
440 #define casin R_casin
441 static double complex casin(double complex z)
442 {
443  /* A&S 4.4.37 */
444  double alpha, t1, t2, x = creal(z), y = cimag(z), ri;
445  t1 = 0.5 * hypot(x + 1, y);
446  t2 = 0.5 * hypot(x - 1, y);
447  alpha = t1 + t2;
448  ri = log(alpha + sqrt(alpha*alpha - 1));
449  /* This comes from
450  'z_asin() is continuous from below if x >= 1
451  and continuous from above if x <= -1.'
452  */
453  if(y < 0 || (y == 0 && x > 1)) ri *= -1;
454  return asin(t1 - t2) + ri*I;
455 }
456 #endif

468 static double complex catan(double complex z)
469 {
470  double x = creal(z), y = cimag(z), rr, ri;
471  rr = 0.5 * atan2(2 * x, (1 - x * x - y * y));
472  ri = 0.25 * log((x * x + (y + 1) * (y + 1)) /
473  (x * x + (y - 1) * (y - 1)));

494 static double complex z_tan(double complex z)
495 {
496  double y = cimag(z);
497  double complex r = ctan(z);
498  if(R_FINITE(y) && fabs(y) > 25.0) {
499  /* at this point the real part is nearly zero, and the
500  imaginary part is one: but some OSes get the imag as NaN */
501 #if __GNUC__
502  __imag__ r = y < 0 ? -1.0 : 1.0;
503 #else
504  r = creal(r) + (y < 0 ? -1.0 : 1.0) * I;
505 #endif
506  }
507  return r;
508 }

521 static double complex z_asin(double complex z)
522 {
523  if(cimag(z) == 0 && fabs(creal(z)) > 1) {
524  double alpha, t1, t2, x = creal(z), ri;
525  t1 = 0.5 * fabs(x + 1);
526  t2 = 0.5 * fabs(x - 1);
527  alpha = t1 + t2;
528  ri = log(alpha + sqrt(alpha*alpha - 1));
529  if(x > 1) ri *= -1;
530  return asin(t1 - t2) + ri*I;
531  }
532  return casin(z);
533 }

538  return cacos(z);

541 static double complex z_atan(double complex z)
542 {
543  if(creal(z) == 0 && fabs(cimag(z)) > 1) {
544  double y = cimag(z), rr, ri;
545  rr = (y > 0) ? M_PI_2 : -M_PI_2;
546  ri = 0.25 * log(((y + 1) * (y + 1))/((y - 1) * (y - 1)));
547  return rr + ri*I;
548  }
549  return catan(z);
550 }

584 SEXP attribute_hidden complex_math1(SEXP call, SEXP op, SEXP args, SEXP env)
585 {
586  SEXP x, y;
587  R_xlen_t n;
588  Rboolean naflag = FALSE;
589 
590  PROTECT(x = CAR(args));
591  n = XLENGTH(x);
592  PROTECT(y = Rf_allocVector(CPLXSXP, n));
593 
594  switch (PRIMVAL(op)) {
595  case 10003: naflag = cmath1(clog, COMPLEX(x), COMPLEX(y), n); break;
596  case 3: naflag = cmath1(csqrt, COMPLEX(x), COMPLEX(y), n); break;
597  case 10: naflag = cmath1(cexp, COMPLEX(x), COMPLEX(y), n); break;
598  case 20: naflag = cmath1(ccos, COMPLEX(x), COMPLEX(y), n); break;
599  case 21: naflag = cmath1(csin, COMPLEX(x), COMPLEX(y), n); break;
600  case 22: naflag = cmath1(z_tan, COMPLEX(x), COMPLEX(y), n); break;
601  case 23: naflag = cmath1(z_acos, COMPLEX(x), COMPLEX(y), n); break;
602  case 24: naflag = cmath1(z_asin, COMPLEX(x), COMPLEX(y), n); break;
603  case 25: naflag = cmath1(z_atan, COMPLEX(x), COMPLEX(y), n); break;
604  case 30: naflag = cmath1(ccosh, COMPLEX(x), COMPLEX(y), n); break;
605  case 31: naflag = cmath1(csinh, COMPLEX(x), COMPLEX(y), n); break;

890 static void R_cpolyroot(double *opr, double *opi, int *degree,
891  double *zeror, double *zeroi, Rboolean *fail)
892 {
893  static const double smalno = DBL_MIN;
894  static const double base = (double)FLT_RADIX;
895  static int d_n, i, i1, i2;
896  static double zi, zr, xx, yy;
897  static double bnd, xxx;
898  Rboolean conv;
899  int d1;
900  double *tmp;
901  static const double cosr =/* cos 94 */ -0.06975647374412529990;
902  static const double sinr =/* sin 94 */ 0.99756405025982424767;
903  xx = M_SQRT1_2;/* 1/sqrt(2) = 0.707.... */
904 
905  yy = -xx;
906  *fail = FALSE;
907 
908  nn = *degree;
909  d1 = nn - 1;
910 
911  /* algorithm fails if the leading coefficient is zero. */
912 
913  if (opr[0] == 0. && opi[0] == 0.) {
914  *fail = TRUE;
915  return;
916  }
917 
918  /* remove the zeros at the origin if any. */
919 
920  while (opr[nn] == 0. && opi[nn] == 0.) {
921  d_n = d1-nn+1;
922  zeror[d_n] = 0.;
923  zeroi[d_n] = 0.;
924  nn--;
925  }
926  nn++;
927  /*-- Now, global var. nn := #{coefficients} = (relevant degree)+1 */
928 
929  if (nn == 1) return;
930 
931  /* Use a single allocation as these as small */
932  const void *vmax = vmaxget();
933  tmp = (double *) R_alloc((size_t) (10*nn), sizeof(double));
934  pr = tmp; pi = tmp + nn; hr = tmp + 2*nn; hi = tmp + 3*nn;
935  qpr = tmp + 4*nn; qpi = tmp + 5*nn; qhr = tmp + 6*nn; qhi = tmp + 7*nn;
936  shr = tmp + 8*nn; shi = tmp + 9*nn;
937 
938  /* make a copy of the coefficients and shr[] = | p[] | */
939  for (i = 0; i < nn; i++) {
940  pr[i] = opr[i];
941  pi[i] = opi[i];
942  shr[i] = hypot(pr[i], pi[i]);
943  }
944 
945  /* scale the polynomial with factor 'bnd'. */
946  bnd = cpoly_scale(nn, shr, eta, infin, smalno, base);
947  if (bnd != 1.) {
948  for (i=0; i < nn; i++) {
949  pr[i] *= bnd;
950  pi[i] *= bnd;
951  }
952  }
953 
954  /* start the algorithm for one zero */
955 
956  while (nn > 2) {
957 
958  /* calculate bnd, a lower bound on the modulus of the zeros. */
959 
960  for (i=0 ; i < nn ; i++)
961  shr[i] = hypot(pr[i], pi[i]);
962  bnd = cpoly_cauchy(nn, shr, shi);
963 
964  /* outer loop to control 2 major passes */
965  /* with different sequences of shifts */
966 
967  for (i1 = 1; i1 <= 2; i1++) {
968 
969  /* first stage calculation, no shift */
970 
971  noshft(5);
972 
973  /* inner loop to select a shift */
974  for (i2 = 1; i2 <= 9; i2++) {
975 
976  /* shift is chosen with modulus bnd */
977  /* and amplitude rotated by 94 degrees */
978  /* from the previous shift */
979 
980  xxx= cosr * xx - sinr * yy;
981  yy = sinr * xx + cosr * yy;
982  xx = xxx;
983  sr = bnd * xx;
984  si = bnd * yy;
985 
986  /* second stage calculation, fixed shift */
987 
988  conv = fxshft(i2 * 10, &zr, &zi);
989  if (conv)
990  goto L10;
991  }
992  }
993 
994  /* the zerofinder has failed on two major passes */
995  /* return empty handed */
996 
997  *fail = TRUE;
998  vmaxset(vmax);
999  return;
1000 
1001  /* the second stage jumps directly to the third stage iteration.
1002  * if successful, the zero is stored and the polynomial deflated.
1003  */
1004  L10:
1005  d_n = d1+2 - nn;
1006  zeror[d_n] = zr;
1007  zeroi[d_n] = zi;
1008  --nn;
1009  for (i=0; i < nn ; i++) {
1010  pr[i] = qpr[i];
1011  pi[i] = qpi[i];
1012  }
1013  }/*while*/
1014 
1015  /* calculate the final zero and return */
1016  cdivid(-pr[1], -pi[1], pr[0], pi[0], &zeror[d1], &zeroi[d1]);
1017 
1018  vmaxset(vmax);
1019  return;
1020 }

1023 /* Computes the derivative polynomial as the initial
1024  * polynomial and computes l1 no-shift h polynomials. */

1026 static void noshft(int l1)
1027 {
1028  int i, j, jj, n = nn - 1, nm1 = n - 1;
1029 
1030  double t1, t2, xni;
1031 
1032  for (i=0; i < n; i++) {
1033  xni = (double)(nn - i - 1);
1034  hr[i] = xni * pr[i] / n;
1035  hi[i] = xni * pi[i] / n;
1036  }
1037 
1038  for (jj = 1; jj <= l1; jj++) {
1039 
1040  if (hypot(hr[n-1], hi[n-1]) <=
1041  eta * 10.0 * hypot(pr[n-1], pi[n-1])) {
1042  /* If the constant term is essentially zero, */
1043  /* shift h coefficients. */
1044 
1045  for (i = 1; i <= nm1; i++) {
1046  j = nn - i;
1047  hr[j-1] = hr[j-2];
1048  hi[j-1] = hi[j-2];
1049  }
1050  hr[0] = 0.;
1051  hi[0] = 0.;
1052  }
1053  else {
1054  cdivid(-pr[nn-1], -pi[nn-1], hr[n-1], hi[n-1], &tr, &ti);
1055  for (i = 1; i <= nm1; i++) {
1056  j = nn - i;
1057  t1 = hr[j-2];
1058  t2 = hi[j-2];
1059  hr[j-1] = tr * t1 - ti * t2 + pr[j-1];
1060  hi[j-1] = tr * t2 + ti * t1 + pi[j-1];
1061  }
1062  hr[0] = pr[0];
1063  hi[0] = pi[0];
1064  }

1069 /* Computes l2 fixed-shift h polynomials and tests for convergence.
1070  * initiates a variable-shift iteration and returns with the
1071  * approximate zero if successful.
1072  */
1073 static Rboolean fxshft(int l2, double *zr, double *zi)
1074 {
1075 /* l2 - limit of fixed shift steps
1076  * zr,zi - approximate zero if convergence (result TRUE)
1077  *
1078  * Return value indicates convergence of stage 3 iteration
1079  *
1080  * Uses global (sr,si), nn, pr[], pi[], .. (all args of polyev() !)
1081 */
1082 
1083  Rboolean pasd, bool_, test;
1084  static double svsi, svsr;
1085  static int i, j, n;
1086  static double oti, otr;
1087 
1088  n = nn - 1;
1089 
1090  /* evaluate p at s. */
1091 
1092  polyev(nn, sr, si, pr, pi, qpr, qpi, &pvr, &pvi);
1093 
1094  test = TRUE;
1095  pasd = FALSE;
1096 
1097  /* calculate first t = -p(s)/h(s). */
1098 
1099  calct(&bool_);
1100 
1101  /* main loop for one second stage step. */
1102 
1103  for (j=1; j<=l2; j++) {
1104 
1105  otr = tr;
1106  oti = ti;
1107 
1108  /* compute next h polynomial and new t. */
1109 
1110  nexth(bool_);
1111  calct(&bool_);
1112  *zr = sr + tr;
1113  *zi = si + ti;
1114 
1115  /* test for convergence unless stage 3 has */
1116  /* failed once or this is the last h polynomial. */
1117 
1118  if (!bool_ && test && j != l2) {
1119  if (hypot(tr - otr, ti - oti) >= hypot(*zr, *zi) * 0.5) {
1120  pasd = FALSE;
1121  }
1122  else if (! pasd) {
1123  pasd = TRUE;
1124  }
1125  else {
1126 
1127  /* the weak convergence test has been */
1128  /* passed twice, start the third stage */
1129  /* iteration, after saving the current */
1130  /* h polynomial and shift. */
1131 
1132  for (i = 0; i < n; i++) {
1133  shr[i] = hr[i];
1134  shi[i] = hi[i];
1135  }
1136  svsr = sr;
1137  svsi = si;
1138  if (vrshft(10, zr, zi)) {
1139  return TRUE;
1140  }
1141 
1142  /* the iteration failed to converge. */
1143  /* turn off testing and restore */
1144  /* h, s, pv and t. */
1145 
1146  test = FALSE;
1147  for (i=1 ; i<=n ; i++) {
1148  hr[i-1] = shr[i-1];
1149  hi[i-1] = shi[i-1];
1150  }
1151  sr = svsr;
1152  si = svsi;
1153  polyev(nn, sr, si, pr, pi, qpr, qpi, &pvr, &pvi);
1154  calct(&bool_);
1155  }
1156  }
1157  }
1158 
1159  /* attempt an iteration with final h polynomial */
1160  /* from second stage. */
1161 
1162  return(vrshft(10, zr, zi));
1163 }

1166 /* carries out the third stage iteration.
1167  */
1168 static Rboolean vrshft(int l3, double *zr, double *zi)
1169 {
1170 /* l3 - limit of steps in stage 3.
1171  * zr,zi - on entry contains the initial iterate;
1172  * if the iteration converges it contains
1173  * the final iterate on exit.
1174  * Returns TRUE if iteration converges
1175  *
1176  * Assign and uses GLOBAL sr, si
1177 */
1178  Rboolean bool_, b;
1179  static int i, j;
1180  static double r1, r2, mp, ms, tp, relstp;
1181  static double omp;
1182 
1183  b = FALSE;
1184  sr = *zr;
1185  si = *zi;
1186 
1187  /* main loop for stage three */
1188 
1189  for (i = 1; i <= l3; i++) {
1190 
1191  /* evaluate p at s and test for convergence. */
1192  polyev(nn, sr, si, pr, pi, qpr, qpi, &pvr, &pvi);
1193 
1194  mp = hypot(pvr, pvi);
1195  ms = hypot(sr, si);
1196  if (mp <= 20. * errev(nn, qpr, qpi, ms, mp, /*are=*/eta, mre)) {
1197  goto L_conv;
1198  }
1199 
1200  /* polynomial value is smaller in value than */
1201  /* a bound on the error in evaluating p, */
1202  /* terminate the iteration. */
1203 
1204  if (i != 1) {
1205 
1206  if (!b && mp >= omp && relstp < .05) {
1207 
1208  /* iteration has stalled. probably a */
1209  /* cluster of zeros. do 5 fixed shift */
1210  /* steps into the cluster to force */
1211  /* one zero to dominate. */
1212 
1213  tp = relstp;
1214  b = TRUE;
1215  if (relstp < eta)
1216  tp = eta;
1217  r1 = sqrt(tp);
1218  r2 = sr * (r1 + 1.) - si * r1;
1219  si = sr * r1 + si * (r1 + 1.);
1220  sr = r2;
1221  polyev(nn, sr, si, pr, pi, qpr, qpi, &pvr, &pvi);
1222  for (j = 1; j <= 5; ++j) {
1223  calct(&bool_);
1224  nexth(bool_);
1225  }
1226  omp = infin;
1227  goto L10;
1228  }
1229  else {
1230 
1231  /* exit if polynomial value */
1232  /* increases significantly. */
1233 
1234  if (mp * .1 > omp)
1235  return FALSE;
1236  }
1237  }
1238  omp = mp;
1239 
1240  /* calculate next iterate. */
1241 
1242  L10:
1243  calct(&bool_);
1244  nexth(bool_);
1245  calct(&bool_);
1246  if (!bool_) {
1247  relstp = hypot(tr, ti) / hypot(sr, si);
1248  sr += tr;

1363 {
1364  /* Computes a lower bound on the moduli of the zeros of a polynomial
1365  * pot[1:nn] is the modulus of the coefficients.
1366  */
1367  double f, x, delf, dx, xm;
1368  int i, n1 = n - 1;
1369 
1370  pot[n1] = -pot[n1];
1371 
1372  /* compute upper estimate of bound. */
1373 
1374  x = exp((log(-pot[n1]) - log(pot[0])) / (double) n1);
1375 
1376  /* if newton step at the origin is better, use it. */
1377 
1378  if (pot[n1-1] != 0.) {
1379  xm = -pot[n1] / pot[n1-1];
1380  if (xm < x)
1381  x = xm;
1382  }
1383 
1384  /* chop the interval (0,x) unitl f le 0. */
1385 
1386  for(;;) {
1387  xm = x * 0.1;
1388  f = pot[0];
1389  for (i = 1; i < n; i++)
1390  f = f * xm + pot[i];
1391  if (f <= 0.0) {
1392  break;
1393  }
1394  x = xm;
1395  }
1396 
1397  dx = x;
1398 
1399  /* do Newton iteration until x converges to two decimal places. */
1400 
1401  while (fabs(dx / x) > 0.005) {
1402  q[0] = pot[0];
1403  for(i = 1; i < n; i++)
1404  q[i] = q[i-1] * x + pot[i];
1405  f = q[n1];
1406  delf = q[0];
1407  for(i = 1; i < n1; i++)
1408  delf = delf * x + q[i];
1409  dx = f / delf;
1410  x -= dx;
1411  }
