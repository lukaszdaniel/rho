50 struct NameWalkData {
51  SEXP ans;
52  int UniqueNames;
53  int IncludeFunctions;
54  int StoreValues;
55  int ItemCounts;
56  int MaxCount;
57 };

59 static void namewalk(SEXP s, NameWalkData *d)
60 {
61  SEXP name;
62 
63  switch(TYPEOF(s)) {
64  case SYMSXP:
65  name = PRINTNAME(s);
66  /* skip blank symbols */
67  if(R_CHAR(name)[0] == '\0') goto ignore;
68  if(d->ItemCounts < d->MaxCount) {
69  if(d->StoreValues) {
70  if(d->UniqueNames) {
71  for(int j = 0 ; j < d->ItemCounts ; j++) {
72  if(STRING_ELT(d->ans, j) == name)
73  goto ignore;
74  }
75  }
76  SET_STRING_ELT(d->ans, d->ItemCounts, name);
77  }
78  d->ItemCounts++;
79  }
80  ignore:
81  break;
82  case LANGSXP:
83  if(!d->IncludeFunctions) s = CDR(s);
84  while(s != nullptr) {
85  namewalk(CAR(s), d);
86  s = CDR(s);
87  }
88  break;
89  case EXPRSXP:
90  for(R_xlen_t i = 0 ; i < XLENGTH(s) ; i++)
91  namewalk(XVECTOR_ELT(s, i), d);
92  break;
93  default:
94  /* it seems the intention is to do nothing here! */
95  break;
96  }
97 }

101 HIDDEN SEXP do_allnames(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* expr_, rho::RObject* functions_, rho::RObject* max_names_, rho::RObject* unique_)
102 {
103  SEXP expr;
104  int i, savecount;
105  NameWalkData data = {nullptr, 0, 0, 0, 0, 0};
106 
107  expr = expr_;
108 
109  data.IncludeFunctions = Rf_asLogical(functions_);
110  if(data.IncludeFunctions == R_NaLog)
111  data.IncludeFunctions = 0;
112 
113  data.MaxCount = Rf_asInteger(max_names_);
114  if(data.MaxCount == -1) data.MaxCount = INT_MAX;
115  if(data.MaxCount < 0 || data.MaxCount == R_NaInt)
116  data.MaxCount = 0;
117 
118  data.UniqueNames = Rf_asLogical(unique_);
119  if(data.UniqueNames == R_NaLog)
120  data.UniqueNames = 1;
121 
122  namewalk(expr, &data);
123  savecount = data.ItemCounts;
124 
125  data.ans = Rf_allocVector(STRSXP, data.ItemCounts);
126 
127  data.StoreValues = 1;
128  data.ItemCounts = 0;
129  namewalk(expr, &data);
130 
131  if(data.ItemCounts != savecount) {
132  PROTECT(expr = data.ans);
133  data.ans = Rf_allocVector(STRSXP, data.ItemCounts);
134  for(i = 0 ; i < data.ItemCounts ; i++)
135  SET_STRING_ELT(data.ans, i, STRING_ELT(expr, i));
136  UNPROTECT(1);
137  }
138 
139  return data.ans;
140 }
