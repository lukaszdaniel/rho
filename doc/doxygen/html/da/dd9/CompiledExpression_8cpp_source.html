53 CompiledExpression::CompiledExpression(const Closure* closure)
54 {
55  llvm::InitializeNativeTarget();
56  llvm::InitializeNativeTargetAsmPrinter();
57  EnsureGlobalsInitialized();
58 
59  const RObject* body = closure->body();
60 
61  // Create a module to compile the code in. MCJIT requires that each
62  // separate invocation of the JIT compiler uses its own module.
63  static llvm::LLVMContext MyGlobalContext;
64  llvm::LLVMContext& context = MyGlobalContext;
65  std::unique_ptr<Module> module = Runtime::createModule(context);
66 
67  // Create a function with signature RObject* (*f)(Environment* environment)
68  llvm::Function* function = llvm::Function::Create(
69  llvm::TypeBuilder<RObject*(Environment*), false>::get(context),
70  llvm::Function::ExternalLinkage,
71  "anonymous_function", // TODO: give it a useful name
72  module.get());
73 
74  Value* environment = &*(function->args().begin());
75  environment->setName("environment");
76 
77  // Setup the compiler and generate code.
78  std::unique_ptr<MCJITMemoryManager> memory_manager(
79  new MCJITMemoryManager(module.get()));
80  CompilerContext compiler_context(closure, environment, function,
81  memory_manager.get());
82  Compiler compiler(&compiler_context);
83 #if (LLVM_VERSION > 306)
84  function->setPersonalityFn(
85  Runtime::getExceptionPersonalityFunction(&compiler));
86 #endif
87  Value* return_value = compiler.emitEval(body);
88 
89  if (!llvm::isa<llvm::UndefValue>(return_value)) {
90  if (!return_value->hasName())
91  return_value->setName("return_value");
92  compiler.CreateRet(return_value);
93  }
94 
95  // Empty dead-code blocks are ill-formed. Add terminators.
96  for (llvm::BasicBlock& block : *function) {
97  if (block.empty()) {
98  compiler.SetInsertPoint(&block);
99  compiler.CreateUnreachable();
100  }
101  }
102 
103  // function->dump(); // So we can see what's going on while developing.
104  llvm::verifyFunction(*function);
105 
106  // Perform some basic intra-procedural optimization.
107  BasicFunctionPass().runOnFunction(*function);
108  llvm::verifyFunction(*function);
109 
110  // The IR is now complete. Compile to native code.
111  llvm::TargetOptions options;
112  // TODO(kmillar): set options dynamically.
113  options.EnableFastISel = true;
114 
115  m_engine.reset(
116 #if (LLVM_VERSION < 306)
117  llvm::EngineBuilder(module.release())
118  .setMCJITMemoryManager(memory_manager.release())
119  .setUseMCJIT(true)
120 #else
121  llvm::EngineBuilder(std::move(module))
122  .setMCJITMemoryManager(std::move(memory_manager))
123 #endif
124  .setOptLevel(llvm::CodeGenOpt::None)
125  .setTargetOptions(options)
126  .create());
127  assert(m_engine);
128 
129  m_engine->finalizeObject();
130  auto ptr = m_engine->getFunctionAddress(function->getName());
131  assert(ptr && "JIT compilation failed");
132 
133  m_function = reinterpret_cast<CompiledExpressionPointer>(ptr);
134  m_frame_descriptor = compiler_context.m_frame_descriptor;
135 }
