56 SEXP do_internal(SEXP call, SEXP op, SEXP args, SEXP env)
57 {
58  PairList* argspl = SEXP_downcast<PairList*>(args);
59  Environment* envir = SEXP_downcast<Environment*>(env);
60  Expression* innercall = dynamic_cast<Expression*>(argspl->car());
61  if (!innercall)
62  Rf_errorcall(call, _("invalid .Internal() argument"));
63  Symbol* funsym = dynamic_cast<Symbol*>(innercall->car());
64  if (!funsym)
65  Rf_errorcall(call, _("invalid .Internal() argument"));
66  BuiltInFunction* func = BuiltInFunction::obtainInternal(funsym);
67  if (!func)
68  Rf_errorcall(call, _("there is no .Internal function \"%s\""),
69  funsym->name()->c_str());
