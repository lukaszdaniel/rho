52 /*
53  .Internal(paste (args, sep, collapse))
54  .Internal(paste0(args, collapse))
55 
56  * do_paste uses two passes to paste the arguments (in CAR(args)) together.
57  * The first pass calculates the width of the paste buffer,
58  * then it is alloc-ed and the second pass stuffs the information in.
59  */

61 /* Note that NA_STRING is not handled separately here. This is
62  deliberate -- see ?paste -- and implicitly coerces it to "NA"
63 */
64 SEXP attribute_hidden do_paste(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::Environment* env, rho::RObject* const* args, int num_args, const rho::PairList* tags)
65 {
66  SEXP ans, collapse, sep, x;
67  int sepw, u_sepw;
68  R_xlen_t i, j, k, maxlen, nx, pwidth;
69  cetype_t ienc;
70  const char *s, *cbuf, *csep=nullptr, *u_csep=nullptr;
71  char *buf;
72  bool allKnown, anyKnown, use_UTF8, use_Bytes,
73  sepASCII = TRUE, sepUTF8 = FALSE, sepBytes = FALSE, sepKnown = FALSE,
74  use_sep = (op->variant() == 0);
75  const void *vmax;
76 
77  /* We use formatting and so we must initialize printing. */
78 
79  Rf_PrintDefaults();
80 
81  /* Check the arguments */
82 
83  x = args[0];
84  if (!Rf_isVectorList(x))
85  Rf_error(_("invalid first argument"));
86  nx = Rf_xlength(x);
87 
88  if(use_sep) { /* paste(..., sep, .) */
89  sep = args[1];
90  if (!Rf_isString(sep) || LENGTH(sep) <= 0 || STRING_ELT(sep, 0) == NA_STRING)
91  Rf_error(_("invalid separator"));
92  sep = STRING_ELT(sep, 0);
93  csep = Rf_translateChar(sep);
94  u_sepw = sepw = int(strlen(csep)); // will be short
95  sepASCII = Rf_strIsASCII(csep);
96  sepKnown = ENC_KNOWN(sep) > 0;
97  sepUTF8 = IS_UTF8(sep);
98  sepBytes = IS_BYTES(sep);
99  collapse = args[2];
100  } else { /* paste0(..., .) */
101  u_sepw = sepw = 0; sep = R_NilValue;/* -Wall */
102  collapse = args[1];
103  }
104  if (!Rf_isNull(collapse))
105  if(!Rf_isString(collapse) || LENGTH(collapse) <= 0 ||
106  STRING_ELT(collapse, 0) == NA_STRING)
107  Rf_error(_("invalid '%s' argument"), "collapse");
108  if(nx == 0)
109  return (!Rf_isNull(collapse)) ? Rf_mkString("") : Rf_allocVector(STRSXP, 0);
110 
111 
112  /* Maximum argument length, coerce if needed */
113 
114  maxlen = 0;
115  for (j = 0; j < nx; j++) {
116  if (!Rf_isString(VECTOR_ELT(x, j))) {
117  /* formerly in R code: moved to C for speed */
118  SEXP call, xj = VECTOR_ELT(x, j);
119  if(OBJECT(xj)) { /* method dispatch */
120  PROTECT(call = Rf_lang2(R_AsCharacterSymbol, xj));
121  SET_VECTOR_ELT(x, j, Rf_eval(call, env));
122  UNPROTECT(1);
123  } else if (Rf_isSymbol(xj))
124  SET_VECTOR_ELT(x, j, Rf_ScalarString(PRINTNAME(xj)));
125  else
126  SET_VECTOR_ELT(x, j, Rf_coerceVector(xj, STRSXP));
127 
128  if (!Rf_isString(VECTOR_ELT(x, j)))
129  Rf_error(_("non-string argument to internal 'paste'"));
130  }
131  if(Rf_xlength(VECTOR_ELT(x, j)) > maxlen)
132  maxlen = Rf_xlength(VECTOR_ELT(x, j));
133  }
134  if(maxlen == 0)
135  return (!Rf_isNull(collapse)) ? Rf_mkString("") : Rf_allocVector(STRSXP, 0);
136 
137  PROTECT(ans = Rf_allocVector(STRSXP, maxlen));
138 
139  for (i = 0; i < maxlen; i++) {
140  /* Strategy for marking the encoding: if all inputs (including
141  * the separator) are ASCII, so is the output and we don't
142  * need to mark. Otherwise if all non-ASCII inputs are of
143  * declared encoding, we should mark.
144  * Need to be careful only to include separator if it is used.
145  */
146  anyKnown = FALSE; allKnown = TRUE; use_UTF8 = FALSE; use_Bytes = FALSE;
147  if(nx > 1) {
148  allKnown = sepKnown || sepASCII;
149  anyKnown = sepKnown;
150  use_UTF8 = sepUTF8;
151  use_Bytes = sepBytes;
152  }
153 
154  pwidth = 0;
155  for (j = 0; j < nx; j++) {
156  k = Rf_xlength(VECTOR_ELT(x, j));
157  if (k > 0) {
158  SEXP cs = STRING_ELT(VECTOR_ELT(x, j), i % k);
159  if(IS_UTF8(cs)) use_UTF8 = TRUE;
160  if(IS_BYTES(cs)) use_Bytes = TRUE;
161  }
162  }
163  if (use_Bytes) use_UTF8 = FALSE;
164  vmax = vmaxget();
165  for (j = 0; j < nx; j++) {
166  k = Rf_xlength(VECTOR_ELT(x, j));
167  if (k > 0) {
168  if(use_Bytes)
169  pwidth += strlen(R_CHAR(STRING_ELT(VECTOR_ELT(x, j), i % k)));
170  else if(use_UTF8)
171  pwidth += strlen(Rf_translateCharUTF8(STRING_ELT(VECTOR_ELT(x, j), i % k)));
172  else
173  pwidth += strlen(Rf_translateChar(STRING_ELT(VECTOR_ELT(x, j), i % k)));
174  vmaxset(vmax);
175  }
176  }
177  if(use_sep) {
178  if (use_UTF8 && !u_csep) {
179  u_csep = Rf_translateCharUTF8(sep);
180  u_sepw = int(strlen(u_csep)); // will be short
181  }
182  pwidth += (nx - 1) * (use_UTF8 ? u_sepw : sepw);
183  }
184  if (pwidth > INT_MAX)
185  Rf_error(_("result would exceed 2^31-1 bytes"));
186  cbuf = buf = static_cast<char*>(R_AllocStringBuffer(pwidth, &cbuff));
187  vmax = vmaxget();
188  for (j = 0; j < nx; j++) {
189  k = Rf_xlength(VECTOR_ELT(x, j));
190  if (k > 0) {
191  SEXP cs = STRING_ELT(VECTOR_ELT(x, j), i % k);
192  if (use_UTF8) {
193  s = Rf_translateCharUTF8(cs);
194  strcpy(buf, s);
195  buf += strlen(s);
196  } else {
197  s = use_Bytes ? R_CHAR(cs) : Rf_translateChar(cs);
198  strcpy(buf, s);
199  buf += strlen(s);
200  allKnown = allKnown && (Rf_strIsASCII(s) || (ENC_KNOWN(cs)> 0));
201  anyKnown = anyKnown || (ENC_KNOWN(cs)> 0);
202  }
203  }
204  if (sepw != 0 && j != nx - 1) {
205  if (use_UTF8) {
206  strcpy(buf, u_csep);
207  buf += u_sepw;
208  } else {
209  strcpy(buf, csep);
210  buf += sepw;
211  }
212  }
213  vmax = vmaxget();
214  }
215  ienc = CE_NATIVE;
216  if(use_UTF8) ienc = CE_UTF8;
217  else if(use_Bytes) ienc = CE_BYTES;
218  else if(anyKnown && allKnown) {
219  if(known_to_be_latin1) ienc = CE_LATIN1;
220  if(known_to_be_utf8) ienc = CE_UTF8;
221  }
222  SET_STRING_ELT(ans, i, Rf_mkCharCE(cbuf, ienc));
223  }
224 
225  /* Now collapse, if required. */
226 
227  if(collapse != R_NilValue && (nx = XLENGTH(ans)) > 0) {
228  sep = STRING_ELT(collapse, 0);
229  use_UTF8 = IS_UTF8(sep);
230  use_Bytes = IS_BYTES(sep);
231  for (i = 0; i < nx; i++) {
232  if(IS_UTF8(STRING_ELT(ans, i))) use_UTF8 = TRUE;
233  if(IS_BYTES(STRING_ELT(ans, i))) use_Bytes = TRUE;
234  }
235  if(use_Bytes) {
236  csep = R_CHAR(sep);
237  use_UTF8 = FALSE;
238  } else if(use_UTF8)
239  csep = Rf_translateCharUTF8(sep);
240  else
241  csep = Rf_translateChar(sep);
242  sepw = int(strlen(csep));
243  anyKnown = ENC_KNOWN(sep) > 0;
244  allKnown = anyKnown || Rf_strIsASCII(csep);
245  pwidth = 0;
246  vmax = vmaxget();
247  for (i = 0; i < nx; i++)
248  if(use_UTF8) {
249  pwidth += strlen(Rf_translateCharUTF8(STRING_ELT(ans, i)));
250  vmaxset(vmax);
251  } else /* already translated */
252  pwidth += strlen(R_CHAR(STRING_ELT(ans, i)));
253  pwidth += (nx - 1) * sepw;
254  if (pwidth > INT_MAX)
255  Rf_error(_("result would exceed 2^31-1 bytes"));
256  cbuf = buf = static_cast<char*>(R_AllocStringBuffer(pwidth, &cbuff));
257  vmax = vmaxget();
258  for (i = 0; i < nx; i++) {
259  if(i > 0) {
260  strcpy(buf, csep);
261  buf += sepw;
262  }
263  if(use_UTF8)
264  s = Rf_translateCharUTF8(STRING_ELT(ans, i));
265  else /* already translated */
266  s = R_CHAR(STRING_ELT(ans, i));
267  strcpy(buf, s);
268  while (*buf)
269  buf++;
270  allKnown = allKnown &&
271  (Rf_strIsASCII(s) || (ENC_KNOWN(STRING_ELT(ans, i)) > 0));
272  anyKnown = anyKnown || (ENC_KNOWN(STRING_ELT(ans, i)) > 0);
273  if(use_UTF8) vmaxset(vmax);
274  }
275  UNPROTECT(1);
276  ienc = CE_NATIVE;
277  if(use_UTF8) ienc = CE_UTF8;
278  else if(use_Bytes) ienc = CE_BYTES;
279  else if(anyKnown && allKnown) {
280  if(known_to_be_latin1) ienc = CE_LATIN1;
281  if(known_to_be_utf8) ienc = CE_UTF8;
282  }
283  PROTECT(ans = Rf_ScalarString(Rf_mkCharCE(cbuf, ienc)));
284  }
285  R_FreeStringBufferL(&cbuff);
286  UNPROTECT(1);
287  return ans;
288 }

290 SEXP attribute_hidden do_filepath(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* sep_)
291 {
292  SEXP ans, sep, x;
293  int i, j, k, ln, maxlen, nx, nzero, pwidth, sepw;
294  const char *s, *csep, *cbuf;
295  char *buf;
296 
297  /* Check the arguments */
298 
299  x = x_;
300  if (!Rf_isVectorList(x))
301  Rf_error(_("invalid first argument"));
302  nx = Rf_length(x);
303  if(nx == 0) return Rf_allocVector(STRSXP, 0);
304 
305 
306  sep = sep_;
307  if (!Rf_isString(sep) || LENGTH(sep) <= 0 || STRING_ELT(sep, 0) == NA_STRING)
308  Rf_error(_("invalid separator"));
309  sep = STRING_ELT(sep, 0);
310  csep = R_CHAR(sep);
311  sepw = int(strlen(csep)); /* hopefully 1 */
312 
313  /* Any zero-length argument gives zero-length result */
314  maxlen = 0; nzero = 0;
315  for (j = 0; j < nx; j++) {
316  if (!Rf_isString(VECTOR_ELT(x, j))) {
317  /* formerly in R code: moved to C for speed */
318  SEXP call, xj = VECTOR_ELT(x, j);
319  if(OBJECT(xj)) { /* method dispatch */
320  PROTECT(call = Rf_lang2(R_AsCharacterSymbol, xj));
321  SET_VECTOR_ELT(x, j, Rf_eval(call, R_BaseEnv));
322  UNPROTECT(1);
323  } else if (Rf_isSymbol(xj))
324  SET_VECTOR_ELT(x, j, Rf_ScalarString(PRINTNAME(xj)));
325  else
326  SET_VECTOR_ELT(x, j, Rf_coerceVector(xj, STRSXP));
327 
328  if (!Rf_isString(VECTOR_ELT(x, j)))
329  Rf_error(_("non-string argument to Internal paste"));
330  }
331  ln = Rf_length(VECTOR_ELT(x, j));
332  if(ln > maxlen) maxlen = ln;
333  if(ln == 0) {nzero++; break;}
334  }
335  if(nzero || maxlen == 0) return Rf_allocVector(STRSXP, 0);
336 
337  PROTECT(ans = Rf_allocVector(STRSXP, maxlen));
338 
339  for (i = 0; i < maxlen; i++) {
340  pwidth = 0;
341  for (j = 0; j < nx; j++) {
342  k = Rf_length(VECTOR_ELT(x, j));
343  pwidth += int(strlen(Rf_translateChar(STRING_ELT(VECTOR_ELT(x, j), i % k))));
344  }
345  pwidth += (nx - 1) * sepw;
346  cbuf = buf = static_cast<char*>(R_AllocStringBuffer(pwidth, &cbuff));
347  for (j = 0; j < nx; j++) {
348  k = Rf_length(VECTOR_ELT(x, j));
349  if (k > 0) {
350  s = Rf_translateChar(STRING_ELT(VECTOR_ELT(x, j), i % k));
351  strcpy(buf, s);
352  buf += strlen(s);
353  }
354  if (j != nx - 1 && sepw != 0) {
355  strcpy(buf, csep);
356  buf += sepw;
357  }
358  }
359 #ifdef Win32
360  // Trailing seps are invalid for file paths except for / and d:/
361  if(streql(csep, "/") || streql(csep, "\\")) {
362  if(buf > cbuf) {
363  buf--;
364  if(*buf == csep[0] && buf > cbuf &&
365  (buf != cbuf+2 || cbuf[1] != ':')) *buf = '\0';
366  }
367  }
368 #endif
369  SET_STRING_ELT(ans, i, Rf_mkChar(cbuf));
370  }
371  R_FreeStringBufferL(&cbuff);
372  UNPROTECT(1);
373  return ans;
374 }

376 /* format.default(x, trim, digits, nsmall, width, justify, na.encode,
377  scientific, decimal.mark) */
378 SEXP attribute_hidden do_format(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* trim_, rho::RObject* digits_, rho::RObject* nsmall_, rho::RObject* width_, rho::RObject* justify_, rho::RObject* na_encode_, rho::RObject* scientific_, rho::RObject* decimal_mark_)
379 {
380  SEXP l, x, y, swd;
381  int il, digits, trim = 0, nsmall = 0, wd = 0, adj = -1, na, sci = 0;
382  int w, d, e;
383  int wi, di, ei, scikeep;
384  const char *strp;
385  R_xlen_t i, n;
386 
387  Rf_PrintDefaults();
388  scikeep = R_print.scipen;
389 
390  if (Rf_isEnvironment(x = x_)) {
391  return Rf_mkString(EncodeEnvironment(x));
392  }
393  else if (!Rf_isVector(x))
394  Rf_error(_("first argument must be atomic"));
395 
396  trim = Rf_asLogical(trim_);
397  if (trim == NA_INTEGER)
398  Rf_error(_("invalid '%s' argument"), "trim");
399 
400  if (!Rf_isNull(digits_)) {
401  digits = Rf_asInteger(digits_);
402  if (digits == NA_INTEGER || digits < R_MIN_DIGITS_OPT
403  || digits > R_MAX_DIGITS_OPT)
404  Rf_error(_("invalid '%s' argument"), "digits");
405  R_print.digits = digits;
406  }
407 
408  nsmall = Rf_asInteger(nsmall_);
409  if (nsmall == NA_INTEGER || nsmall < 0 || nsmall > 20)
410  Rf_error(_("invalid '%s' argument"), "nsmall");
411 
412  if (Rf_isNull(swd = width_)) wd = 0; else wd = Rf_asInteger(swd);
413  if(wd == NA_INTEGER)
414  Rf_error(_("invalid '%s' argument"), "width");
415 
416  adj = Rf_asInteger(justify_);
417  if(adj == NA_INTEGER || adj < 0 || adj > 3)
418  Rf_error(_("invalid '%s' argument"), "justify");
419 
420  na = Rf_asLogical(na_encode_);
421  if(na == NA_LOGICAL)
422  Rf_error(_("invalid '%s' argument"), "na.encode");
423 
424  if(LENGTH(scientific_) != 1)
425  Rf_error(_("invalid '%s' argument"), "scientific");
426  if(Rf_isLogical(scientific_)) {
427  int tmp = LOGICAL(scientific_)[0];
428  if(tmp == NA_LOGICAL) sci = NA_INTEGER;
429  else sci = tmp > 0 ?-100 : 100;
430  } else if (Rf_isNumeric(scientific_)) {
431  sci = Rf_asInteger(scientific_);
432  } else
433  Rf_error(_("invalid '%s' argument"), "scientific");
434  if(sci != NA_INTEGER) R_print.scipen = sci;
435 
436  // copy/paste from "OutDec" part of ./options.c
437  if (TYPEOF(decimal_mark_) != STRSXP || LENGTH(decimal_mark_) != 1)
438  Rf_error(_("invalid '%s' argument"), "decimal.mark");
439  const char *my_OutDec;
440  if(STRING_ELT(decimal_mark_, 0) == NA_STRING)
441  my_OutDec = OutDec; // default
442  else {
443  static char sdec[11];
444 // not warning here by default for now
445 #ifdef _WARN_decimal_mark_non_1
446  if(R_nchar(STRING_ELT(decimal_mark_, 0), Chars,
447  /* allowNA = */ FALSE, /* keepNA = */ FALSE,
448  "decimal.mark") != 1) // will become an error
449  Rf_warning(_("'decimal.mark' must be a string of one character"));
450 #endif
451  strncpy(sdec, R_CHAR(STRING_ELT(decimal_mark_, 0)), 10);
452  sdec[10] = '\0';
453  my_OutDec = sdec;
454  }
455 
456  if ((n = XLENGTH(x)) <= 0) {
457  PROTECT(y = Rf_allocVector(STRSXP, 0));
458  } else {
459  switch (TYPEOF(x)) {
460 
461  case LGLSXP:
462  PROTECT(y = Rf_allocVector(STRSXP, n));
463  if (trim) w = 0; else formatLogical(LOGICAL(x), n, &w);
464  w = max(w, wd);
465  for (i = 0; i < n; i++) {
466  strp = EncodeLogical(LOGICAL(x)[i], w);
467  SET_STRING_ELT(y, i, Rf_mkChar(strp));
468  }
469  break;
470 
471  case INTSXP:
472  PROTECT(y = Rf_allocVector(STRSXP, n));
473  if (trim) w = 0;
474  else formatInteger(INTEGER(x), n, &w);
475  w = max(w, wd);
476  for (i = 0; i < n; i++) {
477  strp = EncodeInteger(INTEGER(x)[i], w);
478  SET_STRING_ELT(y, i, Rf_mkChar(strp));
479  }
480  break;
481 
482  case REALSXP:
483  formatReal(REAL(x), n, &w, &d, &e, nsmall);
484  if (trim) w = 0;
485  w = max(w, wd);
486  PROTECT(y = Rf_allocVector(STRSXP, n));
487  for (i = 0; i < n; i++) {
488  strp = EncodeReal0(REAL(x)[i], w, d, e, my_OutDec);
489  SET_STRING_ELT(y, i, Rf_mkChar(strp));
490  }
491  break;
492 
493  case CPLXSXP:
494  formatComplex(COMPLEX(x), n, &w, &d, &e, &wi, &di, &ei, nsmall);
495  if (trim) wi = w = 0;
496  w = max(w, wd); wi = max(wi, wd);
497  PROTECT(y = Rf_allocVector(STRSXP, n));
498  for (i = 0; i < n; i++) {
499  strp = EncodeComplex(COMPLEX(x)[i], w, d, e, wi, di, ei, my_OutDec);
500  SET_STRING_ELT(y, i, Rf_mkChar(strp));
501  }
502  break;
503 
504  case STRSXP:
505  {
506  /* this has to be different from formatString/EncodeString as
507  we don't actually want to encode here */
508  const char *s;
509  char *q;
510  int b, b0, cnt = 0, j;
511  SEXP s0, xx;
512 
513  /* This is clumsy, but it saves rewriting and re-testing
514  this complex code */
515  PROTECT(xx = Rf_duplicate(x));
516  for (i = 0; i < n; i++) {
517  SEXP tmp = STRING_ELT(xx, i);
518  if(IS_BYTES(tmp)) {
519  const char *p = R_CHAR(tmp), *q;
520  char *pp = R_alloc(4*strlen(p)+1, 1), *qq = pp, buf[5];
521  for (q = p; *q; q++) {
522  unsigned char k = static_cast<unsigned char>(*q);
523  if (k >= 0x20 && k < 0x80) {
524  *qq++ = *q;
525  } else {
526  snprintf(buf, 5, "\\x%02x", k);
527  for(int j = 0; j < 4; j++) *qq++ = buf[j];
528  }
529  }
530  *qq = '\0';
531  s = pp;
532  } else s = Rf_translateChar(tmp);
533  if(s != R_CHAR(tmp)) SET_STRING_ELT(xx, i, Rf_mkChar(s));
534  }
535 
536  w = wd;
537  if (adj != Rprt_adj_none) {
538  for (i = 0; i < n; i++)
539  if (STRING_ELT(xx, i) != NA_STRING)
540  w = max(w, Rstrlen(STRING_ELT(xx, i), 0));
541  else if (na) w = max(w, R_print.na_width);
542  } else w = 0;
543  /* now calculate the buffer size needed, in bytes */
544  for (i = 0; i < n; i++)
545  if (STRING_ELT(xx, i) != NA_STRING) {
546  il = Rstrlen(STRING_ELT(xx, i), 0);
547  cnt = max(cnt, LENGTH(STRING_ELT(xx, i)) + max(0, w-il));
548  } else if (na)
549  cnt = max(cnt, R_print.na_width + max(0, w-R_print.na_width));
550  R_CheckStack2(cnt+1);
551  vector<char> buffv(cnt+1);
552  char* buff = &buffv[0];
