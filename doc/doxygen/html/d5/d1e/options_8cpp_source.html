45 /* Interface to the (polymorphous!) options(...) command.
46  *
47  * We have two kind of options:
48  * 1) those used exclusively from R code,
49  * typically initialized in Rprofile.
50 
51  * Their names need not appear here, but may, when we want
52  * to make sure that they are assigned `valid' values only.
53  *
54  * 2) Those used (and sometimes set) from C code;
55  * Either accessing and/or setting a global C variable,
56  * or just accessed by e.g. Rf_GetOption1(Rf_install("pager"))
57  *
58  * A (complete?!) list of these (2):
59  *
60  * "prompt"
61  * "continue"
62  * "expressions"
63  * "width"
64  * "digits"
65  * "echo"
66  * "verbose"
67  * "keep.source"
68  * "keep.source.pkgs"
69  * "browserNLdisabled"
70 
71  * "de.cellwidth" ../unix/X11/ & ../gnuwin32/dataentry.cpp
72  * "device"
73  * "pager"
74  * "paper.size" ./devPS.c
75 
76  * "timeout" ./connections.cpp
77 
78  * "check.bounds"
79  * "error"
80  * "error.messages"
81  * "show.error.messages"
82  * "warn"
83  * "warning.length"
84  * "warning.expression"
85  * "nwarnings"
86 
87  * "matprod"
88  * "PCRE_study"
89  * "PCRE_use_JIT"
90 
91  *
92  * S additionally/instead has (and one might think about some)
93  * "free", "keep"
94  * "length", "memory"
95  * "object.size"
96  * "reference", "show"
97  * "scrap"
98  */

109 {
110  for ( ; lst != R_NilValue ; lst = CDR(lst)) {

190 static SEXP SetOption(SEXP tag, SEXP value)
191 {
192  SEXP opt, old, t;
193  PROTECT(value);
194  t = opt = SYMVALUE(Options());
195  if (!Rf_isList(opt))
196  Rf_error(_("corrupted options list"));
197  opt = FindTaggedItem(opt, tag);
198 
199  /* The option is being removed. */
200  if (value == R_NilValue) {
201  for ( ; t != R_NilValue ; t = CDR(t))
202  if (TAG(CDR(t)) == tag) {
203  old = CAR(CDR(t));
204  SETCDR(t, CDDR(t));
205  UNPROTECT(1); /* value */
206  return old;
207  }
208  UNPROTECT(1); /* value */
209  return R_NilValue;
210  }
211  /* If the option is new, a new slot */
212  /* is added to the end of .Options */
213  if (opt == R_NilValue) {
214  while (CDR(t) != R_NilValue)
215  t = CDR(t);
216  SETCDR(t, Rf_allocList(1));
217  opt = CDR(t);
218  SET_TAG(opt, tag);
219  }
220  old = CAR(opt);
221  SETCAR(opt, value);
222  UNPROTECT(1); /* value */
223  return old;
224 }

385 /* This needs to manage R_Visible */
386 SEXP attribute_hidden do_options(SEXP call, SEXP op, SEXP args, SEXP rho)
387 {
388  SEXP names, value, options;
389 
390  /* Locate the options values in the symbol table.
391  This will need to change if options are to live in the session
392  frame.
393  */
394 
395  options = SYMVALUE(Options());
396 
397  /* This code is not re-entrant and people have used it in
398  finalizers.
399 
400  If a re-entrancy lock needs to be added, note that it
401  would apply to R_SetOption* too.
402  */
403 
404  checkArity(op, args);
405  if (args == R_NilValue) {
406  /* This is the zero argument case.
407  We alloc up a vector list and write the system values into it.
408  */
409  int n = Rf_length(options);
410  PROTECT(value = Rf_allocVector(VECSXP, n));
411  PROTECT(names = Rf_allocVector(STRSXP, n));
412  for (int i = 0; i < n; i++) {
413  SET_STRING_ELT(names, i, PRINTNAME(TAG(options)));
414  SET_VECTOR_ELT(value, i, Rf_duplicate(CAR(options)));
415  options = CDR(options);
416  }
417  SEXP sind = PROTECT(Rf_allocVector(INTSXP, n));
418  int *indx = INTEGER(sind);
419  for (int i = 0; i < n; i++) indx[i] = i;
420  orderVector1(indx, n, names, TRUE, FALSE, R_NilValue);
421  SEXP value2 = PROTECT(Rf_allocVector(VECSXP, n));
422  SEXP names2 = PROTECT(Rf_allocVector(STRSXP, n));
423  for(int i = 0; i < n; i++) {
424  SET_STRING_ELT(names2, i, STRING_ELT(names, indx[i]));
425  SET_VECTOR_ELT(value2, i, VECTOR_ELT(value, indx[i]));
426  }
427  Rf_setAttrib(value2, R_NamesSymbol, names2);
428  UNPROTECT(5);
429  R_Visible = TRUE;
430  return value2;
431  }
432 
433  /* The arguments to "options" can either be a sequence of
434  name = value form, or can be a single list.
435  This means that we must code so that both forms will work.
436  [ Vomits quietly onto shoes ... ]
437  */
438 
439  int n = Rf_length(args);
440  if (n == 1 && (Rf_isPairList(CAR(args)) || Rf_isVectorList(CAR(args)))
441  && TAG(args) == R_NilValue ) {
442  args = CAR(args);
443  n = Rf_length(args);
444  }
445  PROTECT(value = Rf_allocVector(VECSXP, n));
446  PROTECT(names = Rf_allocVector(STRSXP, n));
447 
448  SEXP argnames = R_NilValue;
449  switch (TYPEOF(args)) {
450  case NILSXP:
451  case LISTSXP:
452  break;
453  case VECSXP:
454  if(n > 0) {
455  argnames = Rf_getAttrib(args, R_NamesSymbol);
456  if(LENGTH(argnames) != n)
457  Rf_error(_("list argument has no valid names"));
458  }
459  break;
460  default:
461  UNIMPLEMENTED_TYPE("options", args);
462  }
463 
464  R_Visible = FALSE;
465  for (int i = 0 ; i < n ; i++) { /* i-th argument */
466  SEXP argi = R_NilValue, namei = R_NilValue;
467  switch (TYPEOF(args)) {
468  case LISTSXP:
469  argi = CAR(args);
470  namei = Rf_EnsureString(TAG(args)); /* gives "" for no tag */
471  args = CDR(args);
472  break;
473  case VECSXP:
474  argi = VECTOR_ELT(args, i);
475  namei = STRING_ELT(argnames, i);
476  break;
477  default: /* already checked, but be safe here */
478  UNIMPLEMENTED_TYPE("options", args);
479  }
480 
481  if (*R_CHAR(namei)) { /* name = value ---> assignment */
482  SEXP tag = Rf_installTrChar(namei);
483  if (streql(R_CHAR(namei), "width")) {
484  int k = Rf_asInteger(argi);
485  if (k < R_MIN_WIDTH_OPT || k > R_MAX_WIDTH_OPT)
486  Rf_error(_("invalid 'width' parameter, allowed %d...%d"),
487  R_MIN_WIDTH_OPT, R_MAX_WIDTH_OPT);
488  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
489  }
490  else if (streql(R_CHAR(namei), "deparse.cutoff")) {
491  int k = Rf_asInteger(argi);
492  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
493  }
494  else if (streql(R_CHAR(namei), "digits")) {
495  int k = Rf_asInteger(argi);
496  if (k < R_MIN_DIGITS_OPT || k > R_MAX_DIGITS_OPT)
497  Rf_error(_("invalid 'digits' parameter, allowed %d...%d"),
498  R_MIN_DIGITS_OPT, R_MAX_DIGITS_OPT);
499  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
500  }
501  else if (streql(R_CHAR(namei), "expressions")) {
502  int k = Rf_asInteger(argi);
503  StackChecker::setDepthLimit(k);
504  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
505  }
506  else if (streql(R_CHAR(namei), "keep.source")) {
507  if (TYPEOF(argi) != LGLSXP || LENGTH(argi) != 1)
508  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
509  int k = Rf_asLogical(argi);
510  R_KeepSource = RHOCONSTRUCT(Rboolean, k);
511  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarLogical(k)));
512  }
513  else if (streql(R_CHAR(namei), "editor") && Rf_isString(argi)) {
514  SEXP s = Rf_asChar(argi);
515  if (s == NA_STRING || LENGTH(s) == 0)
516  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
517  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarString(s)));
518  }
519  else if (streql(R_CHAR(namei), "continue")) {
520  SEXP s = Rf_asChar(argi);
521  if (s == NA_STRING || LENGTH(s) == 0)
522  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
523  /* We want to make sure these are in the native encoding */
524  SET_VECTOR_ELT(value, i,
525  SetOption(tag, Rf_mkString(Rf_translateChar(s))));
526  }
527  else if (streql(R_CHAR(namei), "prompt")) {
528  SEXP s = Rf_asChar(argi);
529  if (s == NA_STRING || LENGTH(s) == 0)
530  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
531  /* We want to make sure these are in the native encoding */
532  SET_VECTOR_ELT(value, i,
533  SetOption(tag, Rf_mkString(Rf_translateChar(s))));
534  }
535  else if (streql(R_CHAR(namei), "contrasts")) {
536  if (TYPEOF(argi) != STRSXP || LENGTH(argi) != 2)
537  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
538  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
539  }
540  else if (streql(R_CHAR(namei), "check.bounds")) {
541  if (TYPEOF(argi) != LGLSXP || LENGTH(argi) != 1)
542  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
543  int k = Rf_asLogical(argi);
544  /* R_CheckBounds = k; */
545  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarLogical(k)));
546  }
547  else if (streql(R_CHAR(namei), "warn")) {
548  if (!Rf_isNumeric(argi) || LENGTH(argi) != 1)
549  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
550  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
551  }
552  else if (streql(R_CHAR(namei), "warning.length")) {
553  int k = Rf_asInteger(argi);
554  if (k < 100 || k > 8170)
555  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
556  R_WarnLength = k;
557  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
558  }
559  else if (streql(R_CHAR(namei), "warning.expression")) {
560  if( !Rf_isLanguage(argi) && !Rf_isExpression(argi) )
561  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
562  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
563  }
564  else if (streql(R_CHAR(namei), "max.print")) {
565  int k = Rf_asInteger(argi);
566  if (k < 1) Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
567  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
568  }
569  else if (streql(R_CHAR(namei), "nwarnings")) {
570  int k = Rf_asInteger(argi);
571  if (k < 1) Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
572  R_nwarnings = k;
573  R_CollectWarnings = 0; /* force a reset */
574  SET_VECTOR_ELT(value, i, SetOption(tag, Rf_ScalarInteger(k)));
575  }
576  else if (streql(R_CHAR(namei), "error")) {
577  if(Rf_isFunction(argi))
578  argi = makeErrorCall(argi);
579  else if( !Rf_isLanguage(argi) && !Rf_isExpression(argi) )
580  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
581  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
582  }
583 /* handle this here to avoid GetOption during error handling */
584  else if (streql(R_CHAR(namei), "show.error.messages")) {
585  if( !Rf_isLogical(argi) && LENGTH(argi) != 1 )
586  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
587  SET_VECTOR_ELT(value, i, SetOption(tag, argi));
588  R_ShowErrorMessages = LOGICAL(argi)[0];
589  }
590  else if (streql(R_CHAR(namei), "echo")) {
591  if (TYPEOF(argi) != LGLSXP || LENGTH(argi) != 1)
592  Rf_error(_("invalid value for '%s'"), R_CHAR(namei));
593  int k = Rf_asLogical(argi);
594  /* Should be quicker than checking options(echo)
595  every time R prompts for input:
