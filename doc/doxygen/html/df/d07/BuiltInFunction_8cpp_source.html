62 // BuiltInFunction::apply() creates a FunctionContext only if
63 // m_transparent is false. This affects the location at which
64 // Rf_error() reports an error as having occurred, and also determines
65 // whether a function is reported within traceback().
66 //
67 // Since functions called via .Internal are not visible to the R user,
68 // it seems clear that such functions should be 'transparent'. One
69 // approach would be to leave it at that, so that errors within
70 // internal functions would be attributed to the surrounding call of
71 // .Internal. However, rho (currently at least) goes further than
72 // this, with a view to attributing an error arising within an
73 // internal function to the documented R function which it implements.
74 // To this end do_internal itself and various 'syntactical' functions
75 // such as do_begin are also flagged as transparent.
76 //
77 // The flip side of this is that if an error really does occur within
78 // one of the 'syntactical' functions (rather than within some inner
79 // but transparent scope), it may be desirable to report the error
80 // using Rf_errorcall() rather than Rf_error(), so that it can
81 // specifically be attributed to the 'syntactical' function.

93 {

95  m_function.pairlist = cfun;
96  if (cfun == do_External

98  || cfun == do_begin
99  || cfun == do_break
100  || cfun == do_dotcall
101  || cfun == do_for
102  || cfun == do_if
103  || cfun == do_internal
104  || cfun == do_repeat
105  || cfun == do_return
106  || cfun == do_while) {
107  m_transparent = true;
108  }
109  if (cfun == do_set) {
110  m_transparent = false;
111  }

301 BuiltInFunction::RealInternalDispatch(const Expression* call,
302  Environment* env,
303  ArgList&& args) const
304 {
305  switch(m_dispatch_type) {
306  case DispatchType::INTERNAL:
307  return Rf_Dispatch(call, this, args, env);
308  case DispatchType::GROUP_MATH:
309  case DispatchType::GROUP_OPS:
310  case DispatchType::GROUP_COMPLEX:
311  case DispatchType::GROUP_SUMMARY:
312  return Rf_DispatchGroup(GetInternalGroupDispatchName(),
313  call, this, std::move(args), env);

315  default:
316  Rf_error("Internal error: Unexepcted group dispatch type");
317  }
318  return std::make_pair(false, nullptr);
