202 static int range, xmin; // used by both icount and do_radixsort
203 static void setRange(int *x, int n)
204 {
205  xmin = NA_INTEGER;
206  int xmax = NA_INTEGER;
207  double overflow;
208 
209  int i = 0;
210  while(i < n && x[i] == NA_INTEGER) i++;
211  if (i < n) xmax = xmin = x[i];
212  for (; i < n; i++) {
213  int tmp = x[i];
214  if (tmp == NA_INTEGER)
215  continue;
216  if (tmp > xmax)
217  xmax = tmp;
218  else if (tmp < xmin)
219  xmin = tmp;
220  }
221  // all NAs, nothing to do
222  if (xmin == NA_INTEGER) {
223  range = NA_INTEGER;
224  return;
225  }
226  // ex: x=c(-2147483647L, NA_integer_, 1L) results in overflowing int range.
227  overflow = (double) xmax - (double) xmin + 1;
228  // detect and force iradix here, since icount is out of the picture
229  if (overflow > INT_MAX) {
230  range = INT_MAX;
231  return;
232  }
233 
234  range = xmax - xmin + 1;
235 
236  return;
237 }

241 static inline int icheck(int x)
242 {
243  // if nalast == 1, NAs must go last.

249 static void icount(int *x, int *o, int n)
250 /* Counting sort:
251  1. Places the ordering into o directly, overwriting whatever was there
252  2. Doesn't change x
253  3. Pushes group sizes onto stack
254 */
255 {
256  int napos = range; // NA's always counted in last bin
257  // static is IMPORTANT, counting sort is called repetitively.
258  static unsigned int counts[N_RANGE + 1] = { 0 };
259  /* counts are set back to 0 at the end efficiently. 1e5 = 0.4MB i.e
260  tiny. We'll only use the front part of it, as large as range. So it's
261  just reserving space, not using it. Have defined N_RANGE to be 100000.*/
262  if (range > N_RANGE)
263  Error("Internal error: range = %d; isorted cannot handle range > %d",
264  range, N_RANGE);
265  for (int i = 0; i < n; i++) {
266  // For nalast=NA case, we won't remove/skip NAs, rather set 'o' indices
267  // to 0. subset will skip them. We can't know how many NAs to skip
268  // beforehand - i.e. while allocating "ans" vector
269  if (x[i] == NA_INTEGER)
270  counts[napos]++;
271  else
272  counts[x[i] - xmin]++;
273  }
274 
275  int tmp = 0;
276  if (nalast != 1 && counts[napos]) {
277  push(counts[napos]);
278  tmp += counts[napos];
279  }
280  int w = (order==1) ? 0 : range-1;
281  for (int i = 0; i < range; i++)
282  /* no point in adding tmp < n && i <= range, since range includes max,
283  need to go to max, unlike 256 loops elsewhere in radixsort.c */
284  {
285  if (counts[w]) {
286  // cumulate but not through 0's.
287  // Helps resetting zeros when n < range, below.
288  push(counts[w]);
289  counts[w] = (tmp += counts[w]);
290  }
291  w += order; // order is +1 or -1
292  }
293  if (nalast == 1 && counts[napos]) {
294  push(counts[napos]);
295  counts[napos] = (tmp += counts[napos]);
296  }
297  for (int i = n - 1; i >= 0; i--) {
298  // This way na.last=TRUE/FALSE cases will have just a
299  // single if-check overhead.
300  o[--counts[(x[i] == NA_INTEGER) ? napos :
301  x[i] - xmin]] = (int) (i + 1);
302  }
303  // nalast = 1, -1 are both taken care already.
304  if (nalast == 0)
305  // nalast = 0 is dealt with separately as it just sets o to 0
306  for (int i = 0; i < n; i++)
307  o[i] = (x[o[i] - 1] == NA_INTEGER) ? 0 : o[i];
308  // at those indices where x is NA. x[o[i]-1] because x is not modifed here.
309 
310  /* counts were cumulated above so leaves non zero.
311  Faster to clear up now ready for next time. */
312  if (n < range) {
313  /* Many zeros in counts already. Loop through n instead,
314  doesn't matter if we set to 0 several times on any repeats */
315  counts[napos] = 0;
316  for (int i = 0; i < n; i++) {
317  if (x[i] != NA_INTEGER)
318  counts[x[i] - xmin] = 0;
319  }
320  } else
321  memset(counts, 0, (range + 1) * sizeof(int));
322  return;
323 }

325 static void iinsert(int *x, int *o, int n)
326 /* orders both x and o by reference in-place. Fast for small vectors,
327  low overhead. don't be tempted to binsearch backwards here, have
328  to shift anyway; many memmove would have overhead and do the same
329  thing. */
330 /* when nalast == 0, iinsert will be called only from within iradix,
331  where o[.] = 0 for x[.]=NA is already taken care of */
332 {
333  for (int i = 1; i < n; i++) {
334  int xtmp = x[i];
335  if (xtmp < x[i - 1]) {
336  int j = i - 1;
337  int otmp = o[i];
338  while (j >= 0 && xtmp < x[j]) {
339  x[j + 1] = x[j];
340  o[j + 1] = o[j];
341  j--;
342  }
343  x[j + 1] = xtmp;
344  o[j + 1] = otmp;
345  }
346  }
347  int tt = 0;
348  for (int i = 1; i < n; i++)
349  if (x[i] == x[i - 1])
350  tt++;
351  else {
352  push(tt + 1);
353  tt = 0;
354  }
355  push(tt + 1);
356 }

358 /*
359  iradix is a counting sort performed forwards from MSB to LSB, with
360  some tricks and short circuits building on Terdiman and Herf.
361  http://codercorner.com/RadixSortRevisited.htm
362  http://stereopsis.com/radix.html
363 
364  ~ Note they are LSD, but we do MSD here which is more complicated,
365  for efficiency.
366  ~ NAs need no special treatment as NA is the most negative integer
367  in R (checked in init.c once, for efficiency) so NA naturally sort
368  to the front.
369  ~ Using 4-pass 1-byte radix for the following reasons :
370 
371  * 11-bit (Herf) reduces to 3-passes (3*11=33) yes, and LSD need
372  random access to o vector in each pass 1:n so reduction in passes is
373  good, but Terdiman's idea to skip a radix if all values are equal
374  occurs less the wider the radix. A narrower radix benefits more from that.
375  * That's detected here using a single 'if', an improvement on
376  Terdiman's exposition of a single loop to find if any count==n
377  * The pass through counts bites when radix is wider,
378  because we repetitively call this iradix from fastorder forwards.
379  * Herf's parallel histogramming is neat. In 4-pass 1-byte it needs
380  4*256 storage, that's tiny, and can be static. 4*256 << 3*2048.
381  4-pass 1-byte is simpler and tighter code than 3-pass 11-bit,
382  giving modern optimizers and modern CPUs a better chance.
383  We may get lucky anyway, if one or two of the 4-passes are skipped.
384 
385  Recall: there are no comparisons at all in counting and radix,
386  there is wide random access in each LSD radix pass, though.
387 */

428 static void iradix(int *x, int *o, int n)
429 /* As icount :
430  Places the ordering into o directly, overwriting whatever was there
431  Doesn't change x
432  Pushes group sizes onto stack */
433 {
434  int nextradix, itmp, thisgrpn, maxgrpn;
435  unsigned int thisx = 0, shift, *thiscounts;
436 
437  for (int i = 0; i < n;i++) {
438  /* parallel histogramming pass; i.e. count occurrences of
439  0:255 in each byte. Sequential so almost negligible. */
440  // relies on overflow behaviour. And shouldn't -INT_MIN be up in iradix?
441  thisx = (unsigned int) (icheck(x[i])) - INT_MIN;
442  // unrolled since inside n-loop
443  radixcounts[0][thisx & 0xFF]++;
444  radixcounts[1][thisx >> 8 & 0xFF]++;
445  radixcounts[2][thisx >> 16 & 0xFF]++;
446  radixcounts[3][thisx >> 24 & 0xFF]++;
447  }
448  for (int radix = 0; radix < 4; radix++) {
449  /* any(count == n) => all radix must have been that value =>
450  last x (still thisx) was that value */
451  int i = thisx >> (radix*8) & 0xFF;
452  skip[radix] = radixcounts[radix][i] == unsigned(n);
453  // clear it now, the other counts must be 0 already
454  if (skip[radix])
455  radixcounts[radix][i] = 0;
456  }
457 
458  int radix = 3; // MSD
459  while (radix >= 0 && skip[radix]) radix--;
460  if (radix == -1) { // All radix are skipped; one number repeated n times.
461  if (nalast == 0 && x[0] == NA_INTEGER)
462  // all values are identical. return 0 if nalast=0 & all NA
463  // because of 'return', have to take care of it here.
464  for (int i = 0; i < n; i++)
465  o[i] = 0;
466  else
467  for (int i = 0; i < n; i++)
468  o[i] = (i + 1);
469  push(n);
470  return;
471  }
472  for (int i = radix - 1; i >= 0; i--) {
473  if (!skip[i])
474  memset(radixcounts[i], 0, 257 * sizeof(unsigned int));
475  /* clear the counts as we only needed the parallel pass for skip[]
476  and we're going to use radixcounts again below. Can't use parallel
477  lower counts in MSD radix, unlike LSD. */
478  }
479  thiscounts = radixcounts[radix];
480  shift = radix * 8;
481 
482  itmp = thiscounts[0];
483  maxgrpn = itmp;
484  for (int i = 1; itmp < n && i < 256; i++) {
485  thisgrpn = thiscounts[i];
486  if (thisgrpn) {
487  // don't cummulate through 0s, important below.
488  if (thisgrpn > maxgrpn)
489  maxgrpn = thisgrpn;
490  thiscounts[i] = (itmp += thisgrpn);
491  }
492  }
493  for (int i = n - 1; i >= 0; i--) {
494  thisx = ((unsigned int) (icheck(x[i])) - INT_MIN) >> shift & 0xFF;
495  o[--thiscounts[thisx]] = i + 1;
496  }
497 
498  if (radix_xsuballoc < size_t(maxgrpn)) {
499  // The largest group according to the first non-skipped radix,
500  // so could be big (if radix is needed on first arg)
501  // TO DO: could include extra bits to divide the first radix
502  // up more. Often the MSD has groups in just 0-4 out of 256.
503  // free'd at the end of do_radixsort once we're done calling iradix
504  // repetitively
505  radix_xsub = (int *) realloc(radix_xsub, maxgrpn * sizeof(double));
506  if (!radix_xsub)
507  Error("Failed to realloc working memory %d*8bytes (xsub in iradix), radix=%d",
508  maxgrpn, radix);
509  radix_xsuballoc = maxgrpn;
510  }
511 
512  // TO DO: can we leave this to do_radixsort and remove these calls??
513  alloc_otmp(maxgrpn);
514  // TO DO: doesn't need to be sizeof(double) always, see inside
515  alloc_xtmp(maxgrpn);
516 
517  nextradix = radix - 1;
518  while (nextradix >= 0 && skip[nextradix]) nextradix--;
519  if (thiscounts[0] != 0)
520  Error("Internal error. thiscounts[0]=%d but should have been decremented to 0. dradix=%d",
521  thiscounts[0], radix);
522  thiscounts[256] = n;
523  itmp = 0;
524  for (int i = 1; itmp < n && i <= 256; i++) {
525  if (thiscounts[i] == 0) continue;
526  // undo cumulate; i.e. diff
527  thisgrpn = thiscounts[i] - itmp;
528  if (thisgrpn == 1 || nextradix == -1) {
529  push(thisgrpn);
530  } else {
531  for (int j = 0; j < thisgrpn; j++)
532  // this is why this xsub here can't be the same memory as
533  // xsub in do_radixsort.
534  ((int *)radix_xsub)[j] = icheck(x[o[itmp+j]-1]);
535  // changes xsub and o by reference recursively.
536  iradix_r(static_cast<int *>(radix_xsub), o+itmp, thisgrpn, nextradix);
537  }
538  itmp = thiscounts[i];
539  thiscounts[i] = 0;
540  }
541  if (nalast == 0) // nalast = 1, -1 are both taken care already.
542  // nalast = 0 is dealt with separately as it just sets o to 0
543  for (int i = 0; i < n; i++)
544  o[i] = (x[o[i] - 1] == NA_INTEGER) ? 0 : o[i];
545  // at those indices where x is NA. x[o[i]-1] because x is not
546  // modified by reference unlike iinsert or iradix_r
547 }

549 static void iradix_r(int *xsub, int *osub, int n, int radix)
550 // xsub is a recursive offset into xsub working memory above in
551 // iradix, reordered by reference. osub is a an offset into the main
552 // answer o, reordered by reference. radix iterates 3,2,1,0
553 {
554  int j, itmp, thisx, thisgrpn, nextradix, shift;
555  unsigned int *thiscounts;
556 
557  // N_SMALL=200 is guess based on limited testing. Needs
558  // calibrate(). Was 50 based on sum(1:50)=1275 worst -vs- 256
559  // cummulate + 256 memset + allowance since reverse order is
560  // unlikely. when nalast==0, iinsert will be called only from
561  // within iradix.
562  if (n < N_SMALL) {
563  iinsert(xsub, osub, n);
564  return;
565  }
566 
567  shift = radix * 8;
568  thiscounts = radixcounts[radix];
569 
570  for (int i = 0; i < n; i++) {
571  thisx = (unsigned int) xsub[i] - INT_MIN; // sequential in xsub
572  thiscounts[thisx >> shift & 0xFF]++;
573  }
574  itmp = thiscounts[0];
575  for (int i = 1; itmp < n && i < 256; i++)
576  // don't cummulate through 0s, important below
577  if (thiscounts[i])
578  thiscounts[i] = (itmp += thiscounts[i]);
579  for (int i = n - 1; i >= 0; i--) {
580  thisx = ((unsigned int) xsub[i] - INT_MIN) >> shift & 0xFF;
581  j = --thiscounts[thisx];
582  otmp[j] = osub[i];
583  ((int *) xtmp)[j] = xsub[i];
584  }
585  memcpy(osub, otmp, n * sizeof(int));
586  memcpy(xsub, xtmp, n * sizeof(int));

786 static void dinsert(unsigned long long *x, int *o, int n)
787 // orders both x and o by reference in-place. Fast for small vectors,
788 // low overhead. don't be tempted to binsearch backwards here, have
789 // to shift anyway; many memmove would have overhead and do the same
790 // thing 'dinsert' will not be called when nalast = 0 and o[0] = -1.
791 {
792  int otmp, tt;
793  unsigned long long xtmp;
794  for (int i = 1; i < n; i++) {
795  xtmp = x[i];
796  if (xtmp < x[i - 1]) {
797  int j = i - 1;
798  otmp = o[i];
799  while (j >= 0 && xtmp < x[j]) {
800  x[j + 1] = x[j];
801  o[j + 1] = o[j];
802  j--;
803  }
804  x[j + 1] = xtmp;
805  o[j + 1] = otmp;
806  }
807  }

809  for (int i = 1; i < n; i++)
810  if (x[i] == x[i - 1])
811  tt++;
812  else {
813  push(tt + 1);
814  tt = 0;
815  }
816  push(tt + 1);
