63 void
64 Subscripting::Indices::initialize(const RObject* subscripts,
65  std::size_t range_size,
66  const StringVector* range_names)
67 {
68  if (!subscripts)
69  return;
70  m_max_index = 0;
71  switch (subscripts->sexptype()) {
72  case LGLSXP:
73  initialize(static_cast<const LogicalVector*>(subscripts), range_size);
74  break;
75  case INTSXP:
76  initialize(static_cast<const IntVector*>(subscripts), range_size);
77  break;
78  case REALSXP:
79  initialize(static_cast<const RealVector*>(subscripts), range_size);
80  break;
81  case STRSXP:
82  initialize(static_cast<const StringVector*>(subscripts), range_size,
83  range_names);
84  break;
85  case SYMSXP:
86  {
87  const Symbol* sym = static_cast<const Symbol*>(subscripts);
88  if (sym != Symbol::missingArgument())
89  Rf_error(_("invalid subscript type '%s'"), sym->typeName());
90  // If a subscript argument is missing, we take this as
91  // meaning 'select everything':
92  resize(range_size);
93  m_max_index = range_size;
94  for (std::size_t i = 0; i < range_size; ++i)
95  (*this)[i] = i + 1;
96  }
97  break;
98  default:
99  Rf_error(_("invalid subscript type '%s'"), subscripts->typeName());
100  }
101 }

103 void Subscripting::Indices::initialize(const IntVector* raw_indices,
104  std::size_t range_size)
105 {
106  const std::size_t rawsize = raw_indices->size();
107  bool anyNA = false;
108  bool anyneg = false;
109  std::size_t zeroes = 0;
110  m_max_index = 0;
111  for (std::size_t i = 0; i < rawsize; ++i) {
112  int index = (*raw_indices)[i];
113  if (isNA(index))
114  anyNA = true;
115  else if (index < 0)
116  anyneg = true;
117  else if (index == 0)
118  ++zeroes;
119  else if (index > int(m_max_index))
120  m_max_index = index;
121  }
122  if (!anyneg) {
123  resize(rawsize - zeroes);
124  std::size_t iout = 0;
125  // Suppress zeroes, and replace NA by zero:
126  for (std::size_t iin = 0; iin < rawsize; ++iin) {
127  int index = (*raw_indices)[iin];
128  if (index != 0)
129  (*this)[iout++] = (isNA(index) ? 0 : index);
130  }
131  m_min_lhssize = m_max_index;
132  } else { // Negative subscripts
133  if (anyNA || m_max_index > 0)
134  Rf_error(_("only 0's may be mixed with negative subscripts"));
135  // Create a vector whose elements are non-zero for elements to
136  // be retained:
137  std::vector<char> retvec(range_size, 1);
138  std::size_t removed = 0;
139  for (std::size_t i = 0; i < rawsize; ++i) {
140  std::size_t index = std::size_t(-(*raw_indices)[i]);
141  if (index != 0 && index <= range_size) {
142  char& elt = retvec[index - 1];
143  if (elt != 0)
144  ++removed;
145  elt = 0;
146  }
147  }
148  // Now set up vector and copy in the elements to be retained:
149  resize(range_size - removed);
150  m_max_index = 0;
151  std::size_t iout = 0;
152  for (std::size_t iin = 0; iin < range_size; ++iin)
153  if (retvec[iin]) {
154  m_max_index = iin + 1;
155  (*this)[iout++] = m_max_index;
156  }
157  m_min_lhssize = range_size;
158  }

220 void Subscripting::Indices::initialize(const LogicalVector* raw_indices,
221  std::size_t range_size)
222 {
223  const std::size_t rawsize = raw_indices->size();
224  if (rawsize == 0)
225  return;
226  m_min_lhssize = std::max(range_size, rawsize);
227  // Determine size of answer:
228  std::size_t anssize = 0;
229  for (std::size_t i = 0; i < m_min_lhssize; ++i)
230  {
231  Logical included = (*raw_indices)[i%rawsize];
232  if (included.isTrue() || included.isNA())
233  ++anssize;
234  }
235  // Create Indices vector:
236  resize(anssize);
237  m_max_index = 0;
238  std::size_t iout = 0;
239  for (std::size_t iin = 0; iin < m_min_lhssize; ++iin) {
240  Logical logical = (*raw_indices)[iin % rawsize];
241  if (logical.isNA())
242  (*this)[iout++] = 0;
243  else if (logical.isTrue()) {
244  m_max_index = iin + 1;
245  (*this)[iout++] = m_max_index;
246  }
247  }

250 void Subscripting::Indices::initialize(const StringVector* raw_indices,
251  std::size_t range_size,
252  const StringVector* range_names)
253 {
254  const std::size_t rawsize = raw_indices->size();
255  typedef std::unordered_map<GCRoot<String>, std::size_t> Nmap;
256 
257  Nmap names_map;
258  m_max_index = (range_names ? 0 : range_size);
259  resize(rawsize);
260  // Process the supplied subscripts in order:
261  for (std::size_t iraw = 0; iraw < rawsize; ++iraw) {
262  String* subscript = (*raw_indices)[iraw];
263  if (subscript == String::NA())
264  (*this)[iraw] = 0;
265  else {
266  GCRoot<String> csubscript(SEXP_downcast<String*>(subscript));
267  // Coerce to UTF8 if necessary:
268  if (csubscript->encoding() != CE_UTF8) {
269  RAllocStack::Scope scope;
270  const char* utf8s = Rf_translateCharUTF8(csubscript);
271  csubscript = String::obtain(utf8s, CE_UTF8);
272  }
273  // Have we met this name already?
274  Nmap::const_iterator it = names_map.find(csubscript);
275  if (it != names_map.end()) // Yes
276  (*this)[iraw] = (*it).second;
277  else { // No
278  // If we haven't yet worked right through range_names,
279  // continue searching for a match from where we left
280  // off, in the process continuing to build a mapping
281  // from names to indices:
282  bool found = false;
283  while (m_max_index < range_size && !found) {
284  String* name = (*range_names)[m_max_index++];
285  if (name != String::NA()) {
286  GCRoot<String> cname(SEXP_downcast<String*>(name));
287  if (cname != String::blank()) {
288  // Coerce to UTF8 if necessary:
289  if (cname->encoding() != CE_UTF8) {
290  RAllocStack::Scope scope;
291  const char* utf8s = Rf_translateCharUTF8(cname);
292  cname = String::obtain(utf8s, CE_UTF8);
293  }
294  // Insert this name into names_map
295  // provided it isn't already there:
296  Nmap::const_iterator nmit = names_map.find(cname);
297  if (nmit == names_map.end())
298  names_map[cname] = m_max_index;
299  found = (cname == csubscript);
300  }
301  }
302  }
303  if (found)
304  (*this)[iraw] = m_max_index;
305  else { // The supplied subscript is a new name:
306  if (!m_use_names)
307  m_use_names = ListVector::create(rawsize);
308  ++m_max_index;
309  (*m_use_names)[iraw] = csubscript;
310  (*this)[iraw] = m_max_index;
311  names_map[csubscript] = m_max_index;
312  }
313  }
314  }
315  }
316  m_min_lhssize = m_max_index;
317 }

321 std::pair<VectorBase*, std::size_t>
322 Subscripting::canonicalize(const RObject* raw_indices, std::size_t range_size,
323  const StringVector* range_names)
324 {
325  Indices indices;
326  indices.initialize(raw_indices, range_size, range_names);
327  std::size_t maxindex = indices.maximumIndex();
328  if (maxindex > std::size_t(std::numeric_limits<int>::max())) {
329  GCStackRoot<RealVector> canvec(RealVector::create(indices.size()));
330  for (std::size_t i = 0; i < indices.size(); ++i) {
331  std::size_t index = indices[i];
332  (*canvec)[i] = (index == 0 ? NA<double>() : double(index));
333  }
334  return std::make_pair(canvec.get(), maxindex);
335  } else {
336  GCStackRoot<IntVector> canvec(IntVector::create(indices.size()));
337  for (std::size_t i = 0; i < indices.size(); ++i) {
338  std::size_t index = indices[i];
339  (*canvec)[i] = (index == 0 ? NA<int>() : int(index));
340  }
341  return std::make_pair(canvec.get(), maxindex);
342  }
343 }

345 void
346 Subscripting::canonicalizeArraySubscripts(std::vector<Indices>* indicesvec,
347  const VectorBase* v,
348  const PairList* subscripts)
349 {
350  const IntVector* dims = v->dimensions();
351  if (!dims)
352  Rf_error(_ ("not a matrix/array"));
353  std::size_t ndims = dims->size();
354  indicesvec->resize(ndims);
355  const ListVector* dimnames = v->dimensionNames();
356  const PairList* pl = subscripts;
357  for (std::size_t d = 0; d < ndims; ++d) {
358  if (!pl)
359  Rf_error(_("too few subscripts"));
360  std::size_t dimsize = std::size_t((*dims)[d]);
361  const StringVector* names
362  = (dimnames ? static_cast<StringVector*>((*dimnames)[d].get()) : nullptr);
363  Indices& indices = (*indicesvec)[d];
364  indices.initialize(pl->car(), dimsize, names);
365  if (indices.maximumIndex() > dimsize)
366  Rf_error(_("subscript out of bounds"));
367  pl = pl->tail();
368  }
369  if (pl)
370  Rf_error(_("too many subscripts"));
371 }

373 std::size_t
374 Subscripting::createDimIndexers(DimIndexerVector* dimindexers,
375  const IntVector* source_dims,
376  const std::vector<Indices>& indicesvec)
377 {
378  std::size_t ndims = source_dims->size();
379  double dresultsize = 1.0;
380  std::size_t resultsize = 1;
381  for (std::size_t d = 0; d < ndims; ++d) {
382  DimIndexer& di = (*dimindexers)[d];
383  std::size_t dsz = indicesvec[d].size();
384  dresultsize *= double(dsz);
385  if (dresultsize > std::numeric_limits<size_t>::max())
386  Rf_error(_("dimensions would exceed maximum size of array"));
387  resultsize *= dsz;
388  di.indexnum = 0;
389  }
390  (*dimindexers)[0].stride = 1;
391  for (std::size_t d = 1; d < ndims; ++d)
392  (*dimindexers)[d].stride
393  = (*dimindexers)[d - 1].stride * std::size_t((*source_dims)[d - 1]);
394  return resultsize;
395 }

397 bool Subscripting::dropDimensions(VectorBase* v)
398 {
399  GCStackRoot<const IntVector> dims(v->dimensions());
400  if (!dims)
401  return false;
402  std::size_t ndims = dims->size();
403  // Count the number of dimensions with extent != 1 :
404  std::size_t ngooddims = 0;
405  for (std::size_t d = 0; d < ndims; ++d)
406  if ((*dims)[d] != 1)
407  ++ngooddims;
408  if (ngooddims == ndims)
409  return false;
410  GCStackRoot<ListVector> dimnames(const_cast<ListVector*>(v->dimensionNames()));
411  v->setDimensionNames(nullptr);
412  if (ngooddims > 1) {
413  // The result will still be an array/matrix.
414  bool havenames = false;
415  // Set up new dimensions attribute:
416  {
417  GCStackRoot<IntVector> newdims(IntVector::create(ngooddims));
418  StringVector* names = nullptr;
419  if (dims->names()) {
420  names = StringVector::create(ngooddims);
421  newdims->setNames(names);
422  }
423  std::size_t dout = 0;
424  for (std::size_t din = 0; din < ndims; ++din) {
425  std::size_t dsize = std::size_t((*dims)[din]);
426  if (dsize != 1) {
427  if (names)
428  (*names)[dout] = (*dims->names())[din];
429  (*newdims)[dout++] = int(dsize);
430  if (dimnames && (*dimnames)[din])
431  havenames = true;
432  }
433  }
434  v->setDimensions(newdims);
435  }
436  if (havenames) {
437  // Set up new dimnames attribute:
438  StringVector* dimnamesnames
439  = const_cast<StringVector*>(dimnames->names());
440  GCStackRoot<ListVector> newdimnames(ListVector::create(ngooddims));
441  GCStackRoot<StringVector>
442  newdimnamesnames(StringVector::create(ngooddims));
443  std::size_t dout = 0;
444  for (std::size_t din = 0; din < ndims; ++din)
445  if ((*dims)[din] != 1) {
446  (*newdimnames)[dout] = ElementTraits::duplicate_element(
447  (*dimnames)[din].get());
448  if (dimnamesnames)
449  (*newdimnamesnames)[dout] = (*dimnamesnames)[din];
450  ++dout;
451  }
452  if (dimnamesnames)
453  newdimnames->setNames(newdimnamesnames);
454  v->setDimensionNames(newdimnames);
455  }
456  } else if (ngooddims == 1) {
457  // Reduce to a vector.
458  v->setDimensions(nullptr);
459  if (dimnames) {
460  std::size_t d = 0;
461  while ((*dims)[d] == 1)
462  ++d;
463  v->setNames(static_cast<StringVector*>((*dimnames)[d].get()));
464  }
465  } else /* ngooddims == 0 */ {
466  v->setDimensions(nullptr);
467  v->setDimensionNames(nullptr);
468  // In this special case it is ambiguous which dimnames to use
469  // for the sole remaining element, so we set up a name only if
470  // just one dimension has names.
471  if (dimnames) {
472  StringVector* newnames = nullptr;
473  std::size_t count = 0;
474  for (std::size_t d = 0; d < ndims; ++d) {
475  RObject* dnd = (*dimnames)[d];
476  if (dnd) {
477  newnames = static_cast<StringVector*>(dnd);
478  ++count;
479  }
480  }
481  if (count == 1)
482  v->setNames(newnames);
483  }
484  }
485  return true;
486 }
