51 HIDDEN SEXP do_split(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* f_)
52 {
53  SEXP x, f, counts, vec, nm, nmj;
54  Rboolean have_names;
55 
56  x = x_;
57  f = f_;
58  if (!Rf_isVector(x))
59  Rf_error(_("first argument must be a vector"));
60  if (!Rf_isFactor(f))
61  Rf_error(_("second argument must be a factor"));
62  int nlevs = Rf_nlevels(f);
63  R_xlen_t nfac = XLENGTH(f_);
64  R_xlen_t nobs = XLENGTH(x_);
65  if (nfac <= 0 && nobs > 0)
66  Rf_error(_("group length is 0 but data length > 0"));
67  if (nfac > 0 && (nobs % nfac) != 0)
68  Rf_warning(_("data length is not a multiple of split variable"));
69  nm = Rf_getAttrib(x, Symbols::NamesSymbol);
70  have_names = Rboolean(nm != nullptr);
71 
72 #ifdef LONG_VECTOR_SUPPORT
73  if (IS_LONG_VEC(x))
74 # define _L_INTSXP_ REALSXP
75 # define _L_INTEG_ REAL
76 # define _L_int_ R_xlen_t
77 # include "split-incl.cpp"
78 
79 # undef _L_INTSXP_
80 # undef _L_INTEG_
81 # undef _L_int_
82  else
83 #endif
84 
85 # define _L_INTSXP_ INTSXP
86 # define _L_INTEG_ INTEGER
87 # define _L_int_ int
88 # include "split-incl.cpp"
89 
90 # undef _L_INTSXP_
91 # undef _L_INTEG_
92 # undef _L_int_
93 
94  Rf_setAttrib(vec, Symbols::NamesSymbol, Rf_getAttrib(f, Symbols::LevelsSymbol));
95  UNPROTECT(2);
96  return vec;
97 }
