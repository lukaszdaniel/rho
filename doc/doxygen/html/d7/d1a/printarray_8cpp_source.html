134 static void printLogicalMatrix(SEXP sx, int offset, int r_pr, int r, int c,
135  StringVector* rl, SEXP cl,
136  const char *rn, const char *cn,
137  Rboolean print_ij)
138 {
139 /* initialization; particularly of row labels, rl= dimnames(.)[[1]] and
140  * rn = names(dimnames(.))[1] : */
141 #define _PRINT_INIT_rl_rn \
142  int *w = (int *) R_alloc(c, sizeof(int)); \
143  int width, rlabw = -1, clabw = -1; /* -Wall */ \
144  int i, j, jmin = 0, jmax = 0, lbloff = 0; \
145  \
146  if (!isNull(rl)) { \
147  StringVector::const_iterator beg = rl->begin(); \
148  rlabw = std::accumulate(beg, beg + r, 0, stringWidth); \
149  } else \
150  rlabw = IndexWidth(r + 1) + 3; \
151  \
152  if (rn) { \
153  int rnw = strwidth(rn); \
154  if ( rnw < rlabw + R_MIN_LBLOFF ) \
155  lbloff = R_MIN_LBLOFF; \
156  else \
157  lbloff = rnw - rlabw; \
158  \
159  rlabw += lbloff; \
160  }
161 
162 # define _COMPUTE_W2_(_FORMAT_j_, _LAST_j_) \
163  /* compute w[j] = column-width of j(+1)-th column : */ \
164  for (j = 0; j < c; j++) { \
165  if(print_ij) { _FORMAT_j_; } else w[j] = 0; \
166  \
167  if (!isNull(cl)) { \
168  const void *vmax = vmaxget(); \
169  if(STRING_ELT(cl, j) == NA_STRING) \
170  clabw = R_print.na_width_noquote; \
171  else clabw = strwidth(translateChar(STRING_ELT(cl, j))); \
172  vmaxset(vmax); \
173  } else \
174  clabw = IndexWidth(j + 1) + 3; \
175  \
176  if (w[j] < clabw) \
177  w[j] = clabw; \
178  _LAST_j_; \
179  }
180 
181 # define _COMPUTE_W_(F_j) _COMPUTE_W2_(F_j, w[j] += R_print.gap)
182  // _LAST_j ------------------- for all but String
183 
184 # define _PRINT_ROW_LAB \
185  \
186  if (cn != NULL) \
187  Rprintf("%*s%s\n", rlabw, "", cn); \
188  if (rn != NULL) \
189  Rprintf("%*s", -rlabw, rn); \
190  else \
191  Rprintf("%*s", rlabw, "")
192 
193 # define _PRINT_MATRIX_(_W_EXTRA_, DO_COLUMN_LABELS, ENCODE_I_J) \
194  \
195  if (c == 0) { \
196  _PRINT_ROW_LAB; \
197  for (i = 0; i < r; i++) \
198  MatrixRowLabel(rl, i, rlabw, lbloff); \
199  Rprintf("\n"); \
200  } \
201  else while (jmin < c) { \
202  /* print columns jmin:(jmax-1) where jmax has to be determined first */ \
203  \
204  width = rlabw; \
205  /* initially, jmax = jmin */ \
206  do { \
207  width += w[jmax] _W_EXTRA_; \
208  jmax++; \
209  } \
210  while (jmax < c && width + w[jmax] _W_EXTRA_ < R_print.width); \
211  \
212  _PRINT_ROW_LAB; \
213  \
214  DO_COLUMN_LABELS; \
215  \
216  for (i = 0; i < r_pr; i++) { \
217  MatrixRowLabel(rl, i, rlabw, lbloff); /* starting with an "\n" */ \
218  if(print_ij) for (j = jmin; j < jmax; j++) { \
219  ENCODE_I_J; \
220  } \
221  } \
222  Rprintf("\n"); \
223  jmin = jmax; \
224  }
225 
226 # define STD_ColumnLabels \
227  for (j = jmin; j < jmax ; j++) \
228  MatrixColumnLabel(cl, j, w[j])
229 
230  _PRINT_INIT_rl_rn;
231  int *x = LOGICAL(sx) + offset;
232 
233  _COMPUTE_W_(formatLogical(&x[j * r], (R_xlen_t) r, &w[j]));
234 
235  _PRINT_MATRIX_( , STD_ColumnLabels,
236  Rprintf("%s", EncodeLogical(x[i + j * r], w[j])));
237 
238 }

270  StringVector* rl, SEXP cl,

360 attribute_hidden
361 void printMatrix(SEXP x, int offset, SEXP dim, int quote, int right,
362  SEXP rl, SEXP cl, const char *rn, const char *cn)
363 {
364 /* 'rl' and 'cl' are dimnames(.)[[1]] and dimnames(.)[[2]] whereas
365  * 'rn' and 'cn' are the names(dimnames(.))
366  */
367  const void *vmax = vmaxget();
368  int r = INTEGER(dim)[0];
369  int c = INTEGER(dim)[1], r_pr;
370  /* PR#850 */
371  if ((rl != R_NilValue) && (r > Rf_length(rl)))
372  error(_("too few row labels"));
373  if ((cl != R_NilValue) && (c > Rf_length(cl)))
374  error(_("too few column labels"));
375  if (r == 0 && c == 0) { // FIXME? names(dimnames(.)) :
376  Rprintf("<0 x 0 matrix>\n");
377  return;
378  }
379  r_pr = r;
380  StringVector* rlabs = SEXP_downcast<StringVector*>(rl);
381  if(c > 0 && R_print.max / c < r) /* avoid integer overflow */
382  /* using floor(), not ceil(), since 'c' could be huge: */
383  r_pr = R_print.max / c;
384  switch (TYPEOF(x)) {
385  case LGLSXP:
386  printLogicalMatrix(x, offset, r_pr, r, c, rlabs, cl, rn, cn, TRUE);
387  break;
388  case INTSXP:
389  printIntegerMatrix(x, offset, r_pr, r, c, rlabs, cl, rn, cn, TRUE);
390  break;
391  case REALSXP:
392  printRealMatrix (x, offset, r_pr, r, c, rlabs, cl, rn, cn, TRUE);
393  break;
394  case CPLXSXP:
395  printComplexMatrix(x, offset, r_pr, r, c, rlabs, cl, rn, cn, TRUE);
396  break;
397  case STRSXP:
398  {
399  if (quote) quote = '"';
400  const StringVector* sv = SEXP_downcast<StringVector*>(x);
401  printStringMatrix (sv, offset, r_pr, r, c, quote,
402  right, rlabs, cl, rn, cn, TRUE);
403  break;
404  }
405  case RAWSXP:
406  printRawMatrix (x, offset, r_pr, r, c, rlabs, cl, rn, cn, TRUE);
407  break;
408  default:
409  UNIMPLEMENTED_TYPE("printMatrix", x);
410  }
411 #ifdef ENABLE_NLS
412  if(r_pr < r) // number of formats must be consistent here
413  Rprintf(n_(" [ reached getOption(\"max.print\") -- omitted %d row ]\n",
414  " [ reached getOption(\"max.print\") -- omitted %d rows ]\n",
415  r - r_pr),
416  r - r_pr);
417 #else
418  if(r_pr < r)
419  Rprintf(" [ reached getOption(\"max.print\") -- omitted %d rows ]\n",
420  r - r_pr);
421 #endif
422  vmaxset(vmax);
423 }

426 void printArray(SEXP x, SEXP dim, int quote, int right, SEXP dimnames)
427 {
428 /* == printArray(.) */
429  const void *vmax = vmaxget();
430  int ndim = LENGTH(dim);
431  const char *rn = NULL, *cn = NULL;
432 
433  if (ndim == 1)
434  printVector(x, 1, quote);
435  else if (ndim == 2) {
436  SEXP rl, cl;
437  GetMatrixDimnames(x, &rl, &cl, &rn, &cn);
438  printMatrix(x, 0, dim, quote, 0, rl, cl, rn, cn);
439  }
440  else { /* ndim >= 3 */
441  SEXP dn, dnn, dn1;
442  StringVector* dn0;
443  int i, j, nb, nb_pr, nr_last,
444  *dims = INTEGER(dim), nr = dims[0], nc = dims[1],
445  b = nr * nc;
446  bool max_reached, has_dimnames = (dimnames != R_NilValue),
447  has_dnn = has_dimnames;
448 
449  if (!has_dimnames) {
450  dn0 = R_NilValue;
451  dn1 = R_NilValue;
452  dnn = R_NilValue; /* -Wall */
453  }
454  else {
455  dn0 = SEXP_downcast<StringVector*>(VECTOR_ELT(dimnames, 0));
456  dn1 = VECTOR_ELT(dimnames, 1);
457  dnn = getAttrib(dimnames, R_NamesSymbol);
458  has_dnn = !isNull(dnn);
459  if ( has_dnn ) {
460  rn = (char *) translateChar(STRING_ELT(dnn, 0));
461  cn = (char *) translateChar(STRING_ELT(dnn, 1));
462  }
463  }
464  /* nb := #{entries} in a slice such as x[1,1,..] or equivalently,
465  * the number of matrix slices x[ , , *, ..] which
466  * are printed as matrices -- if options("max.print") allows */
467  for (i = 2, nb = 1; i < ndim; i++)
468  nb *= dims[i];
469  max_reached = (b > 0 && R_print.max / b < nb);
470  if (max_reached) { /* i.e., also b > 0, nr > 0, nc > 0, nb > 0 */
471  /* nb_pr := the number of matrix slices to be printed */
472  nb_pr = ceil_DIV(R_print.max, b);
473  /* for the last, (nb_pr)th matrix slice, use only nr_last rows;
474  * using floor(), not ceil(), since 'nc' could be huge: */
475  nr_last = (R_print.max - b * (nb_pr - 1)) / nc;
476  if(nr_last == 0) { nb_pr--; nr_last = nr; }
477  } else {
478  nb_pr = (nb > 0) ? nb : 1; // do print *something* when dim = c(a,b,0)
479  nr_last = nr;
480  }
481  for (i = 0; i < nb_pr; i++) {
482  Rboolean do_ij = Rboolean(nb > 0);
483  Rboolean i_last = Rboolean(i == nb_pr - 1); /* for the last slice */
484  int use_nr = i_last ? nr_last : nr;
485  if(do_ij) {
486  int k = 1;
487  Rprintf(", ");
488  for (j = 2 ; j < ndim; j++) {
489  int l = (i / k) % dims[j] + 1;
490  if (has_dimnames &&
491  ((dn = VECTOR_ELT(dimnames, j)) != R_NilValue)) {
492  if ( has_dnn )
493  Rprintf(", %s = %s",
494  translateChar(STRING_ELT(dnn, j)),
495  translateChar(STRING_ELT(dn, l - 1)));
496  else
497  Rprintf(", %s", translateChar(STRING_ELT(dn, l - 1)));
498  } else
499  Rprintf(", %d", l);
500  k *= dims[j];
501  }
502  Rprintf("\n\n");
503  } else { // nb == 0 -- e.g. <2 x 3 x 0 array of logical>
504  for (i = 0; i < ndim; i++)
505  Rprintf("%s%d", (i == 0) ? "<" : " x ", dims[i]);
506  Rprintf(" array of %s>\n", CHAR(type2str_nowarn(TYPEOF(x))));
507  }
508  switch (TYPEOF(x)) {
509  case LGLSXP:
510  printLogicalMatrix(x, i * b, use_nr, nr, nc, dn0, dn1, rn, cn, do_ij);
511  break;
512  case INTSXP:
513  printIntegerMatrix(x, i * b, use_nr, nr, nc, dn0, dn1, rn, cn, do_ij);
514  break;
515  case REALSXP:
516  printRealMatrix (x, i * b, use_nr, nr, nc, dn0, dn1, rn, cn, do_ij);
517  break;
518  case CPLXSXP:
519  printComplexMatrix(x, i * b, use_nr, nr, nc, dn0, dn1, rn, cn, do_ij);
520  break;
521  case STRSXP:
522  {
523  if (quote) quote = '"';
524  const StringVector* sv = SEXP_downcast<StringVector*>(x);
525  printStringMatrix (sv, i * b, use_nr, nr, nc,
526  quote, right, dn0, dn1, rn, cn, do_ij);
527  break;
528  }
529  case RAWSXP:
530  printRawMatrix (x, i * b, use_nr, nr, nc, dn0, dn1, rn, cn, do_ij);
531  break;
532  default: error(_("invalid type")); break;
533  }
534  Rprintf("\n");
535  }
536 
537  if(max_reached && nb_pr < nb) {
538  Rprintf(" [ reached getOption(\"max.print\") -- omitted");
539  if(nr_last < nr) Rprintf(" %d row(s) and", nr - nr_last);
540  Rprintf(" %d matrix slice(s) ]\n", nb - nb_pr);
541  }
542  }
543  vmaxset(vmax);
544 }
