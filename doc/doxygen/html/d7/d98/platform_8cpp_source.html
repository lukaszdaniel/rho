238 /* retrieves information about the current locale and
239  sets the corresponding variables (known_to_be_utf8,
240  known_to_be_latin1, utf8locale, latin1locale and mbcslocale) */
241 void attribute_hidden R_check_locale(void)
242 {
243  known_to_be_utf8 = utf8locale = FALSE;
244  known_to_be_latin1 = latin1locale = FALSE;
245  mbcslocale = FALSE;
246 #ifdef HAVE_LANGINFO_CODESET
247  {
248  char *p = nl_langinfo(CODESET);
249  /* more relaxed due to Darwin: CODESET is case-insensitive and
250  latin1 is ISO8859-1 */
251  if (R_strieql(p, "UTF-8")) known_to_be_utf8 = utf8locale = TRUE;
252  if (streql(p, "ISO-8859-1")) known_to_be_latin1 = latin1locale = TRUE;
253  if (R_strieql(p, "ISO8859-1")) known_to_be_latin1 = latin1locale = TRUE;
254 # if __APPLE__
255  /* On Darwin 'regular' locales such as 'en_US' are UTF-8 (hence
256  MB_CUR_MAX == 6), but CODESET is "" */
257  if (*p == 0 && MB_CUR_MAX == 6)
258  known_to_be_utf8 = utf8locale = TRUE;
259 # endif
260  }
261 #endif
262  mbcslocale = RHOCONSTRUCT(Rboolean, MB_CUR_MAX > 1);
263 #ifdef Win32
264  {
265  char *ctype = setlocale(LC_CTYPE, NULL), *p;
266  p = strrchr(ctype, '.');
267  if (p && isdigit(p[1])) localeCP = atoi(p+1); else localeCP = 0;
268  /* Not 100% correct, but CP1252 is a superset */
269  known_to_be_latin1 = latin1locale = (localeCP == 1252);
270  }
271 #endif
272 #if defined(SUPPORT_UTF8_WIN32) /* never at present */
273  utf8locale = mbcslocale = TRUE;
274 #endif
275 }

311 {
312  SEXP fn, tl, hd, pg;
313  const char **f, **h, *t, *pager = nullptr /* -Wall */;
314  Rboolean dl;
315  int i, n;
316 
317  fn = files_;
318  hd = header_;
319  tl = title_;
320  dl = Rboolean(Rf_asLogical(delete_file_));
321  pg = pager_;
322  n = 0; /* -Wall */
323  if (!Rf_isString(fn) || (n = Rf_length(fn)) < 1)
324  Rf_error(_("invalid filename specification"));
325  if (!Rf_isString(hd) || Rf_length(hd) != n)
326  Rf_error(_("invalid '%s' argument"), "headers");
327  if (!Rf_isString(tl))
328  Rf_error(_("invalid '%s' argument"), "title");
329  if (!Rf_isString(pg))
330  Rf_error(_("invalid '%s' argument"), "pager");
331  f = static_cast<const char**>(RHO_alloc(n, sizeof(char*)));
332  h = static_cast<const char**>(RHO_alloc(n, sizeof(char*)));
333  for (i = 0; i < n; i++) {
334  SEXP el = STRING_ELT(fn, i);
335  if (!Rf_isNull(el) && el != NA_STRING)
336 #ifdef Win32
337  f[i] = Rf_acopy_string(reEnc(R_CHAR(el), Rf_getCharCE(el), CE_UTF8, 1));
338 #else
339  f[i] = Rf_acopy_string(Rf_translateChar(el));
340 #endif
341  else
342  Rf_error(_("invalid filename specification"));
343  if (STRING_ELT(hd, i) != NA_STRING)
344  h[i] = Rf_acopy_string(Rf_translateChar(STRING_ELT(hd, i)));
345  else
346  Rf_error(_("invalid '%s' argument"), "headers");
347  }
348  if (Rf_isValidStringF(tl))
349  t = Rf_acopy_string(Rf_translateChar(STRING_ELT(tl, 0)));
350  else
351  t = "";
352  if (Rf_isValidStringF(pg)) {
353  SEXP pg0 = STRING_ELT(pg, 0);
354  if (pg0 != NA_STRING)
355  pager = Rf_acopy_string(R_CHAR(pg0));
356  else
357  Rf_error(_("invalid '%s' argument"), "pager");
358  } else
359  pager = "";
360  R_ShowFiles(n, f, h, t, dl, pager);
361  return R_NilValue;
362 }

379 static int R_AppendFile(SEXP file1, SEXP file2)
380 {
381  FILE *fp1, *fp2;
382  char buf[APPENDBUFSIZE];
383  size_t nchar;
384  int status = 0;
385  if ((fp1 = RC_fopen(file1, "ab", TRUE)) == nullptr) return 0;
386  if ((fp2 = RC_fopen(file2, "rb", TRUE)) == nullptr) {
387  fclose(fp1);
388  return 0;
389  }
390  while ((nchar = fread(buf, 1, APPENDBUFSIZE, fp2)) == APPENDBUFSIZE)
391  if (fwrite(buf, 1, APPENDBUFSIZE, fp1) != APPENDBUFSIZE) goto append_error;
392  if (fwrite(buf, 1, nchar, fp1) != nchar) goto append_error;
393  status = 1;
394  append_error:
395  if (status == 0) Rf_warning(_("write error during file append"));
396  fclose(fp1);
397  fclose(fp2);
398  return status;
399 }

401 SEXP attribute_hidden do_fileappend(/*const*/ Expression* call, const BuiltInFunction* op, RObject* file1_, RObject* file2_)
402 {
403  SEXP f1, f2, ans;
404  int n, n1, n2;
405 
406  f1 = file1_; n1 = Rf_length(f1);
407  f2 = file2_; n2 = Rf_length(f2);
408  if (!Rf_isString(f1))
409  Rf_error(_("invalid '%s' argument"), "file1");
410  if (!Rf_isString(f2))
411  Rf_error(_("invalid '%s' argument"), "file2");
412  if (n1 < 1)
413  Rf_error(_("nothing to append to"));
414  if (n2 < 1)
415  return Rf_allocVector(LGLSXP, 0);
416  n = (n1 > n2) ? n1 : n2;
417  PROTECT(ans = Rf_allocVector(LGLSXP, n));
418  for (int i = 0; i < n; i++) LOGICAL(ans)[i] = 0; /* all FALSE */
419  if (n1 == 1) { /* common case */
420  FILE *fp1, *fp2;
421  char buf[APPENDBUFSIZE];
422  size_t nchar;
423  int status = 0;
424  if (STRING_ELT(f1, 0) == NA_STRING ||
425  !(fp1 = RC_fopen(STRING_ELT(f1, 0), "ab", TRUE)))
426  goto done;
427  for (int i = 0; i < n; i++) {
428  status = 0;
429  if (STRING_ELT(f2, i) == NA_STRING ||
430  !(fp2 = RC_fopen(STRING_ELT(f2, i), "rb", TRUE))) continue;
431  while ((nchar = fread(buf, 1, APPENDBUFSIZE, fp2)) == APPENDBUFSIZE)
432  if (fwrite(buf, 1, APPENDBUFSIZE, fp1) != APPENDBUFSIZE)
433  goto append_error;
434  if (fwrite(buf, 1, nchar, fp1) != nchar) goto append_error;
435  status = 1;
436  append_error:
437  if (status == 0)
438  Rf_warning(_("write error during file append"));
439  LOGICAL(ans)[i] = status;
440  fclose(fp2);
441  }
442  fclose(fp1);
443  } else {
444  for (int i = 0; i < n; i++) {
445  if (STRING_ELT(f1, i%n1) == R_NilValue ||
446  STRING_ELT(f2, i%n2) == R_NilValue)
447  LOGICAL(ans)[i] = 0;
448  else
449  LOGICAL(ans)[i] =
450  R_AppendFile(STRING_ELT(f1, i%n1), STRING_ELT(f2, i%n2));
451  }
452  }
453 done:
454  UNPROTECT(1);
455  return ans;
456 }
