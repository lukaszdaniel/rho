87 /* ExtractSubset allocates "result" and does the transfer of elements
88  from "x" to "result" according to the integer/real subscripts given
89  in "indx".
90 
91  The EXTRACT_SUBSET_LOOP macro allows the branches based on index
92  type and vector type to happen outside the loop.
93 
94  This could avoid using data pointers, but there is little point as
95  currently the subscript code forces allocation.
96 */
97 
98 #define EXTRACT_SUBSET_LOOP(STDCODE, NACODE) do { \
99  if (TYPEOF(indx) == INTSXP) { \
100  const int *pindx = INTEGER_RO(indx); \
101  for (i = 0; i < n; i++) { \
102  ii = pindx[i]; \
103  if (0 < ii && ii <= nx) { \
104  ii--; \
105  STDCODE; \
106  } \
107  else /* out of bounds or NA */ \
108  NACODE; \
109  } \
110  } \
111  else { \
112  const double *pindx = REAL_RO(indx); \
113  for (i = 0; i < n; i++) { \
114  double di = pindx[i]; \
115  ii = (R_xlen_t) (di - 1); \
116  if (std::isfinite(di) && \
117  0 <= ii && ii < nx) \
118  STDCODE; \
119  else \
120  NACODE; \
121  } \
122  } \
123  } while (0)

125 HIDDEN SEXP Rf_ExtractSubset(SEXP x, SEXP indx, SEXP call)
126 {
127  if (x == nullptr)
128  return x;
129 
130  SEXP result;
131 
132  if (ALTREP(x)) {
133  result = ALTVEC_EXTRACT_SUBSET(x, indx, call);
134  if (result != nullptr)
135  return result;
136  }
137 
138  R_xlen_t i, ii, n, nx;
139  n = XLENGTH(indx);
140  nx = Rf_xlength(x);
141  SEXPTYPE mode = TYPEOF(x);
142 
143  /* protect allocation in case _ELT operations need to allocate */
144  PROTECT(result = Rf_allocVector(mode, n));
145  switch(mode) {
146  case LGLSXP:
147  EXTRACT_SUBSET_LOOP(LOGICAL0(result)[i] = Rboolean(LOGICAL_ELT(x, ii)),
148  LOGICAL0(result)[i] = Rboolean(R_NaInt));
149  break;
150  case INTSXP:
151  EXTRACT_SUBSET_LOOP(INTEGER0(result)[i] = INTEGER_ELT(x, ii),
152  INTEGER0(result)[i] = R_NaInt);
153  break;
154  case REALSXP:
155  EXTRACT_SUBSET_LOOP(REAL0(result)[i] = REAL_ELT(x, ii),
156  REAL0(result)[i] = R_NaReal);
157  break;
158  case CPLXSXP:
159  {
160  Rcomplex NA_CPLX = { R_NaReal, R_NaReal };
161  EXTRACT_SUBSET_LOOP(COMPLEX0(result)[i] = COMPLEX_ELT(x, ii),
162  COMPLEX0(result)[i] = NA_CPLX);
163  }
164  break;
165  case STRSXP:
166  EXTRACT_SUBSET_LOOP(SET_STRING_ELT(result, i, STRING_ELT(x, ii)),
167  SET_STRING_ELT(result, i, R_NaString));
168  break;
169  case VECSXP:
170  EXTRACT_SUBSET_LOOP(SET_VECTOR_ELT(result, i,
171  VECTOR_ELT_FIX_NAMED(x, ii)),
172  SET_VECTOR_ELT(result, i, nullptr));
173  break;
174  case EXPRSXP:
175  EXTRACT_SUBSET_LOOP(SET_XVECTOR_ELT(result, i,
176  XVECTOR_ELT_FIX_NAMED(x, ii)),
177  SET_XVECTOR_ELT(result, i, nullptr));
178  break;
179  case RAWSXP:
180  EXTRACT_SUBSET_LOOP(RAW0(result)[i] = RAW_ELT(x, ii),
181  RAW0(result)[i] = (Rbyte) 0);
182  break;
183  case LISTSXP:
184  /* cannot happen: pairlists are coerced to lists */
185  case LANGSXP:
186  /* cannot happen: LANGSXPs are coerced to lists */
187  default:
188  Rf_errorcall(call, _("object of type '%s' is not subsettable"), Rf_type2char(SEXPTYPE(mode)));
189  }
190  UNPROTECT(1); /* result */
191  return result;
192 }

197 static SEXP VectorSubset(SEXP x, SEXP sarg, SEXP call)
198 {
199  if (!x)
200  return nullptr;
201  GCStackRoot<> s(sarg);
202 
203  if (s == R_MissingArg)
204  return Rf_duplicate(x);
205 
206  /* Check to see if we have special matrix subscripting. */
207  /* If we do, make a real subscript vector and protect it. */
208  {
209  SEXP attrib = Rf_getAttrib(x, Symbols::DimSymbol);
210  if (Rf_isMatrix(s) && Rf_isArray(x) && Rf_ncols(s) == Rf_length(attrib)) {
211  if (Rf_isString(s)) {
212  s = Rf_strmat2intmat(s, Rf_GetArrayDimnames(x), call);
213  }
214  if (Rf_isInteger(s) || Rf_isReal(s)) {
215  s = Rf_mat2indsub(attrib, s, call);
216  }
217  }
218  }
219 
220  SEXPTYPE mode = TYPEOF(x);
221  switch (mode) {
222  case LGLSXP:
223  return Subscripting::vectorSubset(SEXP_downcast<LogicalVector*>(x), s);
224  case INTSXP:
225  return Subscripting::vectorSubset(SEXP_downcast<IntVector*>(x), s);
226  case REALSXP:
227  return Subscripting::vectorSubset(SEXP_downcast<RealVector*>(x), s);
228  case CPLXSXP:
229  return Subscripting::vectorSubset(SEXP_downcast<ComplexVector*>(x), s);
230  case RAWSXP:
231  return Subscripting::vectorSubset(SEXP_downcast<RawVector*>(x), s);
232  case STRSXP:
233  return Subscripting::vectorSubset(SEXP_downcast<StringVector*>(x), s);
234  case VECSXP:
235  return Subscripting::vectorSubset(SEXP_downcast<ListVector*>(x), s);
236  case EXPRSXP:
237  return Subscripting::vectorSubset(SEXP_downcast<ExpressionVector*>(x), s);
238  case LANGSXP:
239  break;
240  default:
241  Rf_errorcall(call, _("object of type '%s' is not subsettable"), Rf_type2char(SEXPTYPE(mode)));
242  }
243 
244  // If we get to here, this must be a LANGSXP. In rho, this case
245  // needs special handling, not least because Expression doesn't
246  // inherit from VectorBase. What follows is legacy CR code,
247  // bodged as necessary.
248 
249  /* Convert to a vector of integer subscripts */
250  /* in the range 1:length(x). */
251  R_xlen_t stretch = 1;
252  GCStackRoot<> indx(Rf_makeSubscript(x, s, &stretch, call));
253  GCStackRoot<> result(Rf_ExtractSubset(x, indx, call));
254  if (mode == VECSXP || mode == EXPRSXP)
255  /* we do not duplicate the values when extracting the subset,
256  so to be conservative mark the result as NAMED = NAMEDMAX */
257  ENSURE_NAMEDMAX(result);
258 
259  // Fix attributes:
260  {
261  SEXP attrib;
262  if (
263  ((attrib = Rf_getAttrib(x, Symbols::NamesSymbol)) != nullptr) ||
264  ( /* here we might have an array. Use row names if 1D */
265  Rf_isArray(x)
266  && (attrib = Rf_getAttrib(x, Symbols::DimNamesSymbol)) != nullptr
267  && Rf_length(attrib) == 1
268  && (attrib = Rf_GetRowNames(attrib)) != nullptr
269  )
270  ) {
271  GCStackRoot<> nattrib(Rf_ExtractSubset(attrib, indx, call));
272  Rf_setAttrib(result, Symbols::NamesSymbol, nattrib);
273  }
274  if ((attrib = Rf_getAttrib(x, Symbols::SrcrefSymbol)) != nullptr &&
275  TYPEOF(attrib) == VECSXP) {
276  GCStackRoot<> nattrib(Rf_ExtractSubset(attrib, indx, call));
277  Rf_setAttrib(result, Symbols::SrcrefSymbol, nattrib);
278  }
279  }
280  return result;
281 }

284 static SEXP ArraySubset(SEXP x, SEXP s, SEXP call, int drop)
285 {
286  const PairList* subs = SEXP_downcast<PairList*>(s);
287  switch (x->sexptype()) {
288  case LGLSXP:
289  return Subscripting::arraySubset(SEXP_downcast<LogicalVector*>(x),
290  subs, drop);
291  case INTSXP:
292  return Subscripting::arraySubset(SEXP_downcast<IntVector*>(x),
293  subs, drop);
294  case REALSXP:
295  return Subscripting::arraySubset(SEXP_downcast<RealVector*>(x),
296  subs, drop);
297  case CPLXSXP:
298  return Subscripting::arraySubset(SEXP_downcast<ComplexVector*>(x),
299  subs, drop);
300  case STRSXP:
301  return Subscripting::arraySubset(SEXP_downcast<StringVector*>(x),
302  subs, drop);
303  case VECSXP:
304  return Subscripting::arraySubset(SEXP_downcast<ListVector*>(x),
305  subs, drop);
306  case RAWSXP:
307  return Subscripting::arraySubset(SEXP_downcast<RawVector*>(x),
308  subs, drop);
309  default:
310  Rf_errorcall(call, _("array subscripting not handled for this type"));
311  }
312  return nullptr; // -Wall
313 }

316 /* Returns and removes a named argument from argument list args.
317  The search ends as soon as a matching argument is found. If
318  the argument is not found, the argument list is not modified
319  and nullptr is returned.
320  */
321 static SEXP ExtractArg(rho::RObject* args, rho::Symbol* arg_sym)
322 {
323  SEXP arg, prev_arg;
324  int found = 0;
325 
326  for (arg = prev_arg = args; arg != nullptr; arg = CDR(arg)) {
327  if(TAG(arg) == arg_sym) {
328  if (arg == prev_arg) /* found at head of args */
329  args = CDR(args);
330  else
331  SETCDR(prev_arg, CDR(arg));
332  found = 1;
333  break;
334  }
335  else prev_arg = arg;
336  }
337  return found ? CAR(arg) : nullptr;
338 }

350 /* Extracts and, if present, removes the 'exact' argument from the
351  argument list. An integer code giving the desired exact matching
352  behavior is returned:
353  0 not exact
354  1 exact
355  -1 not exact, but warn when partial matching is used
356  */
357 static int ExtractExactArg(SEXP args)
358 {
359  SEXP argval = ExtractArg(args, rho::Symbol::obtain("exact"));
360  int exact;
361  if(Rf_isNull(argval)) return 1; /* Default is true as from R 2.7.0 */

363  if (exact == R_NaLog) exact = -1;

368 HIDDEN SEXP do_subset(SEXP call, SEXP op, SEXP args, SEXP rho)
369 {
370  const Expression* expression = SEXP_downcast<Expression*>(call);
371  const BuiltInFunction* function = SEXP_downcast<BuiltInFunction*>(op);
372  Environment* envx = SEXP_downcast<Environment*>(rho);
373  ArgList arglist(SEXP_downcast<PairList*>(args), ArgList::RAW);
374  auto dispatched = Rf_DispatchOrEval(expression, function, &arglist, envx,
375  MissingArgHandling::Keep);
376  if (dispatched.first) {
377  RObject* ans = dispatched.second;
378  if (NAMED(ans))
379  ENSURE_NAMEDMAX(ans);
380  return(ans);
381  }
382 
383  /* Method dispatch has failed, we now */
384  /* run the generic internal code. */
385  return BuiltInFunction::callBuiltInWithCApi(
386  do_subset_dflt, expression, function, arglist, envx);
387 }

389 R_INLINE static R_xlen_t scalarIndex(SEXP s)
390 {
391  if (ATTRIB(s) == nullptr)
392  switch (TYPEOF(s)) {
393  case INTSXP:
394  {
395  int ival = SCALAR_IVAL(s);
396  if (XLENGTH(s) == 1 && ival != R_NaInt)
397  return ival;
398  else return -1;
399  }
400  case REALSXP:
401  {
402  double rval = SCALAR_DVAL(s);
403  // treat infinite indices as NA, like asInteger
404  if (XLENGTH(s) == 1 && std::isfinite(rval))
405  return R_xlen_t(rval);
406  else return -1;
407  }
408  default: return -1;
409  }
410  else return -1;
411 }

413 HIDDEN SEXP do_subset_dflt(SEXP call, SEXP op, SEXP args, SEXP rho)
414 {
415  GCStackRoot<> argsrt(args);
416 
417  /* By default we drop extents of length 1 */
418 
419  /* Handle cases of extracting a single element from a simple vector
420  or matrix directly to improve speed for these simple cases. */
421  SEXP cdrArgs = CDR(args);
422  SEXP cddrArgs = CDR(cdrArgs);
423  if (cdrArgs != nullptr && cddrArgs == nullptr &&
424  TAG(cdrArgs) == nullptr) {
425  /* one index, not named */
426  SEXP x = CAR(args);
427  if (ATTRIB(x) == nullptr) {
428  SEXP s = CAR(cdrArgs);
429  R_xlen_t i = scalarIndex(s);
430  switch (TYPEOF(x)) {
431  case REALSXP:
432  if (i >= 1 && i <= XLENGTH(x))
433  return Rf_ScalarReal( REAL_ELT(x, i-1) );
434  break;
435  case INTSXP:
436  if (i >= 1 && i <= XLENGTH(x))
437  return Rf_ScalarInteger( INTEGER_ELT(x, i-1) );
438  break;
439  case LGLSXP:
440  if (i >= 1 && i <= XLENGTH(x))
441  return Rf_ScalarLogical( LOGICAL_ELT(x, i-1) );
442  break;
443 // do the more rare cases as well, since we've already prepared everything:
444  case CPLXSXP:
445  if (i >= 1 && i <= XLENGTH(x))
446  return Rf_ScalarComplex( COMPLEX_ELT(x, i-1) );
447  break;
448  case RAWSXP:
449  if (i >= 1 && i <= XLENGTH(x))
450  return Rf_ScalarRaw( RAW_ELT(x, i-1) );
451  break;
452  default: break;
453  }
454  }
455  }
456  else if (cddrArgs != nullptr && CDR(cddrArgs) == nullptr &&
457  TAG(cdrArgs) == nullptr && TAG(cddrArgs) == nullptr) {
458  /* two indices, not named */
459  SEXP x = CAR(args);
460  SEXP attr = ATTRIB(x);
461  if (TAG(attr) == Symbols::DimSymbol && CDR(attr) == nullptr) {
462  /* only attribute of x is 'dim' */
463  SEXP dim = CAR(attr);
464  if (TYPEOF(dim) == INTSXP && LENGTH(dim) == 2) {
465  /* x is a matrix */
466  SEXP si = CAR(cdrArgs);
467  SEXP sj = CAR(cddrArgs);
468  R_xlen_t i = scalarIndex(si);
469  R_xlen_t j = scalarIndex(sj);
470  int nrow = INTEGER_ELT(dim, 0);
471  int ncol = INTEGER_ELT(dim, 1);
472  if (i > 0 && j > 0 && i <= nrow && j <= ncol) {
473  /* indices are legal scalars */
474  R_xlen_t k = i - 1 + nrow * (j - 1);
475  switch (TYPEOF(x)) {
476  case REALSXP:
477  if (k < XLENGTH(x))
478  return Rf_ScalarReal( REAL_ELT(x, k) );
479  break;
480  case INTSXP:
481  if (k < XLENGTH(x))
482  return Rf_ScalarInteger( INTEGER_ELT(x, k) );
483  break;
484  case LGLSXP:
485  if (k < XLENGTH(x))
486  return Rf_ScalarLogical( LOGICAL_ELT(x, k) );
487  break;
488  case CPLXSXP:
489  if (k < XLENGTH(x))
490  return Rf_ScalarComplex( COMPLEX_ELT(x, k) );
491  break;
492  case RAWSXP:
493  if (k < XLENGTH(x))
494  return Rf_ScalarRaw( RAW_ELT(x, k) );
495  break;
496  default: break;
497  }
498  }
499  }
500  }
501  }
502 
503  int drop = 1;
504  ExtractDropArg(args, drop);
505  SEXP x = CAR(args);
506 
507  /* This was intended for compatibility with S, */
508  /* but in fact S does not do this. */
509  /* FIXME: replace the test by isNull ... ? */
510 
511  if (x == nullptr) {
512  return x;
513  }
514  SEXP subs = CDR(args);
515  size_t nsubs = Rf_length(subs);
516 
517  /* Here coerce pair-based objects into generic vectors. */
518  /* All subsetting takes place on the generic vector form. */
519 
520  GCStackRoot<> ax(x);
521  if (Rf_isPairList(x)) {
522  SEXP dim = Rf_getAttrib(x, Symbols::DimSymbol);
523  int ndim = Rf_length(dim);
524  if (ndim > 1) {
525  ax = Rf_allocArray(VECSXP, dim);
526  Rf_setAttrib(ax, Symbols::DimNamesSymbol, Rf_getAttrib(x, Symbols::DimNamesSymbol));
527  Rf_setAttrib(ax, Symbols::NamesSymbol, Rf_getAttrib(x, Symbols::DimNamesSymbol));
528  }
529  else {
530  ax = Rf_allocVector(VECSXP, Rf_length(x));
531  Rf_setAttrib(ax, Symbols::NamesSymbol, Rf_getAttrib(x, Symbols::NamesSymbol));
532  }
533  int i = 0;
534  for (SEXP px = x; px != nullptr; px = CDR(px))
535  SET_VECTOR_ELT(ax, i++, CAR(px));
536  }
537  else if (!Rf_isVector(x))
538  Rf_errorcall(call, _("object of type '%s' is not subsettable"), Rf_type2char(TYPEOF(x)));
539 
540  /* This is the actual subsetting code. */
541 
542  GCStackRoot<> ans;
543  SEXP sub1 = CAR(subs); // null if nsubs == 0
544  if (nsubs == 1 && sub1 == R_MissingArg) { // Handle x[] correctly.
545  nsubs = 0;
546  sub1 = nullptr;
547  }
548 
549  const IntVector* dims = SEXP_downcast<VectorBase*>(ax.get())->dimensions();
550  if (dims) {
551  size_t ndim = dims->size();
552  // Check for single matrix subscript:
553  if (nsubs == 1 && Rf_isMatrix(sub1)
554  && Rf_isArray(ax) && Rf_ncols(sub1) == int(ndim))
555  ans = VectorSubset(ax, sub1, call);
556  else if (ndim == nsubs) // regular array subscripting, inc. 1-dim
557  ans = ArraySubset(ax, subs, call, drop);
558  }
559  if (!ans) {
560  if (nsubs < 2) // vector subscripting
561  ans = VectorSubset(ax, (nsubs == 1 ? sub1 : R_MissingArg), call);
562  else Rf_errorcall(call, _("incorrect number of dimensions"));
563  }
564 
565  /* Note: we do not coerce back to pair-based lists. */
566  /* They are "defunct" in this version of R. */
567 
568  if (TYPEOF(x) == LANGSXP) {
569  ax = ans;
570  ans = nullptr;
571  size_t len = LENGTH(ax);
572  if (len > 0) {
573  GCStackRoot<PairList> tl(PairList::make(len - 1));
574  ans = new CachingExpression(nullptr, tl);
575 
576  int i = 0;
577  for (SEXP px = ans; px != nullptr; px = CDR(px))
578  SETCAR(px, VECTOR_ELT(ax, i++));
579  Rf_setAttrib(ans, Symbols::DimSymbol, Rf_getAttrib(ax, Symbols::DimSymbol));
580  Rf_setAttrib(
581  ans, Symbols::DimNamesSymbol, Rf_getAttrib(ax, Symbols::DimNamesSymbol));
582  Rf_setAttrib(ans, Symbols::NamesSymbol, Rf_getAttrib(ax, Symbols::NamesSymbol));
583  RAISE_NAMED(ans, NAMED(ax)); /* PR#7924 */
584  }
585  }
586  if (ATTRIB(ans) != nullptr) { /* remove probably erroneous attr's */
587  Rf_setAttrib(ans, Symbols::TspSymbol, nullptr);
588 #ifdef _S4_subsettable
589  if(!IS_S4_OBJECT(x))
590 #endif
591  Rf_setAttrib(ans, Symbols::ClassSymbol, nullptr);
592  }
593  return ans;
594 }

598 
599 HIDDEN SEXP do_subset2(SEXP call, SEXP op, SEXP args, SEXP rho)
600 {
601  const Expression* expression = SEXP_downcast<Expression*>(call);
602  const BuiltInFunction* function = SEXP_downcast<BuiltInFunction*>(op);
603  Environment* envx = SEXP_downcast<Environment*>(rho);
604  ArgList arglist(SEXP_downcast<PairList*>(args), ArgList::RAW);

618 HIDDEN SEXP do_subset2_dflt(SEXP call, SEXP op,
619  SEXP argsarg, SEXP rho)
620 {
621  GCStackRoot<> args(argsarg);
622  SEXP ans;
623  int drop = 1;
624  ExtractDropArg(args, drop);
625  /* Is partial matching ok? When the exact arg is NA, a warning is
626  issued if partial matching occurs.
627  */
628  int exact = ExtractExactArg(args);
629  int pok;
630  if (exact == -1)
631  pok = exact;
632  else
633  pok = !exact;
634 
635  SEXP x = CAR(args);
636 
637  /* This code was intended for compatibility with S, */
638  /* but in fact S does not do this. Will anyone notice? */
639  if (!x)
640  return nullptr;
641 
642  /* Get the subscripting and dimensioning information */
643  /* and check that any array subscripting is compatible. */
644 
645  SEXP subs = CDR(args);
646  int nsubs = Rf_length(subs);
647  if (nsubs == 0)
648  Rf_errorcall(call, _("no index specified"));
649  SEXP dims = Rf_getAttrib(x, Symbols::DimSymbol);
650  int ndims = Rf_length(dims);
651  if (nsubs > 1 && nsubs != ndims)
652  Rf_errorcall(call, _("incorrect number of subscripts"));
653 
654  /* code to allow classes to extend environment */
655  if (TYPEOF(x) == S4SXP) {
656  x = R_getS4DataSlot(x, ANYSXP);
657  if (x == nullptr)
658  Rf_errorcall(call, _("this S4 class is not subsettable"));
659  }
660 
661  /* split out ENVSXP for now */
662  if ( TYPEOF(x) == ENVSXP ) {
663  if( nsubs != 1 || !Rf_isString(CAR(subs)) || Rf_length(CAR(subs)) != 1 )
664  Rf_errorcall(call, _("wrong arguments for subsetting an environment"));
665  GCStackRoot<>
666  ans(Rf_findVarInFrame(x,
667  rho::Symbol::obtain(Rf_translateChar(STRING_ELT(CAR(subs),
668  0)))));
669  if ( TYPEOF(ans) == PROMSXP )
670  ans = Rf_eval(ans, R_GlobalEnv);
671  else ENSURE_NAMEDMAX(ans);
672 
673  if (ans == R_UnboundValue )
674  return(nullptr);
675  if (NAMED(ans))
676  ENSURE_NAMEDMAX(ans);
677  return ans;
678  }
679 
680  /* back to the regular program */
681  if (!(Rf_isVector(x) || Rf_isList(x) || Rf_isLanguage(x)))
682  Rf_errorcall(call, _("object of type '%s' is not subsettable"), Rf_type2char(TYPEOF(x)));
683 
684  int named_x = NAMED(x); /* x may change below; save this now. See PR#13411 */
685 
686  R_xlen_t offset = 0;
687  if (nsubs == 1) { /* vector indexing */
688  SEXP thesub = CAR(subs);
689  int len = Rf_length(thesub);
690 
691  if (len > 1) {
692 #ifdef SWITCH_TO_REFCNT
693  if (IS_GETTER_CALL(call)) {
694  /* this is (most likely) a getter call in a complex
695  assighment so we duplicate as needed. The original
696  x should have been duplicated if it might be
697  shared */
698  if (MAYBE_SHARED(x))
699  Rf_error("getter call used outside of a complex assignment.");
700  x = vectorIndex(x, thesub, 0, len-1, pok, call, TRUE);
701  }
702  else
703  x = vectorIndex(x, thesub, 0, len-1, pok, call, FALSE);
704 #else
705  x = vectorIndex(x, thesub, 0, len-1, pok, call, FALSE);
706 #endif
707  named_x = NAMED(x);
708  }
709 
710  offset = Rf_get1index(thesub, Rf_getAttrib(x, Symbols::NamesSymbol),
711  Rf_xlength(x), pok, len > 1 ? len-1 : -1, call);
712  if (offset < 0 || offset >= Rf_xlength(x)) {
713  /* a bold attempt to get the same behaviour for $ and [[ */
714  if (offset < 0 && (Rf_isNewList(x) ||
715  Rf_isExpression(x) ||
716  Rf_isList(x) ||
717  Rf_isLanguage(x)))
718  return nullptr;
719  else Rf_errorcall(call, _("subscript out of bounds"));
720  }
721  } else { /* matrix indexing */
722  /* Here we use the fact that: */
723  /* CAR(nullptr) = nullptr */
724  /* CDR(nullptr) = nullptr */
725 
726  int ndn; /* Number of dimnames. Unlikely to be anything but
727  0 or nsubs, but just in case... */
728 
729  GCStackRoot<> indx(Rf_allocVector(INTSXP, nsubs));
730  int *pindx = INTEGER(indx);
731  const int *pdims = INTEGER_RO(dims);
732  SEXP dimnames = Rf_getAttrib(x, Symbols::DimNamesSymbol);
733  ndn = Rf_length(dimnames);
734  for (int i = 0; i < nsubs; i++) {
735  pindx[i] = int(
736  Rf_get1index(CAR(subs),
737  (i < ndn) ? VECTOR_ELT(dimnames, i) : nullptr,
738  pindx[i], pok, -1, call));
739  subs = CDR(subs);
740  if (pindx[i] < 0 || pindx[i] >= pdims[i])
741  Rf_errorcall(call, _("subscript out of bounds"));
742  }
743  offset = 0;
744  for (int i = (nsubs - 1); i > 0; i--)
745  offset = (offset + pindx[i]) * pdims[i - 1];
746  offset += pindx[0];
747  }
748 
749  if (Rf_isPairList(x)) {
750 #ifdef LONG_VECTOR_SUPPORT
751  if (offset > R_SHORT_LEN_MAX)
752  Rf_error("invalid subscript for pairlist");
753 #endif
754  ans = CAR(Rf_nthcdr(x, int(offset)));
755  RAISE_NAMED(ans, named_x);
756  } else if (Rf_isVectorList(x)) {
757  /* did unconditional duplication before 2.4.0 */
758  if (x->sexptype() == EXPRSXP)
759  ans = XVECTOR_ELT(x, offset);
760  else ans = VECTOR_ELT(x, offset);
761  RAISE_NAMED(ans, named_x);
762  } else {
763  ans = Rf_allocVector(TYPEOF(x), 1);
764  switch (TYPEOF(x)) {
765  case LGLSXP:
766  LOGICAL0(ans)[0] = Rboolean(LOGICAL_ELT(x, offset));
767  break;
768  case INTSXP:
769  INTEGER0(ans)[0] = INTEGER_ELT(x, offset);
770  break;
771  case REALSXP:
772  REAL0(ans)[0] = REAL_ELT(x, offset);
773  break;
774  case CPLXSXP:
775  COMPLEX0(ans)[0] = COMPLEX_ELT(x, offset);
776  break;
777  case STRSXP:
778  SET_STRING_ELT(ans, 0, STRING_ELT(x, offset));
779  break;
780  case RAWSXP:
781  RAW0(ans)[0] = RAW_ELT(x, offset);
782  break;
783  default:
784  UNIMPLEMENTED_TYPE("do_subset2", x);
785  }
786  }
787  return ans;
788 }

915  PROTECT(input);

944  break;

966  return y;
967  }

970  else if (Rf_isVectorList(x)) {

994  break;

997  }

1003  switch (TYPEOF(target)) {

1009  break;
1010  default:
1011  throw std::logic_error("Unexpected SEXPTYPE");
1012  }
1013  Rf_warningcall(call, _("partial match of '%s' to '%s'"),
1014  Rf_translateChar(input), st);
1015  }
1016  y = VECTOR_ELT(x, imatch);
1017  RAISE_NAMED(y, NAMED(x));
1018  return y;
1019  }
1020  return nullptr;
1021  }
1022  else if( Rf_isEnvironment(x) ){

1025  PROTECT(y);
