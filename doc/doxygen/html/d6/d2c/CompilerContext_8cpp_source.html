80 FunctionBase* CompilerContext::staticallyResolveFunction(const Symbol* symbol)
81 {
82  // A symbol can be statically resolved if:
83  // - it is a control flow operator and AssumeSaneControlFlowOperators is set.
84  // - OR it is an assignment and AssumeSaneAssignementOperators is set.
85  // In both cases we still need to check for shadowing.
86  bool isSaneControlFlowOp
87  = (getOptimizationOptions().AssumeSaneControlFlowOperators
88  && controlFlowOperatorNames().count(symbol));
89  bool isSaneAssignmentOp
90  = (getOptimizationOptions().AssumeSaneAssignmentOperators
91  && assignmentOperatorNames().count(symbol));
92 
93  if (!isSaneControlFlowOp && !isSaneAssignmentOp) {
94  return nullptr;
95  }
96 
97  if (m_frame_descriptor->getLocation(symbol) != -1) {
98  // The symbol is shadowed in the local frame.
99  return nullptr;
100  }
101 
102  const Frame::Binding* binding
103  = getEnclosingEnvironment()->findBinding(symbol);
104  if (binding->frame() != Environment::base()->frame()
105  && binding->frame() != Environment::baseNamespace()->frame()) {
106  // Lookup returned a binding that isn't in base, so the symbol is
107  // shadowed somewhere else.
108  return nullptr;
109  }
110 
111  FunctionBase* builtin_definition = BuiltInFunction::obtainPrimitive(
112  symbol->name()->stdstring());
113  if (binding->rawValue() != builtin_definition) {
114  // The value in base has been changed.
115  return nullptr;
116  }
117 
118  return builtin_definition;
119 }

128 bool CompilerContext::canInlineControlFlow()
129 {
130  // Control flow can be inlined if all the control flow operators can be
131  // statically resolved.
132  for (const Symbol* symbol : controlFlowOperatorNames())
133  {
134  if (!staticallyResolveFunction(symbol))
135  {
136  return false;
137  }
138  }
139  return true;
140 }

159 void CompilerContext::pushLoopContext(BasicBlock* continue_block,
160  BasicBlock* loop_header,
161  Compiler* compiler)
162 {
163  // TODO(kmillar): this needs to set and reset m_environment->m_in_loop.
164  m_break_destinations.push(continue_block);
165  m_next_destinations.push(loop_header);
166 
167  // The interpreter handles 'next' and 'break' by throwing a LoopException.
168  // Setup an exception handler to handle the exception correctly.
169  PHINode* handler = compiler->emitLoopExceptionHandler(
170  continue_block, loop_header);
171 
172  pushExceptionHandlerContext(&typeid(LoopException), handler, compiler);
173 }

184 void CompilerContext::pushExceptionHandlerContext(const std::type_info* type,
185  PHINode* handler,
186  Compiler* compiler) {
187  PHINode* fallthrough = nullptr;
188  if (!m_exception_handlers.empty()) {
189  fallthrough = m_exception_handlers.top();
190  } else {
191  fallthrough = compiler->emitRethrowException();
192  }
193 
194  PHINode* dispatch = compiler->emitDispatchToExceptionHandler(type,
195  handler,
196  fallthrough);
197  BasicBlock* landing_pad = compiler->emitLandingPad(dispatch);
198  m_exception_handlers.push(dispatch);
199  m_exception_landing_pads.push(landing_pad);
200 }

202 void CompilerContext::popExceptionHandlerContext() {
203  assert(!m_exception_handlers.empty());
204  assert(!m_exception_landing_pads.empty());
205  m_exception_handlers.pop();
206  m_exception_landing_pads.pop();
