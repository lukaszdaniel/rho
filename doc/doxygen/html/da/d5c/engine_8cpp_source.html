159 /* Record the state and callback information for a new graphics
160  * system.
161  * This is called when a graphics system is loaded.
162  * Return the index of the system's information in the graphic
163  * engine's register.
164  */
165 void GEregisterSystem(GEcallback cb, int *systemRegisterIndex) {
166  int i, devNum;
167  GEDevDesc* gdd;
168  if (numGraphicsSystems + 1 == MAX_GRAPHICS_SYSTEMS)
169  Rf_error(_("too many graphics systems registered"));
170  /* Set the system register index so that, if there are existing
171  * devices, it will know where to put the system-specific
172  * information in those devices
173  * If a graphics system has been unregistered, there might
174  * be "holes" in the list of graphics systems, so start
175  * from zero and look for the first NULL
176  */
177  *systemRegisterIndex = 0;
178  while (registeredSystems[*systemRegisterIndex] != nullptr) {
179  (*systemRegisterIndex)++;
180  }
181  /* Run through the existing devices and add the new information
182  * to any GEDevDesc's
183  */
184  i = 1;
185  if (!NoDevices()) {
186  devNum = curDevice();
187  while (i++ < NumDevices()) {
188  gdd = GEgetDevice(devNum);
189  registerOne(gdd, *systemRegisterIndex, cb);
190  devNum = nextDevice(devNum);
191  }
192  }
193  /* Store the information for adding to any new devices
194  */
195  registeredSystems[*systemRegisterIndex] =
196  static_cast<GESystemDesc*>(calloc(1, sizeof(GESystemDesc)));
197  if (registeredSystems[*systemRegisterIndex] == nullptr)
198  Rf_error(_("unable to allocate memory (in GEregister)"));
199  registeredSystems[*systemRegisterIndex]->callback = cb;
200  numGraphicsSystems += 1;
201 }

208 void GEunregisterSystem(int registerIndex)
209 {
210  int i, devNum;
211  GEDevDesc* gdd;
212 
213  /* safety check if called before Ginit() */
214  if(registerIndex < 0) return;
215  if (numGraphicsSystems == 0) {
216  /* This gets called from KillAllDevices, which is called
217  during shutdown. Prior to 2.14.0 it gave an error, which
218  would inhibit shutdown. This should not happen, but
219  apparently it did after a segfault:
220  https://stat.ethz.ch/pipermail/r-devel/2011-June/061153.html
221  */
222  Rf_warning(_("no graphics system to unregister"));
223  return;
224  }
225  /* Run through the existing devices and remove the information
226  * from any GEDevDesc's
227  */
228  i = 1;
229  if (!NoDevices()) {
230  devNum = curDevice();
231  while (i++ < NumDevices()) {
232  gdd = GEgetDevice(devNum);
233  unregisterOne(gdd, registerIndex);
234  devNum = nextDevice(devNum);
235  }
236  }
237  /* Remove the information from the global record
238  * NOTE that there is no systemSpecific information stored
239  * in the global record -- just the system callback pointer.
240  */
241  if (registeredSystems[registerIndex] != nullptr) {
242  free(registeredSystems[registerIndex]);
243  registeredSystems[registerIndex] = nullptr;
244  }
245  numGraphicsSystems -= 1;
246 }

466  Rf_error(_("invalid line end"));

468  code = (code-1) % nlineend + 1;
469  return lineend[code].end;
470  }
471  else if(Rf_isReal(value)) {
472  rcode = REAL(value)[ind];
473  if(!std::isfinite(rcode) || rcode < 0)
474  Rf_error(_("invalid line end"));
475  code = int(rcode);
476  if (code > 0)
477  code = (code-1) % nlineend + 1;
478  return lineend[code].end;
479  }
480  else {
481  Rf_error(_("invalid line end")); /*NOTREACHED, for -Wall : */ return R_GE_lineend(0);

483 }

611 /****************************************************************
612  * GESetClip
613  ****************************************************************
614  */
615 void GESetClip(double x1, double y1, double x2, double y2, GEDevDesc* dd)
616 {
617  pDevDesc d = dd->dev;
618  double dx1 = d->left, dx2 = d->right, dy1 = d->bottom, dy2 = d->top;
619 
620  /* clip to device region */
621  if (dx1 <= dx2) {
622  x1 = std::max(x1, dx1);
623  x2 = std::min(x2, dx2);
624  } else {
625  x1 = std::min(x1, dx1);
626  x2 = std::max(x2, dx2);
627  }
628  if (dy1 <= dy2) {
629  y1 = std::max(y1, dy1);
630  y2 = std::min(y2, dy2);
631  } else {
632  y1 = std::min(y1, dy1);
633  y2 = std::max(y2, dy2);
634  }
635  d->clip(x1, x2, y1, y2, dd->dev);
636  /*
637  * Record the current clip rect settings so that calls to
638  * getClipRect get the up-to-date values.
639  */
640  d->clipLeft = std::min(x1, x2);
641  d->clipRight = std::max(x1, x2);
642  d->clipTop = std::max(y1, y2);
643  d->clipBottom = std::min(y1, y2);
644 }

685  cliprect *cr, int *clipped1, int *clipped2,

692  *clipped2 = 0;

939 static void intersect(Edge b, double x1, double y1, double x2, double y2,
940  double *ix, double *iy, GClipRect *clip)
941 {
942  double m = 0;
943 
944  if (x1 != x2) m = (y1 - y2) / (x1 - x2);
945  switch (b) {
946  case Left:
947  *ix = clip->xmin;
948  *iy = y2 + (clip->xmin - x2) * m;
949  break;
950  case Right:
951  *ix = clip->xmax;
952  *iy = y2 + (clip->xmax - x2) * m;
953  break;
954  case Bottom:
955  *iy = clip->ymin;
956  if (x1 != x2) *ix = x2 + (clip->ymin - y2) / m;
957  else *ix = x2;
958  break;
959  case Top:
960  *iy = clip->ymax;
961  if (x1 != x2) *ix = x2 + (clip->ymax - y2) / m;
962  else *ix = x2;
963  break;
964  }
965 }

967 static void clipPoint(Edge b, double x, double y,
968  double *xout, double *yout, int *cnt, int store,
969  GClipRect *clip, GClipState *cs)
970 {
971  double ix = 0.0, iy = 0.0 /* -Wall */;
972 
973  if (!cs[b].first) {
974  /* No previous point exists for this edge. */
975  /* Save this point. */
976  cs[b].first = 1;
977  cs[b].fx = x;
978  cs[b].fy = y;
979  }
980  else
981  /* A previous point exists. */
982  /* If 'p' and previous point cross edge, find intersection. */
983  /* Clip against next boundary, if any. */
984  /* If no more edges, add intersection to output list. */
985  if (cross (b, x, y, cs[b].sx, cs[b].sy, clip)) {
986  intersect (b, x, y, cs[b].sx, cs[b].sy, &ix, &iy, clip);
987  if (b < Top)
988  clipPoint((Edge) (b + 1), ix, iy, xout, yout, cnt, store,
989  clip, cs);
990  else {
991  if (store) {
992  xout[*cnt] = ix;
993  yout[*cnt] = iy;
994  }
995  (*cnt)++;
996  }
997  }
998 
999  /* Save as most recent point for this edge */
1000  cs[b].sx = x;
1001  cs[b].sy = y;
1002 
1003  /* For all, if point is 'inside' */
1004  /* proceed to next clip edge, if any */
1005  if (inside (b, x, y, clip)) {
1006  if (b < Top)
1007  clipPoint((Edge) (b + 1), x, y, xout, yout, cnt, store, clip, cs);
1008  else {
1009  if (store) {
1010  xout[*cnt] = x;
1011  yout[*cnt] = y;
1012  }
1013  (*cnt)++;
1014  }
1015  }
1016 }

1134 /* Convert a circle into a polygon with specified number of vertices */
1135 static void convertCircle(double x, double y, double r,
1136  int numVertices, double *xc, double *yc)
1137 {
1138  int i;
1139  double theta = 2*M_PI/numVertices;
1140  for (i=0; i<numVertices; i++) {
1141  xc[i] = x + r*sin(theta*i);
1142  yc[i] = y + r*cos(theta*i);
1143  }
1144  xc[numVertices] = x;
1145  yc[numVertices] = y+r;
1146 }

1148 /* Takes a specification of a circle as input and returns a code indicating
1149  how the circle should be clipped.
1150  The return value will be -1 if the circle is to
1151  be totally clipped out of existence, -2 if the circle is to be
1152  totally left alone, 0 and above if the circle has been converted
1153  into a polygon (in which case, the return value indicates the
1154  number of vertices of the polygon and the function convertCircle()
1155  should be called to obtain the vertices of the polygon). */
1156 static int clipCircleCode(double x, double y, double r,
1157  int toDevice, GEDevDesc* dd)
1158 {
1159  int result;
1160  /* determine clipping region */
1161  double xmin, xmax, ymin, ymax;
1162  if (toDevice)
1163  getClipRectToDevice(&xmin, &ymin, &xmax, &ymax, dd);
1164  else
1165  getClipRect(&xmin, &ymin, &xmax, &ymax, dd);
1166 
1167  /* if circle is all within clipping rect */
1168  if (x-r > xmin && x+r < xmax && y-r > ymin && y+r < ymax) {
1169  result = -2;
1170  }
1171  /* if circle is all outside clipping rect */
1172  else {
1173  double distance = r*r;
1174  if (x-r > xmax || x+r < xmin || y-r > ymax || y+r < ymin ||
1175  (x < xmin && y < ymin &&
1176  ((x-xmin)*(x-xmin)+(y-ymin)*(y-ymin) > distance)) ||
1177  (x > xmax && y < ymin &&
1178  ((x-xmax)*(x-xmax)+(y-ymin)*(y-ymin) > distance)) ||
1179  (x < xmin && y > ymax &&
1180  ((x-xmin)*(x-xmin)+(y-ymax)*(y-ymax) > distance)) ||
1181  (x > xmax && y > ymax &&
1182  ((x-xmax)*(x-xmax)+(y-ymax)*(y-ymax) > distance))) {
1183  result = -1;
1184  }
1185  /* otherwise, convert circle to polygon */
1186  else {
1187  /* Replace circle with polygon.
1188 
1189  Heuristic for number of vertices is to use theta so
1190  that cos(theta)*r ~ r - 1 in device units. This is
1191  roughly const * sqrt(r) so there'd be little point in
1192  enforcing an upper limit. */
1193 
1194  result = (r <= 6) ? 10 : int(2 * M_PI/acos(1 - 1/r));
1195  }
1196  }
1197  return result;
1198 }

1201  * GECircle
1202  ****************************************************************
1203  */
1204 void GECircle(double x, double y, double radius, const pGEcontext gc, GEDevDesc* dd)
1205 {
1206  const void *vmax;
1207  double *xc, *yc;
1208  int result;
1209 
1210  /* There is no point in trying to plot a circle of zero radius */
1211  if (radius <= 0.0) return;
1212 
1213  if (gc->lwd == R_PosInf || gc->lwd < 0.0)
1214  Rf_error(_("'lwd' must be non-negative and finite"));
1215  if (std::isnan(gc->lwd) || gc->lty == LTY_BLANK)
1216  /* "transparent" border */

1218  /*
1219  * If the device can clip, then we just clip to the device
1220  * boundary and let the device do clipping within that.
1221  * We do this to avoid problems where writing WAY off the
1222  * device can cause problems for, e.g., ghostview
1223  *
1224  * If the device can't clip, we have to do all the clipping
1225  * ourselves.
1226  */
1227  result = clipCircleCode(x, y, radius, dd->dev->canClip, dd);
1228 
1229  switch (result) {
1230  case -2: /* No clipping; draw all of circle */
1231  /*
1232  * If we did the clipping, then the circle is entirely
1233  * within the current clipping rect.
1234  *
1235  * If the device can clip then we just clipped to the device
1236  * boundary so the circle is entirely within the device; the
1237  * device will perform the clipping to the current clipping rect.
1238  */
1239  dd->dev->circle(x, y, radius, gc, dd->dev);
1240  break;
1241  case -1: /* Total clipping; draw nothing */
1242  /*
1243  * If we did the clipping, then the circle is entirely outside
1244  * the current clipping rect, so there is nothing to draw.
1245  *
1246  * If the device can clip then we just determined that the
1247  * circle is entirely outside the device, so again there is
1248  * nothing to draw
1249  */
1250  break;
1251  default: /* Partial clipping; draw poly[line|gon] */
1252  /*
1253  * If we did the clipping this means that the circle
1254  * intersects the current clipping rect and we need to
1255  * convert to a poly[line|gon] and draw that.
1256  *
1257  * If the device can clip then we just determined that the
1258  * circle intersects the device boundary. We assume that the
1259  * circle is not so big that other parts may be WAY off the
1260  * device and just draw a circle.
1261  */
1262  if (dd->dev->canClip) {
1263  dd->dev->circle(x, y, radius, gc, dd->dev);
1264  }
1265  else {
1266  vmax = vmaxget();
1267  xc = static_cast<double*>(RHO_alloc(result+1, sizeof(double)));
1268  yc = static_cast<double*>(RHO_alloc(result+1, sizeof(double)));
1269  convertCircle(x, y, radius, result, xc, yc);
1270  if (R_TRANSPARENT(gc->fill)) {
1271  GEPolyline(result+1, xc, yc, gc, dd);
1272  }
1273  else {
1274  int npts;
1275  double *xcc, *ycc;
1276  xcc = ycc = nullptr; /* -Wall */
1277  npts = clipPoly(xc, yc, result, 0, !dd->dev->canClip,
1278  xcc, ycc, dd);
1279  if (npts > 1) {
1280  xcc = static_cast<double*>(RHO_alloc(npts, sizeof(double)));
1281  ycc = static_cast<double*>(RHO_alloc(npts, sizeof(double)));
1282  npts = clipPoly(xc, yc, result, 1, !dd->dev->canClip,
1283  xcc, ycc, dd);
1284  dd->dev->polygon(npts, xcc, ycc, gc, dd->dev);
1285  }
1286  }
1287  vmaxset(vmax);
1288  }
1289  }

1481  double width, double height, double rot, double hadj,
1482  const pGEcontext gc, int toDevice, GEDevDesc* dd)

1485  double length, theta2;

1487  double theta1 = M_PI/2 - angle;

1490  if (!std::isfinite(width)) width = GEStrWidth(str, enc, gc, dd);
1491  if (!std::isfinite(height)) height = GEStrHeight(str, enc, gc, dd);
