656  * Note that CTXT_LOOP is already the or of CTXT_NEXT and CTXT_BREAK.
657  * Only functions should have the third bit turned on;
658  * this allows us to move up the context stack easily
659  * with either RETURN's or GENERIC's or RESTART's.
660  * If you add a new context type for functions make sure
661  * CTXT_NEWTYPE & CTXT_FUNCTION > 0

664  CTXT_TOPLEVEL = 0,
665  CTXT_NEXT = 1,
666  CTXT_BREAK = 2,

668  CTXT_FUNCTION = 4,

670  CTXT_RETURN = 12,
671  CTXT_BROWSER = 16,
672  CTXT_GENERIC = 20,
673  CTXT_RESTART = 32,
674  CTXT_BUILTIN = 64, /* used in profiling */
