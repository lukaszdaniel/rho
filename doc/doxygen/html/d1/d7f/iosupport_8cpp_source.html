58 /* Move the iob->write_buf pointer to the next */
59 /* BufferListItem in the chain. If there no next */
60 /* buffer item, then one is added. */
61 
62 static int NextWriteBufferListItem(IoBuffer *iob)
63 {
64  if (iob->write_buf->next) {
65  iob->write_buf = iob->write_buf->next;
66  }
67  else {
68  BufferListItem *_new;
69  if (!(_new = static_cast<BufferListItem*>(malloc(sizeof(BufferListItem)))))
70  return 0;
71  _new->next = nullptr;
72  iob->write_buf->next = _new;
73  iob->write_buf = iob->write_buf->next;
74  }
75  iob->write_ptr = iob->write_buf->buf;
76  iob->write_offset = 0;
77  return 1;
78 }

131 /* Free any BufferListItem's associated with an IoBuffer */
132 /* This resets pointers to NULL, which could be detected */
133 /* in other calls. */
134 
135 int attribute_hidden R_IoBufferFree(IoBuffer *iob)
136 {
137  BufferListItem *thisItem, *nextItem;
138  if (iob == nullptr || iob->start_buf == nullptr)
139  return 0;
140  thisItem = iob->start_buf;
141  while (thisItem) {
142  nextItem = thisItem->next;
143  free(thisItem);
144  thisItem = nextItem;
145  }
146  return 1;
147 }

175 int attribute_hidden R_IoBufferGetc(IoBuffer *iob)
176 {
177  if (iob->read_buf == iob->write_buf &&
178  iob->read_offset >= iob->write_offset)
179  return EOF;
180  if (iob->read_offset == IOBSIZE) NextReadBufferListItem(iob);
181  iob->read_offset++;
182  return *(iob->read_ptr)++;
183 }

186 
187 int attribute_hidden R_IoBufferReadOffset(IoBuffer *iob)
188 {
189  int result = iob->read_offset;
190  BufferListItem* buf = iob->start_buf;
191  while(buf && buf != iob->read_buf) {
192  result += IOBSIZE;
193  buf = buf->next;
194  }
195  return result;

207 int attribute_hidden R_TextBufferInit(TextBuffer *txtb, SEXP text)
208 {
209  int i, k, l, n;
210  if (Rf_isString(text)) {
211  // translateChar might allocate
212  void *vmax = vmaxget();
213  n = Rf_length(text);
214  l = 0;
215  for (i = 0; i < n; i++) {
216  if (STRING_ELT(text, i) != R_NilValue) {
217  k = int(strlen(Rf_translateChar(STRING_ELT(text, i))));
218  if (k > l)
219  l = k;
220  }
221  }
222  vmaxset(vmax);
223  txtb->vmax = vmax;
224  txtb->buf = static_cast<unsigned char *>(RHO_alloc(l+2, sizeof(char))); /* '\n' and '\0' */
225  txtb->bufp = txtb->buf;
226  txtb->text = text;
227  txtb->ntext = n;
228  txtb->offset = 0;
229  transferChars(txtb->buf,
230  Rf_translateChar(STRING_ELT(txtb->text, txtb->offset)));
231  txtb->offset++;
232  return 1;
233  }
234  else {
235  txtb->vmax = vmaxget();
236  txtb->buf = nullptr;
237  txtb->bufp = nullptr;
238  txtb->text = R_NilValue;
239  txtb->ntext = 0;
240  txtb->offset = 1;
241  return 0;
242  }
243 }

256 {

267  txtb->bufp = txtb->buf;

271  }
