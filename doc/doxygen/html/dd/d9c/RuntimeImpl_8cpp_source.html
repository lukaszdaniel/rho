40 /*
41  * This file contains functions that are available in the runtime module.
42  * It gets compiled into LLVM bytecode as part of the compilation process, and
43  * the JIT compiler loads the bytecode and makes it available to the functions
44  * it builds.
45  * Because it is available as LLVM IR at runtime, LLVM is able to inline these
46  * directly into the code that it generates.
47  *
48  * clang -emit-llvm -c -o foo.bc # emits bitcode
49  * clang -emit-llvm -S -o foo.ll # emits human readable IR
50  */

67 /*
68  * Lookup a symbol in a CompiledFrame.
69  * Note that this function doesn't handle the cases where the symbol is
70  * ..., ..n, or missingArg(). rho_runtime_lookupSymbol should be used in
71  * those cases.
72  */
73 RObject* rho_runtime_lookupSymbolInCompiledFrame(const Symbol* symbol,
74  Environment* environment,
75  int position)
76 {
77  assert(environment->frame() != nullptr);
78  assert(symbol != DotsSymbol);
79  assert(!symbol->isDotDotSymbol());
80  assert(symbol != R_MissingArg);
81  assert(position >= 0);
82 
83  Frame* frame = environment->frame();
84  assert(frame != nullptr);
85 
86  Frame::Binding* binding = frame->binding(position);
87  if (binding) {
88  // NB: this logic handles the case where the binding is a promise that
89  // resolves to an unboundValue or missingArgument slightly differently
90  // than Symbol::evaluate() does.
91  std::pair<RObject*, bool> pair = binding->forcedValue2();
92  RObject* value = pair.first;
93  if (value
94  && value != Symbol::missingArgument()
95  && value != Symbol::unboundValue())
96  {
97  if (pair.second) {
98  ENSURE_NAMEDMAX(value);
99  }
100  else if (NAMED(value) < 1) {
101  SET_NAMED(value, 1);
102  }
103  return value;
104  }
105  }
106  // Fallback to the interpreter.
107  return rho_runtime_lookupSymbol(symbol, environment);
108 }
