53 /* R function qsort(x, index.return) */
54 HIDDEN SEXP do_qsort(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* index_return_)
55 {
56  SEXP x, sx;
57  int indx_ret;
58  double *vx = nullptr;
59  int *ivx = nullptr;
60  Rboolean x_real, x_int;
61 
62  x = x_;
63  if (!Rf_isNumeric(x))
64  Rf_error(_("argument is not a numeric vector"));
65  x_real= Rboolean(TYPEOF(x) == REALSXP);
66  x_int = Rboolean(!x_real && (TYPEOF(x) == INTSXP || TYPEOF(x) == LGLSXP));
67  PROTECT(sx = (x_real || x_int) ? Rf_duplicate(x) : Rf_coerceVector(x, REALSXP));
68  sx->clearAttributes();
69  indx_ret = Rf_asLogical(index_return_);
70  R_xlen_t n = XLENGTH(x);
71 #ifdef LONG_VECTOR_SUPPORT
72  Rboolean isLong = Rboolean(n > INT_MAX);
73 #endif
74  if(x_int) ivx = INTEGER(sx); else vx = REAL(sx);
75  if(indx_ret) {
76  SEXP ans, ansnames, indx;
77  /* answer will have x = sorted x , ix = index :*/
78  PROTECT(ans = Rf_allocVector(VECSXP, 2));
79  PROTECT(ansnames = Rf_allocVector(STRSXP, 2));
80 #ifdef LONG_VECTOR_SUPPORT
81  if (isLong) {
82  PROTECT(indx = Rf_allocVector(REALSXP, n));
83  double *ix = REAL(indx);
84  for(R_xlen_t i = 0; i < n; i++) ix[i] = double((i+1));
85  if(x_int) R_qsort_int_R(ivx, ix, 1, n);
86  else R_qsort_R(vx, ix, 1, n);
87  } else
88 #endif
89  {
90  PROTECT(indx = Rf_allocVector(INTSXP, n));
91  int *ix = INTEGER(indx);
92  int nn = int(n);
93  for(int i = 0; i < nn; i++) ix[i] = i+1;
94  if(x_int) R_qsort_int_I(ivx, ix, 1, nn);
95  else R_qsort_I(vx, ix, 1, nn);
96  }
97  SET_VECTOR_ELT(ans, 0, sx);
98  SET_VECTOR_ELT(ans, 1, indx);
99  SET_STRING_ELT(ansnames, 0, Rf_mkChar("x"));
100  SET_STRING_ELT(ansnames, 1, Rf_mkChar("ix"));
101  Rf_setAttrib(ans, Symbols::NamesSymbol, ansnames);
102  UNPROTECT(4);
103  return ans;
104  } else {
105  if(x_int)
106  R_qsort_int(ivx, 1, n);
107  else
108  R_qsort(vx, 1, n);
109 
110  UNPROTECT(1);
111  return sx;
112  }
113 }
