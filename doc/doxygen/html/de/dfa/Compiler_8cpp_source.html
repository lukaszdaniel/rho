106 void Compiler::emitErrorUnless(Value* condition,
107  const char* error_msg,
108  llvm::ArrayRef<Value*> extra_args) {
109  BasicBlock* error_block = createBasicBlock("error_msg");
110  BasicBlock* continue_block = createBasicBlock("continue", error_block);
111 
112  llvm::Type* type = condition->getType();
113  if (type != llvm::Type::getInt1Ty(getContext())) {
114  condition = CreateIsNotNull(condition);
115  }
116  CreateCondBr(condition, continue_block, error_block);
117 
118  SetInsertPoint(error_block);
119  Runtime::emitError(error_msg, extra_args, this);
120 
121  SetInsertPoint(continue_block);
122 }

124 std::vector<llvm::Value*> Compiler::castFunctionArguments(
125  llvm::ArrayRef<llvm::Value*> args,
126  llvm::Function* function)
127 {
128  assert(function->isVarArg() || function->arg_size() == args.size());
129 
130  // Upcast any arguments that require it.
131  std::vector<llvm::Value*> type_checked_args;
132  int i = 0;
133  for (const llvm::Argument& argument : function->args()) {
134  type_checked_args.push_back(
135  // TODO(kmillar): these really ought to be upcasts,
136  // in which case we can statically check that the cast is valid.
137  CreatePointerCast(args[i], argument.getType()));
138  ++i;
139  }
140  return type_checked_args;

162 Value* Compiler::emitEval(const RObject* object)
163 {
164  emitSetVisibility(true);
165 
166  Value* result = emitEvalInternal(object);
167 
168  if (GetInsertBlock()->getTerminator()) {
169  // Any subsequent code is dead. Create a dummy basic block for it.
170  SetInsertPoint(createBasicBlock("dead_code"));
171 
172  if (llvm::isa<llvm::TerminatorInst>(result)) {
173  // There is no current value either. Use undef as a placeholder.
174  return llvm::UndefValue::get(getType<RObject*>());
175  }
176  }
177 
178  return result;
179 }

181 Value* Compiler::emitEvalInternal(const RObject* object)
182 {
183  // This has a non-trivial implementation for all the objects which have
184  // non-default object->evaluate() implementations.
185  if (!object) {
186  return emitNullValue();
187  }
188 
189  switch (object->sexptype()) {
190  case SYMSXP:
191  return emitSymbolEval(SEXP_downcast<const Symbol*>(object));
192  case LANGSXP:
193  return emitExpressionEval(SEXP_downcast<const Expression*>(object));
194  case DOTSXP:
195  return emitDotsEval(SEXP_downcast<const DottedArgs*>(object));
196  case PROMSXP:
197  assert(0 && "Unexpected eval of a promise in JIT compilation.");
198  return nullptr;
199  default:
200  return emitConstantPointer(object);
201  }
202 }

204 Value* Compiler::emitSymbolEval(const Symbol* symbol)
205 {
206  assert(m_context->m_frame_descriptor != nullptr);
207  // Optimize the lookup in the likely case that this is a regular symbol
208  // found in the local environment.
209  int location = m_context->m_frame_descriptor->getLocation(symbol);
210  if (location != -1
211  && symbol != DotsSymbol
212  && !symbol->isDotDotSymbol()
213  && symbol != Symbol::missingArgument()) {
214  // Lookup the symbol directly.
215  return Runtime::emitLookupSymbolInCompiledFrame(
216  emitSymbol(symbol), m_context->getEnvironment(), location,
217  this);
218  }
219  // Otherwise fallback to the interpreter for now.
220  return Runtime::emitLookupSymbol(emitSymbol(symbol),
221  m_context->getEnvironment(), this);
222 }

224 // This function is the JIT version of Expression::evaluate().
225 Value* Compiler::emitExpressionEval(const Expression* expression)
226 {
227  RObject* function = expression->car();
228  Value* resolved_function;
229 
230  // Evaluate the function argument and get a prediction of its likely value.
231  FunctionBase* likely_function = dynamic_cast<FunctionBase*>(function);
232  if (likely_function) {
233  // The first element of the expression is a literal function.
234  resolved_function = emitConstantPointer(likely_function);
235  } else if (Symbol* symbol = dynamic_cast<Symbol*>(function)) {
236  // The first element is a symbol. Look it up.
237  resolved_function = emitFunctionLookup(symbol, &likely_function);
238  // Check that the lookup succeeded, unless the function was resolved
239  // at compile time.
240  if (!llvm::isa<llvm::Constant>(resolved_function)) {
241  emitErrorUnless(resolved_function,
242  _("could not find function \"%s\""),
243  emitConstantPointer(symbol->name()->c_str()));
244  }
245  } else {
246  // The first element is a (function-valued) expression. Fallback
247  // to the interpreter for now.
248  return Runtime::emitEvaluate(emitConstantPointer(expression),
249  m_context->getEnvironment(), this);
250  }
251 
252  if (likely_function) {
253  Value* result = emitInlineableBuiltinCall(
254  expression, resolved_function, likely_function);
255  if (result) {
256  return result;
257  }
258  }
259 
260  // The function wasn't inlined, so emit a call to the interpreter.
261  return Runtime::emitCallFunction(
262  resolved_function, emitConstantPointer(expression->tail()),
263  emitConstantPointer(expression), m_context->getEnvironment(), this);
264 }

273 Value* Compiler::emitFunctionLookup(const Symbol* symbol,
274  FunctionBase** expected_result) {
275  // Resolve the function statically if possible.
276  *expected_result = m_context->staticallyResolveFunction(symbol);
277  if (*expected_result) {
278  return emitConstantPointer(*expected_result);
279  }
280 
281  // Otherwise do a dynamic function lookup.
282  *expected_result = findFunction(symbol,
283  m_context->getClosure()->environment());
284  llvm::Value* fn = emitSymbol(symbol);
285  // TODO(kmillar): emit optimized code for this.
286  return Runtime::emitLookupFunction(fn, m_context->getEnvironment(),
287  this);
288 }

354 Compiler::EmitBuiltinFn Compiler::getInlinedBuiltInEmitter(
355  BuiltInFunction* builtin)
356 {
357  // Only functions that are primitive can be inlined directly. Functions
358  // that require argument matching or create a new function context are not
359  // inlined here.
360  if (!builtin || builtin->viaDotInternal()) {
361  return nullptr;
362  }
363  for (auto inlineable_builtin : getInlineableBuiltins()) {
364  if (builtin == inlineable_builtin.first) {
365  return inlineable_builtin.second;
366  }
367  }
368  return nullptr;
369 }

371 Value* Compiler::emitInlineableBuiltinCall(const Expression* expression,
372  Value* resolved_function,
373  FunctionBase* likely_function)
374 {
375  BuiltInFunction* builtin = dynamic_cast<BuiltInFunction*>(likely_function);
376  EmitBuiltinFn emit_builtin = getInlinedBuiltInEmitter(builtin);
377  if (!emit_builtin) {
378  return nullptr;
379  }
380 
381  if (llvm::isa<llvm::Constant>(resolved_function)) {
382  // When the resolved function is a compile-time constant, everything is
383  // simple.
384  return (this->*emit_builtin)(expression);
385  }
386 
387  InsertPoint incoming_block = saveIP();
388 
389  // TODO(kmillar): write an 'emitGuardedCode' function, and use that here.
390  BasicBlock* inlined_builtin_block = createBasicBlock(builtin->name());
391  BasicBlock* merge_block = createBasicBlock("continue");
392  BasicBlock* fallback_block = createBasicBlock("fallback");
393 
394  // Setup the merge point.
395  SetInsertPoint(merge_block);
396  llvm::Type* robject_type = getType<RObject*>();
397  llvm::PHINode* result = CreatePHI(robject_type, 2);
398 
399  // Emit inlined code.
400  SetInsertPoint(inlined_builtin_block);
401 
402  llvm::Value* inlined_builtin_value = (this->*emit_builtin)(expression);
403  if (!inlined_builtin_value) {
404  // Code generation failed.
405  inlined_builtin_block->eraseFromParent();
406  merge_block->eraseFromParent();
407  fallback_block->eraseFromParent();
408  restoreIP(incoming_block);
409  return nullptr;
410  }
411  if (GetInsertBlock()->getTerminator() == nullptr) {
412  assert(inlined_builtin_value->getType() == robject_type);
413  CreateBr(merge_block);
414  result->addIncoming(inlined_builtin_value, GetInsertBlock());
415  }
416 
417  // Code to check if the function is the predicted one.
418  restoreIP(incoming_block);
419  Value* likely_fn_value = m_context->getMemoryManager()
420  ->getBuiltIn(builtin);
421  likely_fn_value = CreateBitCast(likely_fn_value,
422  llvm::TypeBuilder<FunctionBase*, false>::get(getContext()));
423  Value* is_expected_builtin = CreateICmpEQ(resolved_function,
424  likely_fn_value);
425  CreateCondBr(is_expected_builtin, inlined_builtin_block, fallback_block,
426  CreateBranchWeightsLikelyTaken(m_context->getLLVMContext()));
427 
428  // If the function isn't the one we expected, fall back to the interpreter.
429  // TODO(kmillar): do OSR or similar on guard failure to improve fast
430  // codepath performance and reduce the time spent compiling unlikely
431  // codepaths.
432  // TODO(kmillar): allow this check to be skipped at some optimization
433  // levels.
434  SetInsertPoint(fallback_block);
435  Value* fallback_value
436  = Runtime::emitCallFunction(resolved_function,
437  emitConstantPointer(expression->tail()),
438  emitConstantPointer(expression),
439  m_context->getEnvironment(), this);
440  if (GetInsertBlock()->getTerminator() == nullptr) {
441  CreateBr(merge_block);
442  result->addIncoming(fallback_value, GetInsertBlock());
443  }
444 
445  // Continuing code should be in the merged block.
446  SetInsertPoint(merge_block);
447  return result;
448 }

450 Value* Compiler::emitInlinedAssign(const Expression* expression)
451 {
452  // This corresponds to do_set() for '<-' or '='. Not '<<-'.
453  if (listLength(expression) != 3) {
454  // This is probably a syntax error. Let the interpreter handle it.
455  return nullptr;
456  }
457 
458  // Get the symbol to assign to.
459  const RObject* lhs_expr = expression->tail()->car();
460  const Symbol* symbol = dynamic_cast<const Symbol*>(lhs_expr);
461  if (!symbol) {
462  // If passed a string vector, convert that to a name.
463  const StringVector* symbol_name
464  = dynamic_cast<const StringVector*>(lhs_expr);
465  if (symbol_name && symbol_name->size() == 1) {
466  symbol = Symbol::obtain((*symbol_name)[0]);
467  } else if (dynamic_cast<const Expression*>(lhs_expr)) {
468  // Complex assignment.
469  emitSetVisibility(false);
470  auto applydefine =
471  Runtime::getDeclaration("rho_runtime_applydefine", this);
472  assert(applydefine != nullptr);
473  Value* call_ptr = emitConstantPointer(expression);
474  Value* applydefine_ptr =
475  emitConstantPointer(BuiltInFunction::obtainPrimitive(
476  SEXP_downcast<const Symbol*>(expression->getFunction())));
477  assert(applydefine_ptr != nullptr);
478  Value* args_ptr = emitConstantPointer(expression->getArgs());
479  assert(args_ptr != nullptr);
480  assert(m_context->getEnvironment() != nullptr);
481  return emitCallOrInvoke(applydefine,
482  {call_ptr, applydefine_ptr, args_ptr,
483  m_context->getEnvironment()});
484  } else {
485  // Probably a syntax error. Send it to the interpreter.
486  return nullptr;
487  }
488  }
489  int location = m_context->m_frame_descriptor->getLocation(symbol);
490  if (location == -1) {
491  // For some reason, the symbol isn't in the frame descriptor. Fallback
492  // to the interpreter.
493  return nullptr;
494  }
495 
496  // Get the value.
497  const RObject* value = expression->tail()->tail()->car();
498  Value* evaluated_value = emitEval(value);
499 
500  // Make runtime calls to actually do the assignment.
501  Runtime::emitIncrementNamed(evaluated_value, this);
502  Runtime::emitAssignSymbolInCompiledFrame(emitSymbol(symbol),
503  m_context->getEnvironment(),
504  location,
505  evaluated_value,
506  this);
507  emitSetVisibility(false);
508  return evaluated_value;
509 }

559 Value* Compiler::emitInlinedIf(const Expression* expression)
560 {
561  int length = listLength(expression);
562  if (length != 3 && length != 4) {
563  // This is probably a syntax error. Let the interpreter handle it.
564  return nullptr;
565  }
566 
567  // Evaluate the condition and branch.
568  Value* r_condition = emitEval(CADR(const_cast<Expression*>(expression)));
569  // TODO(kmillar): r_condition needs GC protection.
570  Value* boolean_condition = Runtime::emitCoerceToTrueOrFalse(
571  r_condition, expression, this);
572  InsertPoint branch_point = saveIP();
573 
574  // Create the merge point.
575  BasicBlock* continue_block = createBasicBlock("continue");
576  SetInsertPoint(continue_block);
577  llvm::Type* robject_type = getType<RObject*>();
578  llvm::PHINode* result_value = CreatePHI(robject_type, 2);
579 
580  // Create the if_true branch.
581  BasicBlock* if_true_block = createBranch(
582  "if_true",
583  CADDR(const_cast<Expression*>(expression)),
584  result_value, continue_block);
585 
586  // Generate code to handle the 'else' case.
587  BasicBlock* if_false_block;
588  if (length == 3) {
589  // No 'else' branch. Return an invisible NULL.
590  if_false_block = createBasicBlock("if_false");
591  SetInsertPoint(if_false_block);
592  emitSetVisibility(false);
593  CreateBr(continue_block);
594  result_value->addIncoming(emitNullValue(),
595  GetInsertBlock());
596  } else {
597  // Create the else branch.
598  if_false_block = createBranch(
599  "if_false",
600  CADDDR(const_cast<Expression*>(expression)),
601  result_value, continue_block);
602  }
603 
604  restoreIP(branch_point);
605  CreateCondBr(boolean_condition, if_true_block, if_false_block);
606 
607  SetInsertPoint(continue_block);
608  return result_value;
609 }

611 Value* Compiler::emitInlinedRepeat(const Expression* expression)
612 {
613  if (listLength(expression) != 2) {
614  // This is probably a syntax error. Let the interpreter handle it.
615  return nullptr;
616  }
617 
618  const RObject* body = expression->tail()->car();
619 
620  BasicBlock* loop_body = createBasicBlock("repeat_body");
621  BasicBlock* continue_block = createBasicBlock("continue");
622 
623  CreateBr(loop_body);
624 
625  SetInsertPoint(loop_body);
626  {
627  // The loop scope ensures that 'break' and 'next' work correctly.
628  LoopScope loop(m_context,
629  continue_block, loop_body,
630  this);
631  emitEval(body);
632  }
633  createBackEdge(loop_body);
634 
635  SetInsertPoint(continue_block);
636 
637  return emitInvisibleNullValue();

640 Value* Compiler::emitInlinedWhile(const Expression* expression)
641 {
642  if (listLength(expression) != 3) {
643  // This is probably a syntax error. Let the interpreter handle it.
644  return nullptr;
645  }
646 
647  const RObject* condition = expression->tail()->car();
648  const RObject* body = expression->tail()->tail()->car();
649 
650  BasicBlock* loop_header = createBasicBlock("while_header");
651  BasicBlock* loop_body = createBasicBlock("while_body");
652  BasicBlock* continue_block = createBasicBlock("continue");
653 
654  CreateBr(loop_header);
655 
656  SetInsertPoint(loop_header);
657  llvm::Value* evaluated_condition = emitEval(condition);
658  llvm::Value* condition_as_bool = Runtime::emitCoerceToTrueOrFalse(
659  evaluated_condition, expression, this);
660  CreateCondBr(condition_as_bool, loop_body, continue_block);
661 
662  SetInsertPoint(loop_body);
663  {
664  LoopScope loop(m_context,
665  continue_block, loop_header,
666  this);
667  emitEval(body);
668  }
669  createBackEdge(loop_header);
670 
671  SetInsertPoint(continue_block);
672  return emitInvisibleNullValue();
673 }

675 Value* Compiler::emitInlinedBreak(const Expression* expression) {
676  if (listLength(expression) != 1) {
677  // This is probably a syntax error. Let the interpreter handle it.
678  return nullptr;
679  }
680 
681  BasicBlock* dest = m_context->getBreakDestination();
682  if (dest) {
683  return CreateBr(dest);
684  } else {
