156 SEXP Rf_asChar(SEXP x)
157 {
158  if (Rf_isVectorAtomic(x) && XLENGTH(x) >= 1) {
159  int w, d, e, wi, di, ei;
160  char buf[MAXELTSIZE]; /* Probably 100 would suffice */
161 
162  switch (TYPEOF(x)) {
163  case LGLSXP:
164  if (LOGICAL(x)[0] == NA_LOGICAL)
165  return NA_STRING;
166  if (LOGICAL(x)[0])
167  sprintf(buf, "TRUE");
168  else
169  sprintf(buf, "FALSE");
170  return Rf_mkChar(buf);
171  case INTSXP:
172  if (INTEGER(x)[0] == NA_INTEGER)
173  return NA_STRING;
174  snprintf(buf, MAXELTSIZE, "%d", INTEGER(x)[0]);
175  return Rf_mkChar(buf);
176  case REALSXP:
177  Rf_PrintDefaults();
178  formatReal(REAL(x), 1, &w, &d, &e, 0);
179  return Rf_mkChar(EncodeReal0(REAL(x)[0], w, d, e, OutDec));
180  case CPLXSXP:
181  Rf_PrintDefaults();
182  formatComplex(COMPLEX(x), 1, &w, &d, &e, &wi, &di, &ei, 0);
183  return Rf_mkChar(EncodeComplex(COMPLEX(x)[0], w, d, e, wi, di, ei, OutDec));
184  case STRSXP:
185  return STRING_ELT(x, 0);
186  default:
187  return NA_STRING;
188  }
189  } else if(TYPEOF(x) == CHARSXP) {
190  return x;
191  } else if(TYPEOF(x) == SYMSXP)
192  return PRINTNAME(x);
193  return NA_STRING;
194 }

399 size_t mbcsToUcs2(const char *in, ucs2_t *out, int nout, int enc)
400 {
401  void *cd = nullptr ;
402  const char *i_buf;
403  char *o_buf;
404  size_t i_len, o_len, status, wc_len;
405  /* out length */
406  wc_len = (enc == CE_UTF8)? Rf_utf8towcs(nullptr, in, 0) : mbstowcs(nullptr, in, 0);
407  if (out == nullptr || int(wc_len) < 0) return wc_len;
408 
409  if (reinterpret_cast<void*>(-1) == (cd = Riconv_open(UCS2ENC, (enc == CE_UTF8) ? "UTF-8": "")))
410  return size_t(-1);
411 
412  i_buf = RHO_NO_CAST(char *)in;
413  i_len = strlen(in); /* not including terminator */
414  o_buf = reinterpret_cast<char *>(out);
415  o_len = (size_t(nout)) * sizeof(ucs2_t);
416  status = Riconv(cd, &i_buf, RHO_NO_CAST(size_t *)&i_len, &o_buf, RHO_NO_CAST(size_t *)&o_len);
417  int serrno = errno;
418  Riconv_close(cd);
419  if (status == size_t(-1)) {
420  switch(serrno){
421  case EINVAL:
422  return size_t(-2);
423  case EILSEQ:
424  return size_t(-1);
425  case E2BIG:
426  break;
427  default:
428  errno = EILSEQ;
429  return size_t(-1);
430  }
431  }
432  return wc_len; /* status would be better? */
433 }

614 {
615  int i, j, h, iv, v;
616 
617  for (h = 1; h <= n / 9; h = 3 * h + 1);
618  for (; h > 0; h /= 3)
619  for (i = h; i < n; i++) {
620  v = x[i]; iv = indx[i];
621  j = i;
622  while (j >= h && x[j - h] > v)
623  { x[j] = x[j - h]; indx[j] = indx[j-h]; j -= h; }
624  x[j] = v; indx[j] = iv;
625  }
