38 rho::AllocationTable::AllocationTable(unsigned num_bits): m_num_bits(num_bits) {

57 void rho::AllocationTable::insertUnchecked(const Allocation& allocation) {
58  m_capacity -= requiredCapacity(allocation);
59  bool first = true;
60  size_t key_end = endKey(allocation.dataPointer(), allocation.sizeLog2());
61  for (size_t key = startKey(allocation.dataPointer()); key < key_end; ++key) {
62  insertSingleKey(allocation, key << LOW_BITS, first);
63  first = false;
64  }
65 }

67 void rho::AllocationTable::insertSingleKey(const Allocation& allocation,
68  size_t key, bool first) {
69  size_t hash = pointerHash(key);
70  // Insert will always succed because we have a precondition testing that
71  // there is enough space in the table.
72  for (int i = 0; true;
73  hash = probeFunc(hash, i),
74  m_num_insert_collisions += 1,
75  i += 1) {
76  Allocation& bucket = m_buckets[hash];
77  if (bucket.isEmpty() || bucket.isDeleted()) {
78  // Can insert here.
79  m_buckets[hash] = allocation;
80  if (first) {
81  // Set the first flag for this entry.
82  m_buckets[hash].m_data |= Allocation::s_first_flag;
83  }
84  return;
85  }
86  }
87 }

89 void rho::AllocationTable::erase(uintptr_t pointer, size_t size) {
90  assert(pointer != Allocation::s_empty_key
91  && pointer != Allocation::s_deleted_key
92  && "Trying to remove invalid key from alloctable.");
93  pointer &= ~uintptr_t{3}; // Mask out the flag bits for equality comparison.
94  size_t key_end = endKey(pointer, size);
95  for (size_t key = startKey(pointer); key < key_end; ++key) {
96  eraseSingleKey(pointer, key);
97  }
98 }

100 void rho::AllocationTable::eraseSingleKey(uintptr_t pointer, size_t key) {
101  size_t hash = pointerHash(key << LOW_BITS);
102  // We don't have precondition ensuring the key is in the table before trying
103  // to delete, so we can rely on always finding the key. A fatal error is
104  // generated if the key is not found.
105  for (unsigned int i = 0; i < m_num_buckets;
106  hash = probeFunc(hash, i),
107  m_num_erase_collisions += 1,
108  i += 1) {
109  Allocation& bucket = m_buckets[hash];
110  if (bucket.dataPointer() == pointer) {
111  m_capacity += 1;
112  bucket.m_data = Allocation::s_deleted_key;
113  return;
114  }
115  if (bucket.isEmpty()) {
116  // We didn't find the correct hash entry.
117  allocerr("trying to erase non-existent allocation entry");
118  }
119  }
120  allocerr("trying to erase non-existent allocation entry");
121 }

123 rho::AllocationTable::Allocation* rho::AllocationTable::search(
124  uintptr_t candidate) const {
125  size_t hash = pointerHash(candidate);
126  for (unsigned int i = 0; i < m_num_buckets;
127  hash = probeFunc(hash, i),
128  m_num_search_collisions += 1,
129  i += 1) {
130  Allocation* bucket = &m_buckets[hash];
131  if (bucket->isEmpty()) {
132  return nullptr;
133  }
134  if (bucket->isDeleted()) {
135  continue;
136  }
137  if (bucket->isSuperblock()) {
138  AllocatorSuperblock* superblock = bucket->asSuperblock();
139  if (superblock->isInternalPointer(candidate)) {
140  // Pointer is inside this superblock. Test if it is after the header.
141  if (superblock->isBlockPointer(candidate)) {
142  return bucket;
143  } else {
144  // The pointer mapped to this superblock, but was inside the header.
145  return nullptr;
146  }
147  }
148  } else if (bucket->isInternalPointer(candidate)) {
149  return bucket;
150  }
151  }
152  return nullptr;
153 }

155 void rho::AllocationTable::resize(unsigned new_alloctable_bits) {
156  assert(new_alloctable_bits > m_num_bits
157  && "can not shrink this table");
158  AllocationTable new_table(new_alloctable_bits);
159  // Build new table.
160  for (unsigned int i = 0; i < m_num_buckets; ++i) {
161  const Allocation& bucket = m_buckets[i];
162  if (!bucket.isEmpty() && !bucket.isDeleted() && bucket.isFirst()) {
163  // This is the initial hash bucket for the corresponding allocation.
164  // Rebuild the hash entries in the new table based only on this entry.
165  new_table.insertUnchecked(bucket);
166  }
167  }
168  // Copy everything except collision counts from the new table.
169  m_capacity = new_table.m_capacity;
170  m_num_bits = new_table.m_num_bits;
171  m_num_buckets = new_table.m_num_buckets;
172  m_hash_mask = new_table.m_hash_mask;
173 
174  // Swap to ensure the old bucket array is freed with the temporary table.
175  std::swap(m_buckets, new_table.m_buckets);
176 }

178 void rho::AllocationTable::applyToAllAllocations(
179  std::function<void(void*)> fun) const {
180  for (unsigned int i = 0; i < m_num_buckets; ++i) {
181  Allocation& bucket = m_buckets[i];
182  if (!bucket.isEmpty() && !bucket.isDeleted()) {
183  if (bucket.isFirst()) {
184  if (bucket.isSuperblock()) {
185  // This is a large superblock.
186  AllocatorSuperblock* superblock = bucket.asSuperblock();
187  superblock->applyToBlocks(fun);
188  } else {
189 #ifdef ALLOCATION_CHECK
190  // Extra consistency check.
191  if (!GCNodeAllocator::lookupPointer(bucket.asPointer())) {
192  allocerr("apply to all blocks iterating over non-alloc'd pointer");
193  }
194 #endif
195  fun(bucket.asPointer());
196  }
197  }
198  }
199  }
200 }

224 size_t rho::AllocationTable::pointerHash(uintptr_t pointer) const {
225  pointer >>= LOW_BITS;
226  pointer <<= 4; // Spread out hash keys to leave room for collision buckets.
227  size_t low = pointer & 0xFFFFFFFF;
228  size_t hi = pointer >> 32;
229  size_t hash = low ^ hi;
230  low = hash & 0xFFFF;
231  hi = (hash >> 16) & 0xFFFF;
232  hash = (low >> 16) ^ low ^ hi ^ (pointer & (~0xFFFF));
233  return hash & m_hash_mask;
234 }

244 size_t rho::AllocationTable::endKey(uintptr_t pointer, size_t size_log2) {
245  // We add (1 << LOW_BITS) - 1 to ensure that the end key is one-past the last
246  // valid internal pointer hash key.
247  return (pointer + (1 << size_log2) + ((1 << LOW_BITS) - 1)) >> LOW_BITS;
248 }
