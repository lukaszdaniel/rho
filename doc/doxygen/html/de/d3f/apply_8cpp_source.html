40 
41 /* This is a special .Internal, so has unevaluated arguments. It is
42  called from a closure wrapper, so X and FUN are promises.
43 
44  FUN must be unevaluated for use in e.g. bquote .
45 */
46 HIDDEN SEXP do_lapply(SEXP call, SEXP op, SEXP args, SEXP rho)
47 {
48  PROTECT_INDEX px;
49 
50  SEXP X, XX, FUN;
51  PROTECT_WITH_INDEX(X = CAR(args), &px);
52  XX = PROTECT(Rf_eval(CAR(args), rho));
53  R_xlen_t n = Rf_xlength(XX); // a vector, so will be valid.
54  FUN = CADR(args);
55  Rboolean realIndx = Rboolean(n > INT_MAX);
56 
57  SEXP ans = PROTECT(Rf_allocVector(VECSXP, n));
58  SEXP names = Rf_getAttrib(XX, Symbols::NamesSymbol);
59  if(!Rf_isNull(names)) Rf_setAttrib(ans, Symbols::NamesSymbol, names);
60 
61  /* Build call: FUN(XX[[<ind>]], ...) */
62 
63  SEXP ind = PROTECT(Rf_allocVector(realIndx ? REALSXP : INTSXP, 1));
64  static Symbol* isym = Symbol::obtain("i");
65  Rf_defineVar(isym, ind, rho);
66  INCREMENT_NAMED(ind);
67 
68  Expression* item = new Expression(Symbols::Bracket2Symbol,
69  { X, isym });
70  Expression* R_fcall = new CachingExpression(FUN, { item, Symbols::DotsSymbol });
71 
72  for(R_xlen_t i = 0; i < n; i++) {
73  if (realIndx) REAL(ind)[0] = (double)(i + 1);
74  else INTEGER(ind)[0] = (int)(i + 1);
75  SEXP tmp = R_forceAndCall(R_fcall, 1, rho);
76  if (MAYBE_REFERENCED(tmp)) tmp = Rf_lazy_duplicate(tmp);
77  SET_VECTOR_ELT(ans, i, tmp);
78  }
79 
80  UNPROTECT(4);
81  return ans;
82 }

87 HIDDEN SEXP do_vapply(SEXP call, SEXP op, SEXP args, SEXP rho)
88 {
89  SEXP ans, names = nullptr, rowNames = nullptr,
90  X, XX, FUN, value, dim_v;
91  R_xlen_t i, n;
92  int commonLen;
93  int useNames, rnk_v = -1; // = array_rank(value) := length(dim(value))
94  Rboolean array_value;
95  SEXPTYPE commonType;
96  PROTECT_INDEX index = 0; /* initialize to avoid a warning */
97 
98  PROTECT(X = CAR(args));
99  PROTECT(XX = Rf_eval(CAR(args), rho));
100  FUN = CADR(args); /* must be unevaluated for use in e.g. bquote */
101  PROTECT(value = Rf_eval(CADDR(args), rho));
102  if (!Rf_isVector(value)) Rf_error(_("'FUN.VALUE' must be a vector"));
103  useNames = Rf_asLogical(Rf_eval(CADDDR(args), rho));
104  if (useNames == R_NaLog) Rf_error(_("invalid '%s' value"), "USE.NAMES");
105 
106  n = Rf_xlength(XX);
107  if (n == R_NaInt) Rf_error(_("invalid length"));
108  Rboolean realIndx = Rboolean(n > INT_MAX);
109 
110  commonLen = Rf_length(value);
111  if (commonLen > 1 && n > INT_MAX)
112  Rf_error(_("long vectors are not supported for matrix/array results"));
113  commonType = TYPEOF(value);
114  // check once here
115  if (commonType != CPLXSXP && commonType != REALSXP &&
116  commonType != INTSXP && commonType != LGLSXP &&
117  commonType != RAWSXP && commonType != STRSXP &&
118  commonType != VECSXP)
119  Rf_error(_("type '%s' is not supported"), Rf_type2char(commonType));
120  dim_v = Rf_getAttrib(value, Symbols::DimSymbol);
121  array_value = Rboolean((TYPEOF(dim_v) == INTSXP && LENGTH(dim_v) >= 1));
122  PROTECT(ans = Rf_allocVector(commonType, n*commonLen));
123  if (useNames) {
124  PROTECT(names = Rf_getAttrib(XX, Symbols::NamesSymbol));
125  if (Rf_isNull(names) && TYPEOF(XX) == STRSXP) {
126  UNPROTECT(1);
127  PROTECT(names = XX);
128  }
129  PROTECT_WITH_INDEX(rowNames = Rf_getAttrib(value,
130  array_value ? Symbols::DimNamesSymbol
131  : Symbols::NamesSymbol),
132  &index);
133  }
134  /* The R level code has ensured that XX is a vector.
135  If it is atomic we can speed things up slightly by
136  using the evaluated version.
137  */
138  {
139  SEXP ind;
140  /* Build call: FUN(XX[[<ind>]], ...) */
141 
142  static Symbol* isym = Symbol::obtain("i");
143  PROTECT(ind = Rf_allocVector(INTSXP, 1));
144  Rf_defineVar(isym, ind, rho);
145  INCREMENT_NAMED(ind);
146 
147  Expression* item = new Expression(Symbols::Bracket2Symbol, { X, isym });
148  Expression* R_fcall = new Expression(FUN, { item, Symbols::DotsSymbol });
149 
150  int common_len_offset = 0;
151  for(i = 0; i < n; i++) {
152  SEXP val; SEXPTYPE valType;
153  PROTECT_INDEX indx;
154  if (realIndx) REAL(ind)[0] = (double)(i + 1);
155  else INTEGER(ind)[0] = (int)(i + 1);
156  val = R_forceAndCall(R_fcall, 1, rho);
157  if (MAYBE_REFERENCED(val))
158  val = Rf_lazy_duplicate(val); // Need to duplicate? Copying again anyway
159  PROTECT_WITH_INDEX(val, &indx);
160  if (Rf_length(val) != commonLen)
161  Rf_error(_("values must be length %d,\n but FUN(X[[%d]]) result is length %d"),
162  commonLen, i+1, Rf_length(val));
163  valType = TYPEOF(val);
164  if (valType != commonType) {
165  bool okay = FALSE;
166  switch (commonType) {
167  case CPLXSXP: okay = (valType == REALSXP) || (valType == INTSXP)
168  || (valType == LGLSXP); break;
169  case REALSXP: okay = (valType == INTSXP) || (valType == LGLSXP); break;
170  case INTSXP: okay = (valType == LGLSXP); break;
171  default:
172  Rf_error(_("Internal error: unexpected SEXPTYPE"));
173  }
174  if (!okay)
175  Rf_error(_("values must be type '%s',\n but FUN(X[[%d]]) result is type '%s'"),
176  Rf_type2char(commonType), i+1, Rf_type2char(valType));
177  REPROTECT(val = Rf_coerceVector(val, commonType), indx);
178  }
179  /* Take row names from the first result only */
180  if (i == 0 && useNames && Rf_isNull(rowNames))
181  REPROTECT(rowNames = Rf_getAttrib(val,
182  array_value ? Symbols::DimNamesSymbol : Symbols::NamesSymbol),
183  index);
184  // two cases - only for efficiency
185  if(commonLen == 1) { // common case
186  switch (commonType) {
187  case CPLXSXP: COMPLEX(ans)[i] = COMPLEX(val)[0]; break;
188  case REALSXP: REAL(ans) [i] = REAL (val)[0]; break;
189  case INTSXP: INTEGER(ans)[i] = INTEGER(val)[0]; break;
190  case LGLSXP: LOGICAL(ans)[i] = LOGICAL(val)[0]; break;
191  case RAWSXP: RAW(ans) [i] = RAW (val)[0]; break;
192  case STRSXP: SET_STRING_ELT(ans, i, STRING_ELT(val, 0)); break;
193  case VECSXP: SET_VECTOR_ELT(ans, i, VECTOR_ELT(val, 0)); break;
194  default: Rf_error(_("invalid type")); break;
195  }
196  } else { // commonLen > 1 (typically, or == 0) :
197  switch (commonType) {
198  case REALSXP:
199  memcpy(REAL(ans) + common_len_offset,
200  REAL(val), commonLen * sizeof(double)); break;
201  case INTSXP:
202  memcpy(INTEGER(ans) + common_len_offset,
203  INTEGER(val), commonLen * sizeof(int)); break;
204  case LGLSXP:
205  memcpy(LOGICAL(ans) + common_len_offset,
206  LOGICAL(val), commonLen * sizeof(int)); break;
207  case RAWSXP:
208  memcpy(RAW(ans) + common_len_offset,
209  RAW(val), commonLen * sizeof(Rbyte)); break;
210  case CPLXSXP:
211  memcpy(COMPLEX(ans) + common_len_offset,
212  COMPLEX(val), commonLen * sizeof(Rcomplex)); break;
213  case STRSXP:
214  for (int j = 0; j < commonLen; j++)
215  SET_STRING_ELT(ans, common_len_offset + j, STRING_ELT(val, j));
216  break;
217  case VECSXP:
218  for (int j = 0; j < commonLen; j++)
219  SET_VECTOR_ELT(ans, common_len_offset + j, VECTOR_ELT(val, j));
220  break;
221  default: Rf_error(_("invalid type")); break;
222  }
223  common_len_offset += commonLen;
224  }
225  UNPROTECT(1);
226  }
227  UNPROTECT(1);
228  }
229 
230  if (commonLen != 1) {
231  SEXP dim;
232  rnk_v = array_value ? LENGTH(dim_v) : 1;
233  PROTECT(dim = Rf_allocVector(INTSXP, rnk_v+1));
234  if(array_value)
235  for(int j = 0; j < rnk_v; j++)
236  INTEGER(dim)[j] = INTEGER(dim_v)[j];
237  else
238  INTEGER(dim)[0] = commonLen;
239  INTEGER(dim)[rnk_v] = int(n); // checked above
240  Rf_setAttrib(ans, Symbols::DimSymbol, dim);
241  UNPROTECT(1);
242  }
243 
244  if (useNames) {
245  if (commonLen == 1) {
246  if(!Rf_isNull(names)) Rf_setAttrib(ans, Symbols::NamesSymbol, names);
247  } else {
248  if (!Rf_isNull(names) || !Rf_isNull(rowNames)) {
249  SEXP dimnames;
250  PROTECT(dimnames = Rf_allocVector(VECSXP, rnk_v+1));
251  if(array_value && !Rf_isNull(rowNames)) {
252  if(TYPEOF(rowNames) != VECSXP || LENGTH(rowNames) != rnk_v)
253  // should never happen ..
254  Rf_error(_("dimnames(<value>) is neither NULL nor list of length %d"),
255  rnk_v);
256  for(int j = 0; j < rnk_v; j++)
257  SET_VECTOR_ELT(dimnames, j, VECTOR_ELT(rowNames, j));
258  } else
259  SET_VECTOR_ELT(dimnames, 0, rowNames);
260 
261  SET_VECTOR_ELT(dimnames, rnk_v, names);
262  Rf_setAttrib(ans, Symbols::DimNamesSymbol, dimnames);
263  UNPROTECT(1);
264  }
265  }
266  }
267  UNPROTECT(useNames ? 6 : 4); /* X, XX, value, ans, and maybe names and rowNames */
268  return ans;
269 }

271 static RObject* do_one(RObject* X, RObject* FUN, RObject* classes, RObject* deflt,
272  Rboolean replace, RObject* rho)
273 {
274  SEXP ans, names, klass;
275  int i, j, n;
276  bool matched = false;
277 
278  /* if X is a list, recurse. Otherwise if it matches classes call f */
279  if(Rf_isNewList(X)) {
280  n = Rf_length(X);
281  if (replace) {
282  PROTECT(ans = Rf_shallow_duplicate(X));
283  } else {
284  PROTECT(ans = Rf_allocVector(VECSXP, n));
285  names = Rf_getAttrib(X, Symbols::NamesSymbol);
286  if(!Rf_isNull(names)) Rf_setAttrib(ans, Symbols::NamesSymbol, names);
287  }
288  for(i = 0; i < n; i++)
289  SET_VECTOR_ELT(ans, i, do_one(VECTOR_ELT(X, i), FUN, classes,
290  deflt, replace, rho));
291  UNPROTECT(1);
292  return ans;
293  }
294  if(streql(R_CHAR(STRING_ELT(classes, 0)), "ANY")) /* ASCII */
295  matched = true;
296  else {
297  PROTECT(klass = R_data_class(X, FALSE));
298  for(i = 0; i < LENGTH(klass); i++)
299  for(j = 0; j < Rf_length(classes); j++)
300  if(Rf_Seql(STRING_ELT(klass, i), STRING_ELT(classes, j)))
301  matched = true;
302  UNPROTECT(1);
303  }
304  if(matched) {
305  /* This stores value to which the function is to be applied in
306  a variable X in the environment of the rapply closure call
307  that calls into the rapply .Internal. */
308  SEXP R_fcall; /* could allocate once and preserve for re-use */
309  SEXP Xsym = Rf_install("X");
310  Rf_defineVar(Xsym, X, rho);
311  INCREMENT_NAMED(X);
312  /* PROTECT(R_fcall = Rf_lang2(FUN, Xsym)); */
313  PROTECT(R_fcall = Rf_lang3(FUN, Xsym, Symbols::DotsSymbol));
314  ans = R_forceAndCall(R_fcall, 1, rho);
315  if (MAYBE_REFERENCED(ans))
316  ans = Rf_lazy_duplicate(ans);
317  UNPROTECT(1);
318  return(ans);
319  } else if(replace) return Rf_lazy_duplicate(X);
320  else return Rf_lazy_duplicate(deflt);
321 }

324 {

328

353 static Rboolean islistfactor(RObject* X)
354 {
355  int i, n = Rf_length(X);
356 
357  switch(TYPEOF(X)) {
358  case VECSXP:
359  if(n == 0) return Rboolean(R_NaLog);
360  for(i = 0; i < LENGTH(X); i++)
361  if(!islistfactor(VECTOR_ELT(X, i))) return FALSE;
362  return TRUE;
363  break;
364  case EXPRSXP:
365  if(n == 0) return Rboolean(R_NaLog);
366  for(i = 0; i < LENGTH(X); i++)
367  if(!islistfactor(XVECTOR_ELT(X, i))) return FALSE;
368  return TRUE;
369  break;
370  default: // -Wswitch
371  break;
372  }
373  return Rf_isFactor(X);
374 }

379 HIDDEN SEXP do_islistfactor(/*const*/ Expression* call, const BuiltInFunction* op, RObject* x_, RObject* recursive_)
380 {
381  SEXP X;
382  Rboolean lans = TRUE, recursive;
383  int i, n;
384 
385  X = x_;
386  recursive = Rboolean(Rf_asLogical(recursive_));
387  n = Rf_length(X);
388  if(n == 0 || !Rf_isVectorList(X)) {
389  lans = FALSE;
390  return Rf_ScalarLogical(lans);
391  }
392  if(!recursive) {
393  for(i = 0; i < LENGTH(X); i++)
394  if(!Rf_isFactor(VECTOR_ELT(X, i))) {
395  lans = FALSE;
396  break;
397  }
398  } else {
399  switch(TYPEOF(X)) {
400  case VECSXP:
401  lans = FALSE;
402  for(i = 0; i < LENGTH(X); i++) {
403  Rboolean isfactor = islistfactor(VECTOR_ELT(X, i));
404  if(!isfactor) {
405  lans = FALSE;
406  break;
407  } else if (isfactor == TRUE)
408  lans = TRUE;
409  }
410  break;
411  case EXPRSXP:
412  lans = FALSE;
413  for(i = 0; i < LENGTH(X); i++) {
414  Rboolean isfactor = islistfactor(XVECTOR_ELT(X, i));
415  if(!isfactor) {
416  lans = FALSE;
417  break;
418  } else if (isfactor == TRUE)
419  lans = TRUE;
420  }
421  break;
422  default:
423  break;
424  }
425  }
426  return Rf_ScalarLogical(lans);
