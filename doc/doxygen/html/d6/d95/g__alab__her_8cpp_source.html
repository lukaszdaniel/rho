144 static void _draw_stroke (vfontContext *vc, const pGEcontext gc, pGEDevDesc dd,
145  bool pendown, double deltax, double deltay)
146 {
147  double dx, dy;
148  double theta;
149 
150  theta = M_PI * vc->angle / 180.0;
151 
152  dx = cos(theta) * deltax - sin(theta) * deltay;
153  dy = sin(theta) * deltax + cos(theta) * deltay;
154 
155  if (pendown)
156  linerel(dx, dy, vc, gc, dd);
157  else
158  moverel(dx, dy, vc);
159 }

404 /* _label_width_hershey() computes the width (total delta x) of a
405  controlified character string to be rendered in a vector font, in user
406  units */
407 static double _label_width_hershey (const pGEcontext gc, pGEDevDesc dd,
408  const unsigned short *label)
409 {
410  const unsigned short *ptr = label;
411  unsigned short c;
412  double charsize = 1.0; /* relative char size, 1.0 means full size */
413  double saved_charsize = 1.0;
414  double width = 0.0; /* label width */
415  double saved_width = 0.0;
416 
417  /* loop through unsigned shorts in label */
418  while ((c = (*ptr)) != (unsigned short)'\0')
419  {
420  int glyphnum; /* glyph in Hershey array */
421  const unsigned char *glyph;
422 
423  if (c & RAW_HERSHEY_GLYPH)
424  /* glyph was spec'd via an escape, not as a char in a font */
425  {
426  glyphnum = c & GLYPH_SPEC;
427  glyph = (const unsigned char *)(_occidental_hershey_glyphs[glyphnum]);
428 
429  if (*glyph != '\0') /* nonempty glyph */
430  /* 1st two chars are bounds */
431  width += charsize * ((int)glyph[1] - (int)glyph[0]);
432  }
433  else if (c & RAW_ORIENTAL_HERSHEY_GLYPH)
434  /* glyph was spec'd via an escape, not as a char in a font */
435  {
436  glyphnum = c & GLYPH_SPEC;
437  glyph = (const unsigned char *)_oriental_hershey_glyphs[glyphnum];
438 
439  if (*glyph != '\0') /* nonempty glyph */
440  /* 1st two chars are bounds */
441  width += charsize * ((int)glyph[1] - (int)glyph[0]);
442  }
443  else if (c & CONTROL_CODE) /* parse control code */
444  {
445  switch (c & ~CONTROL_CODE)
446  {
447  case C_BEGIN_SUBSCRIPT:
448  case C_BEGIN_SUPERSCRIPT :
449  charsize *= SCRIPTSIZE;
450  break;
451 
452  case C_END_SUBSCRIPT:
453  case C_END_SUPERSCRIPT:
454  charsize /= SCRIPTSIZE;
455  break;
456 
457  case C_PUSH_LOCATION:
458  saved_width = width;
459  saved_charsize = charsize;
460  break;
461 
462  case C_POP_LOCATION:
463  width = saved_width;
464  charsize = saved_charsize;
465  break;
466 
467  case C_RIGHT_ONE_EM:
468  width += charsize * HERSHEY_EM;
469  break;
470 
471  case C_RIGHT_HALF_EM:
472  width += charsize * HERSHEY_EM / 2.0;
473  break;
474 
475  case C_RIGHT_QUARTER_EM:
476  width += charsize * HERSHEY_EM / 4.0;
477  break;
478 
479  case C_RIGHT_SIXTH_EM:
480  width += charsize * HERSHEY_EM / 6.0;
481  break;
482 
483  case C_RIGHT_EIGHTH_EM:
484  width += charsize * HERSHEY_EM / 8.0;
485  break;
486 
487  case C_RIGHT_TWELFTH_EM:
488  width += charsize * HERSHEY_EM / 12.0;
489  break;
490 
491  case C_LEFT_ONE_EM:
492  width -= charsize * HERSHEY_EM;
493  break;
494 
495  case C_LEFT_HALF_EM:
496  width -= charsize * HERSHEY_EM / 2.0;
497  break;
498 
499  case C_LEFT_QUARTER_EM:
500  width -= charsize * HERSHEY_EM / 4.0;
501  break;
502 
503  case C_LEFT_SIXTH_EM:
504  width -= charsize * HERSHEY_EM / 6.0;
505  break;
506 
507  case C_LEFT_EIGHTH_EM:
508  width -= charsize * HERSHEY_EM / 8.0;
509  break;
510 
511  case C_LEFT_TWELFTH_EM:
512  width -= charsize * HERSHEY_EM / 12.0;
513  break;
514 
515  /* unrecognized control code */
516  default:
517  break;
518  }
519  }
520  else /* yow, an actual character */
521  {
522  int raw_fontnum;
523 
524  /* compute index of font, in table in g_fontdb.c */
525  raw_fontnum = (c >> FONT_SHIFT) & ONE_BYTE;
526 
527  c &= ~FONT_SPEC; /* extract character proper */
528  glyphnum = (_hershey_font_info[raw_fontnum].chars)[c];
529 
530  /* could be a pseudo glyph number, e.g. an indication that
531  character is composite */
532  if (glyphnum == ACC0 || glyphnum == ACC1 || glyphnum == ACC2)
533  {
534  unsigned char composite, character = '\0' /* -Wall */, accent;
535 
536  /* if so, use 1st element of composite character */
537  composite = (unsigned char)c;
538  if (_composite_char (&composite, &character, &accent))
539  glyphnum = (_hershey_font_info[raw_fontnum].chars)[character];
540  else
541  glyphnum = UNDE; /* hope this won't happen */
542  }
543 
544  /* could also be a glyph number displaced by KS, to indicate
545  that this is a small kana */
546  if (glyphnum & KS)
547  glyphnum -= KS;
548 
549  glyph = (const unsigned char *)(_occidental_hershey_glyphs[glyphnum]);
550  if (*glyph != '\0') /* nonempty glyph */
551  /* 1st two chars are bounds */
552  width += charsize * ((int)glyph[1] - (int)glyph[0]);
553  }
554 
555  ptr++; /* bump pointer in string */
556  }
557 
558  return HERSHEY_X_UNITS_TO_USER_UNITS (width);
559 }

580 /* _draw_hershey_glyph() invokes move() and cont() to draw a raw Hershey
581  glyph, specified by index in the occidental or oriental glyph arrays.
582  Size scaling and obliquing (true/false) are specified. */
583 static
584 void _draw_hershey_glyph (vfontContext *vc, const pGEcontext gc, pGEDevDesc dd,
585  int glyphnum,
586  double charsize, int type, bool oblique)
587 {
588  double xcurr, ycurr;
589  double xfinal, yfinal;
590  bool pendown = false;
591  const unsigned char *glyph;
592  double dx, dy;
593  double shear;
594 
595  shear = oblique ? (SHEAR) : 0.0;
596  switch (type)
597  {
598  case OCCIDENTAL:
599  default:
600  glyph = (const unsigned char *)(_occidental_hershey_glyphs[glyphnum]);
601  break;
602  case ORIENTAL:
603  glyph = (const unsigned char *)(_oriental_hershey_glyphs[glyphnum]);
604  break;
605  }
606 
607  if (*glyph != '\0') /* nonempty glyph */
608  {
609  xcurr = charsize * (double)glyph[0];
610  xfinal = charsize * (double)glyph[1];
611  ycurr = yfinal = 0.0;
612  glyph += 2;
613  while (*glyph)
614  {
615  int xnewint;
616 
617  xnewint = (int)glyph[0];
618 
619  if (xnewint == (int)' ')
620  pendown = false;
621  else
622  {
623  double xnew, ynew;
624 
625  xnew = (double)charsize * xnewint;
626  ynew = (double)charsize
627  * ((int)'R'
628  - ((int)glyph[1] + (double)HERSHEY_BASELINE));
629  dx = xnew - xcurr;
630  dy = ynew - ycurr;
631  _draw_hershey_stroke (vc, gc, dd,
632  pendown, dx + shear * dy, dy);
633  xcurr = xnew, ycurr = ynew;
634  pendown = true;
635  }
636 
637  glyph +=2; /* on to next pair */
638  }
639 
640  /* final penup stroke, to end where we should */
641  dx = xfinal - xcurr;
642  dy = yfinal - ycurr;
643  _draw_hershey_stroke (vc, gc, dd, false, dx + shear * dy, dy);
644  }
645 }

660  {

662  glyph number, (2) an oriental Hershey glyph specified by glyph

664  which will be mapped to a Hershey glyph by one of the tables in

673  }

675  c & GLYPH_SPEC, charsize, OCCIDENTAL, false);

683  line_width_type = 2;
684  }
685  _draw_hershey_glyph (vc, gc, dd,
686  c & GLYPH_SPEC, charsize, ORIENTAL, false);
687  }

693  _draw_hershey_stroke (vc, gc, dd,
694  false,
695  SUPERSCRIPT_DX * charsize * HERSHEY_EM,
696  SUPERSCRIPT_DY * charsize * HERSHEY_EM);
697  charsize *= SCRIPTSIZE;
698  break;
699

702  _draw_hershey_stroke (vc, gc, dd,
703  false,
704  - SUPERSCRIPT_DX * charsize * HERSHEY_EM,
705  - SUPERSCRIPT_DY * charsize * HERSHEY_EM);
706  break;
707 
708  case C_BEGIN_SUBSCRIPT:
709  _draw_hershey_stroke (vc, gc, dd,
710  false,
711  SUBSCRIPT_DX * charsize * HERSHEY_EM,
712  SUBSCRIPT_DY * charsize * HERSHEY_EM);
713  charsize *= SCRIPTSIZE;
714  break;
715 
716  case C_END_SUBSCRIPT:
717  charsize /= SCRIPTSIZE;
718  _draw_hershey_stroke (vc, gc, dd,
719  false,

721  - SUBSCRIPT_DY * charsize * HERSHEY_EM);
722  break;

725  /* saved_charsize = charsize;
726  saved_position_x = _plotter->drawstate->pos.x;
727  saved_position_y = _plotter->drawstate->pos.y; */
728  break;

732  _plotter->fmove (R___(_plotter)

734  break;
735 
736  case C_RIGHT_ONE_EM:
737  _draw_hershey_stroke (vc, gc, dd,
738  false, charsize * HERSHEY_EM, 0.0);
739  break;
740 
741  case C_RIGHT_HALF_EM:
742  _draw_hershey_stroke (vc, gc, dd,
743  false, charsize * HERSHEY_EM / 2.0, 0.0);
744  break;
745 
746  case C_RIGHT_QUARTER_EM:
747  _draw_hershey_stroke (vc, gc, dd,
748  false, charsize * HERSHEY_EM / 4.0, 0.0);
749  break;

752  _draw_hershey_stroke (vc, gc, dd,
753  false, charsize * HERSHEY_EM / 6.0, 0.0);
754  break;
755 
756  case C_RIGHT_EIGHTH_EM:
757  _draw_hershey_stroke (vc, gc, dd,
758  false, charsize * HERSHEY_EM / 8.0, 0.0);
759  break;
760 
761  case C_LEFT_ONE_EM:
762  _draw_hershey_stroke (vc, gc, dd,
763  false, - charsize * HERSHEY_EM, 0.0);
764  break;
765 
766  case C_LEFT_HALF_EM:
767  _draw_hershey_stroke (vc, gc, dd,
768  false, - charsize * HERSHEY_EM / 2.0, 0.0);
769  break;
770 
771  case C_LEFT_QUARTER_EM:
772  _draw_hershey_stroke (vc, gc, dd,
773  false, - charsize * HERSHEY_EM / 4.0, 0.0);
774  break;
775 
776  case C_LEFT_SIXTH_EM:
777  _draw_hershey_stroke (vc, gc, dd,
778  false, - charsize * HERSHEY_EM / 6.0, 0.0);
779  break;
780 
781  case C_LEFT_EIGHTH_EM:
782  _draw_hershey_stroke (vc, gc, dd,
783  false, - charsize * HERSHEY_EM / 8.0, 0.0);
784  break;
785

787  default:
788  break;
789  }
790

792  /* yow, an actual font character! Several possibilities: could be
793  a composite (accented) character, could be a small Kana, or
794  could be a garden-variety character. */
795  {
796  int raw_fontnum;
797  int glyphnum; /* glyph in Hershey array */
798  int char_glyphnum, accent_glyphnum; /* for composite chars */
799  int char_width, accent_width; /* for composite chars */
800  const unsigned char *char_glyph, *accent_glyph;
801  unsigned char composite, character = '\0', accent = '\0' /* -Wall */;
802  bool oblique, small_kana = false;
803

805  raw_fontnum = (c >> FONT_SHIFT) & ONE_BYTE;
806  /* shear font? (for HersheySans-Oblique, etc.) */
807  oblique = _hershey_font_info[raw_fontnum].obliquing;
808

811

816  }
817

821  search font table in g_fontdb.c for it */

823  case ACC1:
824  case ACC2:

846  else

848

851  accent_width = (int)accent_glyph[1] - (int)accent_glyph[0];
852  else

856  if (line_width_type != 1)

859  line_width_type = 1;
860  }
861  _draw_hershey_glyph (vc, gc, dd,
862  char_glyphnum, charsize,
863  OCCIDENTAL, oblique);
864  /* back up to draw accent */
865  _draw_hershey_penup_stroke (vc, gc, dd,
866  -0.5 * (double)char_width
867  -0.5 * (double)accent_width,
868  0.0, charsize, oblique);
869 
870  /* repositioning for uppercase and uppercase italic */
871  if (glyphnum == ACC1)
872  _draw_hershey_penup_stroke (vc, gc, dd,
873  0.0,
874  (double)(ACCENT_UP_SHIFT),
875  charsize, oblique);
876  else if (glyphnum == ACC2)
877  _draw_hershey_penup_stroke (vc, gc, dd,
878  (double)(ACCENT_RIGHT_SHIFT),
879  (double)(ACCENT_UP_SHIFT),
880  charsize, oblique);
881 
882  /* draw the accent */
883  _draw_hershey_glyph (vc, gc, dd,
884  accent_glyphnum, charsize,
885  OCCIDENTAL, oblique);
886 
887  /* undo special repositioning if any */
888  if (glyphnum == ACC1)
889  _draw_hershey_penup_stroke (vc, gc, dd,
890  0.0,
891  -(double)(ACCENT_UP_SHIFT),
892  charsize, oblique);
893  else if (glyphnum == ACC2)
894  _draw_hershey_penup_stroke (vc, gc, dd,
895  -(double)(ACCENT_RIGHT_SHIFT),
896  -(double)(ACCENT_UP_SHIFT),
897  charsize, oblique);
898 
899  /* move forward, to end composite char where we should */
900  _draw_hershey_penup_stroke (vc, gc, dd,
901  0.5 * (double)char_width
902  -0.5 * (double)accent_width,
903  0.0, charsize, oblique);
904  break;

906  /* not a composite (accented) character; just an ordinary
907  glyph from occidental+Kana array (could be a Kana, in
908  particular, could be a small Kana) */
909  default:
910  if (small_kana)
911  {

914  double shift = 0.5 * (1.0 - (SMALL_KANA_SIZE));
915 
916  kana_glyph =
917  (const unsigned char *)_occidental_hershey_glyphs[glyphnum];
918  kana_width = (int)kana_glyph[1] - (int)kana_glyph[0];
919 
920  /* draw small Kana, preceded and followed by a penup
921  stroke in order to traverse the full width of an
922  ordinary Kana */
923  _draw_hershey_penup_stroke (vc, gc, dd,
924  shift * (double)kana_width,
925  0.0, charsize, oblique);
926  if (line_width_type != 2)
927  {
928  gc->lwd = HERSHEY_LINE_WIDTH_TO_LWD (HERSHEY_STROKE_WIDTH);
929  line_width_type = 2;
930  }
931  _draw_hershey_glyph (vc, gc, dd,
932  glyphnum,
933  (SMALL_KANA_SIZE) * charsize,
934  OCCIDENTAL, oblique);
935  _draw_hershey_penup_stroke (vc, gc, dd,
936  shift * (double)kana_width,
937  0.0, charsize, oblique);
938  }
939  else
940  /* whew! just an ordinary glyph from the occidental array
941  (could be a Kana however, since they're confusingly
942  placed in that array, at the end) */
943  {
944  if (glyphnum >= BEGINNING_OF_KANA)
945  {
946  if (line_width_type != 2)
947  {
948  gc->lwd = HERSHEY_LINE_WIDTH_TO_LWD (HERSHEY_ORIENTAL_STROKE_WIDTH);
949  line_width_type = 2;
950  }
951  }
952  else
953  if (line_width_type != 1)
954  {
955  gc->lwd = HERSHEY_LINE_WIDTH_TO_LWD (HERSHEY_STROKE_WIDTH);
956  line_width_type = 1;
957  }
958  _draw_hershey_glyph (vc, gc, dd,
959  glyphnum, charsize,
960  OCCIDENTAL, oblique);
961  }
962  break;
963  } /* end of case statement that switches based on glyphnum */
964 
965  } /* end of font character case */
966 
967  } /* end of loop through unsigned shorts in the codestring */
968 
969  return;
