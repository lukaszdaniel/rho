42 /* used in graphics and grid */
43 SEXP CreateAtVector(double *axp, double *usr, int nint, Rboolean logflag)
44 {
45 /* Create an 'at = ...' vector for axis(.)
46  * i.e., the vector of tick mark locations,
47  * when none has been specified (= default).
48  *
49  * axp[0:2] = (x1, x2, nInt), where x1..x2 are the extreme tick marks
50  * {unless in log case, where nInt \in {1,2,3 ; -1,-2,....}
51  * and the `nint' argument is used *instead*.}
52 
53  * The resulting REAL vector must have length >= 1, ideally >= 2
54  */
55  SEXP at = R_NilValue;/* -Wall*/
56  double umin, umax, dn, rng, small;
57  int i, n, ne;
58  if (!logflag || axp[2] < 0) { /* --- linear axis --- Only use axp[] arg. */
59  n = int(fabs(axp[2]) + 0.25);/* >= 0 */
60  dn = imax2(1, n);
61  rng = axp[1] - axp[0];
62  small = fabs(rng)/(100.*dn);
63  at = allocVector(REALSXP, n + 1);
64  for (i = 0; i <= n; i++) {
65  REAL(at)[i] = axp[0] + (i / dn) * rng;
66  if (fabs(REAL(at)[i]) < small)
67  REAL(at)[i] = 0;
68  }
69  }
70  else { /* ------ log axis ----- */
71  Rboolean reversed = FALSE;
72 
73  n = int(axp[2] + 0.5);
74  /* {xy}axp[2] for 'log': GLpretty() [./graphics.cpp] sets
75  n < 0: very small scale ==> linear axis, above, or
76  n = 1,2,3. see switch() below */
77  umin = usr[0];
78  umax = usr[1];
79  if (umin > umax) {
80  reversed = RHOCONSTRUCT(Rboolean, (axp[0] > axp[1]));
81  if (reversed) {
82  /* have *reversed* log axis -- whereas
83  * the switch(n) { .. } below assumes *increasing* values
84  * --> reverse axis direction here, and reverse back at end */
85  umin = usr[1];
86  umax = usr[0];
87  dn = axp[0]; axp[0] = axp[1]; axp[1] = dn;
88  }
89  else {
90  /* can the following still happen... ? */
91  warning("CreateAtVector \"log\"(from axis()): "
92  "usr[0] = %g > %g = usr[1] !", umin, umax);
93  }
94  }
95  /* allow a fuzz since we will do things like 0.2*dn >= umin */
96  umin *= 1 - 1e-12;
97  umax *= 1 + 1e-12;
98 
99  dn = axp[0];
100  if (dn < DBL_MIN) {/* was 1e-300; now seems too cautious */
101  warning("CreateAtVector \"log\"(from axis()): axp[0] = %g !", dn);
102  if (dn <= 0) /* real trouble (once for Solaris) later on */
103  error("CreateAtVector [log-axis()]: axp[0] = %g < 0!", dn);
104  }
105 
106  /* You get the 3 cases below by
107  * for (y in 1e-5*c(1,2,8)) plot(y, log = "y")
108  */
109  switch(n) {
110  case 1: /* large range: 1 * 10^k */
111  i = int(floor(log10(axp[1])) - ceil(log10(axp[0])) + 0.25);
112  ne = i / nint + 1;
113 #ifdef DEBUG_axis
114  REprintf("CreateAtVector [log-axis(), case 1]: (nint, ne) = (%d,%d)\n",
115  nint, ne);
116 #endif
117  if (ne < 1)
118  error("log - axis(), 'at' creation, _LARGE_ range: "
119  "ne = %d <= 0 !!\n"
120  "\t axp[0:1]=(%g,%g) ==> i = %d; nint = %d",
121  ne, axp[0],axp[1], i, nint);
122  rng = Rexp10((double)ne); /* >= 10 */
123  n = 0;
124  while (dn < umax) {
125  n++;
126  dn *= rng;
127  }
128  if (!n)
129  error("log - axis(), 'at' creation, _LARGE_ range: "
130  "invalid {xy}axp or par; nint=%d\n"
131  " axp[0:1]=(%g,%g), usr[0:1]=(%g,%g); i=%d, ni=%d",
132  nint, axp[0],axp[1], umin,umax, i,ne);
133  at = allocVector(REALSXP, n);
134  dn = axp[0];
135  n = 0;
136  while (dn < umax) {
137  REAL(at)[n++] = dn;
138  dn *= rng;
139  }
140  break;
141 
142  case 2: /* medium range: 1, 5 * 10^k */
143  n = 0;
144  if (0.5 * dn >= umin) n++;
145  for (;;) {
146  if (dn > umax) break;
147  n++;
148  if (5 * dn > umax) break;
149  n++;
150  dn *= 10;
151  }
152  if (!n)
153  error("log - axis(), 'at' creation, _MEDIUM_ range: "
154  "invalid {xy}axp or par;\n"
155  " axp[0]= %g, usr[0:1]=(%g,%g)",
156  axp[0], umin,umax);
157 
158  at = allocVector(REALSXP, n);
159  dn = axp[0];
160  n = 0;
161  if (0.5 * dn >= umin) REAL(at)[n++] = 0.5 * dn;
162  for (;;) {
163  if (dn > umax)
164  break;
165  REAL(at)[n++] = dn;
166  if (5 * dn > umax)
167  break;
168  REAL(at)[n++] = 5 * dn;
169  dn *= 10;
170  }
171  break;
172 
173  case 3: /* small range: 1,2,5,10 * 10^k */
174  n = 0;
175  if (0.2 * dn >= umin) n++;
176  if (0.5 * dn >= umin) n++;
177  for (;;) {
178  if (dn > umax)
179  break;
180  n++;
181  if (2 * dn > umax)
182  break;
183  n++;
184  if (5 * dn > umax)
185  break;
186  n++;
187  dn *= 10;
188  }
189  if (!n)
190  error("log - axis(), 'at' creation, _SMALL_ range: "
191  "invalid {xy}axp or par;\n"
192  " axp[0]= %g, usr[0:1]=(%g,%g)",
193  axp[0], umin,umax);
194  at = allocVector(REALSXP, n);
195  dn = axp[0];
196  n = 0;
197  if (0.2 * dn >= umin) REAL(at)[n++] = 0.2 * dn;
198  if (0.5 * dn >= umin) REAL(at)[n++] = 0.5 * dn;
199  for (;;) {
200  if (dn > umax)
201  break;
202  REAL(at)[n++] = dn;
203  if (2 * dn > umax)
204  break;
205  REAL(at)[n++] = 2 * dn;
206  if (5 * dn > umax)
207  break;
208  REAL(at)[n++] = 5 * dn;
209  dn *= 10;
210  }
211  break;
212  default:
213  error("log - axis(), 'at' creation: INVALID {xy}axp[3] = %g",
214  axp[2]);
215  }
216 
217  if (reversed) {/* reverse back again - last assignment was at[n++]= . */
218  for (i = 0; i < n/2; i++) { /* swap( at[i], at[n-i-1] ) : */
219  dn = REAL(at)[i];
220  REAL(at)[i] = REAL(at)[n-i-1];
221  REAL(at)[n-i-1] = dn;
222  }
223  }
224  } /* linear / log */
225  return at;
226 }
