836  template <class VL, class VR>
837  VL* Subscripting::arraySubassign(VL* lhs,
838  const std::vector<Indices>& indicesvec,
839  const VR* rhs)
840  {
841  typedef typename VL::value_type Lval;
842  typedef typename VR::value_type Rval;
843  const IntVector* vdims = lhs->dimensions();
844  std::size_t ndims = vdims->size();
845  DimIndexerVector dimindexer(ndims);
846  std::size_t ni = createDimIndexers(&dimindexer, vdims, indicesvec);
847  std::size_t rhs_size = rhs->size();
848  if (rhs_size > 1) {
849  // TODO: Move NA-detection back into the canonicalisation
850  // process.
851  for (std::size_t d = 0; d < ndims; ++d) {
852  const Indices& dimindices = indicesvec[d];
853  std::size_t dsz = dimindices.size();
854  for (std::size_t i = 0; i < dsz; ++i)
855  if (dimindices[i] == 0)
856  Rf_error(_("NAs are not allowed in subscripted assignments"));
857  }
858  }
859  GCStackRoot<VL> ans(lhs);
860  // If necessary, make a copy to be sure we don't modify rhs.
861  // (FIXME: ideally this should be a shallow copy for
862  // HandleVectors.)
863  const VectorBase* ansvb = static_cast<VectorBase*>(ans.get());
864  if (ansvb == rhs)
865  ans = ans->clone();
866  // Dispose of 'no indices' case:
867  if (ni == 0)
868  return ans;
869  if (rhs_size == 0)
870  Rf_error(_("replacement has length zero"));
871  if (ni%rhs_size != 0)
872  Rf_warning(_("number of items to replace is not"
873  " a multiple of replacement length"));
874  // Copy elements across:
875  for (std::size_t irhs = 0; irhs < ni; ++irhs) {
876  bool naindex = false;
877  std::size_t iout = 0;
878  for (std::size_t d = 0; d < ndims; ++d) {
879  const DimIndexer& di = dimindexer[d];
880  const Indices& dimindices = indicesvec[d];
881  std::size_t index = dimindices[di.indexnum];
882  if (index == 0) {
883  naindex = true;
884  break;
885  }
886  iout += (index - 1)*di.stride;
887  }
888  if (!naindex) {
889  const Rval& rval = (*rhs)[irhs % rhs_size];
890  if (isNA(rval)) {
891  assignDuplicateElement((*ans)[iout], NA<Lval>());
892  } else {
893  assignDuplicateElement((*ans)[iout], rval);
894  }
895  }
896 
897  // Advance the index selection:
898  {
899  std::size_t d = 0;
900  bool done;
901  do {
902  done = true;
903  DimIndexer& di = dimindexer[d];
904  if (++di.indexnum >= indicesvec[d].size()) {
905  di.indexnum = 0;
906  done = (++d >= ndims);
907  }
908  } while (!done);
909  }
910  }
911  return ans;
912  }

926  // Copy elements across:

930  V* vnc = const_cast<V*>(v);
931  for (std::size_t iout = 0; iout < resultsize; ++iout) {
932  bool naindex = false;
933  std::size_t iin = 0;
934  for (std::size_t d = 0; d < ndims; ++d) {
935  const DimIndexer& di = dimindexer[d];

993  template <class VL, class VR>
994  VL* Subscripting::vectorSubassign(VL* lhs,
995  const Indices& indices,
996  const VR* rhs)
997  {
998  typedef typename VL::value_type Lval;
999  typedef typename VR::value_type Rval;
1000  std::size_t ni = indices.size();
1001  std::size_t rhs_size = rhs->size();
1002  if (rhs_size > 1) {
1003  for (std::size_t i = 0; i < ni; ++i)
1004  if (indices[i] == 0)
1005  Rf_error(_("NAs are not allowed in subscripted assignments"));
1006  }
1007  GCStackRoot<VL> ans(lhs);
1008  std::size_t minsize = indices.minimumLHSSize();
1009  if (minsize > lhs->size())
1010  ans = VectorBase::resize(lhs, minsize);
1011  // If necessary, make a copy to be sure we don't modify rhs or
1012  // indices. (FIXME: ideally this should be a shallow copy for
1013  // HandleVectors.)
1014  const VectorBase* ansvb = static_cast<VectorBase*>(ans.get());
1015  if (ansvb == rhs)
1016  ans = ans->clone();
1017  // Dispose of 'no indices' case:
1018  if (ni == 0)
1019  return ans;
1020  if (rhs_size == 0)
1021  Rf_error(_("replacement has length zero"));
1022  if (ni%rhs_size != 0)
1023  Rf_warning(_("number of items to replace is not"
1024  " a multiple of replacement length"));
1025  for (std::size_t i = 0; i < ni; ++i) {
1026  std::size_t index = indices[i];
1027  if (index != 0) {
1028  const Rval& rval = (*rhs)[i % rhs_size];
1029  if (isNA(rval)) {
1030  assignDuplicateElement((*ans)[index - 1], NA<Lval>());
1031  } else {
1032  assignDuplicateElement((*ans)[index - 1], rval);
1033  }
1034  }
1035  }
1036  indices.applyNewNames(ans);
1037  return ans;
1038  }

1040  template <class V>
1041  V* Subscripting::vectorSubset(const V* v, const Indices& indices)
1042  {
1043  std::size_t ni = indices.size();
1044  GCStackRoot<V> ans(V::create(ni));
1045  std::size_t vsize = v->size();
1046  // ***** FIXME ***** Currently needed because Handle's
1047  // assignment operator takes a non-const RHS:
1048  V* vnc = const_cast<V*>(v);
1049  for (std::size_t i = 0; i < ni; ++i) {
1050  std::size_t index = indices[i];
1051  // Note that zero and negative indices ought not to occur.
1052  if (index == 0 || index > vsize) {
1053  (*ans)[i] = ElementTraits::duplicate_element(
1054  NA<typename V::value_type>());
1055  } else {
1056  (*ans)[i] = ElementTraits::duplicate_element(
1057  (*vnc)[index - 1]);
1058  }
1059  }
1060  setVectorAttributes(ans, v, indices);
1061  return ans;
1062  }
