41 SEXP attribute_hidden do_debug(SEXP call, SEXP op, SEXP args, SEXP rho)
42 {
43  SEXP ans = R_NilValue;
44 
45 #define find_char_fun \
46  if (isValidString(CAR(args))) { \
47  SEXP s; \
48  PROTECT(s = installTrChar(STRING_ELT(CAR(args), 0))); \
49  SETCAR(args, findFun(s, rho)); \
50  UNPROTECT(1); \
51  }
52  find_char_fun
53 
54  if (TYPEOF(CAR(args)) != CLOSXP &&
55  TYPEOF(CAR(args)) != SPECIALSXP &&
56  TYPEOF(CAR(args)) != BUILTINSXP)
57  Rf_error(_("argument must be a function"));
58  switch(PRIMVAL(op)) {
59  case 0: // debug()
60  SET_RDEBUG(CAR(args), RHO_TRUE);
61  break;
62  case 1: // undebug()
63  if( RDEBUG(CAR(args)) != 1 )
64  Rf_warning("argument is not being debugged");
65  SET_RDEBUG(CAR(args), RHO_FALSE);
66  break;
67  case 2: // isdebugged()
68  ans = Rf_ScalarLogical(RDEBUG(CAR(args)));
69  break;
70  case 3: // debugonce()
71  SET_RSTEP(CAR(args), 1);
72  break;
73  }
74  return ans;
75 }

101 SEXP attribute_hidden do_traceOnOff(/*const*/ Expression* call, const BuiltInFunction* op, RObject* on_)
102 {
103  SEXP onOff = on_;
104  bool trace = op->variant() == 0; // Otherwise it's debug.
105  Rboolean prev = Rboolean(trace ? FunctionBase::tracingEnabled()
106  : Closure::debuggingEnabled());
107  if(Rf_length(onOff) > 0) {
108  Rboolean _new = RHOCONSTRUCT(Rboolean, Rf_asLogical(onOff));
109  if(_new == TRUE || _new == FALSE) {
110  if (trace)
111  FunctionBase::enableTracing(_new);
112  else
113  Closure::enableDebugging(_new);
114  }
115  else
116  Rf_error(_("Value for '%s' must be TRUE or FALSE"),
117  trace ? "tracingState" : "debuggingState");
118  }
119  return Rf_ScalarLogical(prev);
120 }

147  Rf_warningcall(call,
148  _("'tracemem' is not useful for weak reference or external pointer objects"));

150  object->setMemoryTracing(true);
151  snprintf(buffer, 21, "<%p>", (void *) object);
152  return mkString(buffer);

180 static void memtrace_stack_dump(void)
181 {
182  Evaluator::Context *cptr;
183 
184  for (cptr = Evaluator::Context::innermost();
185  cptr; cptr = cptr->nextOut()) {
186  Evaluator::Context::Type type = cptr->type();
187  if (type == Evaluator::Context::FUNCTION
188  || type == Evaluator::Context::CLOSURE) {
189  FunctionContext* fctxt = static_cast<FunctionContext*>(cptr);
190  SEXP fun = fctxt->call()->car();
191  Rprintf("%s ",
192  TYPEOF(fun) == SYMSXP ? Rf_translateChar(PRINTNAME(fun)) :
193  "<Anonymous>");
194  }
195  }
196  Rprintf("\n");
197 }

226 SEXP do_retracemem(SEXP call, SEXP op, SEXP arg, SEXP rho)
227 {
228 #ifdef R_MEMORY_PROFILING
229  SEXP object, previous, ans;
230  char buffer[21];
231 
232  static GCRoot<ArgMatcher> matcher = new ArgMatcher({ "x", "previous" });
233  matcher->match(&arglist, { &object, &previous });
234  if (object == R_MissingArg)
235  object = R_NilValue;
236  if (previous == R_MissingArg)
237  previous = R_NilValue;
238 
239  if(!Rf_isNull(previous) && (!Rf_isString(previous) || LENGTH(previous) != 1))
240  Rf_errorcall(call, _("invalid '%s' argument"), "previous");
241 
242  if (object->memoryTraced()){
243  snprintf(buffer, 21, "<%p>", (void *) object);
244  ans = mkString(buffer);
245  } else {
246  R_Visible = RHO_FALSE;
247  ans = R_NilValue;
248  }
249 
250  if (previous != R_NilValue){
251  object->setMemoryTracing(true);
252  if (R_current_trace_state()) {
253  /* FIXME: previous will have <0x....> whereas other values are
254  without the < > */
255  Rprintf("tracemem[%s -> %p]: ",
256  Rf_translateChar(STRING_ELT(previous, 0)), (void *) object);
257  memtrace_stack_dump();
258  }
259  }
260  return ans;
261 #else
262  R_Visible = RHO_FALSE; /* for consistency with other case */
263  return R_NilValue;
264 #endif
