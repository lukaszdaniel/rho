55 HIDDEN SEXP do_rawToChar(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* multiple_)
56 {
57  SEXP ans, x = x_;
58 
59  if (!isRaw(x))
60  Rf_error(_("argument 'x' must be a raw vector"));
61  int multiple = Rf_asLogical(multiple_);
62  if (multiple == R_NaLog)
63  Rf_error(_("argument 'multiple' must be TRUE or FALSE"));
64  if (multiple) {
65  R_xlen_t i, nc = XLENGTH(x);
66  char buf[2];
67  buf[1] = '\0';
68  PROTECT(ans = Rf_allocVector(STRSXP, nc));
69  for (i = 0; i < nc; i++) {
70  buf[0] = char(RAW(x)[i]);
71  SET_STRING_ELT(ans, i, Rf_mkChar(buf));
72  }
73  /* do we want to copy e.g. names here? */
74  UNPROTECT(1);
75  return ans;
76  } else {
77  int i, j, nc = LENGTH(x);
78  /* String is not necessarily 0-terminated and may contain nuls.
79  Strip trailing nuls */
80  for (i = 0, j = -1; i < nc; i++) if(RAW(x)[i]) j = i;
81  nc = j + 1;
82  return Rf_ScalarString(
83  Rf_mkCharLenCE(reinterpret_cast<const char *>(RAW(x)), j+1, CE_NATIVE));
84  }
85 }

97  Rf_error(_("argument 'shift' must be a small integer"));

105  UNPROTECT(1);

146 HIDDEN SEXP do_packBits(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* type_)
147 {
148  SEXP ans, x = x_, stype = type_;
149  Rboolean useRaw;
150  R_xlen_t i, len = XLENGTH(x), slen;
151  int fac;
152 
153  if (TYPEOF(x) != RAWSXP && TYPEOF(x) != LGLSXP && TYPEOF(x) != INTSXP)
154  Rf_error(_("argument 'x' must be raw, integer or logical"));
155  if (!Rf_isString(stype) || LENGTH(stype) != 1)
156  Rf_error(_("argument '%s' must be a character string"), "type");
157  useRaw = Rboolean(strcmp(R_CHAR(STRING_ELT(stype, 0)), "integer"));
158  fac = useRaw ? 8 : 32;
159  if (len% fac)
160  Rf_error(_("argument 'x' must be a multiple of %d long"), fac);
161  slen = len/fac;
162  PROTECT(ans = Rf_allocVector(useRaw ? RAWSXP : INTSXP, slen));
163  for (i = 0; i < slen; i++)
164  if (useRaw) {
165  Rbyte btmp = 0;
166  for (int k = 7; k >= 0; k--) {
167  btmp <<= 1;
168  if (isRaw(x))
169  btmp |= RAW(x)[8*i + k] & 0x1;
170  else if (Rf_isLogical(x) || Rf_isInteger(x)) {
171  int j = INTEGER(x)[8*i+k];
172  if (j == R_NaInt)
173  Rf_error(_("argument 'x' must not contain NAs"));
174  btmp |= j & 0x1;
175  }
176  }
177  RAW(ans)[i] = btmp;
178  } else {
179  unsigned int itmp = 0;
180  for (int k = 31; k >= 0; k--) {
181  itmp <<= 1;
182  if (isRaw(x))
183  itmp |= RAW(x)[32*i + k] & 0x1;
184  else if (Rf_isLogical(x) || Rf_isInteger(x)) {
185  int j = INTEGER(x)[32*i+k];
186  if (j == R_NaInt)
187  Rf_error(_("argument 'x' must not contain NAs"));
188  itmp |= j & 0x1;
189  }
190  }
191  INTEGER(ans)[i] = int(itmp);
192  }
193  UNPROTECT(1);
194  return ans;
195 }

197 /* Simplified version for RFC3629 definition of UTF-8 */
198 static int mbrtoint(int *w, const char *s)
199 {
200  unsigned int byte;
201  byte = *(reinterpret_cast<unsigned char *>(const_cast<char *>(s)));
202 
203  if (byte == 0) {
204  *w = 0;
205  return 0;
206  } else if (byte < 0xC0) {
207  *w = int(byte);
208  return 1;
209  } else if (byte < 0xE0) {
210  if (!s[1]) return -2;
211  if ((s[1] & 0xC0) == 0x80) {
212  *w = int((((byte & 0x1F) << 6) | (s[1] & 0x3F)));
213  return 2;
214  } else return -1;
215  } else if (byte < 0xF0) {
216  if (!s[1] || !s[2]) return -2;
217  if (((s[1] & 0xC0) == 0x80) && ((s[2] & 0xC0) == 0x80)) {
218  *w = int(((byte & 0x0F) << 12)
219  | ((s[1] & 0x3F) << 6) | (s[2] & 0x3F));
220  byte = *w;
221  if (byte >= 0xD800 && byte <= 0xDFFF) return -1; /* surrogate */
222  // Following Corrigendum 9, these are valid in UTF-8
223 // if (byte == 0xFFFE || byte == 0xFFFF) return -1;
224  return 3;
225  } else return -1;
226  } else if (byte <= 0xF4) { // for RFC3629
227  if (!s[1] || !s[2] || !s[3]) return -2;
228  if (((s[1] & 0xC0) == 0x80)
229  && ((s[2] & 0xC0) == 0x80)
230  && ((s[3] & 0xC0) == 0x80)) {
231  *w = int(((byte & 0x07) << 18)
232  | ((s[1] & 0x3F) << 12)
233  | ((s[2] & 0x3F) << 6)
234  | (s[3] & 0x3F));
235  byte = *w;
236  return (byte <= 0x10FFFF) ? 4 : -1;
237  } else return -1;
238  } else return -1;
239  /* return -2; not reached */
240 }

242 HIDDEN SEXP do_utf8ToInt(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_)
243 {
244  SEXP ans, x = x_;
245  int tmp, used = 0; /* -Wall */
246  R_xlen_t i, j, nc;
247 
248  if (!Rf_isString(x) || LENGTH(x) == 0)
249  Rf_error(_("argument must be a character vector of length 1"));
250  if (LENGTH(x) > 1)
251  Rf_warning(_("argument should be a character vector of length 1\nall but the first element will be ignored"));
252  if (STRING_ELT(x, 0) == R_NaString) return Rf_ScalarInteger(R_NaInt);
253  const char *s = R_CHAR(STRING_ELT(x, 0));
254  if (!utf8Valid(s)) return Rf_ScalarInteger(R_NaInt);
255  nc = XLENGTH(STRING_ELT(x, 0)); /* ints will be shorter */
256  int *ians = static_cast<int *>(RHO_alloc(nc, sizeof(int)));
257  for (i = 0, j = 0; i < nc; i++) {
258  used = mbrtoint(&tmp, s);
259  if (used <= 0) break;
260  ians[j++] = tmp;
261  s += used;
262  }
263  if (used < 0) Rf_error(_("invalid UTF-8 string"));
264  ans = Rf_allocVector(INTSXP, j);
265  if (j) memcpy(INTEGER(ans), ians, sizeof(int) * j);
266  return ans;
267 }

269 /* Based on PCRE, but current Unicode only needs 4 bytes with maximum 0x10ffff */
270 static const int utf8_table1[] = { 0x7f, 0x7ff, 0xffff, 0x1fffff };
271 static const int utf8_table2[] = { 0, 0xc0, 0xe0, 0xf0 };
272 
273 static size_t inttomb(char *s, const int wc)
274 {
275  int i, j;
276  unsigned int cvalue = wc;
277  char buf[10], *b;
278 
279  b = s ? s : buf;
280  if (cvalue == 0) {*b = 0; return 0;}
281  for (i = 0; i < int(sizeof(utf8_table1)/sizeof(int)); i++)
282  if (int(cvalue) <= utf8_table1[i]) break;
283  b += i;
284  for (j = i; j > 0; j--) {
285  *b-- = char((0x80 | (cvalue & 0x3f)));
286  cvalue >>= 6;
287  }
288  *b = char((utf8_table2[i] | cvalue));
289  return i + 1;
290 }

294 HIDDEN SEXP do_intToUtf8(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_, rho::RObject* multiple_, rho::RObject* allow_surrogate_pairs_)
295 {
296  SEXP ans, x;
297  int multiple, s_pair;
298  size_t used, len;
299  char buf[10], *tmp;
300 
301  PROTECT(x = Rf_coerceVector(x_, INTSXP));
302  if (!Rf_isInteger(x))
303  Rf_error(_("argument 'x' must be an integer vector"));
304  multiple = Rf_asLogical(multiple_);
305  if (multiple == R_NaLog)
306  Rf_error(_("argument 'multiple' must be TRUE or FALSE"));
307  s_pair = Rf_asLogical(allow_surrogate_pairs_);
308  if (s_pair == R_NaLog)
309  Rf_error(_("argument 'allow_surrogate_pairs' must be TRUE or FALSE"));
310  if (multiple) {
311  if (s_pair)
312  Rf_warning("allow_surrogate_pairs = TRUE is incompatible with multiple = TRUE and will be ignored");
313  R_xlen_t i, nc = XLENGTH(x);
314  PROTECT(ans = Rf_allocVector(STRSXP, nc));
315  for (i = 0; i < nc; i++) {
316  int this_ = INTEGER(x)[i];
317  if (this_ == R_NaInt
318  || (this_ >= 0xD800 && this_ <= 0xDFFF)
319  || this_ > 0x10FFFF)
320  SET_STRING_ELT(ans, i, R_NaString);
321  else {
322  used = inttomb(buf, this_);
323  buf[used] = '\0';
324  SET_STRING_ELT(ans, i, Rf_mkCharCE(buf, CE_UTF8));
325  }
326  }
327  /* do we want to copy e.g. names here? */
328  } else {
329  int i, nc = LENGTH(x);
330  Rboolean haveNA = FALSE;
331  /* Note that this gives zero length for input '0', so it is omitted */
332  for (i = 0, len = 0; i < nc; i++) {
333  int this_ = INTEGER(x)[i];
334  if (this_ == R_NaInt
335  || (this_ >= 0xDC00 && this_ <= 0xDFFF)
336  || this_ > 0x10FFFF) {
337  haveNA = TRUE;
338  break;
339  }
340  else if (this_ >= 0xD800 && this_ <= 0xDBFF) {
341  if(!s_pair || i >= nc-1) {haveNA = TRUE; break;}
342  int next = INTEGER(x)[i+1];
343  if(next >= 0xDC00 && next <= 0xDFFF) i++;
344  else {haveNA = TRUE; break;}
345  len += 4; // all points not in the basic plane have length 4
346  }
347  else
348  len += inttomb(NULL, this_);
349  }
350  if (haveNA) {
351  UNPROTECT(1);
352  return Rf_ScalarString(R_NaString);
353  }
354  if (len >= 10000) {
355  tmp = Calloc(len+1, char);
356  } else {
357  R_CheckStack2(len+1);
358  tmp = static_cast<char *>(alloca(len+1)); tmp[len] = '\0';
359  }
360  for (i = 0, len = 0; i < nc; i++) {
361  int this_ = INTEGER(x)[i];
362  if(s_pair && (this_ >= 0xD800 && this_ <= 0xDBFF)) {
363  // all the validity checking has already been done.
364  int next = INTEGER(x)[++i];
365  unsigned int hi = this_ - 0xD800, lo = next - 0xDC00;
366  this_ = 0x10000 + (hi << 10) + lo;
367  }
368  used = inttomb(buf, this_);
369  strncpy(tmp + len, buf, used);
370  len += used;
371  }
372  ans = PROTECT(Rf_ScalarString(Rf_mkCharLenCE(tmp, int(len), CE_UTF8)));
373  if(len >= 10000) Free(tmp);
374  }
375  UNPROTECT(2);
376  return ans;
377 }
