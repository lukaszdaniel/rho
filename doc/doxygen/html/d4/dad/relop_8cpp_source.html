91  template <class V>
92  LogicalVector* relop(const V* vl, const V* vr, RELOP_TYPE code)
93  {
94  typedef typename V::value_type Value;
95 
96  switch (code) {
97  case EQOP:
98  return relop_aux(vl, vr,
99  [](Value lhs, Value rhs) { return lhs == rhs; });
100  case NEOP:
101  return relop_aux(vl, vr,
102  [](Value lhs, Value rhs) { return lhs != rhs; });
103  case LTOP:
104  return relop_aux(vl, vr,
105  [](Value lhs, Value rhs) { return lhs < rhs; });
106  case GTOP:
107  return relop_aux(vl, vr,
108  [](Value lhs, Value rhs) { return lhs > rhs; });
109  case LEOP:
110  return relop_aux(vl, vr,
111  [](Value lhs, Value rhs) { return lhs <= rhs; });
112  case GEOP:
113  return relop_aux(vl, vr,
114  [](Value lhs, Value rhs) { return lhs >= rhs; });
115  }
116  return nullptr; // -Wall
117  }

138 SEXP do_relop(/*const*/ Expression* call,
139  const BuiltInFunction* op,
140  RObject* xarg, RObject* yarg)
141 {
142  GCStackRoot<> x(xarg), y(yarg);
143 
144  R_xlen_t
145  nx = Rf_xlength(x),
146  ny = Rf_xlength(y);
147 
148  /* That symbols and calls were allowed was undocumented prior to
149  R 2.5.0. We deparse them as deparse() would, minus attributes */
150  bool iS;
151  if ((iS = Rf_isSymbol(x)) || TYPEOF(x) == LANGSXP) {
152  x = Rf_ScalarString(
153  (iS) ? PRINTNAME(x) :
154  STRING_ELT(Rf_deparse1(x, RHO_FALSE, DEFAULTDEPARSE), 0));
155  }
156  if ((iS = Rf_isSymbol(y)) || TYPEOF(y) == LANGSXP) {
157  y = Rf_ScalarString(
158  (iS) ? PRINTNAME(y) :
159  STRING_ELT(Rf_deparse1(y, RHO_FALSE, DEFAULTDEPARSE), 0));
160  }
161 
162  if (Rf_isNull(x)) x = Rf_allocVector(INTSXP,0);
163  if (Rf_isNull(y)) y = Rf_allocVector(INTSXP,0);
164  if (!Rf_isVector(x) || !Rf_isVector(y))
165  Rf_errorcall(call,
166  _("comparison (%d) is possible only for atomic and list types"),
167  op->variant());
168 
169  if (TYPEOF(x) == EXPRSXP || TYPEOF(y) == EXPRSXP)
170  Rf_errorcall(call, _("comparison is not allowed for expressions"));
171 
172  /* ELSE : x and y are both atomic or list */
173  checkOperandsConformable(static_cast<VectorBase*>(xarg), static_cast<VectorBase*>(yarg));
174 
175  if (nx > 0 && ny > 0) {
176  RELOP_TYPE opcode = RELOP_TYPE(op->variant());
177  if (Rf_isString(x) || Rf_isString(y)) {
178  // This case has not yet been brought into line with the
179  // general rho pattern.
180  VectorBase* xv = static_cast<VectorBase*>(Rf_coerceVector(x, STRSXP));
181  VectorBase* yv = static_cast<VectorBase*>(Rf_coerceVector(y, STRSXP));
182  if (((nx > ny) ? nx % ny : ny % nx) != 0) // mismatch
183  Rf_warningcall(call, _("longer object length is not a multiple of shorter object length"));
184  GCStackRoot<VectorBase>
185  ans(static_cast<VectorBase*>(string_relop(opcode, xv, yv)));
186  GeneralBinaryAttributeCopier::copyAttributes(ans, xv, yv);
187  return ans;
188  }
189  else if (Rf_isComplex(x) || Rf_isComplex(y)) {
190  GCStackRoot<ComplexVector>
191  vl(static_cast<ComplexVector*>(Rf_coerceVector(x, CPLXSXP)));
192  GCStackRoot<ComplexVector>
193  vr(static_cast<ComplexVector*>(Rf_coerceVector(y, CPLXSXP)));
194  return relop_no_order(vl.get(), vr.get(), opcode);
195  }
196  else if ((Rf_isNumeric(x) || Rf_isLogical(x)) && (Rf_isNumeric(y) || Rf_isLogical(y))) {
197  GCStackRoot<RealVector>
198  vl(static_cast<RealVector*>(Rf_coerceVector(x, REALSXP)));
199  GCStackRoot<RealVector>
200  vr(static_cast<RealVector*>(Rf_coerceVector(y, REALSXP)));
201  return relop(vl.get(), vr.get(), opcode);
202  } // rest of cases only apply when 'x' or 'y' is raw
203  else if (Rf_isReal(x) || Rf_isReal(y)) {
204  GCStackRoot<RealVector>
205  vl(static_cast<RealVector*>(Rf_coerceVector(x, REALSXP)));
206  GCStackRoot<RealVector>
207  vr(static_cast<RealVector*>(Rf_coerceVector(y, REALSXP)));
208  return relop(vl.get(), vr.get(), opcode);
209  }
210  else if (Rf_isInteger(x) || Rf_isInteger(y)) {
211  GCStackRoot<IntVector>
212  vl(static_cast<IntVector*>(Rf_coerceVector(x, INTSXP)));
213  GCStackRoot<IntVector>
214  vr(static_cast<IntVector*>(Rf_coerceVector(y, INTSXP)));
215  return relop(vl.get(), vr.get(), opcode);
216  }
217  else if (Rf_isLogical(x) || Rf_isLogical(y)) {
218  // TODO(kmillar): do this without promoting to integer.
219  GCStackRoot<IntVector>
220  vl(static_cast<IntVector*>(Rf_coerceVector(x, INTSXP)));
221  GCStackRoot<IntVector>
222  vr(static_cast<IntVector*>(Rf_coerceVector(y, INTSXP)));
223  return relop(vl.get(), vr.get(), opcode);
224  }
225  else if (TYPEOF(x) == RAWSXP || TYPEOF(y) == RAWSXP) {
226  GCStackRoot<RawVector>
227  vl(static_cast<RawVector*>(Rf_coerceVector(x, RAWSXP)));
228  GCStackRoot<RawVector>
229  vr(static_cast<RawVector*>(Rf_coerceVector(y, RAWSXP)));
230  return relop(vl.get(), vr.get(), opcode);
231  } else Rf_errorcall(call, _("comparison of these types is not implemented"));
232  } else {
233  GCStackRoot<> val(Rf_allocVector(LGLSXP, 0));
234  GeneralBinaryAttributeCopier::copyAttributes(
235  SEXP_downcast<VectorBase*>(val.get()),
236  static_cast<VectorBase*>(xarg),
237  static_cast<VectorBase*>(yarg));
238  return val;
239  }
240  return nullptr; // -Wall
241 }

245 static SEXP string_relop(RELOP_TYPE code, SEXP s1, SEXP s2)
246 {
247  R_xlen_t i, n, n1, n2, res;
248  SEXP ans, c1, c2;
249  const void *vmax = vmaxget(); // for Scollate
250 
251  n1 = XLENGTH(s1);
252  n2 = XLENGTH(s2);
253  n = std::max(n1, n2);
254  PROTECT(s1);
255  PROTECT(s2);
256  PROTECT(ans = Rf_allocVector(LGLSXP, n));
257 
258  switch (code) {
259  case EQOP:
260  for (i = 0; i < n; i++) {
261 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
262  c1 = STRING_ELT(s1, i % n1);
263  c2 = STRING_ELT(s2, i % n2);
264  if (c1 == NA_STRING || c2 == NA_STRING)
265  LOGICAL(ans)[i] = NA_LOGICAL;
266  else
267  LOGICAL(ans)[i] = Seql(c1, c2) ? 1 : 0;
268  }
269  break;
270  case NEOP:
271  for (i = 0; i < n; i++) {
272 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
273  c1 = STRING_ELT(s1, i % n1);
274  c2 = STRING_ELT(s2, i % n2);
275  if (c1 == NA_STRING || c2 == NA_STRING)
276  LOGICAL(ans)[i] = NA_LOGICAL;
277  else
278  LOGICAL(ans)[i] = Seql(c1, c2) ? 0 : 1;
279  }
280  break;
281  case LTOP:
282  for (i = 0; i < n; i++) {
283 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
284  c1 = STRING_ELT(s1, i % n1);
285  c2 = STRING_ELT(s2, i % n2);
286  if (c1 == NA_STRING || c2 == NA_STRING)
287  LOGICAL(ans)[i] = NA_LOGICAL;
288  else if (c1 == c2)
289  LOGICAL(ans)[i] = 0;
290  else {
291  errno = 0;
292  res = Scollate(c1, c2);
293  if(errno)
294  LOGICAL(ans)[i] = NA_LOGICAL;
295  else
296  LOGICAL(ans)[i] = (res < 0) ? 1 : 0;
297  }
298  }
299  break;
300  case GTOP:
301  for (i = 0; i < n; i++) {
302 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
303  c1 = STRING_ELT(s1, i % n1);
304  c2 = STRING_ELT(s2, i % n2);
305  if (c1 == NA_STRING || c2 == NA_STRING)
306  LOGICAL(ans)[i] = NA_LOGICAL;
307  else if (c1 == c2)
308  LOGICAL(ans)[i] = 0;
309  else {
310  errno = 0;
311  res = Scollate(c1, c2);
312  if(errno)
313  LOGICAL(ans)[i] = NA_LOGICAL;
314  else
315  LOGICAL(ans)[i] = (res > 0) ? 1 : 0;
316  }
317  }
318  break;
319  case LEOP:
320  for (i = 0; i < n; i++) {
321 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
322  c1 = STRING_ELT(s1, i % n1);
323  c2 = STRING_ELT(s2, i % n2);
324  if (c1 == NA_STRING || c2 == NA_STRING)
325  LOGICAL(ans)[i] = NA_LOGICAL;
326  else if (c1 == c2)
327  LOGICAL(ans)[i] = 1;
328  else {
329  errno = 0;
330  res = Scollate(c1, c2);
331  if(errno)
332  LOGICAL(ans)[i] = NA_LOGICAL;
333  else
334  LOGICAL(ans)[i] = (res <= 0) ? 1 : 0;
335  }
336  }
337  break;
338  case GEOP:
339  for (i = 0; i < n; i++) {
340 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
341  c1 = STRING_ELT(s1, i % n1);
342  c2 = STRING_ELT(s2, i % n2);
343  if (c1 == NA_STRING || c2 == NA_STRING)
344  LOGICAL(ans)[i] = NA_LOGICAL;
345  else if (c1 == c2)
346  LOGICAL(ans)[i] = 1;
347  else {
348  errno = 0;
349  res = Scollate(c1, c2);
350  if(errno)
351  LOGICAL(ans)[i] = NA_LOGICAL;
352  else
353  LOGICAL(ans)[i] = (res >= 0) ? 1 : 0;
354  }
355  }
356  break;
357  }
358  UNPROTECT(3);
359  vmaxset(vmax);
360  return ans;
361 }

400 static SEXP bitwiseShiftL(SEXP a, SEXP b)
401 {
402  int np = 0;
403  if(Rf_isReal(a)) {a = PROTECT(Rf_coerceVector(a, INTSXP)); np++;}
404  if(!Rf_isInteger(b)) {b = PROTECT(Rf_coerceVector(b, INTSXP)); np++;}
405  R_xlen_t i, m = XLENGTH(a), n = XLENGTH(b),
406  mn = (m && n) ? std::max(m, n) : 0;
407  SEXP ans = Rf_allocVector(TYPEOF(a), mn);
408  switch(TYPEOF(a)) {
409  case INTSXP:
410  for(i = 0; i < mn; i++) {
411  int aa = INTEGER(a)[i%m], bb = INTEGER(b)[i%n];
412  INTEGER(ans)[i] =
413  (aa == NA_INTEGER || bb == NA_INTEGER || bb < 0 || bb > 31) ? NA_INTEGER : (static_cast<unsigned int>(aa) << bb);
414  }
415  break;
416  default:
417  UNIMPLEMENTED_TYPE("bitShiftL", a);
418  }
419  if(np) UNPROTECT(np);
420  return ans;
421 }
