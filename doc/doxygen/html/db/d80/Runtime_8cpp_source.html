52 static llvm::Function* getDeclaration(const std::string& name,
53  llvm::Module* module)
54 {
55  llvm::Function* resolved_function = module->getFunction(name);
56  if (!resolved_function) {
57  // Copy the declaration from the runtime to the module.
58  // TODO(kmillar): omitting the function body eliminates opportunities
59  // for inlining, which will be required to generate fast code.
60  Function* runtime_function = getRuntimeModule(module->getContext())
61  ->getFunction(name);
62  assert(runtime_function);
63  resolved_function = Function::Create(runtime_function->getFunctionType(),
64  Function::ExternalLinkage,
65  runtime_function->getName(),
66  module);
67  assert(resolved_function);
68  resolved_function->setAttributes(runtime_function->getAttributes());
69  }
70  return resolved_function;
71 }

268 static void cleanupRuntimeModule(Module* module)
269 {
270  // The Runtime module contains a lot of functions that have already been
271  // compiled into the current process. There's no point in recompiling those
272  // functions, so replace them with their declarations.
273  for (Function& function : *module) {
274  if (llvm::sys::DynamicLibrary::SearchForAddressOfSymbol(
275  function.getName()))
276  {
277  function.deleteBody();
278  }
279  }
280 }

282 static std::unique_ptr<Module> createRuntimeModule(LLVMContext& context)
283 {
284  std::string module_filename = std::string(R_Home) + "/jit/RuntimeImpl.bc";
285 
286  llvm::SMDiagnostic err;
287  std::unique_ptr<Module> runtime_module
288  = llvm::parseIRFile(module_filename, err, context);
289  if (!runtime_module) {
290  // TODO(kmillar): better error handling
291  printf("parse failed\n");
292  exit(1);
293  }
294  cleanupRuntimeModule(runtime_module.get());
295 
296  return runtime_module;
297 }

324 std::string getName(FunctionId function)
325 {
326  switch (function) {
327  case NOT_A_RUNTIME_FUNCTION:
328  assert(0 && "Invalid FunctionId value passed.");
329  return nullptr; // TODO: throw an exception.
330  case EVALUATE:
331  return "rho_runtime_evaluate";
332  case LOOKUP_SYMBOL:
333  return "rho_runtime_lookupSymbol";
334  case LOOKUP_SYMBOL_IN_COMPILED_FRAME:
335  return "rho_runtime_lookupSymbolInCompiledFrame";
336  case ASSIGN_SYMBOL_IN_COMPILED_FRAME:
337  return "rho_runtime_assignSymbolInCompiledFrame";
338  case LOOKUP_FUNCTION:
339  return "rho_runtime_lookupFunction";
340  case CALL_FUNCTION:
341  return "rho_runtime_callFunction";
342  case DO_BREAK:
343  return "rho_runtime_do_break";
344  case DO_NEXT:
345  return "rho_runtime_do_next";
346  case COERCE_TO_TRUE_OR_FALSE:
347  return "rho_runtime_coerceToTrueOrFalse";
348  case SET_VISIBILITY:
349  return "rho_runtime_setVisibility";
350  case INCREMENT_NAMED:
351  return "rho_runtime_incrementNamed";
