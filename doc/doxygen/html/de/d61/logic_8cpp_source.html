50  LogicalVector* binaryLogic(int opcode, const LogicalVector* l,
51  const LogicalVector* r)
52  {
53  switch (opcode) {
54  case 1:
55  {
56  return applyBinaryOperator(
57  [](Logical l, Logical r) { return l && r; },
58  GeneralBinaryAttributeCopier(),
59  l, r);
60  }
61  case 2:
62  {
63  return applyBinaryOperator(
64  [](Logical l, Logical r) { return l || r; },
65  GeneralBinaryAttributeCopier(),
66  l, r);
67  }
68  }
69  return nullptr; // -Wall

93  RObject* lbinary(const BuiltInFunction* op, RObject* x, RObject* y)
94  {
95 
96  if (x && x->sexptype() == RAWSXP && y && y->sexptype() == RAWSXP) {
97  } else if (!(Rf_isNull(x) || Rf_isNumber(x))
98  || !(Rf_isNull(y) || Rf_isNumber(y)))
99  Rf_error(_("operations are possible only for numeric, logical or complex types"));
100 
101  R_xlen_t nx = Rf_xlength(x), ny = Rf_xlength(y);
102 
103  checkOperandsConformable(
104  SEXP_downcast<VectorBase*>(x), SEXP_downcast<VectorBase*>(y));
105 
106  if (nx > 0 && ny > 0) {
107  /* logical binary : "&" or "|" */
108  if (x && x->sexptype() == RAWSXP && y && y->sexptype() == RAWSXP) {
109  // Bitwise operations:
110  RawVector* vl = SEXP_downcast<RawVector*>(x);
111  RawVector* vr = SEXP_downcast<RawVector*>(y);
112  return bitwiseBinary(op->variant(), vl, vr);
113  }
114  GCStackRoot<LogicalVector> vl;
115  GCStackRoot<LogicalVector> vr;
116  if (Rf_isNull(x))
117  vl = SEXP_downcast<LogicalVector*>(
118  Rf_allocVector(SEXPTYPE::LGLSXP, 0));
119  else // Rf_isNumeric(x)
120  vl = SEXP_downcast<LogicalVector*>(Rf_coerceVector(x, LGLSXP));
121  if (Rf_isNull(y))
122  vr = SEXP_downcast<LogicalVector*>(
123  Rf_allocVector(SEXPTYPE::LGLSXP, 0));
124  else // Rf_isNumeric(x)
125  vr = SEXP_downcast<LogicalVector*>(Rf_coerceVector(y, LGLSXP));
126  return binaryLogic(op->variant(), vl, vr);
127  } else { // nx == 0 || ny == 0
128  GCStackRoot<> val(Rf_allocVector((isRaw(x) && isRaw(y)) ? RAWSXP : LGLSXP, 0));
129  GeneralBinaryAttributeCopier::copyAttributes(
130  SEXP_downcast<VectorBase*>(val.get()),
131  SEXP_downcast<VectorBase*>(x),
132  SEXP_downcast<VectorBase*>(y));
133  return val;
134  }
135  }

138  RObject* lnot(RObject* arg)
139  {
140  if (arg && arg->sexptype() == RAWSXP) {
141  // Bit inversion:
142  return applyUnaryOperator([](Rbyte x) { return Rbyte(~x); },
143  CopyAllAttributes(),
144  SEXP_downcast<RawVector*>(arg));
145  } else if (arg && arg->sexptype() == LGLSXP) {
146  // Logical negation:
147  return applyUnaryOperator(
148  [](Logical x) { return !x; },
149  CopyAllAttributes(),
150  SEXP_downcast<LogicalVector*>(arg));
151  } else if (!Rf_isNumber(arg)) {
152  if (Rf_length(arg) == 0U) // For back-compatibility
153  return LogicalVector::create(0);
154  Rf_error(_("invalid argument type"));
155  } else {
156  // Logical negation:
157  GCStackRoot<LogicalVector>
158  lv(SEXP_downcast<LogicalVector*>(Rf_coerceVector(arg, LGLSXP)));
159  return applyUnaryOperator(
160  [](Logical x) { return !x; },
161  // Note: in other cases all attributes are copied.
162  CopyLayoutAttributes(),
163  lv.get());
164  }
165  }

169 HIDDEN SEXP do_logic(/*const*/ Expression* call, const BuiltInFunction* op, int num_args, ...)
170 {
171  switch (op->variant()) {
172  case 1:
173  case 2:
174  {
175  op->checkNumArgs(num_args, 2, call);
176  UNPACK_VA_ARGS(num_args, (lhs) (rhs));
177  return lbinary(op, lhs, rhs);
178  }
179  case 3:
180  {
181  op->checkNumArgs(num_args, 1, call);
182  UNPACK_VA_ARGS(num_args, (arg1));
183  return lnot(arg1);
184  }
185  default:
186  Rf_error(_("internal error in do_logic"));
187  }
188  return nullptr; // -Wall

191 /* && || */
192 HIDDEN SEXP do_logic2(SEXP call, SEXP op, SEXP args, SEXP env)
193 {
194 /* && and || */
195  SEXP s1, s2;
196  int x1, x2;
197  int ans;
198 
199  if (Rf_length(args) != 2)
200  Rf_error(_("'%s' operator requires 2 arguments"),
201  PRIMVAL(op) == 1 ? "&&" : "||");
202 
203  s1 = CAR(args);
204  s2 = CADR(args);
205  s1 = Rf_eval(s1, env);
206  if (!Rf_isNumber(s1))
207  Rf_errorcall(call, _("invalid 'x' type in 'x %s y'"),
208  PRIMVAL(op) == 1 ? "&&" : "||");
209  x1 = Rf_asLogical(s1);
210 
211 #define get_2nd \
212  s2 = Rf_eval(s2, env); \
213  if (!Rf_isNumber(s2)) \
214  Rf_errorcall(call, _("invalid 'y' type in 'x %s y'"), \
215  PRIMVAL(op) == 1 ? "&&" : "||"); \
216  x2 = Rf_asLogical(s2);
217 
218  switch (PRIMVAL(op)) {
219  case 1: /* && */
220  if (x1 == FALSE)
221  ans = FALSE;
222  else {
223  get_2nd;
224  if (x1 == R_NaLog)
225  ans = (x2 == R_NaLog || x2) ? R_NaLog : x2;
226  else /* x1 == TRUE */
227  ans = x2;
228  }
229  break;
230  case 2: /* || */
231  if (x1 == TRUE)
232  ans = TRUE;
233  else {
234  get_2nd;
235  if (x1 == R_NaLog)
236  ans = (x2 == R_NaLog || !x2) ? R_NaLog : x2;
237  else /* x1 == FALSE */
238  ans = x2;
239  }
240  break;
241  default:
242  Rf_error(_("internal error in do_logic2"));
243  }
244  return Rf_ScalarLogical(ans);
245 }

252 static Logical checkValues(int op, int na_rm, SEXP x, R_xlen_t n)
253 {
254  R_xlen_t i;
255  int has_na = 0;
256  int *px = LOGICAL(x);
257  for (i = 0; i < n; i++) {
258 // if ((i+1) % NINTERRUPT == 0) R_CheckUserInterrupt();
259  int xi = px[i];
260  if (!na_rm && xi == R_NaLog) has_na = 1;
261  else {
262  if (xi == TRUE && op == _OP_ANY) return true;
263  if (xi == FALSE && op == _OP_ALL) return false;
264  }
265  }
266  switch (op) {
267  case _OP_ANY:
268  return has_na ? Logical::NA() : false;
269  case _OP_ALL:
270  return has_na ? Logical::NA() : true;
271  default:
272  Rf_error("bad op value for do_logic3");
273  }
274  return Logical::NA(); /* -Wall */
275 }

277 /* all, any */
278 HIDDEN SEXP do_logic3(SEXP call, SEXP op, SEXP args, SEXP env)
279 {
280  SEXP ans, s, t;
281  int narm, has_na = 0;
282  /* initialize for behavior on empty vector
283  all(logical(0)) -> TRUE
284  any(logical(0)) -> FALSE
285  */
286  Logical val = PRIMVAL(op) == _OP_ALL ? true : false;
287 
288  PROTECT(args = fixup_NaRm(args));
289  Expression* call2 = new Expression(CAR(call),
290  SEXP_downcast<PairList*>(args));
291 
292  ArgList arglist(SEXP_downcast<PairList*>(args), ArgList::EVALUATED);
293  auto dispatched = SEXP_downcast<BuiltInFunction*>(op)->InternalDispatch(
294  SEXP_downcast<Expression*>(call2),
295  SEXP_downcast<Environment*>(env), std::move(arglist));
296  if (dispatched.first) {
297  UNPROTECT(2);
298  return(dispatched.second);
299  }
300 
301  ans = Rf_matchArgExact(Symbols::NaRmSymbol, &args);
302  narm = Rf_asLogical(ans);
303 
304  for (s = args; s != nullptr; s = CDR(s)) {
305  t = CAR(s);
306  /* Avoid memory waste from coercing empty inputs, and also
307  avoid warnings with empty lists coming from sapply */
308  if(Rf_xlength(t) == 0) continue;
309  /* Rf_coerceVector protects its argument so this actually works
310  just fine */
311  if (TYPEOF(t) != LGLSXP) {
312  /* Coercion of integers seems reasonably safe, but for
313  other types it is more often than not an error.
314  One exception is perhaps the result of lapply, but
315  then sapply was often what was intended. */
316  if(TYPEOF(t) != INTSXP)
317  Rf_warningcall(call,
318  _("coercing argument of type '%s' to logical"),
319  Rf_type2char(TYPEOF(t)));
320  t = Rf_coerceVector(t, LGLSXP);
321  }
322  val = checkValues(PRIMVAL(op), narm, t, XLENGTH(t));
323  if (!val.isNA()) {
324  if ((PRIMVAL(op) == _OP_ANY && val.isTrue())
325  || (PRIMVAL(op) == _OP_ALL && val.isFalse())) {
326  has_na = 0;
327  break;
328  }
329  } else has_na = 1;
330  }
331  UNPROTECT(2);
332  return LogicalVector::createScalar(has_na ? Logical::NA() : val);
333 }
