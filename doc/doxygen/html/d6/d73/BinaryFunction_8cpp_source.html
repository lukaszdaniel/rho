41  const VectorBase* vl,

44  // Handle layout attributes:
45  {
46  RObject* dims = vl ? vl->getAttribute(DimSymbol) : nullptr;
47  RObject* dimnames = nullptr;
48  size_t nx = vl ? vl->size() : 0;
49  size_t ny = vr ? vr->size() : 0;
50 
51  if (dims && (ny != 0 || nx == 0)) {
52  dimnames = vl->getAttribute(DimNamesSymbol);
53  } else if (vr && (nx != 0 || ny == 0)) {
54  dims = vr->getAttribute(DimSymbol);
55  } else
56  dims = nullptr;
57  if (vr && !dimnames)
58  dimnames = vr->getAttribute(DimNamesSymbol);
59  if (dims) {
60  Rf_setAttrib(vout, DimSymbol, dims);
61  // vout->setAttribute(DimSymbol, dims);
62  if (dimnames)
63  vout->setAttribute(DimNamesSymbol, dimnames);
64  } else {
65  // Neither operand is an array. Get the names from the longer
66  // attribute (if present), prefering the first if the lengths are
67  // the same.
68  RObject* vl_names = vl ? vl->getAttribute(NamesSymbol) : nullptr;
69  if (vl_names && vout->size() == nx) {
70  vout->setAttribute(NamesSymbol, vl_names);
71  } else if (vr) {
72  RObject* vr_names = vr->getAttribute(NamesSymbol);
73  if (vr_names && vout->size() == ny)
74  vout->setAttribute(NamesSymbol, vr_names);
75  }
76  }
77  }
78  // Handle attributes related to time series:
79  {
80  RObject* tsp = vl? vl->getAttribute(TspSymbol): nullptr;
81  RObject* klass = nullptr;
82  if (tsp)
83  klass = vl->getAttribute(ClassSymbol);
84  if (!tsp && vr) {
85  tsp = vr->getAttribute(TspSymbol);
86  if (tsp)
87  klass = vr->getAttribute(ClassSymbol);
88  }
89  if (tsp)
