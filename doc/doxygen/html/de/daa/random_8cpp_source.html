59 static void ProbSampleReplace(int n, double *p, int *perm, int nans, int *ans)
60 {
61  double rU;
62  int i, j;
63  int nm1 = n - 1;
64 
65  /* record element identities */
66  for (i = 0; i < n; i++)
67  perm[i] = i + 1;
68 
69  /* sort the probabilities into descending order */
70  Rf_revsort(p, perm, n);
71 
72  /* compute cumulative probabilities */
73  for (i = 1 ; i < n; i++)
74  p[i] += p[i - 1];
75 
76  /* compute the sample */
77  for (i = 0; i < nans; i++) {
78  rU = unif_rand();
79  for (j = 0; j < nm1; j++) {
80  if (rU <= p[j])
81  break;
82  }
83  ans[i] = perm[j];
84  }
85 }

87 /* A version using Walker's alias method, based on Alg 3.13B in
88  Ripley (1987).
89  */

91 constexpr int SMALL = 10000;
92 static void walker_ProbSampleReplace(int n, double *p, int *a, int nans, int *ans)
93 {
94  double *q, rU;
95  int i, j, k;
96  int *HL, *H, *L;
97 
98  /* Create the alias tables.
99  The idea is that for HL[0] ... L-1 label the entries with q < 1
100  and L ... H[n-1] label those >= 1.
101  By rounding error we could have q[i] < 1. or > 1. for all entries.
102  */
103  if(n <= SMALL) {
104  R_CheckStack2(n *(sizeof(int) + sizeof(double)));
105  /* might do this repeatedly, so speed matters */
106  HL = static_cast<int *>(alloca(n * sizeof(int)));
107  q = static_cast<double *>(alloca(n * sizeof(double)));
108  } else {
109  /* Slow enough anyway not to risk overflow */
110  HL = Calloc(n, int);
111  q = Calloc(n, double);
112  }
113  H = HL - 1; L = HL + n;
114  for (i = 0; i < n; i++) {
115  q[i] = p[i] * n;
116  if (q[i] < 1.) *++H = i; else *--L = i;
117  }
118  if (H >= HL && L < HL + n) { /* So some q[i] are >= 1 and some < 1 */
119  for (k = 0; k < n - 1; k++) {
120  i = HL[k];
121  j = *L;
122  a[i] = j;
123  q[j] += q[i] - 1;
124  if (q[j] < 1.) L++;
125  if(L >= HL + n) break; /* now all are >= 1 */
126  }
127  }
128  for (i = 0; i < n; i++) q[i] += i;
129 
130  /* generate sample */
131  for (i = 0; i < nans; i++) {
132  rU = unif_rand() * n;
133  k = int(rU);
134  ans[i] = (rU < q[k]) ? k+1 : a[k]+1;
135  }
136  if(n > SMALL) {
137  Free(HL);
138  Free(q);
139  }
140 }

145 static void ProbSampleNoReplace(int n, double *p, int *perm,
146  int nans, int *ans)
147 {
148  double rT, mass, totalmass;
149  int i, j, k, n1;
150 
151  /* Record element identities */
152  for (i = 0; i < n; i++)
153  perm[i] = i + 1;
154 
155  /* Sort probabilities into descending order */
156  /* Order element identities in parallel */
157  Rf_revsort(p, perm, n);
158 
159  /* Compute the sample */
160  totalmass = 1;
161  for (i = 0, n1 = n-1; i < nans; i++, n1--) {
162  rT = totalmass * unif_rand();
163  mass = 0;
164  for (j = 0; j < n1; j++) {
165  mass += p[j];
166  if (rT <= mass)
167  break;
168  }
169  ans[i] = perm[j];
170  totalmass -= p[j];
171  for(k = j; k < n1; k++) {
172  p[k] = p[k + 1];
173  perm[k] = perm[k + 1];
174  }
175  }
176 }

178 static void FixupProb(double *p, int n, int require_k, Rboolean replace)
179 {
180  double sum = 0.0;
181  int npos = 0;
182  for (int i = 0; i < n; i++) {
183  if (!std::isfinite(p[i]))
184  Rf_error(_("NA in probability vector"));
185  if (p[i] < 0.0)
186  Rf_error(_("negative probability"));
187  if (p[i] > 0.0) {
188  npos++;
189  sum += p[i];
190  }
191  }
192  if (npos == 0 || (!replace && require_k > npos))
193  Rf_error(_("too few positive probabilities"));
194  for (int i = 0; i < n; i++) p[i] /= sum;
195 }

197 /* do_sample - probability sampling with/without replacement.
198  .Internal(sample(n, size, replace, prob))
199 */
200 HIDDEN RObject* do_sample(/*const*/ Expression* call, const BuiltInFunction* op, RObject* x_, RObject* size_, RObject* replace_, RObject* prob_)
201 {
202  SEXP x, y, sn, sk, prob, sreplace;
203 
204  sn = x_;
205  sk = size_;
206  sreplace = replace_;
207  if(Rf_length(sreplace) != 1)
208  Rf_error(_("invalid '%s' argument"), "replace");
209  int replace = Rf_asLogical(sreplace);
210  prob = prob_;
211  if (replace == R_NaLog)
212  Rf_error(_("invalid '%s' argument"), "replace");
213  GetRNGstate();
214  if (!Rf_isNull(prob)) {
215  int n = Rf_asInteger(sn), k = Rf_asInteger(sk);
216  if (n == R_NaInt || n < 0 || (k > 0 && n == 0))
217  Rf_error(_("invalid first argument"));
218  if (k == R_NaInt || k < 0)
219  Rf_error(_("invalid '%s' argument"), "size");
220  if (!replace && k > n)
221  Rf_error(_("cannot take a sample larger than the population when 'replace = FALSE'"));
222  PROTECT(y = Rf_allocVector(INTSXP, k));
223  prob = Rf_coerceVector(prob, REALSXP);
224  if (MAYBE_REFERENCED(prob)) prob = Rf_duplicate(prob);
225  PROTECT(prob);
226  double *p = REAL(prob);
227  if (Rf_length(prob) != n)
228  Rf_error(_("incorrect number of probabilities"));
229  FixupProb(p, n, k, Rboolean(replace));
230  PROTECT(x = Rf_allocVector(INTSXP, n));
231  if (replace) {
232  int i, nc = 0;
233  for (i = 0; i < n; i++) if(n * p[i] > 0.1) nc++;
234  if (nc > 200)
235  walker_ProbSampleReplace(n, p, INTEGER(x), k, INTEGER(y));
236  else
237  ProbSampleReplace(n, p, INTEGER(x), k, INTEGER(y));
238  } else
239  ProbSampleNoReplace(n, p, INTEGER(x), k, INTEGER(y));
240  UNPROTECT(2);
241  }
242  else { // uniform sampling
243  double dn = Rf_asReal(sn);
244  R_xlen_t k = asVecSize(sk);
245  if (!std::isfinite(dn) || dn < 0 || dn > 4.5e15 || (k > 0 && dn == 0))
246  Rf_error(_("invalid first argument"));
247  if (k < 0) Rf_error(_("invalid '%s' argument"), "size");
248  if (!replace && k > dn)
249  Rf_error(_("cannot take a sample larger than the population when 'replace = FALSE'"));
250  if (dn > INT_MAX || k > INT_MAX) {
251  PROTECT(y = Rf_allocVector(REALSXP, k));
252  if (replace) {
253  double *ry = REAL(y);
254  for (R_xlen_t i = 0; i < k; i++) ry[i] = R_unif_index(dn) + 1;
255  } else {
256 #ifdef LONG_VECTOR_SUPPORT
257  R_xlen_t n = R_xlen_t(dn);
258  double *x = static_cast<double *>(RHO_alloc(n, sizeof(double)));
259  double *ry = REAL(y);
260  for (R_xlen_t i = 0; i < n; i++) x[i] = double(i);
261  for (R_xlen_t i = 0; i < k; i++) {
262  R_xlen_t j = (R_xlen_t) R_unif_index(n);
263  ry[i] = x[j] + 1;
264  x[j] = x[--n];
265  }
266 #else
267  Rf_error(_("n >= 2^31, replace = FALSE is only supported on 64-bit platforms"));
268 #endif
269  }
270  } else {
271  int n = int(dn);
272  PROTECT(y = Rf_allocVector(INTSXP, k));
273  int *iy = INTEGER(y);
274  /* avoid allocation for a single sample */
275  if (replace || k < 2) {
276  for (int i = 0; i < k; i++) iy[i] = (int)(R_unif_index(dn) + 1);
277  } else {
278  int *x = static_cast<int *>(RHO_alloc(n, sizeof(int)));
279  for (int i = 0; i < n; i++) x[i] = i;
280  for (int i = 0; i < k; i++) {
281  int j = (int)(R_unif_index(n));
282  iy[i] = x[j] + 1;
283  x[j] = x[--n];
284  }
285  }
286  }
287  }
288  PutRNGstate();
289  UNPROTECT(1);
290  return y;
291 }
