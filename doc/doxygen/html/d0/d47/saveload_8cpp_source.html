59 /* From time to time changes in R, such as the addition of a new SXP,
60  * may require changes in the save file format. Here are some
61  * guidelines on handling format changes:
62  *
63  * Starting with R 1.4.0 there is a version number associated with
64  * save file formats. This version number should be incremented
65  * when the format is changed so older versions of R can recognize
66  * and reject the new format with a meaningful error message.
67  *
68  * R should remain able to write older workspace formats. An error
69  * should be signaled if the contents to be saved is not compatible
70  * with the requested format.
71  *
72  * To allow older versions of R to give useful error messages, the
73  * header now contains the version of R that wrote the workspace
74  * and the oldest version that can read the workspace. These
75  * versions are stored as an integer packed by the R_Version macro
76  * from Rversion.h. Some workspace formats may only exist
77  * temporarily in the development stage. If readers are not
78  * provided in a release version, then these should specify the
79  * oldest reader R version as -1.
80  */

275  while ((c = R_fgetc(fp)) != '"');

292  }
293  }

454 static SEXP OffsetToNode(int offset, NodeInfo *node)
455 {
456  int l, m, r;
457 
458  if (offset == -1) return R_NilValue;
459  if (offset == -2) return R_GlobalEnv;
460  if (offset == -3) return R_UnboundValue;
461  if (offset == -4) return R_MissingArg;
462 
463  /* binary search for offset */
464 
465  l = 0;
466  r = node->NTotal - 1;
467  do {
468  m = (l + r) / 2;
469  if (offset < node->OldOffset[m])
470  r = m - 1;
471  else
472  l = m + 1;
473  }
474  while (offset != node->OldOffset[m] && l <= r);
475  if (offset == node->OldOffset[m]) return VECTOR_ELT(node->NewAddress, m);
476 
477  /* Not supposed to happen: */
478  Rf_warning(_("unresolved node during restore"));
479  return R_NilValue;
480 }

482 static SEXPTYPE FixupType(unsigned int type, int VersionId)
483 {
484  if (VersionId) {
485  switch(VersionId) {
486 
487  case 16:
488  /* In the version 0.16.1 -> 0.50 switch */
489  /* we really introduced complex values */
490  /* and found that numeric/complex numbers */
491  /* had to be contiguous. Hence this switch */
492  if (type == STRSXP)
493  type = CPLXSXP;
494  else if (type == CPLXSXP)
495  type = STRSXP;
496  break;
497 
498  default:
499  Rf_error(_("restore compatibility error - no version %d compatibility"),
500  VersionId);
501  }
502  }
503

505  if (type == 11 || type == 12)
506  type = 13;
507 
508  return RHOCONSTRUCT(SEXPTYPE, type);

511 static void RemakeNextSEXP(FILE *fp, NodeInfo *node, int version, InputRoutines *m, SaveLoadData *d)
512 {
513  unsigned int j, idx;
514  SEXPTYPE type;
515  int len;
516  SEXP s = R_NilValue; /* -Wall */
517 
518  idx = m->InInteger(fp, d);
519  type = FixupType(m->InInteger(fp, d), version);
520 
521  /* skip over OBJECT, LEVELS, and ATTRIB */
522  /* OBJECT(s) = */ m->InInteger(fp, d);
523  /* LEVELS(s) = */ m->InInteger(fp, d);
524  /* ATTRIB(s) = */ m->InInteger(fp, d);
525  switch (type) {
526  case LISTSXP:
527  s = new PairList;
528  break;
529  case LANGSXP:
530  s = new CachingExpression;
531  break;
532  case CLOSXP:
533  case PROMSXP:
534  case ENVSXP:
535  Rf_error("Loading pre-version-1 serialization not (yet) supported in rho");
536  // All this code needs fixing in rho!
537  // s = new RObject(type);
538  /* skip over CAR, CDR, and TAG */
539  /* CAR(s) = */ m->InInteger(fp, d);
540  /* CDR(s) = */ m->InInteger(fp, d);
541  /* TAG(s) = */ m->InInteger(fp, d);
542  break;
543  case SPECIALSXP:
544  case BUILTINSXP:
545  /* skip over length and name fields */
546  /* length = */ m->InInteger(fp, d);
547  R_AllocStringBuffer(MAXELTSIZE - 1, &(d->buffer));
548  s = BuiltInFunction::obtainPrimitive(m->InString(fp, d));
549  break;
550  case CHARSXP:
551  len = m->InInteger(fp, d);
552  R_AllocStringBuffer(len, &(d->buffer));
553  s = String::obtain(m->InString(fp, d));
554  break;
555  case REALSXP:
556  len = m->InInteger(fp, d);
557  s = Rf_allocVector(type, len);
558  /* skip over the vector content */
559  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
560  /*REAL(s)[j] = */ m->InReal(fp, d);
561  break;
562  case CPLXSXP:
563  len = m->InInteger(fp, d);
564  s = Rf_allocVector(type, len);
565  /* skip over the vector content */
566  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
567  /* COMPLEX(s)[j] = */ m->InComplex(fp, d);
568  break;
569  case INTSXP:
570  case LGLSXP:
571  len = m->InInteger(fp, d);;
572  s = Rf_allocVector(type, len);
573  /* skip over the vector content */
574  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
575  /* INTEGER(s)[j] = */ m->InInteger(fp, d);
576  break;
577  case STRSXP:
578  case VECSXP:
579  case EXPRSXP:
580  len = m->InInteger(fp, d);
581  s = Rf_allocVector(type, len);
582  /* skip over the vector content */
583  for (j = 0; j < RHOCONSTRUCT(uint, len); j++) {
584  /* VECTOR(s)[j] = */ m->InInteger(fp, d);
585  }
586  break;
587  default: Rf_error(_("bad SEXP type in data file"));
588  }
589 
590  /* install the new SEXP */
591  SET_VECTOR_ELT(node->NewAddress, idx, s);
592 }

594 static void RestoreSEXP(SEXP s, FILE *fp, InputRoutines *m, NodeInfo *node, int version, SaveLoadData *d)
595 {
596  unsigned int j;
597  SEXPTYPE type;
598  int len;
599 
600  type = FixupType(m->InInteger(fp, d), version);
601  if (type != TYPEOF(s))
602  Rf_error(_("mismatch on types"));
603 
604  SETLEVELS(s, m->InInteger(fp, d));
605  SET_ATTRIB(s, OffsetToNode(m->InInteger(fp, d), node));
606  switch (TYPEOF(s)) {
607  case LISTSXP:
608  case LANGSXP:
609  case CLOSXP:
610  case PROMSXP:
611  case ENVSXP:
612  SETCAR(s, OffsetToNode(m->InInteger(fp, d), node));
613  SETCDR(s, OffsetToNode(m->InInteger(fp, d), node));
614  SET_TAG(s, OffsetToNode(m->InInteger(fp, d), node));
615  break;
616  case SPECIALSXP:
617  case BUILTINSXP:
618  len = m->InInteger(fp, d);
619  R_AllocStringBuffer(MAXELTSIZE - 1, &(d->buffer));
620  // Skip offset field:
621  m->InString(fp, d);
622  break;
623  case CHARSXP:
624  len = m->InInteger(fp, d);
625  R_AllocStringBuffer(len, &(d->buffer));
626  /* Better to use a fresh copy in the cache */
627  // rho FIXME
628  // strcpy(CHAR_RW(s), m->InString(fp, d));
629  m->InString(fp, d); // Just skip the string for the mo.
630  break;
631  case REALSXP:
632  len = m->InInteger(fp, d);
633  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
634  REAL(s)[j] = m->InReal(fp, d);
635  break;
636  case CPLXSXP:
637  len = m->InInteger(fp, d);
638  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
639  COMPLEX(s)[j] = m->InComplex(fp, d);
640  break;
641  case INTSXP:
642  case LGLSXP:
643  len = m->InInteger(fp, d);;
644  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
645  INTEGER(s)[j] = m->InInteger(fp, d);
646  break;
647  case STRSXP:
648  len = m->InInteger(fp, d);
649  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
650  SET_STRING_ELT(s, j, OffsetToNode(m->InInteger(fp, d), node));
651  break;
652  case VECSXP:
653  case EXPRSXP:
654  len = m->InInteger(fp, d);
655  for (j = 0; j < RHOCONSTRUCT(uint, len); j++)
656  SET_VECTOR_ELT(s, j, OffsetToNode(m->InInteger(fp, d), node));
657  break;

812 /* If "item" is a special value (as defined in "NewSaveSpecialHook")
813  * then a negative value is returned.
814  *
815  * If "item" is present in "list" the a positive value is returned
816  * (the 1-based offset into the list).
817  *
818  * Otherwise, a value of zero is returned.
819  *
820  * The "list" is managed with a hash table. This results in
821  * significant speedups for saving large amounts of code. A fixed
822  * hash table size is used; this is not ideal but seems adequate for
823  * now. The hash table representation consists of a (list . vector)
824  * pair. The hash buckets are in the vector. The list holds the
825  * list of keys. This list is in reverse order to the way the keys
826  * were added (i.e. the most recently added key is first). The
827  * indices produced by HashAdd are in order. Since the list is
828  * written out in order, we either have to reverse the list or
829  * reverse the indices; to retain byte for byte compatibility the
830  * function FixHashEntries reverses the indices. FixHashEntries must
831  * be called after filling the tables and before using them to find
832  * indices. LT */

842 #define SET_HASH_TABLE_COUNT(ht, val) SET_TRUELENGTH(CDR(ht), val)
843 
844 #define HASH_TABLE_SIZE(ht) LENGTH(CDR(ht))
845 
846 #define HASH_BUCKET(ht, pos) VECTOR_ELT(CDR(ht), pos)

866 static void HashAdd(SEXP obj, SEXP ht)
867 {
868  R_size_t pos = PTRHASH(obj) % HASH_TABLE_SIZE(ht);
869  int count = HASH_TABLE_COUNT(ht) + 1;
870  SEXP val = Rf_ScalarInteger(count);
871  SEXP cell = CONS(val, HASH_BUCKET(ht, pos));
872 
873  SET_HASH_TABLE_COUNT(ht, count);
874  SET_HASH_BUCKET(ht, pos, cell);
875  SET_TAG(cell, obj);
876  SET_HASH_TABLE_KEYS_LIST(ht, CONS(obj, HASH_TABLE_KEYS_LIST(ht)));
877  SET_TAG(HASH_TABLE_KEYS_LIST(ht), val);
878 }

885  if (item == TAG(cell))

887  return 0;
888 }
