40 static SEXP cumsum(SEXP x, SEXP s)
41 {
42  LDOUBLE sum = 0.;
43  double *rx = REAL(x), *rs = REAL(s);
44  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
45  sum += rx[i]; /* NA and NaN propagated */
46  rs[i] = double(sum);
47  }
48  return s;
49 }

51 /* We need to ensure that overflow gives NA here */
52 static SEXP icumsum(SEXP x, SEXP s)
53 {
54  int *ix = INTEGER(x), *is = INTEGER(s);
55  double sum = 0.0;
56  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
57  if (ix[i] == R_NaInt) break;
58  sum += ix[i];
59  if(sum > INT_MAX || sum < 1 + INT_MIN) { /* INT_MIN is R_NaInt */
60  Rf_warning(_("integer overflow in 'cumsum'; use 'cumsum(as.numeric(.))'"));
61  break;
62  }
63  is[i] = int(sum);
64  }
65  return s;
66 }

68 static SEXP ccumsum(SEXP x, SEXP s)
69 {
70  Rcomplex sum;
71  sum.r = 0;
72  sum.i = 0;
73  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
74  sum.r += COMPLEX(x)[i].r;
75  sum.i += COMPLEX(x)[i].i;
76  COMPLEX(s)[i].r = sum.r;
77  COMPLEX(s)[i].i = sum.i;
78  }
79  return s;
80 }

82 static SEXP cumprod(SEXP x, SEXP s)
83 {
84  LDOUBLE prod;
85  double *rx = REAL(x), *rs = REAL(s);
86  prod = 1.0;
87  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
88  prod *= rx[i]; /* NA and NaN propagated */
89  rs[i] = double(prod);
90  }
91  return s;
92 }

94 static SEXP ccumprod(SEXP x, SEXP s)
95 {
96  Rcomplex prod, tmp;
97  prod.r = 1;
98  prod.i = 0;
99  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
100  tmp.r = prod.r;
101  tmp.i = prod.i;
102  prod.r = COMPLEX(x)[i].r * tmp.r - COMPLEX(x)[i].i * tmp.i;
103  prod.i = COMPLEX(x)[i].r * tmp.i + COMPLEX(x)[i].i * tmp.r;
104  COMPLEX(s)[i].r = prod.r;
105  COMPLEX(s)[i].i = prod.i;
106  }
107  return s;
108 }

110 static SEXP cummax(SEXP x, SEXP s)
111 {
112  double max, *rx = REAL(x), *rs = REAL(s);
113  max = R_NegInf;
114  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++) {
115  if(std::isnan(rx[i]) || std::isnan(max))
116  max = max + rx[i]; /* propagate NA and NaN */
117  else
118  max = (max > rx[i]) ? max : rx[i];
119  rs[i] = max;
120  }
121  return s;
122 }

124 static SEXP cummin(SEXP x, SEXP s)
125 {
126  double min, *rx = REAL(x), *rs = REAL(s);
127  min = R_PosInf; /* always positive, not NA */
128  for (R_xlen_t i = 0 ; i < XLENGTH(x) ; i++ ) {
129  if (std::isnan(rx[i]) || std::isnan(min))
130  min = min + rx[i]; /* propagate NA and NaN */
131  else
132  min = (min < rx[i]) ? min : rx[i];
133  rs[i] = min;
134  }
135  return s;
136 }

138 static SEXP icummax(SEXP x, SEXP s)
139 {
140  int *ix = INTEGER(x);
141  if(ix[0] == R_NaInt)
142  return s; // all NA
143  int *is = INTEGER(s), max = ix[0];
144  is[0] = max;
145  for (R_xlen_t i = 1 ; i < XLENGTH(x) ; i++) {
146  if(ix[i] == R_NaInt) break;
147  is[i] = max = (max > ix[i]) ? max : ix[i];
148  }
149  return s;
150 }

164 /*
165 op = 1 is cumsum
166 op = 2 is cumprod
167 op = 3 is cummax
168 op = 4 is cummin
169 */
170 HIDDEN SEXP do_cum(/*const*/ rho::Expression* call, const rho::BuiltInFunction* op, rho::RObject* x_)
171 {
172  SEXP s, t, ans;
173  R_xlen_t i, n;
174  if (Rf_isComplex(x_)) {
175  t = x_;
176  n = XLENGTH(t);
177  PROTECT(s = Rf_allocVector(CPLXSXP, n));
178  Rf_setAttrib(s, Symbols::NamesSymbol, Rf_getAttrib(t, Symbols::NamesSymbol));
179  UNPROTECT(1);
180  if(n == 0) return s;
181  /* no need to initialize s, ccum* set all elements */
182  switch (op->variant() ) {
183  case 1: /* cumsum */
184  return ccumsum(t, s);
185  break;
186  case 2: /* cumprod */
187  return ccumprod(t, s);
188  break;
189  case 3: /* cummax */
190  Rf_errorcall(call, _("'cummax' not defined for complex numbers"));
191  break;
192  case 4: /* cummin */
193  Rf_errorcall(call, _("'cummin' not defined for complex numbers"));
194  break;
195  default:
196  Rf_errorcall(call, "unknown cumxxx function");
197  }
198  } else if( ( Rf_isInteger(x_) || Rf_isLogical(x_) ) &&
199  op->variant() != 2) {
200  PROTECT(t = Rf_coerceVector(x_, INTSXP));
201  n = XLENGTH(t);
202  PROTECT(s = Rf_allocVector(INTSXP, n));
203  Rf_setAttrib(s, Symbols::NamesSymbol, Rf_getAttrib(t, Symbols::NamesSymbol));
204  if(n == 0) {
205  UNPROTECT(2); /* t, s */
206  return s;
207  }
208  for(i = 0 ; i < n ; i++) INTEGER(s)[i] = R_NaInt;
209  switch (op->variant() ) {
210  case 1: /* cumsum */
211  ans = icumsum(t,s);
212  break;
213  case 3: /* cummax */
214  ans = icummax(t,s);
215  break;
216  case 4: /* cummin */
217  ans = icummin(t,s);
218  break;
219  default:
220  Rf_errorcall(call, _("unknown cumxxx function"));
221  ans = nullptr;
222  }
223  UNPROTECT(2); /* t, s */
224  return ans;
225  } else {
226  PROTECT(t = Rf_coerceVector(x_, REALSXP));
227  n = XLENGTH(t);
228  PROTECT(s = Rf_allocVector(REALSXP, n));
229  Rf_setAttrib(s, Symbols::NamesSymbol, Rf_getAttrib(t, Symbols::NamesSymbol));
230  UNPROTECT(2);
231  if(n == 0) return s;
232  /* no need to initialize s, cum* set all elements */
233  switch (op->variant() ) {
234  case 1: /* cumsum */
235  return cumsum(t,s);
236  break;
237  case 2: /* cumprod */
238  return cumprod(t,s);
239  break;
240  case 3: /* cummax */
241  return cummax(t,s);
242  break;
243  case 4: /* cummin */
244  return cummin(t,s);
245  break;
246  default:
247  Rf_errorcall(call, _("unknown cumxxx function"));
248  }
249  }
250  return nullptr; /* for -Wall */
251 }
