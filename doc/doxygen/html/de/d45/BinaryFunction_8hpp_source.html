211  template<typename Op, typename AttributeCopier,
212  typename LhsType, typename RhsType,
213  typename OutputType = VectorOpReturnType<Op, LhsType, RhsType>>
214  OutputType* applyBinaryOperator(const Op& op,
215  AttributeCopier attribute_copier,
216  const LhsType* lhs,
217  const RhsType* rhs)
218  {
219  size_t lhs_size = lhs->size();
220  size_t rhs_size = rhs->size();
221  size_t size = std::max(lhs->size(), rhs->size());
222  if (lhs_size == 0 || rhs_size == 0) {
223  /* S4-compatibility change: if lhs or rhs are zero length, then
224  the result is zero length. */
225  size = 0;
226  }
227  if (size == 1 && !lhs->hasAttributes() && !rhs->hasAttributes()) {
228  return OutputType::createScalar(op((*lhs)[0], (*rhs)[0]));
229  }
230  OutputType* result = OutputType::create(size);
231  if (size == 1) {
232  (*result)[0] = op((*lhs)[0], (*rhs)[0]);
233  } else if (lhs_size == 1) {
234  // TODO: move these into a separate function so that the scalar
235  // case can be inlined.
236  typename LhsType::value_type lhs_value = (*lhs)[0];
237  for (size_t i = 0; i < size; i++) {
238  (*result)[i] = op(lhs_value, (*rhs)[i]);
239  }
240  } else if (rhs_size == 1) {
241  typename RhsType::value_type rhs_value = (*rhs)[0];
242  for (size_t i = 0; i < size; i++) {
243  (*result)[i] = op((*lhs)[i], rhs_value);
244  }
245  } else if (lhs_size == rhs_size) {
246  for (size_t i = 0; i < size; i++) {
247  (*result)[i] = op((*lhs)[i], (*rhs)[i]);
248  }
249  } else {
250  // Full recycling rule.
251  size_t lhs_i = 0, rhs_i = 0;
252  for (size_t i = 0; i < size; i++)
253  {
254  (*result)[i] = op((*lhs)[lhs_i], (*rhs)[rhs_i]);
255 
256  lhs_i = lhs_i + 1 == lhs_size ? 0 : lhs_i + 1;
257  rhs_i = rhs_i + 1 == rhs_size ? 0 : rhs_i + 1;
258  }
259 
260  if (lhs_i != 0 || rhs_i != 0) {
261  Rf_warning(_("longer object length is not a multiple of shorter object length"));
262  }
263  }
264 
265  attribute_copier.copyAttributes(result, lhs, rhs);
266  return result;
267  }
