138 // This follows CR in adding new attributes at the end of the list,
139 // though it would be easier to add them at the beginning.
140 void RObject::setAttribute(const Symbol* name, RObject* value)
141 {
142  if (!name)
143  Rf_error(_("attributes must be named"));
144  // Update 'has class' bit if necessary:
145  if (name == R_ClassSymbol) {
146  if (value == nullptr)
147  m_type &= static_cast<signed char>(~s_class_mask);
148  else m_type |= static_cast<signed char>(s_class_mask);
149  }
150  // Find attribute:
151  PairList* prev = nullptr;
152  PairList* node = m_attrib;
153  while (node && node->tag() != name) {
154  prev = node;
155  node = node->tail();
156  }
157  if (node) { // Attribute already present
158  // Update existing attribute:
159  if (value)
160  node->setCar(value);
161  // Delete existing attribute:
162  else if (prev)
163  prev->setTail(node->tail());
164  else m_attrib = node->tail();
165  } else if (value) {
166  // Create new node:
167  PairList* newnode = PairList::cons(value, nullptr, name);
168  if (prev)
169  prev->setTail(newnode);
170  else { // No preexisting attributes at all:
171  m_attrib = newnode;
172  }
173  }
174 }

261 /*
262  * Evil lurks here.
263  *
264  * To support SET_TYPEOF (which is unbelievably nasty, but unfortunately
265  * required by some important packages), we need to turn objects of one type
266  * into another, in-place.
267  *
268  * The way that this is done is to store the contents of the object, including
269  * GC related state, into local variables, explicitly destroy the object
270  * and then use placement new to create an object of the new type in its place.
271  * Finally, the values are restored.
272  *
273  * This scheme is somewhat brittle -- changes to the way that objects are
274  * represented may require rewriting parts of this code.
275  */
276 void RObject::Transmute(RObject* source,
277  std::function<RObject*(void*)> constructor)
278 {
279  // Store RObject properties.
280  unsigned char named = source->m_named;
281  bool isS4 = source->isS4Object();
282 #ifdef R_MEMORY_PROFILING
283  bool memory_traced = source->memoryTraced();
284 #endif
285  unsigned missing = source->m_missing;
286  bool active_binding = source->m_active_binding;
287  bool binding_locked = source->m_binding_locked;
288  const PairList* attributes = source->attributes();
289  auto gc_data = source->storeInternalData();
290 
291  // Destroy the object and create the new type in it's place.
292  void* location = source;
293  source->~RObject();
294  RObject* dest = constructor(location);
295 
296  // Restore the RObject properties.
297  dest->restoreInternalData(gc_data);
298  dest->setAttributes(attributes);
299 
300  dest->m_named = named;
301  dest->setS4Object(isS4);
302 #ifdef R_MEMORY_PROFILING
303  dest->setMemoryTracing(memory_traced);
304 #endif
305  dest->m_missing = missing;
306  dest->m_active_binding = active_binding;
307  dest->m_binding_locked = binding_locked;
308 }

330 void RObject::TransmuteConsCell(ConsCell* object, SEXPTYPE dest_type)
331 {
332  static_assert(sizeof(CachingExpression) == sizeof(PaddedPairList),
333  "Expected PaddedPairList and CachingExpression to be the same size");
334  static_assert(sizeof(PairList) == sizeof(Expression),
335  "Expected PairList and Expression to be the same size");
336 
337  // Store the fields from the object.
338  RObject* car = object->car();
339  PairList* tail = object->tail();
340  const RObject* tag = object->tag();
341 
342  // Transmute the object.
343  std::function<ConsCell*(void*)> constructor;
344  if (dest_type == LANGSXP) {
345  constructor = dynamic_cast<PaddedPairList*>(object)
346  ? [](void* p) -> ConsCell* { return new(p) CachingExpression; }
347  : [](void* p) -> ConsCell* { return new(p) Expression; };
348  } else {
349  constructor = dynamic_cast<CachingExpression*>(object)
350  ? [](void* p) -> ConsCell* { return new(p) PaddedPairList; }
351  : [](void* p) -> ConsCell* { return new(p) PairList; };
352  }
353 
354  RObject::Transmute(object, constructor);
355 
356  // Restore the values.
357  object->setCar(car);
358  object->setTail(tail);
359  object->setTag(tag);
360 }

362 void RObject::TransmuteLogicalToInt(RObject* x)
363 {
364  LogicalVector* object = SEXP_downcast<LogicalVector*>(x);
365 
366  size_t length = object->size();
367  size_t truelength = XTRUELENGTH(object);
368 
369  // Store any data values that fall within the memory range of the
370  // object.
371  static const int STORAGE_SIZE = 4;
372  Logical storage[STORAGE_SIZE];
373 
374  Logical* data_start = object->begin();
375  Logical* data_end = object->end();
376  Logical* object_end = reinterpret_cast<Logical*>(object + 1);
377  bool data_start_is_in_object
378  = (void*)object <= data_start && data_start <= object_end;
379 
380  Logical* embedded_data_end = std::min(data_end, object_end);
381  ptrdiff_t stored_length = embedded_data_end - data_start;
382 
383  // Sanity checking.
384  assert(data_start_is_in_object);
385  assert(stored_length <= STORAGE_SIZE);
386  if (!data_start_is_in_object || stored_length > STORAGE_SIZE) {
387  Rf_error("Unexpected LogicalVector layout in SET_TYPEOF");
388  }
389 
390  std::copy(data_start, embedded_data_end, storage);
391 
392  // Replace the original LogicalVector an IntVector in the same memory
393  // location.
394  RObject::Transmute(object,
395  [=](void* p) { return new(p) IntVector(length); });
396 
397  // Restore the truelength and stored values.
398  SET_TRUELENGTH(object, truelength);
399  std::copy(storage, storage + stored_length, data_start);
400 }

402 void SET_TYPEOF(SEXP x, SEXPTYPE dest_type) {
403  SEXPTYPE source_type = x->sexptype();
404  if (source_type == dest_type)
405  return;
406 
407  switch(dest_type) {
408  case LANGSXP:
409  case LISTSXP:
410  if (source_type == LANGSXP || source_type == LISTSXP) {
411  RObject::TransmuteConsCell(SEXP_downcast<ConsCell*>(x), dest_type);
412  return;
413  }
414  case INTSXP:
415  if (source_type == LGLSXP) {
416  RObject::TransmuteLogicalToInt(SEXP_downcast<LogicalVector*>(x));
417  return;
418  }
419  default:
420  break;
421  }
422 
423  Rf_error(
424  "Calling SET_TYPEOF to convert from type %s to type %s is not supported in rho",
425  Rf_type2char(source_type),
426  Rf_type2char(dest_type));
