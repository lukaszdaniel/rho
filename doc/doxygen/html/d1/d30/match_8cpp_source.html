27  * Matching and Partial Matching for Strings
28  *
29  * In theory all string matching code should be placed in this file
30  * At present there are still a couple of rogue matchers about.
31  *
32  *
33  * psmatch(char *, char *, int);
34  *
35  * This code will perform partial matching for list tags. When
36  * exact is 1, and exact match is required (typically after ...)
37  * otherwise partial matching is performed.
38  *
39  * Examples:
40  *
41  * psmatch("aaa", "aaa", 0) -> 1
42  * psmatch("aaa", "aa", 0) -> 1
43  * psmatch("aa", "aaa", 0) -> 0
44  *
45  */

63 Rboolean Rf_NonNullStringMatch(SEXP s, SEXP t)
64 {
65  /* "" or NA string matches nothing */
66  if (s == R_NaString || t == R_NaString) return FALSE;
67  if (R_CHAR(s)[0] && R_CHAR(t)[0] && Rf_Seql(s, t))
68  return TRUE;
69  else
70  return FALSE;
71 }

74 Rboolean Rf_psmatch(const char *f, const char *t, Rboolean exact)
75 {
76  if (exact)
77  return Rboolean(streql(f, t));
78  /* else */
79  while (*t) {
80  if (*t != *f) return FALSE;
81  t++;
82  f++;
83  }
84  return TRUE;
85 }

91 Rboolean Rf_pmatch(SEXP formal, SEXP tag, Rboolean exact)
92 {
93  const char *f, *t;
94  const void *vmax = vmaxget();
95  Rboolean res; // In rho this must be declared outside the span of the gotos
96  switch (TYPEOF(formal)) {
97  case SYMSXP:
98  f = R_CHAR(PRINTNAME(formal));
99  break;
100  case CHARSXP:
101  f = R_CHAR(formal);
102  break;
103  case STRSXP:
104  f = Rf_translateChar(STRING_ELT(formal, 0));
105  break;
106  default:
107  goto fail;
108  }
109  switch(TYPEOF(tag)) {
110  case SYMSXP:
111  t = R_CHAR(PRINTNAME(tag));
112  break;
113  case CHARSXP:
114  t = R_CHAR(tag);
115  break;
116  case STRSXP:
117  t = Rf_translateChar(STRING_ELT(tag, 0));
118  break;
119  default:
120  goto fail;
121  }
122  res = Rf_psmatch(f, t, exact);
123  vmaxset(vmax);
124  return res;
125  fail:
126  Rf_error(_("invalid partial string match"));
127  return FALSE;/* for -Wall */
128 }

130 void ArgMatcher::unusedArgsError(const SuppliedList& supplied_list)
131 {
132  GCStackRoot<PairList> unused_list;
133  // Produce a PairList of the unused args:
134  for (SuppliedList::const_reverse_iterator rit = supplied_list.rbegin();
135  rit != supplied_list.rend(); ++rit) {
136  const SuppliedData& supplied_data = *rit;
137  RObject* value = supplied_data.value;
138  if (value != nullptr && value->sexptype() == PROMSXP)
139  value = const_cast<RObject*>(PREXPR(value));
140  unused_list = PairList::cons(value, unused_list, supplied_data.tag);
141  }
142  unusedArgsError(unused_list);
143 }

158 /* Returns the first exactly matching tag found. */
159 /* Pattern is a symbol. */
160 // Used for matching na.rm in logic.cpp and match.cpp.
161 HIDDEN SEXP Rf_matchArgExact(SEXP tag, SEXP *list)
162 {
163  if (*list == nullptr)
164  return R_MissingArg;
165  else if (TAG(*list) == tag) {
166  SEXP s = *list;
167  *list = CDR(*list);
168  return CAR(s);
169  }
170  else {
171  SEXP last = *list;
172  SEXP next = CDR(*list);
173  while (next != nullptr) {
174  if (TAG(next) == tag) {
175  SETCDR(last, CDR(next));
176  return CAR(next);
177  }
178  else {
179  last = next;
180  next = CDR(next);
181  }
182  }
183  return R_MissingArg;
184  }
