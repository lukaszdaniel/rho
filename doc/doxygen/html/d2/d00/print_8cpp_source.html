180 HIDDEN SEXP do_printfunction(/*const*/ Expression* call, const BuiltInFunction* op, Environment* rho, RObject* const* args, int num_args, const PairList* tags)
181 {
182  op->checkNumArgs(num_args, call);
183  SEXP s = args[0];
184  switch (TYPEOF(s)) {
185  case CLOSXP:
186  PrintLanguageEtc(s, Rboolean(Rf_asLogical(args[1])), /*is closure = */ TRUE);
187  printAttributes(s, rho, FALSE);
188  break;
189  case BUILTINSXP:
190  case SPECIALSXP:
191  PrintSpecial(s);
192  break;
193 
194  default: /* if(!Rf_isFunction(s)) */
195  Rf_error(_("non-function argument to .Internal(print.function(.))"));
196  }
197  return s;
198 }

200 /* PrintLanguage() or PrintClosure() : */
201 static void PrintLanguageEtc(SEXP s, Rboolean useSource, Rboolean isClosure)
202 {
203  int i;
204  SEXP t = Rf_getAttrib(s, Symbols::SrcrefSymbol);
205  Rboolean useSrc = Rboolean(useSource && Rf_isInteger(t));
206  if (useSrc) {
207  PROTECT(t = Rf_lang2(Symbols::AsCharacterSymbol, t));
208  t = Rf_eval(t, R_BaseEnv);
209  UNPROTECT(1);
210  } else {
211  t = Rf_deparse1w(s, Rboolean(0), useSource | DEFAULTDEPARSE);
212  }
213  PROTECT(t);
214  for (i = 0; i < LENGTH(t); i++) {
215  Rprintf("%s\n", Rf_translateChar(STRING_ELT(t, i))); // translate: for srcref part (PR#16732)
216  }
217  UNPROTECT(1);
218  if (isClosure) {
219  if (isByteCode(BODY(s))) Rprintf("<bytecode: %p>\n", BODY(s));
220  t = CLOENV(s);
221  if (t != R_GlobalEnv)
222  Rprintf("%s\n", EncodeEnvironment(t));
223  }
224 }

238 HIDDEN SEXP do_printdefault(/*const*/ Expression* call, const BuiltInFunction* op, Environment* rho, RObject* const* args, int num_args, const PairList* tags)
239 {
240  SEXP x, naprint;
241  int tryS4;
242  Rboolean callShow = FALSE;
243 
244  Rf_PrintDefaults();
245 
246  x = args[0]; args = (args + 1);
247 
248  if(!Rf_isNull(args[0])) {
249  R_print.digits = Rf_asInteger(args[0]);
250  if (R_print.digits == R_NaInt ||
251  R_print.digits < R_MIN_DIGITS_OPT ||
252  R_print.digits > R_MAX_DIGITS_OPT)
253  Rf_error(_("invalid '%s' argument"), "digits");
254  }
255  args = (args + 1);
256 
257  R_print.quote = Rf_asLogical(args[0]);
258  if(R_print.quote == R_NaLog)
259  Rf_error(_("invalid '%s' argument"), "quote");
260  args = (args + 1);
261 
262  naprint = args[0];
263  if(!Rf_isNull(naprint)) {
264  if(!Rf_isString(naprint) || LENGTH(naprint) < 1)
265  Rf_error(_("invalid 'na.print' specification"));
266  R_print.na_string = R_print.na_string_noquote = STRING_ELT(naprint, 0);
267  R_print.na_width = R_print.na_width_noquote =
268  int(strlen(R_CHAR(R_print.na_string)));
269  }
270  args = (args + 1);
271 
272  if(!Rf_isNull(args[0])) {
273  R_print.gap = Rf_asInteger(args[0]);
274  if (R_print.gap == R_NaInt || R_print.gap < 0)
275  Rf_error(_("'gap' must be non-negative integer"));
276  }
277  args = (args + 1);
278 
279  R_print.right = Rprt_adj(Rf_asLogical(args[0])); /* Should this be Rf_asInteger()? */
280  if(R_print.right == R_NaLog)
281  Rf_error(_("invalid '%s' argument"), "right");
282  args = (args + 1);
283 
284  if(!Rf_isNull(args[0])) {
285  R_print.max = Rf_asInteger(args[0]);
286  if(R_print.max == R_NaInt || R_print.max < 0)
287  Rf_error(_("invalid '%s' argument"), "max");
288  else if(R_print.max == INT_MAX) R_print.max--; // so we can add
289  }
290  args = (args + 1);
291 
292  R_print.useSource = Rf_asLogical(args[0]);
293  if(R_print.useSource == R_NaLog)
294  Rf_error(_("invalid '%s' argument"), "useSource");
295  if(R_print.useSource) R_print.useSource = USESOURCE;
296  args = (args + 1);
297 
298  tryS4 = Rf_asLogical(args[0]);
299  if(tryS4 == R_NaLog)
300  Rf_error(_("invalid 'tryS4' internal argument"));
301 
302  if(tryS4 && IS_S4_OBJECT(x) && isMethodsDispatchOn())
303  callShow = TRUE;
304 
305  if(callShow) {
306  /* we need to get show from the methods namespace if it is
307  not visible on the search path. */
308  SEXP call, showS;
309  showS = Rf_findVar(Rf_install("show"), rho);
310  if(showS == R_UnboundValue) {
311  SEXP methodsNS = R_FindNamespace(Rf_mkString("methods"));
312  if(methodsNS == R_UnboundValue)
313  Rf_error("missing methods namespace: this should not happen");
314  PROTECT(methodsNS);
315  showS = Rf_findVarInFrame3(methodsNS, Rf_install("show"), TRUE);
316  UNPROTECT(1);
317  if(showS == R_UnboundValue)
318  Rf_error("missing show() in methods namespace: this should not happen");
319  }
320  PROTECT(call = Rf_lang2(showS, x));
321  Rf_eval(call, rho);
322  UNPROTECT(1);
323  } else {
324  Rf_CustomPrintValue(x, rho);
325  }
326 
327  Rf_PrintDefaults(); /* reset, as na.print etc may have been set */
328  return x;
329 }/* do_printdefault */

336 static void PrintGenericVector(SEXP s, SEXP env)
337 {
338  int i, taglen, ns, w, d, e, wr, dr, er, wi, di, ei;
339  SEXP dims, t, names, newcall, tmp;
340  char pbuf[115], *ptag, save[TAGBUFLEN0];
341 
342  ns = Rf_length(s);
343  if((dims = Rf_getAttrib(s, Symbols::DimSymbol)) != nullptr && Rf_length(dims) > 1) {
344  // special case: array-like list
345  PROTECT(dims);
346  PROTECT(t = Rf_allocArray(STRSXP, dims));
347  /* FIXME: check (ns <= R_print.max +1) ? ns : R_print.max; */
348  for (i = 0; i < ns; i++) {
349  switch(TYPEOF(PROTECT(tmp = VECTOR_ELT(s, i)))) {
350  case NILSXP:
351  snprintf(pbuf, 115, "NULL");
352  break;
353  case LGLSXP:
354  if (LENGTH(tmp) == 1) {
355  const int *x = LOGICAL_RO(tmp);
356  formatLogical(x, 1, &w);
357  snprintf(pbuf, 115, "%s",
358  EncodeLogical(x[0], w));
359  } else
360  snprintf(pbuf, 115, "Logical,%d", LENGTH(tmp));
361  break;
362  case INTSXP:
363  /* factors are stored as integers */
364  if (Rf_inherits(tmp, "factor")) {
365  snprintf(pbuf, 115, "factor,%d", LENGTH(tmp));
366  } else {
367  if (LENGTH(tmp) == 1) {
368  const int *x = INTEGER_RO(tmp);
369  formatInteger(x, 1, &w);
370  snprintf(pbuf, 115, "%s",
371  EncodeInteger(x[0], w));
372  } else
373  snprintf(pbuf, 115, "Integer,%d", LENGTH(tmp));
374  }
375  break;
376  case REALSXP:
377  if (LENGTH(tmp) == 1) {
378  const double *x = REAL_RO(tmp);
379  formatReal(x, 1, &w, &d, &e, 0);
380  snprintf(pbuf, 115, "%s",
381  EncodeReal0(x[0], w, d, e, OutDec));
382  } else
383  snprintf(pbuf, 115, "Numeric,%d", LENGTH(tmp));
384  break;
385  case CPLXSXP:
386  if (LENGTH(tmp) == 1) {
387  const Rcomplex *x = COMPLEX_RO(tmp);
388  if (ISNA(x[0].r) || ISNA(x[0].i))
389  /* formatReal(NA) --> w=R_print.na_width, d=0, e=0 */
390  snprintf(pbuf, 115, "%s",
391  EncodeReal0(R_NaReal, R_print.na_width, 0, 0, OutDec));
392  else {
393  formatComplex(x, 1, &wr, &dr, &er, &wi, &di, &ei, 0);
394  snprintf(pbuf, 115, "%s",
395  EncodeComplex(x[0],
396  wr, dr, er, wi, di, ei, OutDec));
397  }
398  } else
399  snprintf(pbuf, 115, "Complex,%d", LENGTH(tmp));
400  break;
401  case STRSXP:
402  if (LENGTH(tmp) == 1) {
403  const void *vmax = vmaxget();
404  /* This can potentially overflow */
405  const char *ctmp = Rf_translateChar(STRING_ELT(tmp, 0));
406  int len = int(strlen(ctmp));
407  if(len < 100)
408  snprintf(pbuf, 115, "\"%s\"", ctmp);
409  else {
410  snprintf(pbuf, 101, "\"%s\"", ctmp);
411  pbuf[100] = '"'; pbuf[101] = '\0';
412  strcat(pbuf, " [truncated]");
413  }
414  vmaxset(vmax);
415  } else
416  snprintf(pbuf, 115, "Character,%d", LENGTH(tmp));
417  break;
418  case RAWSXP:
419  snprintf(pbuf, 115, "Raw,%d", LENGTH(tmp));
420  break;
421  case LISTSXP:
422  case VECSXP:
423  snprintf(pbuf, 115, "List,%d", Rf_length(tmp));
424  break;
425  case LANGSXP:
426  snprintf(pbuf, 115, "Expression");
427  break;
428  default:
429  snprintf(pbuf, 115, "?");
430  break;
431  }
432  UNPROTECT(1); /* tmp */
433  pbuf[114] = '\0';
434  SET_STRING_ELT(t, i, Rf_mkChar(pbuf));
435  }
436  if (LENGTH(dims) == 2) {
437  SEXP rl, cl;
438  const char *rn, *cn;
439  Rf_GetMatrixDimnames(s, &rl, &cl, &rn, &cn);
440  /* as from 1.5.0: don't quote here as didn't in array case */
441  printMatrix(t, 0, dims, 0, R_print.right, rl, cl, rn, cn);
442  }
443  else {
444  names = Rf_GetArrayDimnames(s);
445  printArray(t, dims, 0, Rprt_adj_left, names);
446  }
447  UNPROTECT(2);
448  }
449  else { // no dim()
450  names = Rf_getAttrib(s, Symbols::NamesSymbol);
451  taglen = int(strlen(tagbuf));
452  ptag = tagbuf + taglen;
453  PROTECT(newcall = new Expression(Rf_install("print"), { nullptr }));
454 
455  if(ns > 0) {
456  int n_pr = (ns <= R_print.max +1) ? ns : R_print.max;
457  /* '...max +1' ==> will omit at least 2 ==> plural in msg below */
458  for (i = 0; i < n_pr; i++) {
459  if (i > 0) Rprintf("\n");
460  if (names != nullptr &&
461  STRING_ELT(names, i) != nullptr &&
462  *R_CHAR(STRING_ELT(names, i)) != '\0') {
463  const void *vmax = vmaxget();
464  /* Bug for L <- list(`a\\b` = 1, `a\\c` = 2) :
465  const char *ss = Rf_translateChar(STRING_ELT(names, i));
466  */
467  const char *ss = Rf_EncodeChar(STRING_ELT(names, i));
468  if (taglen + strlen(ss) > TAGBUFLEN) {
469  if (taglen <= TAGBUFLEN)
470  sprintf(ptag, "$...");
471  } else {
472  /* we need to distinguish character NA from "NA", which
473  is a valid (if non-syntactic) name */
474  if (STRING_ELT(names, i) == R_NaString)
475  sprintf(ptag, "$<NA>");
476  else if( Rf_isValidName(ss) )
477  sprintf(ptag, "$%s", ss);
478  else
479  sprintf(ptag, "$`%s`", ss);
480  }
481  vmaxset(vmax);
482  }
483  else {
484  if (taglen + Rf_IndexWidth(i) > TAGBUFLEN) {
485  if (taglen <= TAGBUFLEN)
486  sprintf(ptag, "$...");
487  } else
488  sprintf(ptag, "[[%d]]", i+1);
489  }
490  Rprintf("%s\n", tagbuf);
491  if(Rf_isObject(VECTOR_ELT(s, i))) {
492  SEXP x = VECTOR_ELT(s, i);
493  int nprot = 0;
494  if (TYPEOF(x) == LANGSXP) {
495  // quote(x) to not accidentally evaluate it with newcall() below:
496  x = PROTECT(Rf_lang2(R_Primitive("quote"), x)); nprot++;
497  }
498  /* need to preserve tagbuf */
499  strcpy(save, tagbuf);
500  SETCADR(newcall, x);
501  Rf_eval(newcall, env);
502  strcpy(tagbuf, save);
503  UNPROTECT(nprot);
504  }
505  else Rf_PrintValueRec(VECTOR_ELT(s, i), env);
506  *ptag = '\0';
507  }
508  Rprintf("\n");
509  if(n_pr < ns)
510  Rprintf(" [ reached getOption(\"max.print\") -- omitted %d entries ]\n",
511  ns - n_pr);
512  }
513  else { /* ns = length(s) == 0 */
514  const void *vmax = vmaxget();
515  /* Formal classes are represented as empty lists */
516  const char *className = nullptr;
517  SEXP klass;
518  if(Rf_isObject(s) && isMethodsDispatchOn()) {
519  klass = Rf_getAttrib(s, Symbols::ClassSymbol);
520  if(Rf_length(klass) == 1) {
521  /* internal version of isClass() */
522  char str[201];
523  const char *ss = Rf_translateChar(STRING_ELT(klass, 0));
524  snprintf(str, 200, ".__C__%s", ss);
525  if(Rf_findVar(rho::Symbol::obtain(str), env) != R_UnboundValue)
526  className = ss;
527  }
528  }
529  if(className) {
530  Rprintf("An object of class \"%s\"\n", className);
531  UNPROTECT(1);
532  printAttributes(s, env, TRUE);
533  vmaxset(vmax);
534  return;
535  }
536  else {
537  if(names != nullptr) Rprintf("named ");
538  Rprintf("list()\n");
539  }
540  vmaxset(vmax);
541  }
542  UNPROTECT(1);
543  }
544  printAttributes(s, env, FALSE);
545 } // PrintGenericVector

555  const char *rn, *cn;
556 
557  if ((dims = Rf_getAttrib(s, Symbols::DimSymbol)) != nullptr && Rf_length(dims) > 1) {

559  PROTECT(dims);
560  PROTECT(t = Rf_allocArray(STRSXP, dims));
561  i = 0;
562  while(s != nullptr) {
563  switch(TYPEOF(CAR(s))) {
564 
565  case NILSXP:
566  snprintf(pbuf, 100, "NULL");

568

572

576  break;
577 
578  case CPLXSXP:
579  snprintf(pbuf, 100, "Complex,%d", LENGTH(CAR(s)));
580  break;
581 
582  case STRSXP:
583  snprintf(pbuf, 100, "Character,%d", LENGTH(CAR(s)));
584  break;
585 
586  case RAWSXP:
587  snprintf(pbuf, 100, "Raw,%d", LENGTH(CAR(s)));
588  break;

866 static void printAttributes(SEXP s, SEXP env, Rboolean useSlots)
867 {

870  char save[TAGBUFLEN0] = "\0";
871 
872  a = ATTRIB(s);
873  if (a != nullptr) {
874  /* guard against cycles through attributes on environments */
875  if (strlen(tagbuf) > TAGBUFLEN0)
876  Rf_error(_("print buffer overflow"));
877  strcpy(save, tagbuf);
878  /* remove the tag if it looks like a list not an attribute */
879  if (strlen(tagbuf) > 0 &&
880  *(tagbuf + strlen(tagbuf) - 1) != ')')
881  tagbuf[0] = '\0';
882  ptag = tagbuf + strlen(tagbuf);
883  while (a != nullptr) {
884  if(useSlots && TAG(a) == Symbols::ClassSymbol)
885  goto nextattr;
886  if(Rf_isArray(s) || Rf_isList(s)) {
887  if(TAG(a) == Symbols::DimSymbol ||
888  TAG(a) == Symbols::DimNamesSymbol)
889  goto nextattr;
890  }
891  if(Rf_inherits(s, "factor")) {
892  if(TAG(a) == Symbols::LevelsSymbol)
893  goto nextattr;
894  if(TAG(a) == Symbols::ClassSymbol)
895  goto nextattr;
896  }
897  if(Rf_isFrame(s)) {
898  if(TAG(a) == Symbols::RowNamesSymbol)
899  goto nextattr;
900  }
901  if(!Rf_isArray(s)) {
902  if (TAG(a) == Symbols::NamesSymbol)
903  goto nextattr;
904  }
905  if(TAG(a) == Symbols::CommentSymbol || TAG(a) == Symbols::SrcrefSymbol
906  || TAG(a) == Symbols::WholeSrcrefSymbol || TAG(a) == Symbols::SrcfileSymbol)
907  goto nextattr;
908  if(useSlots)
909  sprintf(ptag, "Slot \"%s\":", Rf_EncodeChar(PRINTNAME(TAG(a))));
910  else
911  sprintf(ptag, "attr(,\"%s\")", Rf_EncodeChar(PRINTNAME(TAG(a))));
912  Rprintf("%s", tagbuf); Rprintf("\n");
913  if (TAG(a) == Symbols::RowNamesSymbol) {
914  /* need special handling AND protection */
915  SEXP val;
916  PROTECT(val = Rf_getAttrib(s, Symbols::RowNamesSymbol));
917  Rf_PrintValueRec(val, env);
918  UNPROTECT(1);
919  goto nextattr;
920  }

922  SEXP s, showS;
923 
924  showS = Rf_findVar(Rf_install("show"), env);
925  if(showS == R_UnboundValue) {
926  SEXP methodsNS = R_FindNamespace(Rf_mkString("methods"));
927  if(methodsNS == R_UnboundValue)
928  Rf_error("missing methods namespace: this should not happen");
929  PROTECT(showS);
930  PROTECT(methodsNS);
931  showS = Rf_findVarInFrame3(methodsNS, Rf_install("show"), TRUE);
932  UNPROTECT(2);
933  if(showS == R_UnboundValue)
934  Rf_error("missing show() in methods namespace: this should not happen");
935  }
936  PROTECT(s = Rf_lang2(showS, CAR(a)));
937  Rf_eval(s, env);
938  UNPROTECT(1);
939  } else if (Rf_isObject(CAR(a))) {
940  /* Need to construct a call to
941  print(CAR(a), digits)
942  based on the R_print structure, then Rf_eval(call, env).
943  See do_docall for the template for this sort of thing.
944 
945  quote, right, gap should probably be included if
946  they have non-missing values.
947 
948  This will not dispatch to show() as 'digits' is supplied.
949  */
950  SEXP s, na_string = R_print.na_string,
951  na_string_noquote = R_print.na_string_noquote;
952  int quote = R_print.quote,
953  digits = R_print.digits, gap = R_print.gap,
954  na_width = R_print.na_width,
955  na_width_noquote = R_print.na_width_noquote;
956  Rprt_adj right = Rprt_adj(R_print.right);
957 
958  SEXP x = CAR(a);
959  int nprot = 0;
960  if (TYPEOF(x) == LANGSXP) {
961  x = PROTECT(Rf_lang2(R_Primitive("quote"), x)); nprot++;
962  }
963  s = PROTECT(new Expression(Rf_install("print"),
964  { x, Rf_ScalarInteger(digits) })); nprot++;
965  SET_TAG(CDDR(s), Rf_install("digits"));
966  Rf_eval(s, env);
967  UNPROTECT(nprot);
968  R_print.quote = quote;
969  R_print.right = right;
970  R_print.digits = digits;
971  R_print.gap = gap;
972  R_print.na_width = na_width;
973  R_print.na_width_noquote = na_width_noquote;
974  R_print.na_string = na_string;
975  R_print.na_string_noquote = na_string_noquote;
976  } else
977  Rf_PrintValueRec(CAR(a), env);
978  nextattr:
979  *ptag = '\0';
980  a = CDR(a);
981  }
982  strcpy(tagbuf, save);
983  }

994  PROTECT(s);
995  if(Rf_isObject(s) || Rf_isFunction(s)) {
996  /*
