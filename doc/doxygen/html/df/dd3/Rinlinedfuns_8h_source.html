572 INLINE_FUN Rboolean Rf_isVectorizable(SEXP s)
573 {

575  else if (Rf_isNewList(s)) {
576  R_xlen_t i, n;
577

579  for (i = 0 ; i < n; i++)

587  return TRUE;
