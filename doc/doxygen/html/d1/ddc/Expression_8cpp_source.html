89 FunctionBase *Expression::getFunction(Environment *env) const
90 {
91  RObject *head = car();
92  if (head->sexptype() == SYMSXP)
93  {
94  Symbol *symbol = SEXP_downcast<Symbol *>(head);
95  FunctionBase *func = findFunction(symbol, env);
96  if (!func)
97  Rf_error(_("could not find function \"%s\""),
98  symbol->name()->c_str());

100  }
101  else
102  {
103  RObject *val = Evaluator::evaluate(head, env);
104  if (!FunctionBase::isA(val))
105  Rf_error(_("attempt to apply non-function in getFunction"));
106  return SEXP_downcast<FunctionBase *>(val);
107  }

183 RObject *Expression::evaluateNativeBuiltInCall(const BuiltInFunction *func,
184  Environment *env,
185  const ArgList &arglist) const
186 {
187  assert(func->sexptype() == SPECIALSXP || arglist.status() == ArgList::EVALUATED);
188  const PairList *tags = arglist.tags();
189  prepareToInvokeBuiltIn(func);
190  switch (arglist.size())
191  {
192  case 0:
193  return func->invokeNativeCall(this, env, tags);
194 /* This macro expands out to:
195  case 1:
196  return func->invokeNativeCall(this, env, tags, arglist.get(0));
197  case 2:
198  return func->invokeNativeCall(this, env, tags, arglist.get(0), arglist.get(1));
199  ...
200 */
201 #define ARGUMENT_LIST(Z, N, IGNORED) \
202  BOOST_PP_COMMA_IF(N) \
203  arglist.get(N)
204 #define CASE_STATEMENT(Z, N, IGNORED) \
205  case N: \
206  return func->invokeNativeCall(this, env, tags, BOOST_PP_REPEAT(N, ARGUMENT_LIST, 0));
207 
208  BOOST_PP_REPEAT_FROM_TO(1, 20, CASE_STATEMENT, 0);
209 
210 #undef ARGUMENT_LIST
211 #undef CASE_STATEMENT
212 
213  default:
214  Rf_errorcall(const_cast<Expression *>(this), _("too many arguments, sorry"));
215  }
216 }

223 RObject *Expression::evalArgsAndEvaluateNativeBuiltInCall(
224  const BuiltInFunction *func, Environment *env, const ArgList &arglist) const
225 {
226  const PairList *tags = arglist.tags();
227  int arity = arglist.size();
228  if (arity == 0)
229  {
230  prepareToInvokeBuiltIn(func);
231  return func->invokeNativeCall(this, env, tags);
232  }
233  auto arg_iterator = arglist.getArgs().begin();
234 /* This macro expands out to:
235  RObject* arg0 = evalIfNonNull(arg_iterator->car(), env);
236  if (arity == 1) {
237  prepareToInvokeBuiltIn(func);
238  return func->invokeNativeCall(this, env, tags, arg0);
239  }
240  ++arg_iterator;
241  RObject* arg1 = evalIfNonNull(arg_iterator->car(), env);
242  if (arity == 2) {
243  prepareToInvokeBuiltIn(func);
244  return func->invokeNativeCall(this, env, tags, arg0, arg1);
245  }
246  ...
247 */
248 #define ARGUMENT_LIST(Z, N, IGNORED) \
249  BOOST_PP_COMMA_IF(N) \
250  arg##N
251 #define CMD_SEQUENCE(Z, N, IGNORED) \
252  RObject *arg##N = evalIfNonNull(arg_iterator->car(), env); \
253  if (arity == N + 1) \
254  { \
255  prepareToInvokeBuiltIn(func); \
256  return func->invokeNativeCall(this, env, tags, \
257  BOOST_PP_REPEAT(BOOST_PP_ADD(N, 1), ARGUMENT_LIST, 0)); \
258  } \
259  ++arg_iterator;
260 
261  BOOST_PP_REPEAT_FROM_TO(0, 20, CMD_SEQUENCE, 0);
262 
263  Rf_errorcall(const_cast<Expression *>(this), _("too many arguments, sorry"));
264 }
265 #undef CMD_SEQUENCE
266 #undef ARGUMENT_LIST

300 RObject *Expression::evaluateDirectBuiltInCall(
301  const BuiltInFunction *func, Environment *env, const ArgList &arglist) const
302 {
303  bool args_need_evaluation = arglist.status() != ArgList::EVALUATED;
304  if (args_need_evaluation && arglist.has3Dots())
305  {
306  ArgList expanded_args(arglist);
307  expanded_args.evaluate(env);
308  return evaluateDirectBuiltInCall(func, env, expanded_args);
309  }
310 
311  int num_evaluated_args = arglist.size();
312  checkArityAndNamingRequirements(func, num_evaluated_args);
313 
314  auto calling_convention = func->getCallingConvention();
315  if (calling_convention == BuiltInFunction::CallingConvention::FixedNative || calling_convention == BuiltInFunction::CallingConvention::VarArgsNative)
316  {
317  if (args_need_evaluation)
318  {
319  return evalArgsAndEvaluateNativeBuiltInCall(func, env, arglist);
320  }
321  else
322  {
323  return evaluateNativeBuiltInCall(func, env, arglist);
324  }
325  }
326 
327  // Create an array on stack to write arguments to.
328  RObject **evaluated_arg_array = (RObject **)alloca(
329  num_evaluated_args * sizeof(RObject *));
330 
331  // Copy the arguments to the stack, evaluating if necessary.
332  arglist.evaluateToArray(env, num_evaluated_args, evaluated_arg_array);
333 
334  prepareToInvokeBuiltIn(func);
335  return func->invoke(this, env, evaluated_arg_array, num_evaluated_args,
336  arglist.tags());
337 }

387 RObject *Expression::invokeClosureImpl(const Closure *func,
388  Environment *calling_env,
389  const ArgList &parglist,
390  const Frame *method_bindings) const
391 {
392  // We can't modify parglist, as it's on the other side of a
393  // GCStackFrameboundary, so make a copy instead.
394  ArgList arglist(parglist);
395  arglist.wrapInPromises(calling_env, this);
396 
397  Environment *execution_env = func->createExecutionEnv(arglist);
398  matchArgsIntoEnvironment(func, calling_env, arglist, execution_env);
399 
400  // If this is a method call, merge in supplementary bindings and modify
401  // calling_env.
402  if (method_bindings)
403  {
404  importMethodBindings(method_bindings, execution_env->frame());
405  calling_env = getMethodCallingEnv();
406  }
407 
408  RObject *result;
409  {
410  // Evaluate the function.
411  ClosureContext context(this, calling_env, func, execution_env);
412  result = func->execute(execution_env);
413  }
414 
415  Environment::monitorLeaks(result);
416  execution_env->maybeDetachFrame();
417 
418  return result;
419 }
