652  template <typename UnaryPredicate>
653  RObject *findTestedValue(const Symbol *symbol, Environment *env,
654  UnaryPredicate pred, bool inherits)
655  {
656  using namespace std;
657 
658  do
659  {
660  Frame::Binding *bdg;
661  if (inherits && env == Environment::global())
662  {
663  // findBinding() handles the details of the cache correctly.
664  bdg = env->findBinding(symbol);
665  }
666  else
667  {
668  bdg = env->frame()->binding(symbol);
669  }
670  if (bdg)
671  {
672  pair<RObject *, bool> fpr = bdg->forcedValue2();
673  RObject *val = fpr.first;
674  if (pred(val))
675  {
676  // Invoke read monitor (if any) only if
677  // forcedValue() did not force a Promise. (If a
678  // Promise was forced, the read monitor will have
679  // been invoked anyway, and 'bdg' may now be
680  // junk.)
681  if (!fpr.second)
682  bdg->rawValue();
683  return val;
684  }
685  }
686  env = env->enclosingEnvironment();
687  } while (inherits && env);
688  return nullptr;
689  }
