37 /*
38  These structures are for C (and R function) top-level task handlers.
39  Such routines are called at the end of every (successful) top-level task
40  in the regular REPL.
41  */
42 
43 #include <Rinternals.h>
44 /*
45  The signature of the C routine that a callback must implement.
46  expr - the expression for the top-level task that was evaluated.
47  value - the result of the top-level task, i.e. evaluating expr.
48  succeeded - a logical value indicating whether the task completed propertly.
49  visible - a logical value indicating whether the result was printed to the R ``console''/stdout.
50  data - user-level data passed to the registration routine.
51  */
52 typedef Rboolean (*R_ToplevelCallback)(SEXP expr, SEXP value, Rboolean succeeded, Rboolean visible, void *);
53 
54 typedef struct _ToplevelCallback R_ToplevelCallbackEl;
55 /*
56  Linked list element for storing the top-level task callbacks.
57  */
58 struct _ToplevelCallback {
59  R_ToplevelCallback cb; /* the C routine to call. */
60  void *data; /* the user-level data to pass to the call to cb() */
61  void (*finalizer)(void *data); /* Called when the callback is removed. */
62 
63  char *name; /* a name by which to identify this element. */
64 
65  R_ToplevelCallbackEl *next; /* the next element in the linked list. */
66 };
