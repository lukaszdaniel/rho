49 Provenance::Set* Provenance::ancestors(const Set& roots)
50 {
51  Set open = roots;
52  Set* closed = new Set();
53  while (!open.empty()) {
54  const Provenance* n = *(open.begin());
55  std::pair<CommandChronicle::ParentVector::const_iterator,
56  CommandChronicle::ParentVector::const_iterator>
57  pr = n->parents();
58  for (CommandChronicle::ParentVector::const_iterator it = pr.first;
59  it != pr.second; ++it) {
60  const Provenance* prov = *it;
61  if (closed->count(prov) == 0)
62  open.insert(prov);
63  }
64  open.erase(n);
65  closed->insert(n);
66  }
67  return closed;
68 }

94 Provenance::Set* Provenance::descendants(const Set& roots)
95 {
96  Set open = roots;
97  Set* closed = new Set();
98  while (!open.empty()) {
99  const Provenance* n = *(open.begin());
100  const Set& c = n->children();
101  for (const Provenance* s : c) {
102  // If s isn't in closed set, put it in open
103  if (closed->find(s) == closed->end())
104  open.insert(s);
105  }
106  open.erase(n);
107  closed->insert(n);
108  }
109  return closed;
110 }
