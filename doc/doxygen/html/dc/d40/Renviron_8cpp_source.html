49 /* remove leading and trailing space */
50 static char *rmspace(char *s)
51 {
52  ssize_t i; // to be safe
53 
54  for (i = ssize_t(strlen(s)) - 1; i >= 0 && isspace(int(s[i])); i--)
55  s[i] = '\0';
56  for (i = 0; isspace(int(s[i])); i++);
57  return s + i;
58 }

64 static const char *subterm(char *s)
65 {
66  char *p, *q;
67 
68  if(strncmp(s, "${", 2)) return s;
69  if(s[strlen(s) - 1] != '}') return s;
70  /* remove leading ${ and final } */
71  s[strlen(s) - 1] = '\0';
72  s += 2;
73  s = rmspace(s);
74  if(!strlen(s)) return "";
75  p = Rf_strchr(s, '-');
76  if(p) {
77  q = p + 1; /* start of value */
78  if(p - s > 1 && *(p-1) == ':') *(p-1) = '\0'; else *p = '\0';
79  } else q = nullptr;
80  p = getenv(s);
81  if(p && strlen(p)) return p; /* variable was set and non-empty */
82  return q ? subterm(q) : "";
83 }

85 /* skip along until we find an unmatched right brace */
86 static char *findRbrace(char *s)
87 {
88  char *p = s, *pl, *pr;
89  int nl = 0, nr = 0;
90 
91  while(nr <= nl) {
92  pl = Rf_strchr(p, '{');
93  pr = Rf_strchr(p, '}');
94  if(!pr) return nullptr;
95  if(!pl || pr < pl) {
96  p = pr+1; nr++;
97  } else {
98  p = pl+1; nl++;
99  }
100  }
101  return pr;
102 }

106 static const char *findterm(const char *s)
107 {
108  char *p, *q;
109  const char* ss=s;
110  const char* r2;
111  static char ans[BUF_SIZE];
112 
113  if(!strlen(s)) return "";
114  ans[0] = '\0';
115  while(1) {
116  /* Look for ${...}, taking care to look for inner matches */
117  p = Rf_strchr(s, '$');
118  if(!p || p[1] != '{') break;
119  q = findRbrace(p+2);
120  if(!q) break;
121  /* copy over leading part */
122  size_t nans = strlen(ans);
123  strncat(ans, s, size_t (p - s)); ans[nans + p - s] = '\0';
124  vector<char> rv(size_t(q - p + 2));
125  char* r = &rv[0];
126  strncpy(r, p, size_t (q - p + 1));
127  r[q - p + 1] = '\0';
128  r2 = subterm(r);
129  if(strlen(ans) + strlen(r2) < BUF_SIZE) strcat(ans, r2); else return ss;
130  /* now repeat on the tail */
131  s = q+1;
132  }
133  if(strlen(ans) + strlen(s) < BUF_SIZE) strcat(ans, s); else return ss;
134  return ans;
135 }

137 static void Putenv(char *a, const char *b)
138 {
139  char *buf, *value, *q, quote='\0';
140  const char *p;
141  int inquote = 0;
142 
143 #ifdef HAVE_SETENV
144  buf = static_cast<char *>(malloc((strlen(b) + 1) * sizeof(char)));
145  if(!buf) R_Suicide("allocation failure in reading Renviron");
146  value = buf;
147 #else
148  buf = (char *) malloc((strlen(a) + strlen(b) + 2) * sizeof(char));
149  if(!buf) R_Suicide("allocation failure in reading Renviron");
150  strcpy(buf, a); strcat(buf, "=");
151  value = buf+strlen(buf);
152 #endif
153 
154  /* now process the value */
155  for(p = b, q = value; *p; p++) {
156  /* remove quotes around sections, preserve \ inside quotes */
157  if(!inquote && (*p == '"' || *p == '\'')) {
158  inquote = 1;
159  quote = *p;
160  continue;
161  }
162  if(inquote && *p == quote && *(p-1) != '\\') {
163  inquote = 0;
164  continue;
165  }
166  if(!inquote && *p == '\\') {
167  if(*(p+1) == '\n') p++;
168  else if(*(p+1) == '\\') *q++ = *p;
169  continue;
170  }
171  if(inquote && *p == '\\' && *(p+1) == quote) continue;
172  *q++ = *p;
173  }
174  *q = '\0';
175 #ifdef HAVE_SETENV
176  if(setenv(a, buf, 1))
177  Rf_warningcall(nullptr,
178  _("problem in setting variable '%s' in Renviron"), a);
179  free(buf);
180 #elif defined(HAVE_PUTENV)
181  if(putenv(buf))
182  Rf_warningcall(nullptr,
183  _("problem in setting variable '%s' in Renviron"), a);
184  /* no free here: storage remains in use */
185 #else
186  /* pretty pointless, and was not tested prior to 2.3.0 */
187  free(buf);
188 #endif
189 }

194 static int process_Renviron(const char *filename)
195 {
196  FILE *fp;
197  char *s, *p, sm[BUF_SIZE], *lhs, msg[MSG_SIZE+50];
198  const char *rhs;
199  int errs = 0;
200 
201  if (!filename || !(fp = R_fopen(filename, "r"))) return 0;
202  snprintf(msg, MSG_SIZE+50,
203  "\n File %s contains invalid line(s)", filename);
204 
205  while(fgets(sm, BUF_SIZE, fp)) {
206  sm[BUF_SIZE-1] = '\0';
207  s = rmspace(sm);
208  if(strlen(s) == 0 || s[0] == '#') continue;
209  if(!(p = Rf_strchr(s, '='))) {
210  errs++;
211  if(strlen(msg) < MSG_SIZE) {
212  strcat(msg, "\n "); strcat(msg, s);
213  }
214  continue;
215  }
216  *p = '\0';
217  lhs = rmspace(s);
218  rhs = findterm(rmspace(p+1));
219  /* set lhs = rhs */
220  if(strlen(lhs) && strlen(rhs)) Putenv(lhs, rhs);
221  }
222  fclose(fp);
223  if (errs) {
224  strcat(msg, "\n They were ignored\n");
225  R_ShowMessage(msg);
226  }
227  return 1;
228 }

263 {

273  } else {

322 #endif
323  process_Renviron(s);
324 }
