48 R_xlen_t asVecSize(SEXP x)
49 {
50  if (Rf_isVectorAtomic(x) && LENGTH(x) >= 1) {
51  switch (TYPEOF(x)) {
52  case INTSXP:
53  {
54  int res = INTEGER(x)[0];
55  if(res == NA_INTEGER) Rf_error(_("vector size cannot be NA"));
56  return R_xlen_t(res);
57  }
58  case REALSXP:
59  {
60  double d = REAL(x)[0];
61  if(ISNAN(d)) Rf_error(_("vector size cannot be NA/NaN"));
62  if(!R_FINITE(d)) Rf_error(_("vector size cannot be infinite"));
63  if(d > R_XLEN_T_MAX) Rf_error(_("vector size specified is too large"));
64  return R_xlen_t(d);
65  }
66  case STRSXP:
67  {
68  double d = Rf_asReal(x);
69  if(ISNAN(d)) Rf_error(_("vector size cannot be NA/NaN"));
70  if(!R_FINITE(d)) Rf_error(_("vector size cannot be infinite"));
71  if(d > R_XLEN_T_MAX) Rf_error(_("vector size specified is too large"));
72  return R_xlen_t(d);
73  }
74  default:
75  ; // -Wswitch
76  }
77  }
78  return -999; /* which gives error in the caller */
79 }

101 SEXP attribute_hidden do_makelazy(/*const*/ Expression* call, const BuiltInFunction* op, RObject* vars_, RObject* vals_, RObject* expr_, RObject* db_, RObject* envir_)
102 {
103  SEXP names, values, val, expr, eenv, aenv, expr0;
104  R_xlen_t i;
105 
106  names = vars_;
107  if (!Rf_isString(names))
108  Rf_error(_("invalid first argument"));
109  values = vals_;
110  expr = expr_;
111  eenv = db_;
112  if (!Rf_isEnvironment(eenv)) Rf_error(_("invalid '%s' argument"), "eval.env");
113  aenv = envir_;
114  if (!Rf_isEnvironment(aenv)) Rf_error(_("invalid '%s' argument"), "assign.env");
115 
116  for(i = 0; i < XLENGTH(names); i++) {
117  SEXP name = Rf_installTrChar(STRING_ELT(names, i));
118  PROTECT(val = Rf_eval(VECTOR_ELT(values, i), eenv));
119  PROTECT(expr0 = Rf_duplicate(expr));
120  SETCAR(CDR(expr0), val);
121  Rf_defineVar(name, Rf_mkPROMISE(expr0, eenv), aenv);
122  UNPROTECT(2);
123  }
124  return R_NilValue;
125 }

128 SEXP attribute_hidden do_onexit(SEXP call, SEXP op, SEXP args, SEXP rho)
129 {
130  ClosureContext *ctxt;
131  SEXP code, oldcode, tmp, add;
132  int addit = 0;
133 
134  checkArity(op, args);
135  static GCRoot<ArgMatcher> matcher = new ArgMatcher({ "expr", "add" });
136  ArgList arglist(SEXP_downcast<PairList*>(args), ArgList::RAW);
137  matcher->match(arglist, { &code, &add });
138  if (code == R_MissingArg)
139  code = R_NilValue;
140  if (add != R_MissingArg) {
141  addit = Rf_asLogical(Rf_eval(add, rho));
142  if (addit == NA_INTEGER)
143  Rf_errorcall(call, _("invalid '%s' argument"), "add");
144  }
145 
146  ctxt = R_GlobalContext();
147  /* Search for the context to which the on.exit action is to be
148  attached. Lexical scoping is implemented by searching for the
149  first closure call context with an environment matching the
150  expression evaluation environment. */
151  while (ctxt && ctxt->workingEnvironment() != rho)
152  ctxt = ClosureContext::innermost(ctxt->nextOut());
153  if (ctxt)
154  {
155  if (addit && (oldcode = ctxt->onExit()) != R_NilValue ) {
156  if ( CAR(oldcode) != R_BraceSymbol )
157  {
158  PROTECT(tmp = new Expression(R_BraceSymbol,
159  { oldcode, code }));
160  ctxt->setOnExit(tmp);
161  UNPROTECT(1);
162  }
163  else
164  {
165  PROTECT(tmp = Rf_allocList(1));
166  SETCAR(tmp, code);
167  ctxt->setOnExit(Rf_listAppend(Rf_duplicate(oldcode),tmp));
168  UNPROTECT(1);
169  }
170  }
171  else
172  ctxt->setOnExit(code);
173  }
174  UNPROTECT(1);
175  return R_NilValue;
176 }

178 SEXP attribute_hidden do_args(SEXP call, SEXP op, SEXP args, SEXP rho)
179 {
180  SEXP s;
181 
182  if (TYPEOF(CAR(args)) == STRSXP && Rf_length(CAR(args))==1) {
183  PROTECT(s = Rf_installTrChar(STRING_ELT(CAR(args), 0)));
184  SETCAR(args, Rf_findFun(s, rho));
185  UNPROTECT(1);
186  }
187 
188  if (TYPEOF(CAR(args)) == CLOSXP) {
189  return Rf_mkCLOSXP(FORMALS(CAR(args)), nullptr, R_GlobalEnv);
190  }
191 
192  if (TYPEOF(CAR(args)) == BUILTINSXP || TYPEOF(CAR(args)) == SPECIALSXP) {
193  RHOCONST char *nm = PRIMNAME(CAR(args));
194  SEXP env, s2;
195  PROTECT_INDEX xp;
196 
197  PROTECT_WITH_INDEX(env = Rf_findVarInFrame3(R_BaseEnv,
198  Rf_install(".ArgsEnv"), TRUE),
199  &xp);
200 
201  if (TYPEOF(env) == PROMSXP) REPROTECT(env = Rf_eval(env, R_BaseEnv), xp);
202  PROTECT(s2 = Rf_findVarInFrame3(env, Rf_install(nm), TRUE));
203  if(s2 != R_UnboundValue) {
204  s = Rf_duplicate(s2);
205  SET_CLOENV(s, R_GlobalEnv);
206  UNPROTECT(2);
207  return s;
208  }
209  UNPROTECT(1); /* s2 */
210  REPROTECT(env = Rf_findVarInFrame3(R_BaseEnv, Rf_install(".GenericArgsEnv"),
211  TRUE), xp);
212  if (TYPEOF(env) == PROMSXP) REPROTECT(env = Rf_eval(env, R_BaseEnv), xp);
213  PROTECT(s2 = Rf_findVarInFrame3(env, Rf_install(nm), TRUE));
214  if(s2 != R_UnboundValue) {
215  s = Rf_mkCLOSXP(FORMALS(s2), nullptr, R_GlobalEnv);
216  UNPROTECT(2);
217  return s;
218  }
219  UNPROTECT(2);
220  }
221  return R_NilValue;
222 }

319 static Rboolean R_IsImportsEnv(SEXP env)
320 {
321  if (Rf_isNull(env) || !Rf_isEnvironment(env))
322  return FALSE;
323  if (ENCLOS(env) != R_BaseNamespace)
324  return FALSE;
325  SEXP name = Rf_getAttrib(env, R_NameSymbol);
326  if (!Rf_isString(name) || Rf_length(name) != 1)
327  return FALSE;
328 
329  const char *imports_prefix = "imports:";
330  const char *name_string = R_CHAR(STRING_ELT(name, 0));
331 
332  return streqln(name_string, imports_prefix, strlen(imports_prefix));

336 {

355 SEXP attribute_hidden do_envirName(/*const*/ Expression* call, const BuiltInFunction* op, RObject* env_)
356 {
357  SEXP ans=Rf_mkString(""), res;
358 
359  if (!env_) {
360  return ans;
361  }
362 
363  Environment* env = simple_as_environment(env_);
364  if (env) {
365  if (env == R_GlobalEnv) ans = Rf_mkString("R_GlobalEnv");
366  else if (env == R_BaseEnv) ans = Rf_mkString("base");
367  else if (env == R_EmptyEnv) ans = Rf_mkString("R_EmptyEnv");
368  else if (R_IsPackageEnv(env))
369  ans = Rf_ScalarString(STRING_ELT(R_PackageEnvName(env), 0));
370  else if (R_IsNamespaceEnv(env))
371  ans = Rf_ScalarString(STRING_ELT(R_NamespaceEnvSpec(env), 0));
372  else if (!Rf_isNull(res = Rf_getAttrib(env, R_NameSymbol))) ans = res;
373  }
374  return ans;

380 /* Uses R_alloc but called by a .Internal. Result may be R_alloc-ed */
381 static const char *trChar(SEXP x)
382 {
383  size_t n = strlen(R_CHAR(x));
384  cetype_t ienc = Rf_getCharCE(x);
385 
386  if (ienc == CE_BYTES) {
387  const char *p = R_CHAR(x), *q;
388  char *pp = R_alloc(4*n+1, 1), *qq = pp, buf[5];
389  for (q = p; *q; q++) {
390  unsigned char k = static_cast<unsigned char>(*q);
391  if (k >= 0x20 && k < 0x80) {
392  *qq++ = *q;
393  } else {
394  snprintf(buf, 5, "\\x%02x", k);
395  for(int j = 0; j < 4; j++) *qq++ = buf[j];
396  }
397  }
398  *qq = '\0';
399  return pp;
400  } else {
401 #ifdef Win32
402  static char buf[106];
403  char *p;
404  /* Long strings will be rare, and few per cat() call so we
405  can afford to be profligate here: translateChar is */
406  if (n < 100) p = buf; else p = R_alloc(n+7, 1);
407  if (WinUTF8out && ienc == CE_UTF8) {
408  strcpy(p, UTF8in); strcat(p, R_CHAR(x)); strcat(p, UTF8out);
409  return p;
410  } else
411 #endif
412  return Rf_translateChar(x);
413  }
414 }

467 SEXP attribute_hidden do_cat(/*const*/ Expression* call, const BuiltInFunction* op, RObject* dots_, RObject* file_, RObject* sep_, RObject* fill_, RObject* labels_, RObject* append_)
468 {
469  cat_info ci;
470  SEXP objs, file, fill, sepr, labs, s;
471  int ifile;
472  Rconnection con;
473  int append;
474  int i, iobj, n, nobjs, pwidth, width, sepw, lablen, ntot, nlsep, nlines;
475  char buf[512];
476  const char *p = "";
477 
478  /* Use standard printing defaults */
479  Rf_PrintDefaults();
480 
481  objs = dots_;
482 
483  file = file_;
484  ifile = Rf_asInteger(file);
485  con = getConnection(ifile);
486  if(!con->canwrite) /* if it is not open, we may not know yet */
487  Rf_error(_("cannot write to this connection"));
488 
489  sepr = sep_;
490  if (!Rf_isString(sepr))
491  Rf_error(_("invalid '%s' specification"), "sep");
492  nlsep = 0;
493  for (i = 0; i < LENGTH(sepr); i++)
494  if (strstr(R_CHAR(STRING_ELT(sepr, i)), "\n")) nlsep = 1; /* ASCII */
495 
496  fill = fill_;
497  if ((!Rf_isNumeric(fill) && !Rf_isLogical(fill)) || (Rf_length(fill) != 1))
498  Rf_error(_("invalid '%s' argument"), "fill");
499  if (Rf_isLogical(fill)) {
500  if (Rf_asLogical(fill) == 1)
501  pwidth = R_print.width;
502  else
503  pwidth = INT_MAX;
504  }
505  else pwidth = Rf_asInteger(fill);
506  if(pwidth <= 0) {
507  Rf_warning(_("non-positive 'fill' argument will be ignored"));
508  pwidth = INT_MAX;
509  }
510 
511  labs = labels_;
512  if (!Rf_isString(labs) && labs != R_NilValue)
513  Rf_error(_("invalid '%s' argument"), "labels");
514  lablen = Rf_length(labs);
515 
516  append = Rf_asLogical(append_);
517  if (append == NA_LOGICAL)
518  Rf_error(_("invalid '%s' specification"), "append");
519 
520  ci.wasopen = con->isopen;
521 
522  ci.changedcon = switch_stdout(ifile, 0);
523  /* will open new connection if required, and check for writeable */
524 #ifdef Win32
525  /* do this after re-sinking output */
526  WinCheckUTF8();
527 #endif
528 
529  ci.con = con;
530 
531  /* use try-catch to close the connection if there is an error */
532  try {
533  nobjs = Rf_length(objs);
534  width = 0;
535  ntot = 0;
536  nlines = 0;
537  for (iobj = 0; iobj < nobjs; iobj++) {
538  s = VECTOR_ELT(objs, iobj);
539  if (iobj != 0 && !Rf_isNull(s))
540  cat_printsep(sepr, ntot++);
541  n = Rf_length(s);
542  /* 0-length objects are ignored */
543  if (n > 0) {
544  if (labs != R_NilValue && (iobj == 0)
545  && (Rf_asInteger(fill) > 0)) {
546  Rprintf("%s ", trChar(STRING_ELT(labs, nlines % lablen)));
547  /* FIXME -- Rstrlen allows for double-width chars */
548  width += Rstrlen(STRING_ELT(labs, nlines % lablen), 0) + 1;
549  nlines++;
550  }
551  if (Rf_isString(s))
552  p = trChar(STRING_ELT(s, 0));
553  else if (Rf_isSymbol(s)) /* length 1 */
554  p = R_CHAR(PRINTNAME(s));
555  else if (Rf_isVectorAtomic(s)) {
556  /* Not a string, as that is covered above.
557  Thus the maximum size is about 60.
558  The copy is needed as cat_newline might reuse the buffer.
559  Use strncpy is in case these assumptions change.
560  */
561  p = EncodeElement0(s, 0, 0, OutDec);
562  strncpy(buf, p, 512); buf[511] = '\0';
563  p = buf;
564  }
565 #ifdef fixed_cat
566  else if (Rf_isVectorList(s)) {
567  /* FIXME: call EncodeElement() for every element of s.
568 
569  Real Problem: `s' can be large;
570  should do line breaking etc.. (buf is of limited size)
571  */
572  }
573 #endif
574  else
575  Rf_error(_("argument %d (type '%s') cannot be handled by 'cat'"),
576  1+iobj, Rf_type2char(TYPEOF(s)));
577  /* FIXME : cat(...) should handle ANYTHING */
578  size_t w = strlen(p);
579  cat_sepwidth(sepr, &sepw, ntot);
580  if ((iobj > 0) && (width + w + sepw > RHOCONSTRUCT(size_t, pwidth))) {
581  cat_newline(labs, &width, lablen, nlines);
582  nlines++;
583  }
584  for (i = 0; i < n; i++, ntot++) {
585  Rprintf("%s", p);
586  width += int(w + sepw);
587  if (i < (n - 1)) {
588  cat_printsep(sepr, ntot);
589  if (Rf_isString(s))
590  p = trChar(STRING_ELT(s, i+1));
591  else {
592  p = EncodeElement0(s, i+1, 0, OutDec);
593  strncpy(buf, p, 512); buf[511] = '\0';
594  p = buf;
595  }
596  w = int(strlen(p));
597  cat_sepwidth(sepr, &sepw, ntot);
598  /* This is inconsistent with the version above.
599  As from R 2.3.0, fill <= 0 is ignored. */
600  if ((width + w + sepw > RHOCONSTRUCT(size_t, pwidth)) && pwidth) {
601  cat_newline(labs, &width, lablen, nlines);
602  nlines++;
603  }
604  } else ntot--; /* we don't print sep after last, so don't advance */
605  }
606  }
607  }
608  if ((pwidth != INT_MAX) || nlsep)
609  Rprintf("\n");
610  }
611  catch (...) {
612  cat_cleanup(&ci);
613  throw;
614  }
615

938 SEXP attribute_hidden do_switch(SEXP call, SEXP op, SEXP args, SEXP rho)
939 {
940  int argval, nargs = Rf_length(args);
941  SEXP x, y, z, w, ans, dflt = nullptr;
942 
943  PROTECT(x = Rf_eval(CAR(args), rho));
944  if (!Rf_isVector(x) || Rf_length(x) != 1)
945  Rf_errorcall(call, _("EXPR must be a length 1 vector"));
946  if (Rf_isFactor(x))
947  Rf_warningcall(call,
948  _("EXPR is a \"factor\", treated as integer.\n Consider using '%s' instead."),
949  "switch(as.character( * ), ...)");
950  if (nargs > 1) {
951  /* There is a complication: if called from lapply
952  there may be a ... argument */
953  PROTECT(w = expandDots(CDR(args), rho));
954  if (Rf_isString(x)) {
955  for (y = w; y != R_NilValue; y = CDR(y)) {
956  if (TAG(y) != R_NilValue) {
957  if (Rf_pmatch(STRING_ELT(x, 0), TAG(y), RHO_TRUE /* exact */)) {
958  /* Find the next non-missing argument.
959  (If there is none, return NULL.) */
960  while (CAR(y) == R_MissingArg) {
961  y = CDR(y);
962  if (y == R_NilValue) break;
963  if (TAG(y) == R_NilValue) dflt = setDflt(y, dflt);
964  }
965  if (y == R_NilValue) {
966  R_Visible = FALSE;
967  UNPROTECT(2);
968  return R_NilValue;
969  }
970  /* Check for multiple defaults following y. This loop
971  is not necessary to determine the value of the
972  switch(), but it should be fast and will detect
973  typos. */
974  for (z = CDR(y); z != R_NilValue; z = CDR(z))
975  if (TAG(z) == R_NilValue) dflt = setDflt(z, dflt);
976 
977  ans = Rf_eval(CAR(y), rho);
978  UNPROTECT(2);
979  return ans;
980  }
981  } else
982  dflt = setDflt(y, dflt);
983  }
984  if (dflt) {
985  ans = Rf_eval(dflt, rho);
986  UNPROTECT(2);
987  return ans;
988  }
989  /* fall through to error */
990  } else { /* Treat as numeric */
991  argval = Rf_asInteger(x);
992  if (argval != NA_INTEGER && argval >= 1 && argval <= Rf_length(w)) {
993  SEXP alt = CAR(Rf_nthcdr(w, argval - 1));
994  if (alt == R_MissingArg)
995  Rf_error("empty alternative in numeric switch");
996  ans = Rf_eval(alt, rho);
997  UNPROTECT(2);
998  return ans;
999  }
1000  /* fall through to error */
1001  }
1002  UNPROTECT(1); /* w */
1003  } else
1004  Rf_warningcall(call, _("'switch' with no alternatives"));
1005  /* an error */
1006  UNPROTECT(1); /* x */
1007  R_Visible = FALSE;
1008  return R_NilValue;
1009 }
