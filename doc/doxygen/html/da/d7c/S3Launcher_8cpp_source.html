38 void S3Launcher::addMethodBindings(Frame *frame) const
39 {
40  // .Class:
41  if (m_index == 0)
42  frame->bind(DotClassSymbol, m_classes);
43  else
44  {
45  GCStackRoot<StringVector>
46  dotclass(StringVector::create(m_classes->size() - m_index));
47  for (unsigned int j = 0; j < dotclass->size(); ++j)
48  (*dotclass)[j] = (*m_classes)[j + m_index];
49  dotclass->setAttribute(PreviousSymbol, m_classes);
50  frame->bind(DotClassSymbol, dotclass);
51  }
52 
53  // .Method:
54  {
55  String *method_name = const_cast<String *>(m_symbol->name());
56  frame->bind(DotMethodSymbol, StringVector::createScalar(method_name));
57  }
58 
59  // .Group:
60  if (m_using_group)
61  frame->bind(DotGroupSymbol, asStringVector(m_group));
62 
63  // Others:
64  frame->bind(DotGenericSymbol, asStringVector(m_generic));
65  frame->bind(DotGenericCallEnvSymbol, m_call_env);
66  frame->bind(DotGenericDefEnvSymbol, m_table_env);
67 }

79 std::pair<FunctionBase *, bool>
80 S3Launcher::findMethod(const Symbol *symbol, Environment *call_env,
81  Environment *table_env)
82 {
83  FunctionBase *fun = findFunction(symbol, call_env);
84  if (fun)
85  return make_pair(fun, true);
86  if (!table_env)
87  return pair<FunctionBase *, bool>(nullptr, false);
88  Environment *table = nullptr;
89  // Look for S3 methods table:
90  {
91  Frame::Binding *tblbdg = table_env->frame()->binding(S3MethodsTableSymbol);
92  if (tblbdg)
93  {
94  RObject *tblbdgval = tblbdg->forcedValue();
95  if (tblbdgval && tblbdgval->sexptype() == ENVSXP)
96  table = SEXP_downcast<Environment *>(tblbdgval);
97  }
98  }
99  // Look up method in table:
100  if (table)
101  {
102  Frame::Binding *symbdg = table->frame()->binding(symbol);
103  if (symbdg)
104  {
105  RObject *symbdgval = symbdg->forcedValue();
106  // Assume that the result is a FunctionBase:
107  return make_pair(SEXP_downcast<FunctionBase *>(symbdgval), false);
108  }
109  }
110  return pair<FunctionBase *, bool>(nullptr, false);
111 }
