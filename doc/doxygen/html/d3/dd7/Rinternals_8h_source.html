426 #ifdef DISABLE_PROTECT_MACROS
427  /* Danger! You almost certainly don't need to use DISABLE_PROTECT_MACROS for
428  * your code.
429  *
430  * In most cases, rho doesn't require nodes to be explicitly PROTECTed, as it
431  * automatically protects local variables. In that case, any function that
432  * only stores SEXPs as local variables, inside other R SEXPs or protected
433  * by calls to R_Protect/R_PreserveObject() doesn't need to use PROTECT() and
434  * can be compiled with DISABLE_PROTECT_MACROS defined for a modest
435  * performance gain on code that handles a lot of small objects.
436  *
437  * Code that holds SEXPs in static variables, global variables, thread-local
438  * storage or heap memory allocated with malloc or new cannot safely use
439  * this macro, as the garbage collector doesn't scan those locations.
440  */
