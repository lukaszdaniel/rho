60 /*
61  * ed, vi etc have 3 parameters. the data, a file and an editor
62  *
63  * If `file' is specified then the given file is used (and not removed on
64  * exit). If `file' is not specified then a temporary file is used; since
65  * only one temporary file is used for an entire session previous
66  * editing is lost. That file is removed at the end of the R session.
67  *
68  * If `data' is specified then it is passed out to be edited; if `data' is not
69  * specified then either `file' (if specified) or the temporary file is used
70  * (thus errors can be re-edited by calling edit a second time with no
71  * arguments).
72  *
73  * If the editor is specified then the specified editor is invoked if
74  * possible and an error message reported otherwise
75  */

94 SEXP do_edit(SEXP call, SEXP op, SEXP args, SEXP rho)
95 {
96  int i, rc;
97  ParseStatus status;
98  SEXP x, fn, envir, ed, src, srcfile, Rfn;
99  char *filename, *editcmd;
100  const char *cmd;
101  const void *vmaxsave;
102  FILE *fp;
103 #ifdef Win32
104  SEXP ti;
105  char *title;
106 #endif
107 
108  vmaxsave = vmaxget();
109 
110  x = CAR(args); args = CDR(args);
111  if (TYPEOF(x) == CLOSXP) envir = CLOENV(x);
112  else envir = nullptr;
113  PROTECT(envir);
114 
115  fn = CAR(args); args = CDR(args);
116  if (!Rf_isString(fn))
117  Rf_error(_("invalid argument to edit()"));
118 
119  if (LENGTH(STRING_ELT(fn, 0)) > 0) {
120  const char *ss = Rf_translateChar(STRING_ELT(fn, 0));
121  filename = R_alloc(strlen(ss), sizeof(char));
122  strcpy(filename, ss);
123  }
124  else filename = DefaultFileName;
125 
126  if (x != nullptr) {
127  if((fp=R_fopen(R_ExpandFileName(filename), "w")) == nullptr)
128  Rf_errorcall(call, _("unable to open file"));
129  if (LENGTH(STRING_ELT(fn, 0)) == 0) EdFileUsed++;
130  PROTECT(src = Rf_deparse1(x, FALSE, FORSOURCING)); /* deparse for sourcing, not for display */
131  for (i = 0; i < LENGTH(src); i++)
132  fprintf(fp, "%s\n", Rf_translateChar(STRING_ELT(src, i)));
133  UNPROTECT(1); /* src */
134  fclose(fp);
135  }
136 #ifdef Win32
137  ti = CAR(args);
138 #endif
139  args = CDR(args);
140  ed = CAR(args);
141  if (!Rf_isString(ed)) Rf_errorcall(call, _("argument 'editor' type not valid"));
142  cmd = Rf_translateChar(STRING_ELT(ed, 0));
143  if (strlen(cmd) == 0) Rf_errorcall(call, _("argument 'editor' is not set"));
144  editcmd = R_alloc(strlen(cmd) + strlen(filename) + 6, sizeof(char));
145 #ifdef Win32
146  if (streql(cmd,"internal")) {
147  if (!Rf_isString(ti))
148  Rf_error(_("'title' must be a string"));
149  if (LENGTH(STRING_ELT(ti, 0)) > 0) {
150  title = R_alloc(strlen(R_CHAR(STRING_ELT(ti, 0)))+1, sizeof(char));
151  strcpy(title, R_CHAR(STRING_ELT(ti, 0)));
152  } else {
153  title = R_alloc(strlen(filename)+1, sizeof(char));
154  strcpy(title, filename);
155  }
156  Rgui_Edit(filename, CE_NATIVE, title, 1);
157  }
158  else {
159  /* Quote path if necessary */
160  if(cmd[0] != '"' && Rf_strchr(cmd, ' '))
161  sprintf(editcmd, "\"%s\" \"%s\"", cmd, filename);
162  else
163  sprintf(editcmd, "%s \"%s\"", cmd, filename);
164  rc = runcmd(editcmd, CE_NATIVE, 1, 1, NULL, NULL, NULL);
165  if (rc == NOLAUNCH)
166  Rf_errorcall(call, _("unable to run editor '%s'"), cmd);
167  if (rc != 0)
168  Rf_warningcall(call, _("editor ran but returned error status"));
169  }
170 #else
171  if (ptr_R_EditFile)
172  rc = ptr_R_EditFile(filename);
173  else {
174  sprintf(editcmd, "'%s' '%s'", cmd, filename); // allow for spaces
175  rc = R_system(editcmd);
176  }
177  if (rc != 0)
178  Rf_errorcall(call, _("problem with running editor %s"), cmd);
179 #endif
180 
181  if (Rf_asLogical(Rf_GetOption1(Rf_install("keep.source")))) {
182  PROTECT(Rfn = Rf_findFun(Rf_install("readLines"), R_BaseEnv));
183  PROTECT(src = Rf_lang2(Rfn, Rf_ScalarString(Rf_mkChar(R_ExpandFileName(filename)))));
184  PROTECT(src = Rf_eval(src, R_BaseEnv));
185  PROTECT(Rfn = Rf_findFun(Rf_install("srcfilecopy"), R_BaseEnv));
186  PROTECT(srcfile = Rf_lang3(Rfn, Rf_ScalarString(Rf_mkChar("<tmp>")), src));
187  srcfile = Rf_eval(srcfile, R_BaseEnv);
188  UNPROTECT(5);
189  } else
190  srcfile = nullptr;
191  PROTECT(srcfile);
192 
193  /* <FIXME> setup a context to close the file, and parse and eval
194  line by line */
195  if((fp = R_fopen(R_ExpandFileName(filename), "r")) == nullptr)
196  Rf_errorcall(call, _("unable to open file to read"));
197 
198  x = PROTECT(R_ParseFile(fp, -1, &status, srcfile));
199  fclose(fp);
200 
201  if (status != PARSE_OK)
202  Rf_errorcall(call,
203  _("%s occurred on line %d\n use a command like\n x <- edit()\n to recover"), R_ParseErrorMsg, R_ParseError);
204  R_ResetConsole();
205  { /* can't just Rf_eval(x) here */
206  int j, n;
207  SEXP tmp = nullptr;
208 
209  n = LENGTH(x);
210  for (j = 0 ; j < n ; j++)
211  tmp = Rf_eval(XVECTOR_ELT(x, j), R_GlobalEnv);
212  x = tmp;
213  }
214  if (TYPEOF(x) == CLOSXP && envir != nullptr)
215  SET_CLOENV(x, envir);
216  UNPROTECT(3);
217  vmaxset(vmaxsave);
218  return x;
219 }
