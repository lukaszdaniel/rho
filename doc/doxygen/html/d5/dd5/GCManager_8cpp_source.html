80 void GCManager::gc(bool force_full_collection)
81 {
82  if (s_inhibitor_count > 0) {
83  s_gc_pending = true;
84  return;
85  }
86  s_gc_pending = false;
87 
88  // Prevent recursion:
89  if (s_gc_is_running)
90  return;
91  s_gc_is_running = true;
92  ++gc_count;
93 
94  s_max_bytes = std::max(s_max_bytes, MemoryBank::bytesAllocated());
95  s_max_nodes = std::max(s_max_nodes, GCNode::numNodes());
96 
97  if (s_pre_gc) (*s_pre_gc)();
98 
99  GCNode::gc(false);
100 
101  if (force_full_collection || MemoryBank::bytesAllocated() > s_threshold) {
102  GCNode::gc(true);
103  s_threshold = std::max(size_t(0.8*double(s_threshold)),
104  std::max(s_min_threshold,
105  size_t(1.2*MemoryBank::bytesAllocated())));
106  }
107 
108  s_gclite_threshold = std::max(size_t(0.8*double(s_gclite_threshold)),
109  std::max(s_min_threshold,
110  size_t(1.2*MemoryBank::bytesAllocated())));
111 
112  if (s_post_gc) (*s_post_gc)();
113 
114  s_gc_is_running = false;
115 }
