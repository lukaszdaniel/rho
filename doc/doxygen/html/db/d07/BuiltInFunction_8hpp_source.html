357  template<typename... Args>
358  RObject* invokeNativeCall(const Expression* call,
359  Environment* env,
360  const PairList* tags,
361  Args... args) const {
362  assert(getCallingConvention() == CallingConvention::FixedNative
363  || getCallingConvention() == CallingConvention::VarArgsNative);
364  assert(sizeof...(Args) == arity() || arity() == -1);
365  if (functionNeedsInternalDispatch()
366  && argsHaveClassToDispatchOn(args...))
367  {
368  std::initializer_list<RObject*> args_array = { args... };
369  auto dispatched = RealInternalDispatch(
370  call, args_array.size(), args_array.begin(), tags, env);
371  if (dispatched.first)
372  return dispatched.second;
373  }
374  return invokeNativeCallWithoutDispatch(call, args...);
375  }

377  template<typename... Args>
378  RObject* invokeNativeCallWithoutDispatch(const Expression* call,
379  Args... args) const {
380  assert(sizeof...(Args) == arity() || arity() == -1);
381  if (getCallingConvention() == CallingConvention::FixedNative) {
382  auto fn = reinterpret_cast<
383  RObject*(*)(Expression*, const BuiltInFunction*,
384  Args...)>(m_function.fixed_native);
385  return (*fn)(const_cast<Expression*>(call), this, args...);
386  } else {
387  assert(getCallingConvention() == CallingConvention::VarArgsNative);
388  return (m_function.varargs_native)(
389  const_cast<Expression*>(call),
390  this, sizeof...(Args), args...);
391  }

434  /* @brief Attempt to dispatch this call to a method.
435  *
436  * @param call The call being dispatched.
437  * @param env The environment to call the method from.
438  * @param evaluated_args The arguments to pass to the method.
439  *
440  * @return The first value is whether or not the dispatch succeeded.

445  std::pair<bool, RObject*>
446  InternalDispatch(const Expression* call,
447  Environment* env,
448  ArgList&& evaluated_args) const
449  {
450  assert(evaluated_args.status() == ArgList::EVALUATED);
451 
452  // NB: functionNeedsInternalDispatch() might not be true, since
453  // we could be doing dispatch for a special etc.
454  assert(m_dispatch_type != DispatchType::NONE);
455  if (argsHaveClassToDispatchOn(evaluated_args)) {
456  return RealInternalDispatch(call, env,
457  std::move(evaluated_args));
458  }
459  return std::make_pair(false, nullptr);
460  }

601  bool argsHaveClassToDispatchOn(RObject* const* evaluated_args,
602  int num_args) const
603  {
604  switch(num_args) {
605  case 0:
606  return false;
607  case 1:

609  default:
610  return argsHaveClassToDispatchOn(evaluated_args[0],
611  evaluated_args[1]);
612  }

615  bool argsHaveClassToDispatchOn(const ArgList& evaluated_args) const
616  {
617  switch(evaluated_args.size()) {
618  case 0:
619  return false;
620  case 1:
621  return argsHaveClassToDispatchOn(evaluated_args.get(0));
622  default:
623  return argsHaveClassToDispatchOn(evaluated_args.get(0),
624  evaluated_args.get(1));
625  }
626  }

628  // Internal dispatch is used a lot, but there most of the time
629  // no dispatch is needed because no objects are involved.
630  // This quickly detects those cases.
631  bool argsHaveClassToDispatchOn(const RObject* arg1 = nullptr,
632  const RObject* arg2 = nullptr,
633  ...) const
634  {
635  assert(m_dispatch_type != DispatchType::NONE);
636  if (arg1 && arg1->hasClass())
637  return true;
638  // 'Ops' dispatch on the second argument as well.
639  if (m_dispatch_type == DispatchType::GROUP_OPS
640  && arg2 && arg2->hasClass())
641  return true;
642  return false;
643  }
