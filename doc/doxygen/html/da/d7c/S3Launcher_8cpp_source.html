79 std::pair<FunctionBase*, bool>
80 S3Launcher::findMethod(const Symbol* symbol, Environment* call_env,
81  Environment* table_env)
82 {
83  FunctionBase* fun = findFunction(symbol, call_env);
84  if (fun)
85  return make_pair(fun, true);
86  if (!table_env)
87  return pair<FunctionBase*, bool>(nullptr, false);
88  Environment* table = nullptr;
89  // Look for S3 methods table:
90  {
91  Frame::Binding* tblbdg
92  = table_env->frame()->binding(S3MethodsTableSymbol);
93  if (tblbdg) {
94  RObject* tblbdgval = tblbdg->forcedValue();
95  if (tblbdgval && tblbdgval->sexptype() == ENVSXP)
96  table = static_cast<Environment*>(tblbdgval);
97  }
98  }
99  // Look up method in table:
100  if (table) {
101  Frame::Binding* symbdg = table->frame()->binding(symbol);
102  if (symbdg) {
103  RObject* symbdgval = symbdg->forcedValue();
104  // Assume that the result is a FunctionBase:
105  return make_pair(static_cast<FunctionBase*>(symbdgval), false);
106  }
107  }
108  return pair<FunctionBase*, bool>(nullptr, false);
109 }
