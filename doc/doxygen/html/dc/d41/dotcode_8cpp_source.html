104 /*
105  Called from resolveNativeRoutine (and itself).
106 
107  Checks whether the specified object correctly identifies a native routine.
108  op is the supplied value for .NAME. This can be
109  a) a string (when this does nothing).
110  b) an external pointer giving the address of the routine
111  (e.g. getNativeSymbolInfo("foo")$address)
112  c) or a NativeSymbolInfo itself (e.g. getNativeSymbolInfo("foo"))
113 
114  It copies the symbol name to buf.
115 
116  NB: in the last two cases it sets fun and symbol as well!
117  */
118 static void
119 checkValidSymbolId(SEXP op, SEXP call, DL_FUNC *fun,
120  R_RegisteredNativeSymbol *symbol, char *buf)
121 {
122  if (Rf_isValidString(op)) return;
123 
124  if(TYPEOF(op) == EXTPTRSXP) {
125  static SEXP native_symbol = NULL;
126  static SEXP registered_native_symbol = NULL;
127  if (native_symbol == NULL) {
128  native_symbol = Rf_install("native symbol");
129  registered_native_symbol = Rf_install("registered native symbol");
130  }
131  char *p = NULL;
132  if(R_ExternalPtrTag(op) == native_symbol)
133  *fun = R_ExternalPtrAddrFn(op);
134  else if(R_ExternalPtrTag(op) == registered_native_symbol) {
135  R_RegisteredNativeSymbol *tmp;
136  tmp = static_cast<R_RegisteredNativeSymbol *>(R_ExternalPtrAddr(op));
137  if(tmp) {
138  if(symbol->type != R_ANY_SYM && symbol->type != tmp->type)
139  Rf_errorcall(call, _("NULL value passed as symbol address"));
140  /* Check the type of the symbol. */
141  switch(symbol->type) {
142  case R_C_SYM:
143  *fun = tmp->symbol.c->fun;
144  p = tmp->symbol.c->name;
145  break;
146  case R_CALL_SYM:
147  *fun = tmp->symbol.call->fun;
148  p = tmp->symbol.call->name;
149  break;
150  case R_FORTRAN_SYM:
151  *fun = tmp->symbol.fortran->fun;
152  p = tmp->symbol.fortran->name;
153  break;
154  case R_EXTERNAL_SYM:
155  *fun = tmp->symbol.external->fun;
156  p = tmp->symbol.external->name;
157  break;
158  default:
159  /* Something unintended has happened if we get here. */
160  Rf_errorcall(call, _("Unimplemented type %d in createRSymbolObject"),
161  symbol->type);
162  break;
163  }
164  *symbol = *tmp;
165  }
166  }
167  /* This is illegal C */
168  if(*fun == nullptr)
169  Rf_errorcall(call, _("NULL value passed as symbol address"));
170 
171  /* copy the symbol name. */
172  if (p && buf) {
173  if (strlen(p) >= MaxSymbolBytes)
174  Rf_error(_("symbol '%s' is too long"), p);
175  memcpy(buf, p, strlen(p)+1);
176  }
177 
178  return;
179  }
180  else if(isNativeSymbolInfo(op)) {
181  checkValidSymbolId(VECTOR_ELT(op, 1), call, fun, symbol, buf);
182  return;
183  }
184 
185  Rf_errorcall(call,
186  _("first argument must be a string (of length 1) or native symbol reference"));
187  return; /* not reached */
188 }

209 static SEXP
210 resolveNativeRoutine(SEXP args, DL_FUNC *fun,
211  R_RegisteredNativeSymbol *symbol, char *buf,
212  int *nargs, int *naok, SEXP call, SEXP env)
213 {
214  SEXP op;
215  const char *p; char *q;
216  DllReference dll;
217  /* This is used as shorthand for 'all' in R_FindSymbol, but
218  should never be supplied */
219  strcpy(dll.DLLname, "");
220  dll.dll = nullptr; dll.obj = nullptr; dll.type = NOT_DEFINED;
221 
222  op = CAR(args); // value of .NAME =
223  /* NB, this sets fun, symbol and buf and is not just a check! */
224  checkValidSymbolId(op, call, fun, symbol, buf);
225 
226  /* The following code modifies the argument list */
227  /* We know this is ok because do_dotCode is entered */
228  /* with its arguments evaluated. */
229 
230  if(symbol->type == R_C_SYM || symbol->type == R_FORTRAN_SYM) {
231  /* And that also looks for PACKAGE = */
232  args = naokfind(CDR(args), nargs, naok, &dll);
233  if(*naok == NA_LOGICAL)
234  Rf_errorcall(call, _("invalid '%s' value"), "naok");
235  if(*nargs > MAX_ARGS)
236  Rf_errorcall(call, _("too many arguments in foreign function call"));
237  } else {
238  /* This has the side effect of setting dll.type if a PACKAGE=
239  argument if found, but it will only be used if a string was
240  passed in */
241  args = pkgtrim(args, &dll);
242  }
243 
244  /* We were given a symbol (or an address), so we are done. */
245  if (*fun) return args;
246 
247  if (dll.type == FILENAME && !strlen(dll.DLLname))
248  Rf_errorcall(call, _("PACKAGE = \"\" is invalid"));
249 
250  // find if we were called from a namespace
251  SEXP env2 = ENCLOS(env);
252  const char *ns = "";
253  if(R_IsNamespaceEnv(env2))
254  ns = R_CHAR(STRING_ELT(R_NamespaceEnvSpec(env2), 0));
255  else env2 = R_NilValue;
256 
257 #ifdef CHECK_CROSS_USAGE
258  if (dll.type == FILENAME && strcmp(dll.DLLname, "base")) {
259  if(strlen(ns) && strcmp(dll.DLLname, ns) &&
260  !(streql(dll.DLLname, "BioC_graph") && streql(ns, "graph")))
261  Rf_warningcall(call,
262  "using PACKAGE = \"%s\" from namespace '%s'",
263  dll.DLLname, ns);
264  }
265 #endif
266 
267  /* Make up the load symbol */
268  if(TYPEOF(op) == STRSXP) {
269  const void *vmax = vmaxget();
270  p = Rf_translateChar(STRING_ELT(op, 0));
271  if(strlen(p) >= MaxSymbolBytes)
272  Rf_error(_("symbol '%s' is too long"), p);
273  q = buf;
274  while ((*q = *p) != '\0') {
275  if(symbol->type == R_FORTRAN_SYM) *q = char(tolower(*q));
276  p++;
277  q++;
278  }
279  vmaxset(vmax);
280  }
281 
282  if(dll.type != FILENAME && strlen(ns)) {
283  /* no PACKAGE= arg, so see if we can identify a DLL
284  from the namespace defining the function */
285  *fun = R_FindNativeSymbolFromDLL(buf, &dll, symbol, env2);
286  if (*fun) return args;
287  Rf_errorcall(call, "\"%s\" not resolved from current namespace (%s)",
288  buf, ns);
289  }
290 
291  /* NB: the actual conversion to the symbol is done in
292  R_dlsym in Rdynload.c. That prepends an underscore (usually),
293  and may append one or more underscores.
294  */
295 
296  *fun = R_FindSymbol(buf, dll.DLLname, symbol);
297  if (*fun) return args;
298 
299  /* so we've failed and bail out */
300  if(strlen(dll.DLLname)) {
301  switch(symbol->type) {
302  case R_C_SYM:
303  Rf_errorcall(call,
304  _("\"%s\" not available for %s() for package \"%s\""),
305  buf, ".C", dll.DLLname);
306  break;
307  case R_FORTRAN_SYM:
308  Rf_errorcall(call,
309  _("\"%s\" not available for %s() for package \"%s\""),
310  buf, ".Fortran", dll.DLLname);
311  break;
312  case R_CALL_SYM:
313  Rf_errorcall(call,
314  _("\"%s\" not available for %s() for package \"%s\""),
315  buf, ".Call", dll.DLLname);
316  break;
317  case R_EXTERNAL_SYM:
318  Rf_errorcall(call,
319  _("\"%s\" not available for %s() for package \"%s\""),
320  buf, ".External", dll.DLLname);
321  break;
322  case R_ANY_SYM:
323  Rf_errorcall(call,
324  _("%s symbol name \"%s\" not in DLL for package \"%s\""),
325  "C/Fortran", buf, dll.DLLname);
326  break;
327  }
328  } else
329  Rf_errorcall(call, _("%s symbol name \"%s\" not in load table"),
330  symbol->type == R_FORTRAN_SYM ? "Fortran" : "C", buf);
331 
332  return args; /* -Wall */
333 }

369 /* find NAOK and DUP, find and remove PACKAGE */
370 static SEXP naokfind(SEXP args, int * len, int *naok, DllReference *dll)
371 {
372  SEXP s, prev;
373  int nargs=0, naokused=0, dupused=0, pkgused=0;
374  const char *p;
375 
376  *naok = 0;
377  *len = 0;
378  for(s = args, prev=args; s != R_NilValue;) {
379  if(TAG(s) == NaokSymbol) {
380  *naok = Rf_asLogical(CAR(s));
381  if(naokused++ == 1) Rf_warning(_("'%s' used more than once"), "NAOK");
382  } else if(TAG(s) == DupSymbol) {
383  if(dupused++ == 1) Rf_warning(_("'%s' used more than once"), "DUP");
384  } else if(TAG(s) == PkgSymbol) {
385  dll->obj = CAR(s); // really?
386  if(TYPEOF(CAR(s)) == STRSXP) {
387  p = Rf_translateChar(STRING_ELT(CAR(s), 0));
388  if(strlen(p) > PATH_MAX - 1)
389  Rf_error(_("DLL name is too long"));
390  dll->type = FILENAME;
391  strcpy(dll->DLLname, p);
392  if(pkgused++ > 1)
393  Rf_warning(_("'%s' used more than once"), "PACKAGE");
394  /* More generally, this should allow us to process
395  any additional arguments and not insist that PACKAGE
396  be the last argument.
397  */
398  } else {
399  /* Have a DLL object, which is not something documented .... */
400  if(TYPEOF(CAR(s)) == EXTPTRSXP) {
401  dll->dll = static_cast<HINSTANCE>(R_ExternalPtrAddr(CAR(s)));
402  dll->type = DLL_HANDLE;
403  } else if(TYPEOF(CAR(s)) == VECSXP) {
404  dll->type = R_OBJECT;
405  dll->obj = s;
406  strcpy(dll->DLLname,
407  Rf_translateChar(STRING_ELT(VECTOR_ELT(CAR(s), 1), 0)));
408  dll->dll = static_cast<HINSTANCE>(R_ExternalPtrAddr(VECTOR_ELT(s, 4)));
409  } else
410  Rf_error("incorrect type (%s) of PACKAGE argument\n",
411  Rf_type2char(TYPEOF(CAR(s))));
412  }
413  } else {
414  nargs++;
415  prev = s;
416  s = CDR(s);
417  continue;
418  }
419  if(s == args)
420  args = s = CDR(s);
421  else
422  SETCDR(prev, s = CDR(s));
423  }
424  *len = nargs;
425  return args;
426 }

444 static SEXP pkgtrim(SEXP args, DllReference *dll)
445 {
446  SEXP s, ss;
447  int pkgused = 0;
448 
449  if (PkgSymbol == nullptr) PkgSymbol = Rf_install("PACKAGE");
450 
451  for(s = args ; s != R_NilValue;) {
452  ss = CDR(s);
453  /* Look for PACKAGE=. We look at the next arg, unless
454  this is the last one (which will only happen for one arg),
455  and remove it */
456  if(ss == R_NilValue && TAG(s) == PkgSymbol) {
457  if(pkgused++ == 1)
458  Rf_warning(_("'%s' used more than once"), "PACKAGE");
459  setDLLname(s, dll->DLLname);
460  dll->type = FILENAME;
461  return R_NilValue;
462  }
463  if(TAG(ss) == PkgSymbol) {
464  if(pkgused++ == 1)
465  Rf_warning(_("'%s' used more than once"), "PACKAGE");
466  setDLLname(ss, dll->DLLname);
467  dll->type = FILENAME;
468  SETCDR(s, CDR(ss));
469  }
470  s = CDR(s);
471  }
472  return args;

607 SEXP attribute_hidden do_dotcall(SEXP call, SEXP op, SEXP args, SEXP env)
608 {
609  DL_FUNC ofun = NULL;
610  SEXP retval, cargs[MAX_ARGS], pargs;
611  R_RegisteredNativeSymbol symbol = {R_CALL_SYM, {NULL}, NULL};

613  int nargs;
