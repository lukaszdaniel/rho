178 void R_SizeFromEnv(Rstart Rp)
179 {
180  int ierr;
181  R_size_t value;
182  char *p, msg[256];
183 
184  if((p = getenv("R_VSIZE"))) {
185  value = R_Decode2Long(p, &ierr);
186  if(ierr != 0 || value > Max_Vsize)
187  R_ShowMessage("WARNING: invalid R_VSIZE ignored\n");
188  else if(value < Min_Vsize) {
189  snprintf(msg, 256,
190  "WARNING: R_VSIZE smaller than Min_Vsize = %lu is ignored\n",
191  (unsigned long) Min_Vsize);
192  R_ShowMessage(msg);
193  }
194  else
195  Rp->vsize = value;
196  }
197 }

199 static void SetSize(R_size_t vsize)
200 {
201  char msg[1024];
202 
203  /* vsize > 0 to catch long->int overflow */
204  if (vsize < 1000 && vsize > 0) {
205  R_ShowMessage("WARNING: vsize ridiculously low, Megabytes assumed\n");
206  vsize *= R_size_t( Mega);
207  }
208  if(vsize < Min_Vsize || vsize > Max_Vsize) {
209  snprintf(msg, 1024,
210  "WARNING: invalid v(ector heap)size `%lu' ignored\n"
211  "using default = %gM\n", static_cast<unsigned long>(vsize),
212  R_VSIZE / Mega);
213  R_ShowMessage(msg);
214  R_VSize = R_VSIZE;
215  } else
216  R_VSize = vsize;
217 }
