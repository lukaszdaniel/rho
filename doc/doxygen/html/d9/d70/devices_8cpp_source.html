67 /*
68  * DEVICE FUNCTIONS
69  *
70  * R allows there to be (up to 64) multiple devices in
71  * existence at the same time. Only one device is the
72  * active device and all drawing occurs in this device
73  *
74  * Each device has its own set of graphics parameters
75  * so that switching between devices, switches between
76  * their graphical contexts (e.g., if you set the line
77  * width on one device then switch to another device,
78  * don't expect to be using the line width you just set!)
79  *
80  * Each device has additional device-specific graphics
81  * parameters which the device driver (i.e., NOT this
82  * generic graphics code) is wholly responsible for
83  * maintaining (including creating and destroying special
84  * resources such as X11 windows).
85  *
86  * Each device has a display list which records every
87  * graphical operation since the last dpptr(dd)->newPage;
88  * this is used to redraw the output on the device
89  * when it is resized and to copy output from one device
90  * to another (this can be disabled, which is the default
91  * for postscript).
92  *
93  * NOTE: that graphical operations should only be
94  * recorded in the displayList if they are "guaranteed"
95  * to succeed (to avoid heaps of error messages on a
96  * redraw) which means that the recording should be the
97  * last thing done in a graphical operation (see do_*
98  * in plot.cpp).
99  *
100  */
101

104 /*
105  R_MaxDevices is defined in ../include/Defn.h to be 64. Slots are
106  initiialized to be NULL, and returned to NULL when a device is
107  removed.
108 
109  Slot 0 is the null device, and slot 63 is keep empty as a sentinel
110  for over-allocation: if a driver fails to call
111  R_CheckDeviceAvailable and uses this slot the device it allocated
112  will be killed.

115  process of being closed and destroyed. We do this to allow for GUI
116  callbacks starting to kill a device whilst another is being killed.
117  */

119 static Rboolean active[R_MaxDevices];

152 pGEDevDesc GEcurrentDevice(void)
153 {
154  /* If there are no active devices
155  * check the options for a "default device".
156  * If there is one, start it up. */
157  if (NoDevices()) {
158  SEXP defdev = Rf_GetOption1(Rf_install("device"));
159  if (Rf_isString(defdev) && Rf_length(defdev) > 0) {
160  SEXP devName = Rf_installTrChar(STRING_ELT(defdev, 0));
161  /* Not clear where this should be evaluated, since
162  grDevices need not be in the search path.
163  So we look for it first on the global search path.
164  */
165  defdev = Rf_findVar(devName, R_GlobalEnv);
166  if(defdev != R_UnboundValue) {
167  PROTECT(defdev = Rf_lang1(devName));
168  Rf_eval(defdev, R_GlobalEnv);
169  UNPROTECT(1);
170  } else {
171  /* Not globally visible:
172  try grDevices namespace if loaded.
173  The option is unlikely to be set if it is not loaded,
174  as the default setting is in grDevices:::.onLoad.
175  */
176  SEXP ns = Rf_findVarInFrame(R_NamespaceRegistry,
177  Rf_install("grDevices"));
178  PROTECT(ns);
179  if(ns != R_UnboundValue &&
180  Rf_findVar(devName, ns) != R_UnboundValue) {
181  PROTECT(defdev = Rf_lang1(devName));
182  Rf_eval(defdev, ns);
183  UNPROTECT(1);
184  } else
185  Rf_error(_("no active or default device"));
186  UNPROTECT(1);
187  }
188  } else if(TYPEOF(defdev) == CLOSXP) {
189  PROTECT(defdev = Rf_lang1(defdev));
190  Rf_eval(defdev, R_GlobalEnv);
191  UNPROTECT(1);
192  } else
193  Rf_error(_("no active or default device"));
194  if (NoDevices()) // the startup above may have failed
195  Rf_error(_("no active device and default getOption(\"device\") is invalid"));
196  }
197  return R_Devices[R_CurrentDevice];
198 }

211 int nextDevice(int from)
212 {
213  if (R_NumDevices == 1)
214  return 0;
215  else {
216  int i = from;
217  int nextDev = 0;
218  while ((i < (R_MaxDevices-1)) && (nextDev == 0))
219  if (active[++i]) nextDev = i;
220  if (nextDev == 0) {
221  /* start again from 1 */
222  i = 0;
223  while ((i < (R_MaxDevices-1)) && (nextDev == 0))
224  if (active[++i]) nextDev = i;
225  }
226  return nextDev;
227  }
228 }

230 int prevDevice(int from)
231 {
232  if (R_NumDevices == 1)
233  return 0;
234  else {
235  int i = from;
236  int prevDev = 0;
237  if (i < R_MaxDevices)
238  while ((i > 1) && (prevDev == 0))
239  if (active[--i]) prevDev = i;
240  if (prevDev == 0) {
241  /* start again from R_MaxDevices */
242  i = R_MaxDevices;
243  while ((i > 1) && (prevDev == 0))
244  if (active[--i]) prevDev = i;
245  }
246  return prevDev;
247  }

274 int selectDevice(int devNum)
275 {
276  /* Valid to select nullDevice, but that will open a new device.
277  See ?dev.set.
278  */
279  if((devNum >= 0) && (devNum < R_MaxDevices) &&
280  (R_Devices[devNum] != nullptr) && active[devNum])
281  {
282  pGEDevDesc gdd;
283 
284  if (!NoDevices()) {
285  pGEDevDesc oldd = GEcurrentDevice();
286  if (oldd->dev->deactivate) oldd->dev->deactivate(oldd->dev);
287  }
288 
289  R_CurrentDevice = devNum;
290 
291  /* maintain .Device */
292  Rf_gsetVar(R_DeviceSymbol,
293  Rf_elt(getSymbolValue(R_DevicesSymbol), devNum),
294  R_BaseEnv);
295 
296  gdd = GEcurrentDevice(); /* will start a device if current is null */
297  if (!NoDevices()) /* which it always will be */
298  if (gdd->dev->activate) gdd->dev->activate(gdd->dev);
299  return devNum;
300  }
301  else
302  return selectDevice(nextDevice(devNum));
303 }

305 /* historically the close was in the [kK]illDevices.
306  only use findNext= FALSE when shutting R dowm, and .Device[s] are not
307  updated.
308 */
309 static
310 void removeDevice(int devNum, Rboolean findNext)
311 {
312  /* Not vaild to remove nullDevice */
313  if((devNum > 0) && (devNum < R_MaxDevices) &&
314  (R_Devices[devNum] != nullptr) && active[devNum])
315  {
316  int i;
317  SEXP s;
318  pGEDevDesc g = R_Devices[devNum];
319 
320  active[devNum] = FALSE; /* stops it being selected again */
321  R_NumDevices--;
322 
323  if(findNext) {
324  /* maintain .Devices */
325  PROTECT(s = getSymbolValue(R_DevicesSymbol));
326  for (i = 0; i < devNum; i++) s = CDR(s);
327  SETCAR(s, Rf_mkString(""));
328  UNPROTECT(1);
329 
330  /* determine new current device */
331  if (devNum == R_CurrentDevice) {
332  R_CurrentDevice = nextDevice(R_CurrentDevice);
333  /* maintain .Device */
334  Rf_gsetVar(R_DeviceSymbol,
335  Rf_elt(getSymbolValue(R_DevicesSymbol), R_CurrentDevice),
336  R_BaseEnv);
337 
338  /* activate new current device */
339  if (R_CurrentDevice) {
340  pGEDevDesc gdd = GEcurrentDevice();
341  if(gdd->dev->activate) gdd->dev->activate(gdd->dev);
342  }
343  }
344  }
345  g->dev->close(g->dev);
346  GEdestroyDevDesc(g);
347  R_Devices[devNum] = nullptr;
348  }
349 }

418 void GEaddDevice(pGEDevDesc gdd)
419 {
420  int i;
421  Rboolean appnd;
422  SEXP s, t;
423  pGEDevDesc oldd;
424 
425  PROTECT(s = getSymbolValue(R_DevicesSymbol));
426 
427  if (!NoDevices()) {
428  oldd = GEcurrentDevice();
429  if(oldd->dev->deactivate) oldd->dev->deactivate(oldd->dev);
430  }
431 
432  /* find empty slot for new descriptor */
433  i = 1;
434  if (CDR(s) == R_NilValue)
435  appnd = TRUE;
436  else {
437  s = CDR(s);
438  appnd = FALSE;
439  }
440  while (R_Devices[i] != nullptr) {
441  i++;
442  if (CDR(s) == R_NilValue)
443  appnd = TRUE;
444  else
445  s = CDR(s);
446  }
447  R_CurrentDevice = i;
448  R_NumDevices++;
449  R_Devices[i] = gdd;
450  gdd->index = i;
451  active[i] = TRUE;
452 
453  GEregisterWithDevice(gdd);
454  if(gdd->dev->activate) gdd->dev->activate(gdd->dev);
455 
456  /* maintain .Devices (.Device has already been set) */
457  t = PROTECT(Rf_duplicate(getSymbolValue(R_DeviceSymbol)));
458  if (appnd)
459  SETCDR(s, CONS(t, R_NilValue));
460  else
461  SETCAR(s, t);
462 
463  UNPROTECT(2);
464 
465  /* In case a device driver did not call R_CheckDeviceAvailable
466  before starting its allocation, we complete the allocation and
467  then call killDevice here. This ensures that the device gets a
468  chance to deallocate its resources and the current active
469  device is restored to a sane value. */
470  if (i == R_MaxDevices - 1) {
471  killDevice(i);
472  Rf_error(_("too many open devices"));
473  }
474 }
