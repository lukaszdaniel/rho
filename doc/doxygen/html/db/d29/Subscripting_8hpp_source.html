843  template <class VL, class VR>
844  VL *Subscripting::arraySubassign(VL *lhs,
845  const std::vector<Indices> &indicesvec,
846  const VR *rhs)
847  {
848  typedef typename VL::value_type Lval;
849  typedef typename VR::value_type Rval;
850  const IntVector *vdims = lhs->dimensions();
851  std::size_t ndims = vdims->size();
852  DimIndexerVector dimindexer(ndims);
853  std::size_t ni = createDimIndexers(&dimindexer, vdims, indicesvec);
854  std::size_t rhs_size = rhs->size();
855  if (rhs_size > 1)
856  {
857  // TODO: Move NA-detection back into the canonicalisation
858  // process.
859  for (std::size_t d = 0; d < ndims; ++d)
860  {
861  const Indices &dimindices = indicesvec[d];
862  std::size_t dsz = dimindices.size();
863  for (std::size_t i = 0; i < dsz; ++i)
864  if (dimindices[i] == 0)
865  Rf_error(_("NAs are not allowed in subscripted assignments"));
866  }
867  }
868  GCStackRoot<VL> ans(lhs);
869  // If necessary, make a copy to be sure we don't modify rhs.
870  // (FIXME: ideally this should be a shallow copy for
871  // HandleVectors.)
872  const VectorBase *ansvb = static_cast<VectorBase *>(ans.get());
873  if (ansvb == rhs)
874  ans = ans->clone();
875  // Dispose of 'no indices' case:
876  if (ni == 0)
877  return ans;
878  if (rhs_size == 0)
879  Rf_error(_("replacement has length zero"));
880  if (ni % rhs_size != 0)
881  Rf_warning(_("number of items to replace is not a multiple of replacement length"));
882  // Copy elements across:
883  for (std::size_t irhs = 0; irhs < ni; ++irhs)
884  {
885  bool naindex = false;
886  std::size_t iout = 0;
887  for (std::size_t d = 0; d < ndims; ++d)
888  {
889  const DimIndexer &di = dimindexer[d];
890  const Indices &dimindices = indicesvec[d];
891  std::size_t index = dimindices[di.indexnum];
892  if (index == 0)
893  {
894  naindex = true;
895  break;
896  }
897  iout += (index - 1) * di.stride;
898  }
899  if (!naindex)
900  {
901  const Rval &rval = (*rhs)[irhs % rhs_size];
902  if (isNA(rval))
903  {
904  assignDuplicateElement((*ans)[iout], NA<Lval>());
905  }
906  else
907  {
908  assignDuplicateElement((*ans)[iout], rval);
909  }
910  }
911 
912  // Advance the index selection:
913  {
914  std::size_t d = 0;
915  bool done;
916  do
917  {
918  done = true;
919  DimIndexer &di = dimindexer[d];
920  if (++di.indexnum >= indicesvec[d].size())
921  {
922  di.indexnum = 0;
923  done = (++d >= ndims);
924  }
925  } while (!done);
926  }
927  }
928  return ans;
929  }

932  V *Subscripting::arraySubset(const V *v,
933  const std::vector<Indices> &indicesvec,
934  bool drop)
935  {

937  std::size_t ndims = vdims->size();
938  DimIndexerVector dimindexer(ndims);
939  std::size_t resultsize = createDimIndexers(&dimindexer, vdims,
940  indicesvec);
941  GCStackRoot<V> result(V::create(resultsize));
942  // Copy elements across:
943  {
944  // ***** FIXME ***** Currently needed because Handle's
945  // assignment operator takes a non-const RHS:
946  V *vnc = const_cast<V *>(v);
947  for (std::size_t iout = 0; iout < resultsize; ++iout)
948  {
949  bool naindex = false;
950  std::size_t iin = 0;
951  for (std::size_t d = 0; d < ndims; ++d)
952  {
953  const DimIndexer &di = dimindexer[d];
954  const Indices &dimindices = indicesvec[d];
955  std::size_t index = dimindices[di.indexnum];
956  if (index == 0)
957  {
958  naindex = true;
959  break;
960  }
961  iin += (index - 1) * di.stride;
962  }
963  (*result)[iout] = ElementTraits::duplicate_element(
964  naindex ? NA<typename V::value_type>()
965  : (*vnc)[iin]);
966  // Advance the index selection:
967  {
968  std::size_t d = 0;
969  bool done;
970  do
971  {
972  done = true;
973  DimIndexer &di = dimindexer[d];
974  if (++di.indexnum >= indicesvec[d].size())
975  {
976  di.indexnum = 0;
977  done = (++d >= ndims);
978  }
979  } while (!done);

981  }

1013  template <class VL, class VR>
1014  VL *Subscripting::vectorSubassign(VL *lhs,
1015  const Indices &indices,
1016  const VR *rhs)
1017  {
1018  typedef typename VL::value_type Lval;
1019  typedef typename VR::value_type Rval;
1020  std::size_t ni = indices.size();
1021  std::size_t rhs_size = rhs->size();
1022  if (rhs_size > 1)
1023  {
1024  for (std::size_t i = 0; i < ni; ++i)
1025  if (indices[i] == 0)
1026  Rf_error(_("NAs are not allowed in subscripted assignments"));
1027  }
1028  GCStackRoot<VL> ans(lhs);
1029  std::size_t minsize = indices.minimumLHSSize();
1030  if (minsize > lhs->size())
1031  ans = VectorBase::resize(lhs, minsize);
1032  // If necessary, make a copy to be sure we don't modify rhs or
1033  // indices. (FIXME: ideally this should be a shallow copy for
1034  // HandleVectors.)
1035  const VectorBase *ansvb = static_cast<VectorBase *>(ans.get());
1036  if (ansvb == rhs)
1037  ans = ans->clone();
1038  // Dispose of 'no indices' case:
1039  if (ni == 0)
1040  return ans;
1041  if (rhs_size == 0)
1042  Rf_error(_("replacement has length zero"));
1043  if (ni % rhs_size != 0)
1044  Rf_warning(_("number of items to replace is not a multiple of replacement length"));
1045  for (std::size_t i = 0; i < ni; ++i)
1046  {
1047  std::size_t index = indices[i];
1048  if (index != 0)
1049  {
1050  const Rval &rval = (*rhs)[i % rhs_size];
1051  if (isNA(rval))
1052  {
1053  assignDuplicateElement((*ans)[index - 1], NA<Lval>());
1054  }
1055  else
1056  {
1057  assignDuplicateElement((*ans)[index - 1], rval);
1058  }
1059  }
1060  }
1061  indices.applyNewNames(ans);
1062  return ans;
1063  }

1065  template <class V>
1066  V *Subscripting::vectorSubset(const V *v, const Indices &indices)
1067  {
1068  std::size_t ni = indices.size();
1069  GCStackRoot<V> ans(V::create(ni));
1070  std::size_t vsize = v->size();
1071  // ***** FIXME ***** Currently needed because Handle's
1072  // assignment operator takes a non-const RHS:
1073  V *vnc = const_cast<V *>(v);
1074  for (std::size_t i = 0; i < ni; ++i)
1075  {
1076  std::size_t index = indices[i];
1077  // Note that zero and negative indices ought not to occur.
1078  if (index == 0 || index > vsize)
1079  {
1080  (*ans)[i] = ElementTraits::duplicate_element(
1081  NA<typename V::value_type>());
1082  }
1083  else
1084  {
1085  (*ans)[i] = ElementTraits::duplicate_element(
1086  (*vnc)[index - 1]);
1087  }
1088  }
1089  setVectorAttributes(ans, v, indices);
1090  return ans;
