305  template <class V>
306  V* VectorBase::resize(const V* pattern, size_type new_size)
307  {
308  GCStackRoot<V> ans(V::create(new_size));
309  size_type copysz = std::min(pattern->size(), new_size);
310  for (size_type i = 0; i < copysz; i++) {
311  (*ans)[i] = ElementTraits::duplicate_element((*pattern)[i]);
312  }
313  for (size_type i = copysz; i < new_size; i++) {
314  (*ans)[i] = ElementTraits::duplicate_element(
315  NA<typename V::value_type>());
316  }
317  ans->setAttributes(resizeAttributes(pattern->attributes(), new_size));
318  ans->setS4Object(pattern->isS4Object());
319  return ans;
320  }
