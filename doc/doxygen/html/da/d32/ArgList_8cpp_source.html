59 void ArgList::evaluateToArray(Environment* env,
60  unsigned int num_args, RObject** evaluated_args,
61  MissingArgHandling allow_missing) const
62 {
63  assert(allow_missing != MissingArgHandling::Drop);
64 
65  unsigned int arg_number = 0;
66  for (const ConsCell& arg_cell : getExpandedArgs(env)) {
67  RObject* arg = arg_cell.car();
68  RObject* value;
69  if (m_status == EVALUATED) {
70  value = arg;
71  } else {
72  value = evaluateSingleArgument(arg, env, allow_missing,
73  arg_number + 1);
74  }
75  evaluated_args[arg_number] = value;
76  ++arg_number;
77  }
78  assert(arg_number == num_args);

103 RObject* ArgList::evaluateSingleArgument(const RObject* arg,
104  Environment* env,
105  MissingArgHandling allow_missing,
106  int arg_number) const {
107  // assert(allow_missing != MissingArgHandling::Drop);
108 
109  if (m_first_arg_env) {
110  RObject* value = m_first_arg;
111  m_first_arg = nullptr;
112  m_first_arg_env = nullptr;
113  return value;
114  } else if (arg && arg->sexptype() == SYMSXP) {
115  const Symbol* sym = static_cast<const Symbol*>(arg);
116  if (sym == Symbol::missingArgument()) {
117  if (allow_missing == MissingArgHandling::Keep)
118  return Symbol::missingArgument();
119  else Rf_error(_("argument %d is empty"), arg_number);
120  } else if (allow_missing == MissingArgHandling::Keep
121  && isMissingArgument(sym, env->frame())) {
122  return Symbol::missingArgument();
123  // allow_missing = false && isMissingArgument() is handled in
124  // evaluate() below.
125  }
126  }
127  return Evaluator::evaluate(const_cast<RObject*>(arg), env);
128 }

130 void ArgList::merge(const ConsCell* extraargs)
131 {
132  if (m_status != PROMISED)
133  Rf_error("Internal error: ArgList::merge() requires PROMISED ArgList");
134  // Convert extraargs into a doubly linked list:
135  typedef std::list<pair<const RObject*, RObject*> > Xargs;
136  Xargs xargs;
137  for (const ConsCell* cc = extraargs; cc; cc = cc->tail())
138  xargs.push_back(make_pair(cc->tag(), cc->car()));
139  // Apply overriding arg values supplied in extraargs:
140  PairList* last = nullptr;
141  for (PairList* pl = mutable_list(); pl; pl = pl->tail()) {
142  last = pl;
143  const RObject* tag = pl->tag();
144  if (tag) {
145  Xargs::iterator it = xargs.begin();
146  while (it != xargs.end() && (*it).first != tag)
147  ++it;
148  if (it != xargs.end()) {
149  pl->setCar((*it).second);
150  xargs.erase(it);
151  }
152  }
153  }
154  // Append remaining extraargs:
155  for (Xargs::const_iterator it = xargs.begin(); it != xargs.end(); ++it) {
156  last = append(it->second, it->first, last);
157  }
158 }

160 pair<bool, RObject*> ArgList::firstArg(Environment* env)
161 {
162  auto expanded_args = getExpandedArgs(env);
163  if (expanded_args.empty()) {
164  return pair<bool, RObject*>(false, nullptr);
165  }
166  RObject* first_arg = expanded_args.begin()->car();
167  if (m_status == EVALUATED) {
168  return make_pair(true, first_arg);
169  }
170  if (!first_arg)
171  return pair<bool, RObject*>(true, nullptr);
172 
173  m_first_arg = Evaluator::evaluate(first_arg, env);
174  m_first_arg_env = env;
175  return make_pair(true, m_first_arg.get());
176 }

256 void ArgList::wrapInPromises(Environment* env,
257  const Expression* call)
258 {
259  if (m_status == PROMISED)
260  return;
261  if (m_status == EVALUATED) {
262  assert(call != nullptr);
263  ArgList raw_args(call->tail(), RAW);
264  raw_args.wrapInForcedPromises(env, *this);
265  m_status = PROMISED;
266  m_list = raw_args.m_list;
267  return;
268  }
269  assert(env != nullptr);
270 
271  auto expanded_args = getExpandedArgs(env);
272  setList(nullptr);
273  PairList* lastout = nullptr;
274 
275  for (const ConsCell& arg : expanded_args) {
276  RObject* rawvalue = arg.car();
277  const Symbol* tag = tag2Symbol(arg.tag());
278  RObject* value = Symbol::missingArgument();
279  if (m_first_arg_env) {
280  value = Promise::createEvaluatedPromise(rawvalue, m_first_arg);
281  m_first_arg = nullptr;
282  m_first_arg_env = nullptr;
283  } else if (rawvalue != Symbol::missingArgument())
284  value = new Promise(rawvalue, env);
285  lastout = append(value, tag, lastout);
286  }
287 
288  m_status = PROMISED;
289 }

291 void ArgList::wrapInForcedPromises(Environment* env,
292  const ArgList& evaluated_values)
293 {
294  assert(m_status == RAW);
295  assert(evaluated_values.status() == EVALUATED);
296 
297  if (m_first_arg_env) {
298  assert(m_first_arg.get() == evaluated_values.get(0));
299  }
300 
301  auto expanded_args = getExpandedArgs(env);
302  const auto& values = *evaluated_values.list();
303 
304  setList(nullptr);
305  PairList* lastout = nullptr;
306 
307  auto arg = expanded_args.begin();
308  auto value = values.begin();
309  for (; arg != expanded_args.end() && value != values.end();
310  ++arg, ++value) {
311  const RObject* expr = arg->car();
312  const Symbol* tag = tag2Symbol(arg->tag());
313 
314  Promise* promise = Promise::createEvaluatedPromise(expr, value->car());
315  lastout = append(promise, tag, lastout);
316  }
317 
318  // Check to make sure that the lengths matched up OK.
319  if (arg != expanded_args.end() || value != values.end()) {
320  Rf_error(_("dispatch error"));
321  }
322 
323  m_status = PROMISED;
324 }
