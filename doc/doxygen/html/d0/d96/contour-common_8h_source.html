41 static bool ctr_intersect(double z0, double z1, double zc, double *f)
42 {
43 /* Old test was ((z0 - zc) * (z1 - zc) < 0.0), but rounding led to inconsistencies
44  in PR#15454 */
45  if ( (z0 < zc) != (z1 < zc) && z0 != zc && z1 != zc ) {
46  *f = (zc - z0) / (z1 - z0);
47  return true;
48  }
49  return false;
50 }

72 inline bool XMATCH(double x0, double x1) { return (fabs(x0-x1) == 0); };

75 static int ctr_segdir(double xend, double yend, double *x, double *y,
76  int *i, int *j, int nx, int ny)
77 {
78  if (YMATCH(yend, y[*j])) {
79  if (*j == 0)
80  return 0;
81  *j = *j - 1;
82  return 3;
83  }
84  if (XMATCH(xend, x[*i])) {
85  if (*i == 0)
86  return 0;
87  *i = *i - 1;
88  return 4;
89  }
90  if (YMATCH(yend, y[*j + 1])) {
91  if (*j >= ny - 1)
92  return 0;
93  *j = *j + 1;
94  return 1;
95  }
96  if (XMATCH(xend, x[*i + 1])) {
97  if (*i >= nx - 1)
98  return 0;
99  *i = *i + 1;
100  return 2;
101  }
102  return 0;
103 }

105 /* Search seglist for a segment with endpoint (xend, yend). */
106 /* The cell entry direction is dir, and if tail=1/0 we are */
107 /* building the tail/head of a contour. The matching segment */
108 /* is pointed to by seg and the updated segment list (with */
109 /* the matched segment stripped) is returned by the funtion. */
110 
111 static SEG* ctr_segupdate(double xend, double yend, int dir, Rboolean tail,
112  SEG* seglist, SEG** seg)
113 {
114  if (seglist == NULL) {
115  *seg = NULL;
116  return NULL;
117  }
118  switch (dir) {
119  case 1:
120  case 3:
121  if (YMATCH(yend,seglist->y0)) {
122  if (!tail)
123  ctr_swapseg(seglist);
124  *seg = seglist;
125  return seglist->next;
126  }
127  if (YMATCH(yend,seglist->y1)) {
128  if (tail)
129  ctr_swapseg(seglist);
130  *seg = seglist;
131  return seglist->next;
132  }
133  break;
134  case 2:
135  case 4:
136  if (XMATCH(xend,seglist->x0)) {
137  if (!tail)
138  ctr_swapseg(seglist);
139  *seg = seglist;
140  return seglist->next;
141  }
142  if (XMATCH(xend,seglist->x1)) {
143  if (tail)
144  ctr_swapseg(seglist);
145  *seg = seglist;
146  return seglist->next;
147  }
148  break;
149  }
150  seglist->next = ctr_segupdate(xend, yend, dir, tail, seglist->next, seg);
151  return seglist;
152 }

156 /*
157  * Generate a list of segments for a single level
158  *
159  * NB this R_allocs its return value, so callers need to manage R_alloc stack.
160  */
161 static SEG** contourLines(double *x, int nx, double *y, int ny,
162  double *z, double zc, double atom)
163 {
164  double f, xl, xh, yl, yh, zll, zhl, zlh, zhh, xx[4], yy[4];
165  int i, j, k, l, m, nacode;
166  SEG* seglist;
167  SEG** segmentDB;
168  /* Initialize the segment data base */
169  /* Note we must be careful about resetting */
170  /* the top of the stack, otherwise we run out of */
171  /* memory after a sequence of displaylist replays */
172  /*
173  * This reset is done out in GEcontourLines
174  */
175  segmentDB = (SEG**)R_alloc(nx*ny, sizeof(SEG*));
176  for (i = 0; i < nx; i++)
177  for (j = 0; j < ny; j++)
178  segmentDB[i + j * nx] = NULL;
179  for (i = 0; i < nx - 1; i++) {
180  xl = x[i];
181  xh = x[i + 1];
182  for (j = 0; j < ny - 1; j++) {
183  yl = y[j];
184  yh = y[j + 1];
185  k = i + j * nx;
186  zll = z[k];
187  zhl = z[k + 1];
188  zlh = z[k + nx];
189  zhh = z[k + nx + 1];
190 
191  /* If the value at a corner is exactly equal to a contour level,
192  * change that value by a tiny amount */
193 
194  if (zll == zc) zll += atom;
195  if (zhl == zc) zhl += atom;
196  if (zlh == zc) zlh += atom;
197  if (zhh == zc) zhh += atom;
198 #ifdef DEBUG_contour
199  /* Haven't seen this happening (MM): */
200  if (zll == zc) REprintf(" [%d,%d] ll: %g\n",i,j, zll);
201  if (zhl == zc) REprintf(" [%d,%d] hl: %g\n",i,j, zhl);
202  if (zlh == zc) REprintf(" [%d,%d] lh: %g\n",i,j, zlh);
203  if (zhh == zc) REprintf(" [%d,%d] hh: %g\n",i,j, zhh);
204 #endif
205  /* Check for intersections with sides */
206 
207  nacode = 0;
208  if (R_FINITE(zll)) nacode += 1;
209  if (R_FINITE(zhl)) nacode += 2;
210  if (R_FINITE(zlh)) nacode += 4;
211  if (R_FINITE(zhh)) nacode += 8;
212 
213  k = 0;
214  switch (nacode) {
215  case 15:
216  if (ctr_intersect(zll, zhl, zc, &f)) {
217  xx[k] = xl + f * (xh - xl);
218  yy[k] = yl; k++;
219  }
220  if (ctr_intersect(zll, zlh, zc, &f)) {
221  yy[k] = yl + f * (yh - yl);
222  xx[k] = xl; k++;
223  }
224  if (ctr_intersect(zhl, zhh, zc, &f)) {
225  yy[k] = yl + f * (yh - yl);
226  xx[k] = xh; k++;
227  }
228  if (ctr_intersect(zlh, zhh, zc, &f)) {
229  xx[k] = xl + f * (xh - xl);
230  yy[k] = yh; k++;
231  }
232  break;
233  case 14:
234  if (ctr_intersect(zhl, zhh, zc, &f)) {
235  yy[k] = yl + f * (yh - yl);
236  xx[k] = xh; k++;
237  }
238  if (ctr_intersect(zlh, zhh, zc, &f)) {
239  xx[k] = xl + f * (xh - xl);
240  yy[k] = yh; k++;
241  }
242  if (ctr_intersect(zlh, zhl, zc, &f)) {
243  xx[k] = xl + f * (xh - xl);
244  yy[k] = yh + f * (yl - yh);
245  k++;
246  }
247  break;
248  case 13:
249  if (ctr_intersect(zll, zlh, zc, &f)) {
250  yy[k] = yl + f * (yh - yl);
251  xx[k] = xl; k++;
252  }
253  if (ctr_intersect(zlh, zhh, zc, &f)) {
254  xx[k] = xl + f * (xh - xl);
255  yy[k] = yh; k++;
256  }
257  if (ctr_intersect(zll, zhh, zc, &f)) {
258  xx[k] = xl + f * (xh - xl);
259  yy[k] = yl + f * (yh - yl);
260  k++;
261  }
262  break;
263  case 11:
264  if (ctr_intersect(zhl, zhh, zc, &f)) {
265  yy[k] = yl + f * (yh - yl);
266  xx[k] = xh; k++;
267  }
268  if (ctr_intersect(zll, zhl, zc, &f)) {
269  xx[k] = xl + f * (xh - xl);
270  yy[k] = yl; k++;
271  }
272  if (ctr_intersect(zll, zhh, zc, &f)) {
273  xx[k] = xl + f * (xh - xl);
274  yy[k] = yl + f * (yh - yl);
275  k++;
276  }
277  break;
278  case 7:
279  if (ctr_intersect(zll, zlh, zc, &f)) {
280  yy[k] = yl + f * (yh - yl);
281  xx[k] = xl; k++;
282  }
283  if (ctr_intersect(zll, zhl, zc, &f)) {
284  xx[k] = xl + f * (xh - xl);
285  yy[k] = yl; k++;
286  }
287  if (ctr_intersect(zlh, zhl, zc, &f)) {
288  xx[k] = xl + f * (xh - xl);
289  yy[k] = yh + f * (yl - yh);
290  k++;
291  }
292  break;
293  }
294 
295  /* We now have k(=2,4) endpoints */
296  /* Decide which to join */
297 
298  seglist = nullptr;
299 
300  if (k > 0) {
301  if (k == 2) {
302  seglist = ctr_newseg(xx[0], yy[0], xx[1], yy[1], seglist);
303  }
304  else if (k == 4) {
305  for (k = 3; k >= 1; k--) {
306  m = k;
307  xl = xx[k];
308  for (l = 0; l < k; l++) {
309  if (xx[l] > xl) {
310  xl = xx[l];
311  m = l;
312  }
313  }
314  if (m != k) {
315  xl = xx[k];
316  yl = yy[k];
317  xx[k] = xx[m];
318  yy[k] = yy[m];
319  xx[m] = xl;
320  yy[m] = yl;
321  }
322  }
323  seglist = ctr_newseg(xx[0], yy[0], xx[1], yy[1], seglist);
324  seglist = ctr_newseg(xx[2], yy[2], xx[3], yy[3], seglist);
325  }
326  else Rf_error("k = %d, should be 2 or 4", k);
327  }
328  segmentDB[i + j * nx] = seglist;
329  }
330  }
331  return segmentDB;
332 }
