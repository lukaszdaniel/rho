100 RObject *Closure::execute(Environment *env) const
101 {
102  RObject *ans;
103  Evaluator::maybeCheckForUserInterrupts();
104  Environment::ReturnScope returnscope(env);
105  Closure::DebugScope debugscope(this);
106  try
107  {
108  ++m_num_invokes;
109 #ifdef ENABLE_LLVM_JIT
110  if (m_compiled_body && m_compiled_body->hasMatchingFrameLayout(env))
111  {
112  PlainContext boctxt;
113  ans = m_compiled_body->evalInEnvironment(env);
114  }
115  else
116  {
117  if (!m_compiled_body && m_num_invokes == 100)
118  {
119  // Compile the body, but stay in the interpreter because the
120  // frame hasn't been setup for a compiled function.
121  // TODO(kmillar): recompile functions as needed.
122  compile();
123  }
124 #endif
125  BailoutContext boctxt;
126  ans = Evaluator::evaluate(m_body, env);
127 #ifdef ENABLE_LLVM_JIT
128  }
129 #endif
130  if (ans && ans->sexptype() == BAILSXP)
131  {
132  ReturnBailout *rbo = dynamic_cast<ReturnBailout *>(ans);
133  if (!rbo || rbo->environment() != env)
134  abort();
135  R_Visible = Rboolean(rbo->printResult());
136  ans = rbo->value();
137  }
138  }
139  catch (ReturnException &rx)
140  {
141  if (rx.environment() != env)
142  throw;
143  ans = rx.value();
144  }
145  return ans;
146 }
