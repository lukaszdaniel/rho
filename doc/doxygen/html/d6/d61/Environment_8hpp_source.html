635  template <typename UnaryPredicate>
636  RObject* findTestedValue(const Symbol* symbol, Environment* env,
637  UnaryPredicate pred, bool inherits)
638  {
639  using namespace std;
640 
641  do {
642  Frame::Binding *bdg;
643  if (inherits && env == Environment::global()) {
644  // findBinding() handles the details of the cache correctly.
645  bdg = env->findBinding(symbol);
646  } else {
647  bdg = env->frame()->binding(symbol);
648  }
649  if (bdg) {
650  pair<RObject*, bool> fpr = bdg->forcedValue2();
651  RObject* val = fpr.first;
652  if (pred(val)) {
653  // Invoke read monitor (if any) only if
654  // forcedValue() did not force a Promise. (If a
655  // Promise was forced, the read monitor will have
656  // been invoked anyway, and 'bdg' may now be
657  // junk.)
658  if (!fpr.second)
659  bdg->rawValue();
660  return val;
661  }
662  }
663  env = env->enclosingEnvironment();
664  } while (inherits && env);
665  return nullptr;
666  }
