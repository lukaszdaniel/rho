67 Symbol::Symbol(const String *the_name)
68  : RObject(SYMSXP), m_dd_index(-1), m_is_special_symbol(false)
69 {
70  m_name = the_name;
71  // If this is a ..n symbol, extract the value of n.
72  // boost::regex_match (libboost_regex1_36_0-1.36.0-9.5) doesn't
73  // seem comfortable with empty strings, hence the size check.
74  if (m_name && m_name->size() > 2)
75  {
76  // Versions of GCC prior to 4.9 don't support std::regex, so use
77  // boost::regex instead.
78  static const boost::regex *regex = new boost::regex("\\.\\.(\\d+)");
79 
80  string name(m_name->c_str());
81  boost::smatch dd_match;
82  if (boost::regex_match(name, dd_match, *regex))
83  {
84  istringstream iss(dd_match[1]);
85  int n;
86  iss >> n;
87  m_dd_index = n;
88  }
89  }
90 }

113 RObject *Symbol::evaluate(Environment *env)
114 {
115  if (this == DotsSymbol)
116  Rf_error(_("'...' used in an incorrect context"));
117  GCStackRoot<> val;
118  if (isDotDotSymbol())
119  val = Rf_ddfindVar(this, env);
120  else
121  {
122  Frame::Binding *bdg = env->findBinding(this);
123  if (bdg)
124  val = bdg->unforcedValue();
125  else if (this == missingArgument())
126  val = this; // This reproduces CR behaviour
127  else
128  val = unboundValue();
129  }
130  if (!val)
131  return nullptr;
132  if (val == unboundValue())
133  Rf_error(_("object '%s' not found"), name()->c_str());
134  if (val == missingArgument() && !isDotDotSymbol())
135  {
136  if (m_name && m_name != String::blank())
137  Rf_error(_("argument \"%s\" is missing, with no default"), name()->c_str());
138  else
139  Rf_error(_("argument is missing, with no default"));
140  }
141  if (val->sexptype() == PROMSXP)
142  {
143  val = Rf_eval(val, env);
144  ENSURE_NAMEDMAX(val);
145  }
146  else
147  ENSURE_NAMED(val);
148  return val;
149 }
