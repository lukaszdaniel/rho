72 #ifdef ALLOC_STATS
73 // Allocation and free frequency tables for profiling the allocator.
74 // Allocation stats are collected into 8-byte bins, allocations > 256 bytes are
75 // counted in the 256 byte bin.
76 size_t alloc_counts[32];
77 size_t free_counts[32];

79 // Computes the bin to update in an allocation frequency table.
80 static int get_bin_number(size_t bytes)
81 {
82  if (bytes >= 32 * 8)
83  {
84  // Allocations of size >= 256 are mapped to the 256-byte bin.
85  return 31;
86  }
87  else if (bytes >= 8)
88  {
89  // Allocations of size n*8 to n*8-1 are mapped to the n-byte bin.
90  return (31 & (bytes / 8)) - 1;
91  }
92  else
93  {
94  // Allocations of size < 8 are mapped to the 8-byte bin.
95  return 0;
96  }
97 }
98 #endif

109 void *MemoryBank::allocate(size_t bytes)
110 {
111  notifyAllocation(bytes);
112  void *p;
113  if (bytes >= s_new_threshold)
114  p = ::operator new(bytes);
115  else
116  {
117  Pool &pool = s_pools[s_pooltab[(bytes + 7) >> 3]];
118  p = pool.allocate();
119  }
120  return p;
121 }

157 void MemoryBank::notifyAllocation(size_t bytes)
158 {
159 #ifdef R_MEMORY_PROFILING
160  if (s_monitor && bytes >= s_monitor_threshold)
161  s_monitor(bytes);
162 #endif
163  ++s_blocks_allocated;
164  s_bytes_allocated += bytes;
165 #ifdef ALLOC_STATS
166  alloc_counts[get_bin_number(bytes)] += 1;
167 #endif
168 }
