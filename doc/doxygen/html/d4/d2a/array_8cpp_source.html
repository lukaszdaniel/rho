81 HIDDEN SEXP do_matrix(/*const*/ Expression* call, const BuiltInFunction* op, RObject* vals, RObject* nrow, RObject* ncol, RObject* byrow_, RObject* dimnames, RObject* miss_nr_, RObject* miss_nc_)
82 {
83  SEXP ans;
84  int nr = 1, nc = 1;
85 
86  switch(TYPEOF(vals)) {
87  case LGLSXP:
88  case INTSXP:
89  case REALSXP:
90  case CPLXSXP:
91  case STRSXP:
92  case RAWSXP:
93  case EXPRSXP:
94  case VECSXP:
95  break;
96  default:
97  Rf_error(_("'data' must be of a vector type, was '%s'"),
98  Rf_type2char(TYPEOF(vals)));
99  }
100  R_xlen_t lendat = XLENGTH(vals);
101  int byrow = Rf_asLogical(byrow_);
102  if (byrow == R_NaInt)
103  Rf_error(_("invalid '%s' argument"), "byrow");
104  int miss_nr = Rf_asLogical(miss_nr_);
105  int miss_nc = Rf_asLogical(miss_nc_);
106 
107  if (!miss_nr) {
108  if (!Rf_isNumeric(nrow)) Rf_error(_("non-numeric matrix extent"));
109  nr = Rf_asInteger(nrow);
110  if (nr == R_NaInt)
111  Rf_error(_("invalid 'nrow' value (too large or NA)"));
112  if (nr < 0)
113  Rf_error(_("invalid 'nrow' value (< 0)"));
114  }
115  if (!miss_nc) {
116  if (!Rf_isNumeric(ncol)) Rf_error(_("non-numeric matrix extent"));
117  nc = Rf_asInteger(ncol);
118  if (nc == R_NaInt)
119  Rf_error(_("invalid 'ncol' value (too large or NA)"));
120  if (nc < 0)
121  Rf_error(_("invalid 'ncol' value (< 0)"));
122  }
123  if (miss_nr && miss_nc) {
124  if (lendat > INT_MAX) Rf_error("data is too long");
125  nr = int(lendat);
126  } else if (miss_nr) {
127  if (lendat > (double) nc * INT_MAX) Rf_error("data is too long");
128  // avoid division by zero
129  if (nc == 0) {
130  if (lendat) Rf_error(_("nc = 0 for non-null data"));
131  else nr = 0;
132  } else
133  nr = (int) ceil((double) lendat / (double) nc);
134  } else if (miss_nc) {
135  if (lendat > (double) nr * INT_MAX) Rf_error("data is too long");
136  // avoid division by zero
137  if (nr == 0) {
138  if (lendat) Rf_error(_("nr = 0 for non-null data"));
139  else nc = 0;
140  } else
141  nc = (int) ceil((double) lendat / (double) nr);
142  }
143 
144  if(lendat > 0) {
145  R_xlen_t nrc = R_xlen_t(nr) * nc;
146  if (lendat > 1 && nrc % lendat != 0) {
147  if (((lendat > nr) && (lendat / nr) * nr != lendat) ||
148  ((lendat < nr) && (nr / lendat) * lendat != nr))
149  Rf_warning(_("data length [%d] is not a sub-multiple or multiple of the number of rows [%d]"), lendat, nr);
150  else if (((lendat > nc) && (lendat / nc) * nc != lendat) ||
151  ((lendat < nc) && (nc / lendat) * lendat != nc))
152  Rf_warning(_("data length [%d] is not a sub-multiple or multiple of the number of columns [%d]"), lendat, nc);
153  }
154  else if ((lendat > 1) && (nrc == 0)){
155  Rf_warning(_("data length exceeds size of matrix"));
156  }
157  }
158 
159 #ifndef LONG_VECTOR_SUPPORT
160  if (double(nr) * double(nc) > INT_MAX)
161  Rf_error(_("too many elements specified"));
162 #endif
163 
164  PROTECT(ans = Rf_allocMatrix(TYPEOF(vals), nr, nc));
165  if(lendat) {
166  if (Rf_isVector(vals))
167  Rf_copyMatrix(ans, vals, Rboolean(byrow));
168  else
169  Rf_copyListMatrix(ans, vals, Rboolean(byrow));
170  } else if (Rf_isVector(vals)) { /* fill with NAs */
171  R_xlen_t N = R_xlen_t(nr) * nc, i;
172  switch(TYPEOF(vals)) {
173  case STRSXP:
174  for (i = 0; i < N; i++)
175  SET_STRING_ELT(ans, i, R_NaString);
176  break;
177  case LGLSXP:
178  for (i = 0; i < N; i++)
179  LOGICAL(ans)[i] = R_NaLog;
180  break;
181  case INTSXP:
182  for (i = 0; i < N; i++)
183  INTEGER(ans)[i] = R_NaInt;
184  break;
185  case REALSXP:
186  for (i = 0; i < N; i++)
187  REAL(ans)[i] = R_NaReal;
188  break;
189  case CPLXSXP:
190  {
191  Rcomplex na_cmplx;
192  na_cmplx.r = R_NaReal;
193  na_cmplx.i = 0;
194  for (i = 0; i < N; i++)
195  COMPLEX(ans)[i] = na_cmplx;
196  }
197  break;
198  case RAWSXP:
199  if (N) memset(RAW(ans), 0, N);
200  break;
201  default:
202  /* don't fill with anything */
203  ;
204  }
205  }
206  if(!Rf_isNull(dimnames) && Rf_length(dimnames) > 0)
207  ans = Rf_dimnamesgets(ans, dimnames);
208  UNPROTECT(1);
209  return ans;
210 }

412 HIDDEN SEXP do_lengths(/*const*/ Expression* call, const BuiltInFunction* op, Environment* rho, RObject* const* args, int num_args, const PairList* tags)
413 {
414  SEXP x = args[0], ans;
415  R_xlen_t x_len, i;
416  int *ans_elt;
417  int useNames = Rf_asLogical(args[1]);
418  if (useNames == R_NaLog)
419  Rf_error(_("invalid '%s' value"), "use.names");
420  bool isList_ = Rf_isVectorList(x) || Rf_isS4(x);
421  if(!isList_) switch(TYPEOF(x)) {
422  case NILSXP:
423  case CHARSXP:
424  case LGLSXP:
425  case INTSXP:
426  case REALSXP:
427  case CPLXSXP:
428  case STRSXP:
429  case RAWSXP:
430  break;
431  default:
432  Rf_error(_("'%s' must be a list or atomic vector"), "x");
433  }
434  x_len = dispatch_xlength(x, call, rho);
435  PROTECT(ans = Rf_allocVector(INTSXP, x_len));
436  if(isList_) {
437  for (i = 0, ans_elt = INTEGER(ans); i < x_len; i++, ans_elt++) {
438  R_xlen_t x_elt_len = getElementLength(x, i, call, rho);
439 #ifdef LONG_VECTOR_SUPPORT
440  if (x_elt_len > INT_MAX) {
441  ans = do_lengths_long(x, call, rho);
442  UNPROTECT(1);
443  PROTECT(ans);
444  break;
445  }
446 #endif
447  *ans_elt = (int)x_elt_len;
448  }
449  } else { // atomic: every element has length 1
450  for (i = 0, ans_elt = INTEGER(ans); i < x_len; i++, ans_elt++)
451  *ans_elt = 1;
452  }
453  SEXP dim = Rf_getAttrib(x, Symbols::DimSymbol);
454  if(!Rf_isNull(dim)) {
455  Rf_setAttrib(ans, Symbols::DimSymbol, dim);
456  }
457  if(useNames) {
458  SEXP names = Rf_getAttrib(x, Symbols::NamesSymbol);
459  if(!Rf_isNull(names)) Rf_setAttrib(ans, Symbols::NamesSymbol, names);
460  SEXP dimnames = Rf_getAttrib(x, Symbols::DimNamesSymbol);
461  if(!Rf_isNull(dimnames)) Rf_setAttrib(ans, Symbols::DimNamesSymbol, dimnames);
462  }
463  UNPROTECT(1);
464  return ans;
465 }

499 /*
500  Whenever vector x contains NaN or Inf (or -Inf), the function returns TRUE.
501  It can be imprecise: it can return TRUE in other cases as well.
502 
503  A precise version of the function could be implemented as
504 
505  for (R_xlen_t i = 0; i < n; i++)
506  if (!std::isfinite(x[i])) return TRUE;
507  return FALSE;
508 
509  The present version is imprecise, but faster.
510 */
511 static bool mayHaveNaNOrInf(double *x, R_xlen_t n)
512 {
513  if ((n&1) != 0 && !std::isfinite(x[0]))
514  return true;
515  for (R_xlen_t i = n&1; i < n; i += 2)
516  /* A precise version could use this condition:
517  *
518  * !std::isfinite(x[i]+x[i+1]) && (!std::isfinite(x[i]) || !std::isfinite(x[i+1]))
519  *
520  * The present imprecise version has been found to be faster
521  * with GCC and ICC in the common case when the sum of the two
522  * values is always finite.
523  *
524  * The present version is imprecise because the sum of two very
525  * large finite values (e.g. 1e308) may be infinite.
526  */
527  if (!std::isfinite(x[i]+x[i+1]))
528  return true;
529  return false;
530 }

532 /*
533  This is an experimental version that has been observed to run fast on some
534  SIMD hardware with GCC and ICC.
535  Note that the OpenMP reduction assumes associativity of addition, which is
536  safe here, because the result is only used for an imprecise test for
537  the presence of NaN and Inf values.
538 */
539 static Rboolean mayHaveNaNOrInf_simd(double *x, R_xlen_t n)
540 {
541  double s = 0;
542  /* SIMD reduction is supported since OpenMP 4.0. The value of _OPENMP is
543  unreliable in some compilers, so we depend on HAVE_OPENMP_SIMDRED,
544  which is normally set by configure based on a test. */
545  /* _OPENMP >= 201307 */
546 #if defined(_OPENMP) && HAVE_OPENMP_SIMDRED
547  #pragma omp simd reduction(+:s)
548 #endif
549  for (R_xlen_t i = 0; i < n; i++)
550  s += x[i];
551  return Rboolean(!std::isfinite(s));
552 }

554 static bool cmayHaveNaNOrInf(Rcomplex *x, R_xlen_t n)
555 {
556  /* With HAVE_FORTRAN_DOUBLE_COMPLEX set, it should be clear that
557  Rcomplex has no padding, so we could probably use mayHaveNaNOrInf,
558  but better safe than sorry... */
559  if ((n&1) != 0 && (!std::isfinite(x[0].r) || !std::isfinite(x[0].i)))
560  return true;
561  for (R_xlen_t i = n&1; i < n; i += 2)
562  if (!std::isfinite(x[i].r+x[i].i+x[i+1].r+x[i+1].i))
563  return true;
564  return false;
565 }
