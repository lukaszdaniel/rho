61 // This seems based on Markus Kuhn's function but with 1-based 'max'
62 static int wcsearch(int wint, const struct interval *table, int max)
63 {
64  int min = 0;
65  int mid;
66  max--;
67 
68  if (wint < table[0].first || wint > table[max].last)
69  return 0;
70  while (max >= min) {
71  mid = (min + max) / 2;
72  if (wint > table[mid].last)
73  min = mid + 1;
74  else if (wint < table[mid].first)
75  max = mid - 1;
76  else
77  return 1;
78  }
79  return 0;
80 }

82 static int wcwidthsearch(int wint, const struct interval_wcwidth *table,
83  int max, int locale)
84 {
85  int min = 0;
86  int mid;
87  max--;
88 
89  /* This quickly gives one for ASCII characters since the table
90  starts at 0xa0 */
91  if (wint < table[0].first || wint > table[max].last) return 1;
92  while (max >= min) {
93  mid = (min + max) / 2;
94  if (wint > table[mid].last)
95  min = mid + 1;
96  else if (wint < table[mid].first)
97  max = mid - 1;
98  else{
99  return(table[mid].mb[locale]);
100  }
101  }
102  return -1;
103 }

157 // used in character.cpp, ../gnuwin32/console.cpp , ../library/grDevices/src/devP*.cpp :
158 int Ri18n_wcwidth(Rwchar_t c)
159 {
160  char lc_str[128];
161  unsigned int i, j;
162 
163  static const char *lc_cache = "";
164  static int lc = 0;
165 
166  if (0 != strcmp(setlocale(LC_CTYPE, nullptr), lc_cache)) {
167  strncpy(lc_str, setlocale(LC_CTYPE, nullptr), sizeof(lc_str));
168  lc_str[sizeof(lc_str) - 1] = '\0';
169  for (i = 0, j = int(strlen(lc_str)); i < j && i < sizeof(lc_str); i++)
170  lc_str[i] = char(toupper(lc_str[i]));
171  for (i = 0; i < (sizeof(cjk_locale_name)/sizeof(cjk_locale_name_t));
172  i++) {
173  if (0 == strncmp(cjk_locale_name[i].name, lc_str,
174  strlen(cjk_locale_name[i].name))) {
175  lc = cjk_locale_name[i].locale;
176  break;
177  }
178  }
179  }
180 
181  int wd = wcwidthsearch(c, table_wcwidth,
182  (sizeof(table_wcwidth)/sizeof(struct interval_wcwidth)),
183  lc);
184  if (wd >= 0) return wd; // currently all are 1 or 2.
185  int zw = wcsearch(c, zero_width, zero_width_count);
186  return zw ? 0 : 1; // assume unknown chars are width one.
187 }

191 int Ri18n_wcswidth (const wchar_t *s, size_t n)
192 {
193  int rs = 0;
194  while ((n-- > 0) && (*s != L'\0'))
195  {
196  int now = Ri18n_wcwidth (*s);
197  if (now == -1) return -1;
198  rs += now;
199  s++;
200  }
