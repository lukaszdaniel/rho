105 String::~String()
106 {
107  // m_key_val_pr is null for serialization proxies.
108  if (m_key_val_pr)
109  {
110  // Must copy the key, because some implementations may,
111  // having deleted the cache entry pointed to by
112  // m_key_val_pr, continue looking for other entries with
113  // the given key.
114  key k = m_key_val_pr->first;
115  getCache()->erase(k);
116  }
117  // GCNode::~GCNode doesn't know about the string storage space in this
118  // object, so account for it here.
119  size_t bytes = size() + 1;
120  MemoryBank::adjustFreedSize(sizeof(String), sizeof(String) + bytes);
121 }

163 String *String::obtain(const std::string &str, cetype_t encoding)
164 {
165  // This will be checked again when we actually construct the
166  // String, but we precheck now so that we don't create an
167  // invalid cache key:
168  switch (encoding)
169  {
170  case CE_NATIVE:
171  case CE_UTF8:
172  case CE_LATIN1:
173  case CE_BYTES:
174  break;
175  default:
176  Rf_error("unknown encoding: %d", encoding);
177  }
178  bool ascii = rho::isASCII(str);
179  if (ascii)
180  encoding = CE_NATIVE;
181  std::pair<map::iterator, bool> pr = getCache()->insert(map::value_type(key(str, encoding), nullptr));
182  map::iterator it = pr.first;
183  if (pr.second)
184  {
185  try
186  {
187  map::value_type &val = *it;
188  val.second = String::create(str, encoding, ascii);
189  val.second->m_key_val_pr = &*it;
190  }
191  catch (...)
192  {
193  getCache()->erase(it);
194  throw;
195  }
196  }
197  return (*it).second;
198 }
