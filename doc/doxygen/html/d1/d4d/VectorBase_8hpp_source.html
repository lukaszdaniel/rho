305  template <class V>
306  V *VectorBase::resize(const V *pattern, size_type new_size)
307  {
308  GCStackRoot<V> ans(V::create(new_size));
309  size_type copysz = std::min(pattern->size(), new_size);
310  for (size_type i = 0; i < copysz; i++)
311  {
312  (*ans)[i] = ElementTraits::duplicate_element((*pattern)[i]);
313  }
314  for (size_type i = copysz; i < new_size; i++)
315  {
316  (*ans)[i] = ElementTraits::duplicate_element(NA<typename V::value_type>());
317  }
318  ans->setAttributes(resizeAttributes(pattern->attributes(), new_size));
319  ans->setS4Object(pattern->isS4Object());
320  return ans;
321  }
