101 char* S_alloc(long num_elts, int elt_size)
102 {

105  if (elt_size <= 0)
106  Rf_error(_("S_alloc: element size must be positive."));
107  size_t unum_elts = size_t(num_elts);
108  size_t uelt_size = size_t(elt_size);
109  size_t size = unum_elts*uelt_size;

118 char* S_realloc(char *prev_block, long new_sz, long old_sz, int elt_size)
119 {
