97 static unsigned next_log2_32(unsigned size) {
98  if (size == 0) {
99  return 0;
100  }
101 
102  // Feature test macro for GCC builtins. Also available in recent Clangs.
103 #ifndef __has_builtin
104 #define __has_builtin(x) 0
105 #endif
106 
107  // First we figure out the 2-log of the highest set bit.
108  // Use builtin to count leading zeroes if available.
109 #if __has_builtin(__builtin_clz)
110  unsigned log2 = 31 - __builtin_clz(size);
111 #else
112  // Use manual log-2 finding implementation. The builtin version is available
113  // for most compilers, but if not we fall back on this version.
114  //
115  // The algorithm below is inspired by this one for finding the number of
116  // trailing zeroes:
117  // https://graphics.stanford.edu/~seander/bithacks.html#ZerosOnRightBinSearch
118  // The code below works similarly by doing a binary search to find the
119  // number of leading zeroes.
120  unsigned log2 = 0;
121  // This holds remaining bits after shifting out uninteresting bits:
122  unsigned remaining = size;
123  if (remaining & 0xFFFF0000) {
124  // The high 16 bits have at least one bit set.
125  log2 += 16;
126  remaining >>= 16;
127  }
128  if (remaining & 0xFF00) {
129  // The high 8 bits of remaining bits have at least one bit set.
130  log2 += 8;
131  remaining >>= 8;
132  }
133  if (remaining & 0xF0) {
134  // The high 4 bits of remaining bits have at least one bit set.
135  log2 += 4;
136  remaining >>= 4;
137  }
138  // Test the upper 2 bits of the remaining bits.
139  if (remaining & 0xC) {
140  log2 += 2;
141  remaining >>= 2;
142  }
143  // Adjust for the final bits.
144  if (remaining > 0) {
145  log2 += remaining - 1;
146  }
147 #endif // __has_builtin(__builtin_clz)
148 
149  // Now make sure we go one above the highest set bit if the input is larger
150  // than that power of two:
151  if (size > static_cast<unsigned int>(1 << log2)) {
152  log2 += 1;
153  }
154  return log2;
155 }

182 void* rho::GCNodeAllocator::allocate(size_t bytes) {
183  void* result = nullptr;
184 #ifdef HAVE_ADDRESS_SANITIZER
185  // Increase size to include redzones.
186  bytes += 2 * s_redzone_size;
187 #endif
188  unsigned actual_bytes; // The actual allocation size
189  // (may be more than requested).
190  if (bytes <= 256) {
191  int size_class = (bytes + 7) / 8; // Computes ceil(bytes / 8).
192  if (size_class < 4) {
193  // Ensure at least 32-byte blocks for the small-object arena. This is
194  // required both to fit a FreeListNode (20 bytes), and to reduce the
195  // number of bytes needed for the fixed size bitset (as part of the
196  // constant SUPERBLOCK_HEADER_SIZE).
197  size_class = 4;
198  }
199  actual_bytes = size_class * 8;
200  result = removeFromFreelist(size_class);
201  if (!result) {
202  result = AllocatorSuperblock::allocateBlock(actual_bytes);
203  // If allocating in the small object arena fails, we continue
204  // on to using the medium block allocator. The allocation
205  // size will be increased to the minimum allocation size 64 bytes.
206  }
207  }
208  if (!result) {
209  // Now we try to allocate in a large superblock or, if the size
210  // exceeds the superblock threshold, use a separate allocation.
211  // These allocations are rounded up to the next power of two size.
212  unsigned size_log2 = next_log2_32(bytes);
213  if (size_log2 < 6) {
214  // Ensure a minimum allocation size of 64 bytes.
215  size_log2 = 6;
216  }
217  actual_bytes = 1 << size_log2;
218  unsigned size_class = AllocatorSuperblock::sizeClassFromSizeLog2(size_log2);
219  result = removeFromFreelist(size_class);
220  if (!result) {
221  if (size_log2 < s_num_medium_pools) {
222  result = AllocatorSuperblock::allocateLarge(size_log2);
223  } else {
224  result = new double[(1L << size_log2) / sizeof(double)];
225  GCNodeAllocator::s_alloctable->insert(result, size_log2);
226  }
227  // Only update heap bounds if allocating a new block.
228  updateHeapBounds(result, actual_bytes);
229  }
230  }
231  if (!result) {
232  allocerr("failed to allocate object");
233  }
234 
235 #ifdef ALLOCATION_CHECK
236  if (lookup_in_allocation_map(result)) {
237  allocerr("reusing live allocation");
238  }
239  add_to_allocation_map(result, actual_bytes);
240  lookupPointer(result); // Check lookup table consistency.
241 #endif
242 
243 #ifdef HAVE_ADDRESS_SANITIZER
244  // Poison the redzones.
245  // The end redzone can be larger than s_redzone_size if the allocator
246  // grew the allocation larger than the requested size.
247  // The additional size (actual_bytes - bytes) is added to the default
248  // redzone size.
249  void* end_redzone = offsetPointer(result, bytes - s_redzone_size);
250  unsigned end_redzone_size = s_redzone_size + (actual_bytes - bytes);
251  ASAN_POISON_MEMORY_REGION(result, s_redzone_size);
252  ASAN_POISON_MEMORY_REGION(end_redzone, end_redzone_size);
253 
254  // Offset the result pointer past first redzone.
255  result = offsetPointer(result, s_redzone_size);
256 #endif // HAVE_ADDRESS_SANITIZER
257  return result;
258 }

260 void rho::GCNodeAllocator::free(void* pointer) {
261 #ifdef HAVE_ADDRESS_SANITIZER
262  // Adjust for redzone to find the true start of the allocation.
263  pointer = offsetPointer(pointer, -s_redzone_size);
264 
265  // Unpoison the first redzone so we can link it into a freelist.
266  ASAN_UNPOISON_MEMORY_REGION(pointer, sizeof(FreeListNode));
267 #endif
268 #ifdef ALLOCATION_CHECK
269  if (!lookupPointer(pointer)) {
270  allocerr("can not free unknown/already-freed pointer");
271  }
272  remove_from_allocation_map(pointer);
273 #endif
274  uintptr_t pointer_uint = reinterpret_cast<uintptr_t>(pointer);
275  AllocatorSuperblock* superblock =
276  AllocatorSuperblock::arenaSuperblockFromPointer(pointer_uint);
277  if (superblock) {
278  superblock->freeBlock(pointer);
279  } else {
280  // Search for the allocation in the hashtable. If the allocation is in a
281  // superblock, the superblock is left as is but the superblock bitset is
282  // updated to indicate that the block is free.
283  // If the allocation is a large object (ie. not in a superblock), then
284  // the object is taken out of the hashtable and inserted into a freelist.
285  AllocationTable::Allocation* allocation =
286  s_alloctable->search(pointer_uint);
287  if (!allocation) {
288  // Free failed: could not find block to free.
289  allocerr("failed to free pointer - unallocated or double-free problem");
290  }
291  if (allocation->isSuperblock()) {
292  AllocatorSuperblock* superblock = allocation->asSuperblock();
293  superblock->freeBlock(pointer);
294  // Done. Don't erase hash entries for the superblock.
295  } else if (allocation->asPointer() == pointer) {
296  // Erase all entries in hashtable for the allocation.
297  s_alloctable->erase(pointer_uint, allocation->sizeLog2());
298  // Add allocation to free list.
299  addLargeAllocationToFreelist(pointer, allocation->sizeClass());
300  } else {
301  allocerr("failed to free pointer - unallocated or double-free problem");
302  }
303  }
304 }

320 rho::GCNode* rho::GCNodeAllocator::lookupPointer(void* candidate) {
321  uintptr_t candidate_uint = reinterpret_cast<uintptr_t>(candidate);
322  void* result = AllocatorSuperblock::lookupAllocation(candidate_uint);
323  if (!result
324  && (candidate_uint >= s_heap_start && candidate_uint < s_heap_end)) {
325  // Lookup a pointer in the hashtable.
326  AllocationTable::Allocation* allocation =
327  s_alloctable->search(candidate_uint);
328  if (allocation) {
329  if (allocation->isSuperblock()) {
330  AllocatorSuperblock* superblock = allocation->asSuperblock();
331  result = superblock->lookupBlock(candidate_uint);
332  } else {
333  result = allocation->asPointer();
334  }
335  }
336  }
337 #ifdef ALLOCATION_CHECK
338  if (result != lookup_in_allocation_map(candidate)) {
339  if (!result) {
340  allocerr("allocation map mismatch: did not find existing node");
341  } else {
342  allocerr("allocation map mismatch");
343  }
344  }
345 #endif
346 #ifdef HAVE_ADDRESS_SANITIZER
347  // Adjust for redzone.
348  if (result) {
349  result = offsetPointer(result, s_redzone_size);
350  }
351 #endif
352  return static_cast<GCNode*>(result);
353 }

361 void rho::GCNodeAllocator::addToFreelist(FreeListNode* free_node,
362  unsigned size_class) {
363 #ifdef HAVE_ADDRESS_SANITIZER
364  addToQuarantine(free_node, size_class);
365 #else
366  if (size_class >= s_num_small_pools + s_num_medium_pools) {
367  // NOTE: large allocations are freed directly instead of inserting in a
368  // freelist. This is to avoid the memory cost of keeping large allocations
369  // around. This may be changed in the future so that the freelist is used
370  // again for large objects.
371  delete[] free_node;
372  } else {
373  free_node->m_next = s_freelists[size_class];
374  s_freelists[size_class] = free_node;
375  }
376 #endif
377 }

379 void* rho::GCNodeAllocator::removeFromFreelist(unsigned size_class) {
380  FreeListNode* node = s_freelists[size_class];
381  if (node) {
382 #ifdef HAVE_ADDRESS_SANITIZER
383  ASAN_UNPOISON_MEMORY_REGION(node, bytesFromSizeClass(size_class));
384 #endif
385  s_freelists[size_class] = node->m_next;
386 
387  if (node->m_superblock) {
388  // Tag the resued allocation as allocated since it is part of a
389  // superblock.
390  node->m_superblock->tagBlockAllocated(node->m_block);
391  } else {
392  // This is a non-superblock allocation. Insert it back into the
393  // allocation table.
394  s_alloctable->insert(static_cast<void*>(node),
395  sizeLog2FromSizeClass(size_class));
396  }
397  }
398  return static_cast<void*>(node);
399 }

448  void* lookup_in_allocation_map(void* tentative_pointer) {
449  // Find the largest key less than or equal to tentative_pointer.
450  allocation_map::const_iterator next_allocation =
451  allocations.upper_bound(tentative_pointer);
452  if (next_allocation != allocations.begin()) {
453  allocation_map::const_iterator allocation = std::prev(next_allocation);
454 
455  // Check that tentative_pointer is before the end of the allocation.
456  void* allocation_end = allocation->second;
457  if (tentative_pointer < allocation_end) {
458  return allocation->first;
459  }
460  }
461  return nullptr;
462  }
463 }

472 void rho::GCNodeAllocator::addToQuarantine(rho::FreeListNode* free_node,
473  unsigned size_class) {
474  // Add an allocation to the quarantine and poison it.
475  static unsigned quarantine_size = 0;
476 
477  size_t block_size = bytesFromSizeClass(size_class);
478  free_node->m_next = s_quarantine[size_class];
479  s_quarantine[size_class] = free_node;
480  ASAN_POISON_MEMORY_REGION(free_node, block_size);
481 
482  // Clear the quarantine if it has grown too large.
483  quarantine_size += block_size;
484  if (quarantine_size > s_max_quarantine_size) {
485  quarantine_size = 0;
486  clearQuarantine();
487  }
488 }

498 void rho::GCNodeAllocator::clearQuarantineBySizeClass(int size_class) {
499  if (size_class >= s_num_small_pools + s_num_medium_pools) {
500  // NOTE: large allocations are freed instead of inserting in a
501  // freelist. This is to avoid the memory cost of keeping large allocations
502  // around. This may be changed in the future so that the freelist is used
503  // again for large objects.
504  FreeListNode* next_node = s_quarantine[size_class];
505  while (next_node) {
506  FreeListNode* free_node = next_node;
507 
508  // Unpoison the quarantined allocation so we can read the next link.
509  ASAN_UNPOISON_MEMORY_REGION(free_node, sizeof(FreeListNode));
510  next_node = free_node->m_next;
511  delete[] free_node;
512  }
513  s_quarantine[size_class] = nullptr;
514  } else {
515  FreeListNode* tail_node = nullptr;
516  FreeListNode* next_node = s_quarantine[size_class];
517  while (next_node) {
518  tail_node = next_node;
519 
520  // Unpoison the quarantined allocation so we can read the next link.
521  ASAN_UNPOISON_MEMORY_REGION(tail_node, sizeof(FreeListNode));
522  next_node = tail_node->m_next;
523  // Re-poison so the freelist nodes stay poisoned while free.
524  ASAN_POISON_MEMORY_REGION(tail_node, sizeof(FreeListNode));
525  }
526  if (tail_node) {
527  // Unpoison the last node so we can write the next link.
528  ASAN_UNPOISON_MEMORY_REGION(tail_node, sizeof(FreeListNode));
529  tail_node->m_next = s_freelists[size_class];
530  // Re-poison so the freelist nodes stay poisoned while free.
531  ASAN_POISON_MEMORY_REGION(tail_node, sizeof(FreeListNode));
532 
533  s_freelists[size_class] = s_quarantine[size_class];
534  s_quarantine[size_class] = nullptr;
535  }
536  }
537 }
