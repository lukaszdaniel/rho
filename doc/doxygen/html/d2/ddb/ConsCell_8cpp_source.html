106 void rho::ccdump(std::ostream& os, const ConsCell& cc, std::size_t margin)
107 {
108  indent(os, margin);
109  os << Rf_type2char(cc.sexptype()) << '\n';
110  for (const ConsCell* p = &cc; p; p = p->tail()) {
111  // Print tag:
112  indent(os, margin);
113  os << "- ";
114  const RObject* tag = p->tag();
115  if (!tag)
116  os << "(No tag):\n";
117  else if (tag->sexptype() != SYMSXP)
118  os << "(Tag not a SYMSXP):\n";
119  else os << sympname(tag) << ":\n";
120  // Print car:
121  const RObject* car = p->car();
122  if (const ConsCell* ccinner = dynamic_cast<const ConsCell*>(car))
123  ccdump(os, *ccinner, margin + 2);
124  else if (const StringVector* sv
125  = dynamic_cast<const StringVector*>(car))
126  strdump(os, *sv, margin + 2);
127  else {
128  indent(os, margin + 2);
129  if (!car)
130  os << "NILSXP\n";
131  else {
132  SEXPTYPE st = car->sexptype();
133  os << Rf_type2char(st);
134  if (st == SYMSXP)
135  os << ": " << sympname(car);
136  os << '\n';
137  }
138  }
139  }
