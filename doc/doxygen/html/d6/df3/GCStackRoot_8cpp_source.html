77 void GCStackRootBase::visitRoots(GCNode::const_visitor* visitor,
78  const void* start_ptr,
79  const void* end_ptr)
80 {
81  uintptr_t start = start_ptr ? reinterpret_cast<uintptr_t>(start_ptr)
82  : reinterpret_cast<uintptr_t>(getStackBase());
83  uintptr_t end = reinterpret_cast<uintptr_t>(end_ptr);
84 
85 
86 #ifdef STACK_GROWS_UP
87  for (uintptr_t stack_pointer = start; stack_pointer < end;
88  stack_pointer += alignof(void*))
89 #else
90  for (uintptr_t stack_pointer = start; stack_pointer > end;
91  stack_pointer -= alignof(void*))
92 #endif
93  {
94  void* candidate_pointer = *reinterpret_cast<void**>(stack_pointer);
95  GCNode* node = GCNode::asGCNode(candidate_pointer);
96  if (node) {
97  (*visitor)(node);
98  }
99  }
100 }
