45  // While the contents of the stack below the boundary are assumed to not change,
46  // the set of pointers found by the conservative stack scanner in that region
47  // can. To ensure that the same nodes are protected and unprotected, we keep
48  // a separate stack containing just the protected nodes here.
49  // Also note that this only stores the first instance of each node.

91  void GCStackFrameBoundary::advanceBarrier()
92  {
93  BoundaryStack::iterator frame_start = s_boundaries.iterator_to(*s_barrier);
94  BoundaryStack::iterator frame_end = std::next(frame_start);
95  while (frame_end != s_boundaries.end())
96  {
97  advanceBarrierOneFrame(&*frame_start, &*frame_end);
98  frame_start = frame_end;
99  ++frame_end;
100  }
101 
102  // Move the barrier.
103  s_barrier = &s_boundaries.back();
104  }

106  void GCStackFrameBoundary::advanceBarrierOneFrame(
107  GCStackFrameBoundary *frame_start, GCStackFrameBoundary *frame_end)
108  {
109  ProtectPointerVisitor visitor(&frame_end->m_num_protected_pointers);
110 
111  GCStackRootBase::visitRoots(&visitor,
112  frame_start->getStackPointer(),
113  frame_end->getStackPointer());
114  }

116  void GCStackFrameBoundary::applyBarrier()
117  {
118  assert(s_barrier == this);
119 
120  // Pop off the protected nodes.
121  for (int i = 0; i < m_num_protected_pointers; i++)
122  {
123  const GCNode *pointer = s_protected_nodes.top();
124  s_protected_nodes.pop();
125  pointer->clearOnStackBit();
126  }
127 
128  // Move the barrier back.
129  BoundaryStack::iterator current_it = s_boundaries.iterator_to(*this);
130  assert(std::next(current_it) == s_boundaries.end());
131  s_barrier = &*std::prev(current_it);
132  }
